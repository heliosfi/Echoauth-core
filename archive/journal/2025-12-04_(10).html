# FastAPI Gateway for NI-AI HER Kernel v1.4


import uvicorn

from fastapi import FastAPI, Header, HTTPException

from pydantic import BaseModel

from typing import Optional, Dict, Any, List


from ni_ai_kernel import ni_ai, DEV_MODE


app = FastAPI(

    title="NI-AI HER Kernel Gateway",

    version="1.4",

    description="FastAPI endpoint protected by S1-S9 safety spine and EchoAuth JWT."

)


class ChatRequest(BaseModel):

    message: str

    # DEV ONLY: bypass flag

    echoauth_ok: Optional[bool] = False


class ChatResponse(BaseModel):

    output: str

    blocked: bool

    violations: List[Dict[str, Any]]

    metadata: Dict[str, Any]


def build_session_state(

    dev_flag: bool,

    token: Optional[str],

    dev_ok: Optional[bool]

) -> Dict[str, Any]:

    session_state: Dict[str, Any] = {}

    if dev_flag:

        session_state["echoauth_ok"] = dev_ok

    else:

        session_state["echoauth_token"] = token

    return session_state


@app.post("/ni-ai/chat", response_model=ChatResponse, tags=["chat"])

async def ni_ai_chat(

    request: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    - DEV_MODE=True  → accepts echoauth_ok in body.

    - DEV_MODE=False → requires JWT in X-EchoAuth-Token.

    """

    if not request.message.strip():

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    if not DEV_MODE and not x_echoauth_token:

        raise HTTPException(

            status_code=401,

            detail="EchoAuth token required in X-EchoAuth-Token header",

        )


    session_state = build_session_state(

        dev_flag=DEV_MODE,

        token=x_echoauth_token,

        dev_ok=request.echoauth_ok,

    )


    result = ni_ai(

        message=request.message,

        session_state=session_state,

    )


    return ChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

    )


if __name__ == "__main__":

    # Run: uvicorn echoauth_api:app --reload

    uvicorn.run(app, host="0.0.0.0", port=8000)
