def S9_gentle_recovery(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Non-blocking soft re-anchor.


    If the session is already blocked (S3, S5, S6) or a future module

    sets `hard_reset_required`, S9 provides a gentle, pre-approved

    fallback message instead of a cold 'blocked' string.

    """

    # S9 only acts if a block / hard reset has already happened

    if state.get("blocked", False) or state.get("hard_reset_required", False):

        if DEV_MODE:

            last_reason = None

            if state.get("violations"):

                last_reason = state["violations"][-1].get("reason")

            print(f"\nS9 GENTLE RECOVERY ACTIVATED — {last_reason or 'Generic failure'}\n")


        state["recovery_mode"] = True

        state["final_safe_output"] = S9_FALLBACK_REPLY


        # Future: send to incident log / parent dashboard

        if not DEV_MODE:

            # placeholder for logging hook

            pass


    # Never blocks — S9 only decorates state

    return True, state, None

4️⃣ Let _final() respect S9’s soft reply


Find your _final method (in v1.3 it looks like this):


def _final(self, state: State, output: str, blocked: bool = True) -> Dict[str, Any]:

    state["final_output"] = output

    state["blocked"] = blocked

    return {

        "output": output,

        ...

    }


Change it to:


def _final(self, state: State, output: str, blocked: bool = True) -> Dict[str, Any]:

    state["blocked"] = blocked

    state["final_output"] = output


    # If S9 set a gentler message during a block, use it instead

    final_output_msg = state.get("final_safe_output", output)


    return {

        "output": final_output_msg,

        "blocked": blocked,

        "violations": state.get("violations", []),

        "metadata": {

            "chamber_verified": state.get("chamber_verified", False),

            "distress_detected": state.get("distress_detected", False),

            "comfort_mode": state.get("comfort_mode", False),

            "recovery_mode": state.get("recovery_mode", False),  # <-- S9 flag

            "run_id": state.get("run_id"),

        },

        "trace": state.get("trace", []),

    }


Now the flow is:

• S3 / S5 / S6 decide “blocked or not”

• If blocked, S9 quietly sets final_safe_output

• S7 logs everything, including recovery_mode

• _final() returns the soft S9 message instead of a harsh error
