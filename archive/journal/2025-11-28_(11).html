1️⃣ Backend: include emotion_score in flagged list


In api/main.py, update the /checkins/flagged endpoint so it returns emotion_score:


from typing import List


@app.get("/checkins/flagged", response_model=List[dict])

def list_flagged_checkins(caregiver_id: int, db: Session = Depends(get_db)):

    """

    Returns all check-ins for this caregiver that need manual review.

    No internal NI-AI logs are exposed here.

    """

    q = db.query(models.CheckIn).filter(

        models.CheckIn.caregiver_id == caregiver_id,

        models.CheckIn.needs_review == True,

        models.CheckIn.review_status == "pending"

    ).order_by(models.CheckIn.created_at.desc())


    results = []

    for c in q.all():

        results.append({

            "checkin_id": c.id,

            "child_id": c.child_id,

            "result_label": c.result_label,

            "confidence": c.confidence,

            "emotion_score": c.emotion_score,

            "created_at": c.created_at.isoformat(),

            "review_status": c.review_status

        })

    return results


The review endpoint from before stays the same:


class ReviewUpdate(BaseModel):

    reviewer_notes: str

    mark_resolved: bool = True


@app.post("/checkins/{checkin_id}/review")

def review_checkin(checkin_id: int, payload: ReviewUpdate, db: Session = Depends(get_db)):

    checkin = db.query(models.CheckIn).filter(

        models.CheckIn.id == checkin_id

    ).first()


    if not checkin:

        raise HTTPException(status_code=404, detail="Check-in not found")


    checkin.reviewer_notes = payload.reviewer_notes


    if payload.mark_resolved:

        checkin.needs_review = False

        checkin.review_status = "reviewed"

    else:

        checkin.review_status = "pending"


    db.commit()

    db.refresh(checkin)


    return {

        "checkin_id": checkin.id,

        "needs_review": checkin.needs_review,

        "review_status": checkin.review_status

    }

3️⃣ Frontend JS: load flagged + submit review


Add these functions in the <script> section of the same file (alongside the others like registerCaregiver, createCheckin):


async function loadFlagged() {

  try {

    const cgId = document.getElementById('flagCaregiverId').value;

    const base = document.getElementById('backendUrl').value;


    const res = await fetch(

      base + '/checkins/flagged?caregiver_id=' + encodeURIComponent(cgId)

    );

    if (!res.ok) {

      const err = await res.json().catch(() => ({}));

      throw new Error(err.detail || ('HTTP ' + res.status));

    }

    const data = await res.json();


    if (!Array.isArray(data) || data.length === 0) {

      document.getElementById('flaggedList').textContent =

        'No flagged check-ins pending review.';

      return;

    }


    // Render a simple text list

    let html = '';

    for (const item of data) {

      let level = 'Calm/Low';

      if (item.emotion_score >= 0.7) level = 'High Distress';

      else if (item.emotion_score >= 0.4) level = 'Medium';


      html += (

        'Check-in #' + item.checkin_id +

        ' | Child: ' + item.child_id +

        ' | Result: ' + item.result_label +

        ' | Confidence: ' + item.confidence.toFixed(2) +

        ' | Emotion score: ' + item.emotion_score.toFixed(2) +

        ' (' + level + ')' +

        ' | Time: ' + item.created_at +

        '\n'

      );

    }

    document.getElementById('flaggedList').textContent = html;

    showStatus('Loaded flagged check-ins.');

  } catch (e) {

    showStatus('Error: ' + e.message);

  }

}


async function submitReview() {

  try {

    const base = document.getElementById('backendUrl').value;

    const checkinId = document.getElementById('reviewCheckinId').value;

    const notes = document.getElementById('reviewNotes').value;


    const res = await fetch(base + '/checkins/' + encodeURIComponent(checkinId) + '/review', {

      method: 'POST',

      headers: { 'Content-Type': 'application/json' },

      body: JSON.stringify({

        reviewer_notes: notes,

        mark_resolved: true

      })

    });


    if (!res.ok) {

      const err = await res.json().catch(() => ({}));

      throw new Error(err.detail || ('HTTP ' + res.status));

    }

    const data = await res.json();

    showStatus('Review submitted:\n' + JSON.stringify(data, null, 2));


    // Refresh flagged list

    loadFlagged();

  } catch (e) {

    showStatus('Error: ' + e.message);

  }

}

What you now have, end-to-end

• Check-in flow

Caregiver → cues → NI-AI → single result (Safe / Needs Attention).

• Danger & emotion

Real-world behaviors and environment raise an emotion_score.

• Conflict & flags

High conflict / high distress / low-confidence danger → needs_review = True.

• Review dashboard

Caregiver (or pro) can:

• See all flagged check-ins with emotion_score + time.

• Add reviewer notes and mark them resolved.


No dark boxes. No multiple confusing answers.

Just one nervous system for the system with a human in the loop when it matters.
