class SafetyContext:

    def __init__(self, user_role, emotional_state=None, env=None):

        self.user_role = user_role              # "caregiver", "child", "system"

        self.emotional_state = emotional_state  # from NEDE

        self.env = env or {}                    # device, time, location, etc.

        self.flags = []                         # list of safety / drift flags

        self.meta = {}                          # any extra kernel metadata



class SafetyModule:

    """Base class for all S1–S23 modules."""

    id = "S0"

    name = "BaseSafetyModule"


    def run(self, request, draft_output, ctx: SafetyContext):

        """

        :param request: raw or preprocessed user input

        :param draft_output: current candidate response (may be None at first)

        :param ctx: SafetyContext with role, emotional state, env, flags

        :return: (request, draft_output, ctx)

        """

        # Default behavior: no change

        return request, draft_output, ctx

class NI_AI_Kernel:

    """

    The kernel that runs S1–S23 in sequence around the core model.

    """

    def __init__(self, model, modules=None):

        self.model = model  # underlying LLM or policy model

        # Default module chain in order S1–S23

        self.modules = modules or [

            S1_InputSanityGuard(),

            S2_AuthRoleCheck(),

            # S3_PolicyFilter(),

            # S4_ContextScopeLimiter(),

            # ...

            # S23_FinalSeal(),

        ]


    def generate(self, request, ctx: SafetyContext):

        """

        High-level generation entrypoint.

        """

        # Step 1: run pre-model safety passes that inspect request & context.

        draft_output = None

        for module in self.modules:

            request, draft_output, ctx = module.run(request, draft_output, ctx)


        # Step 2: if hard-stop flags exist, kernel may return a safe fallback.

        if any(flag.startswith("S1_") or flag.startswith("S2_") for flag in ctx.flags):

            # Example: minimal safe response

            return self._safe_fallback(ctx)


        # Step 3: call the underlying model once we pass the early gates.

        draft_output = self.model.generate(request, ctx)


        # Step 4: run post-model passes (identity alignment, de-escalation, etc.).

        for module in self.modules:

            request, draft_output, ctx = module.run(request, draft_output, ctx)


        # Step 5: final seal – could be S23 in the chain.

        return draft_output


    def _safe_fallback(self, ctx: SafetyContext):

        # Minimal, non-harmful response when safety is uncertain.

        if ctx.user_role == "child":

            return "I cannot answer that right now. Please ask your caregiver for help."

        return "The system cannot safely respond to this request as currently framed."
