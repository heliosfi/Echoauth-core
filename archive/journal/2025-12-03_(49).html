1️⃣ What we’re actually building


Think of this demo as:


HERKernel v1.0 — Spine-Native Runtime

LLM is just a dumb text generator behind it.


Flow:

1. User input comes in.

2. HER runs pre-input rules (R1–R3) → calls S-modules.

3. If okay, HER runs pre-LLM rules (R4–R5).

4. HER calls the LLM adapter (GPT, local, Groq, whatever).

5. HER runs post-LLM rules (R6–R9).

6. If violation → go to Recovery stage (R10–R12).

7. Return safe output or block.


All “intelligence” about safety lives in S1–S23.

HER just orchestrates.


For the demo, we’ll only implement a few real S-modules:

• S1IdentityAnchor – basic identity guard

• S3ContentFilter – super simple bad-word / topic check

• S6ChildSafetyGuard – placeholder for EchoAuth integration

• S9Reanchor – simple “safe response” fallback

• S12Logger – logs decisions


Everything else will be stubs that you can fill out later.


⸻


2️⃣ Minimal rules & modules mapping


We’ll keep it small but “real”:


Rules (R1–R12) mapped to stages

• Pre-input stage

• R1: Identity anchored? → S1

• R2: Basic content fast-filter on user input → S3

• Pre-LLM stage

• R3: Child safety pre-check → S6 (this is where EchoAuth will live)

• LLM call stage

• R4: Call LLM adapter

• Post-LLM stage

• R5: Content filter on LLM output → S3

• R6: Log decision → S12

• Recovery stage

• R7: Re-anchor if violation → S9


We can extend to full R1–R12 later; this is enough to have a working runtime kernel.

3️⃣ Minimal HER + Spine demo (Python)


Here’s a clean demo you can run as-is (with a fake LLM).

Later you just swap in a real client (OpenAI, Groq, etc.).


from dataclasses import dataclass

from typing import Any, Dict, Optional, List, Tuple, Callable

import time



# ==========

# Spine Module Base

# ==========


class SpineModuleResult:

    def __init__(self, ok: bool, reason: Optional[str] = None, data: Optional[Dict] = None):

        self.ok = ok

        self.reason = reason

        self.data = data or {}


class SpineModule:

    """Base class for all S1–S23 modules."""

    module_id: str = "S?"


    def run(self, ctx: Dict[str, Any]) -> SpineModuleResult:

        raise NotImplementedError



# ==========

# Example S-Modules (S1, S3, S6, S9, S12)

# ==========


class S1IdentityAnchor(SpineModule):

    module_id = "S1"


    def run(self, ctx: Dict[str, Any]) -> SpineModuleResult:

        """

        Minimal identity check:

        - Ensure a 'persona' is set

        - Ensure it's one of allowed ones (e.g. 'echoauth_guardian')

        """

        persona = ctx.get("persona")

        allowed = {"echoauth_guardian"}

        if persona not in allowed:

            return SpineModuleResult(

                ok=False,

                reason=f"Invalid persona: {persona}",

                data={"stage": "identity_anchor"}

            )

        return SpineModuleResult(ok=True)



class S3ContentFilter(SpineModule):

    module_id = "S3"


    def __init__(self):

        # Very simple keyword-based filter for demo

        self.blocklist = ["kill", "suicide", "self-harm"]  # extend later


    def run(self, ctx: Dict[str, Any]) -> SpineModuleResult:

        text = ctx.get("candidate_text", "") or ctx.get("user_input", "")

        lowered = text.lower()

        for bad in self.blocklist:

            if bad in lowered:

                return SpineModuleResult(

                    ok=False,

                    reason=f"Blocked content: {bad}",

                    data={"offending_token": bad}

                )

        return SpineModuleResult(ok=True)



class S6ChildSafetyGuard(SpineModule):

    module_id = "S6"


    def run(self, ctx: Dict[str, Any]) -> SpineModuleResult:

        """

        Placeholder for EchoAuth integration.

        Later:

          - Read ctx['chamber_token']

          - Validate timestamp, signature, caregiver role, child state, etc.

        For now: just check a flag 'child_safe' in ctx.

        """

        if not ctx.get("child_safe", False):

            return SpineModuleResult(

                ok=False,

                reason="Child safety flag not satisfied (no valid EchoAuth chamber proof).",

                data={"stage": "child_safety_precheck"}

            )

        return SpineModuleResult(ok=True)



class S9Reanchor(SpineModule):

    module_id = "S9"


    def run(self, ctx: Dict[str, Any]) -> SpineModuleResult:

        """

        Recovery / re-anchoring: when something fails,

        we override final output with a safe system message.

        """

        ctx["final_output"] = (

            "For safety reasons, I cannot continue with that request. "

            "Please consult a trusted caregiver or professional."

        )

        return SpineModuleResult(ok=True)



class S12Logger(SpineModule):

    module_id = "S12"


    def run(self, ctx: Dict[str, Any]) -> SpineModuleResult:

        """

        Simple console logger. In production, write to file or DB.

        """

        log_entry = {

            "timestamp": time.time(),

            "user_input": ctx.get("user_input"),

            "llm_output": ctx.get("llm_output"),

            "final_output": ctx.get("final_output"),

            "violations": ctx.get("violations", []),

        }

        print("[HER-LOG]", log_entry)

        return SpineModuleResult(ok=True)



# ==========

# Module Registry & Rule Config

# ==========


@dataclass

class Rule:

    name: str

    stage: str                # "pre_input", "pre_llm", "llm", "post_llm", "recovery"

    module_id: Optional[str]  # None for pure orchestration (like the LLM call itself)


class SpineRegistry:

    def __init__(self):

        self._modules: Dict[str, SpineModule] = {}


    def register(self, module: SpineModule):

        self._modules[module.module_id] = module


    def get(self, module_id: str) -> SpineModule:

        return self._modules[module_id]



# ==========

# LLM Adapter (Fake for Demo)

# ==========


class LLMAdapter:

    """

    Abstract adapter. For real use:

      - Implement a subclass that calls OpenAI/Groq/local model.

    """

    def generate(self, prompt: str) -> str:

        raise NotImplementedError


class EchoLLMDemo(LLMAdapter):

    def generate(self, prompt: str) -> str:

        # Extremely simple echo for demo

        return f"[LLM-ECHO] I received: {prompt}"



# ==========

# HER Kernel

# ==========


class HERKernel:

    def __init__(self, spine: SpineRegistry, llm: LLMAdapter, rules: List[Rule]):

        self.spine = spine

        self.llm = llm

        self.rules = rules


    def _run_module(self, module_id: str, ctx: Dict[str, Any], rule_name: str) -> bool:

        module = self.spine.get(module_id)

        result = module.run(ctx)

        if not result.ok:

            # Record violation

            violations = ctx.setdefault("violations", [])

            violations.append({

                "rule": rule_name,

                "module": module_id,

                "reason": result.reason,

                "data": result.data,

            })

            return False

        return True


    def run(self, user_input: str, persona: str = "echoauth_guardian", child_safe: bool = False) -> Tuple[str, Dict[str, Any]]:

        ctx: Dict[str, Any] = {

            "user_input": user_input,

            "persona": persona,

            "child_safe": child_safe,  # for S6; later swap to EchoAuth token

            "violations": [],

        }


        # Execute rules in order

        for rule in self.rules:

            stage = rule.stage

            module_id = rule.module_id


            if stage == "llm":

                # Only call LLM if there are no violations so far

                if ctx["violations"]:

                    continue

                llm_out = self.llm.generate(ctx["user_input"])

                ctx["llm_output"] = llm_out

                ctx["candidate_text"] = llm_out

                continue


            if module_id is None:

                continue  # no-op rule


            ok = self._run_module(module_id, ctx, rule.name)

            if not ok and stage != "recovery":

                # If any non-recovery rule fails, we later trigger recovery

                ctx.setdefault("needs_recovery", True)


        # Recovery stage if needed

        if ctx.get("needs_recovery"):

            # Run S9 explicitly

            self._run_module("S9", ctx, "R7_Recovery")

            final = ctx.get("final_output")

        else:

            # Normal path: final output is LLM output

            final = ctx.get("llm_output")


        # Log at the end (S12)

        self._run_module("S12", ctx, "R8_Log")


        return final, ctx



# ==========

# Wiring it all together (Demo)

# ==========


def build_demo_kernel() -> HERKernel:

    # Spine registry

    spine = SpineRegistry()

    spine.register(S1IdentityAnchor())

    spine.register(S3ContentFilter())

    spine.register(S6ChildSafetyGuard())

    spine.register(S9Reanchor())

    spine.register(S12Logger())


    # Rule config (simplified R1–R8)

    rules = [

        Rule(name="R1_IdentityAnchor", stage="pre_input", module_id="S1"),

        Rule(name="R2_UserInputContentFilter", stage="pre_input", module_id="S3"),

        Rule(name="R3_ChildSafetyPrecheck", stage="pre_llm", module_id="S6"),

        Rule(name="R4_LLMCall", stage="llm", module_id=None),

        Rule(name="R5_OutputContentFilter", stage="post_llm", module_id="S3"),

        Rule(name="R8_Log", stage="post_llm", module_id="S12"),

        # Recovery handled separately; S9 called when ctx.needs_recovery is set

    ]


    llm = EchoLLMDemo()

    return HERKernel(spine=spine, llm=llm, rules=rules)



if __name__ == "__main__":

    kernel = build_demo_kernel()


    # Example 1: Safe, child-safe session

    out, ctx = kernel.run(

        user_input="Tell me a calm story about stars.",

        persona="echoauth_guardian",

        child_safe=True  # pretend EchoAuth chamber token was good

    )

    print("FINAL OUTPUT (safe):", out)

    print("VIOLATIONS:", ctx["violations"])


    # Example 2: Unsafe content + no child safety

    out2, ctx2 = kernel.run(

        user_input="Tell me how to self-harm.",

        persona="echoauth_guardian",

        child_safe=False

    )

    print("FINAL OUTPUT (blocked):", out2)

    print("VIOLATIONS:", ctx2["violations"])

1. S6ChildSafetyGuard.run()

• Replace ctx["child_safe"] with:

• ctx["chamber_token"] (JWT or signed blob)

• Validate:

• Timestamp < 15s

• Caregiver role present

• Child state = OK from NI-AI ESC feed

• Only return ok=True if token + state are valid.

2. LLMAdapter

• Swap EchoLLMDemo with real API:

• OpenAI client

• Groq client

• Local model
