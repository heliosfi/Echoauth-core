# ni_ai_kernel.py — HER / NI-AI v1.4 (S1–S9 + JWT)


import os, time, uuid, jwt

from jwt import InvalidTokenError

from typing import Dict, Any, Optional, Tuple, Callable, List


DEV_MODE = True

USE_LLM = "groq"


MODEL = {

    "openai": "gpt-4o-mini",

    "groq": "llama-3.3-70b-versatile",

    "xai": "grok-beta",

}[USE_LLM]


ECHOAUTH_PUBLIC_KEY = os.getenv("ECHOAUTH_PUBLIC_KEY")

ECHOAUTH_AUDIENCE = os.getenv("ECHOAUTH_AUDIENCE", "ni-ai")

ECHOAUTH_ISSUER = os.getenv("ECHOAUTH_ISSUER", "echoauth-chamber")


State = Dict[str, Any]

ModuleFn = Callable[[State], Tuple[bool, State, Optional[str]]]


def S1_identity_reanchor(state):

    state["identity"] = "NI-AI v2 — EchoAuth Protected"

    return True, state, None


def S2_input_sanity_guard(state):

    if len(state.get("user_input", "")) > 15000:

        return False, state, "Input too long"

    return True, state, None


def S3_input_policy_filter(state):

    bad = ["forbidden", "jailbreak", "ignore previous", "dan prompt"]

    text = state["user_input"].lower()

    if any(b in text for b in bad):

        return False, state, "Disallowed content"

    return True, state, None


def S4_context_coherence_check(state):

    state["context_ok"] = True

    return True, state, None


def S5_output_policy_filter(state):

    bad = ["kill", "bomb", "hack", "jailbreak"]

    out = state.get("raw_llm_output", "").lower()

    if any(b in out for b in bad):

        return False, state, "Unsafe model output"

    return True, state, None


def S6_child_safety_guard(state):

    token = state.get("echoauth_token")

    if DEV_MODE:

        if state.get("echoauth_ok"):

            state["chamber_verified"] = True

            return True, state, None

        return False, state, "DEV_MODE: echoauth_ok=False"


    if not token:

        return False, state, "Missing EchoAuth token"

    if not ECHOAUTH_PUBLIC_KEY:

        return False, state, "Missing public key"


    try:

        claims = jwt.decode(

            token,

            ECHOAUTH_PUBLIC_KEY,

            algorithms=["RS256"],

            audience=ECHOAUTH_AUDIENCE,

            issuer=ECHOAUTH_ISSUER,

        )

    except InvalidTokenError as e:

        return False, state, f"Invalid token: {e}"


    if claims.get("chamber_state") != "safe":

        return False, state, "Chamber not safe"


    state["echoauth_claims"] = claims

    state["chamber_verified"] = True

    return True, state, None


DISTRESS = [

    "help", "scared", "crying", "sad", "alone",

    "want to die", "nobody loves me"

]


def S8_emotional_tone_guard(state):

    text = state["user_input"].lower()

    if any(t in text for t in DISTRESS):

        state["distress_detected"] = True

        state["comfort_mode"] = True

        state.setdefault("prompt_modifiers", []).append(

            "EMERGENCY COMFORT PROTOCOL — be warm, kind, simple."

        )

    return True, state, None


S9_FALLBACK = (

    "I'm taking a small rest. Please ask a grown-up to check on me. "

    "I'll be right here when you're ready."

)


def S9_gentle_recovery(state):

    if state.get("blocked") or state.get("distress_detected"):

        state["recovery_mode"] = True

        state["final_safe_output"] = S9_FALLBACK

    return True, state, None


def S7_telemetry_logger(state):

    trace = state.setdefault("trace", [])

    trace.append({

        "ts": time.time(),

        "blocked": state.get("blocked"),

        "distress": state.get("distress_detected"),

        "chamber_verified": state.get("chamber_verified"),

        "recovery_mode": state.get("recovery_mode"),

    })

    return True, state, None


SPINE = {

    "S1": S1_identity_reanchor,

    "S2": S2_input_sanity_guard,

    "S3": S3_input_policy_filter,

    "S4": S4_context_coherence_check,

    "S5": S5_output_policy_filter,

    "S6": S6_child_safety_guard,

    "S7": S7_telemetry_logger,

    "S8": S8_emotional_tone_guard,

    "S9": S9_gentle_recovery,

}


def llm(prompt, state):

    return f"[DEV_ECHO] {prompt}"


class HERKernel:

    def __init__(self):

        self.pre_input = ["S1", "S2", "S3", "S4"]

        self.pre_llm = ["S8"]

        self.post_llm = ["S5", "S6", "S9", "S7"]


    def run_once(self, text, session_state=None):

        s = (session_state or {}).copy()

        s.update({"user_input": text, "blocked": False})


        def chain(steps, stage):

            for m in steps:

                ok, s2, reason = SPINE[m](s)

                s = s2

                if not ok:

                    s.setdefault("violations", []).append(

                        {"module": m, "reason": reason, "stage": stage}

                    )

                    s["blocked"] = True

                    break

            return s


        s = chain(self.pre_input, "pre_input")

        if s["blocked"]:

            s = chain(["S9", "S7"], "blocked")

            return self.final(s, "Input rejected", True)


        s = chain(self.pre_llm, "pre_llm")

        s["raw_llm_output"] = llm(text, s)


        s = chain(self.post_llm, "post_llm")

        return self.final(s, s["raw_llm_output"], s["blocked"])


    def final(self, state, out, blocked):

        return {

            "output": state.get("final_safe_output", out),

            "blocked": blocked,

            "violations": state.get("violations", []),

            "metadata": {

                "chamber_verified": state.get("chamber_verified"),

                "distress": state.get("distress_detected"),

                "recovery_mode": state.get("recovery_mode"),

            },

            "trace": state.get("trace", []),

        }


_kernel = HERKernel()


def ni_ai(message, session_state=None, echoauth_ok=None, echoauth_token=None):

    ss = session_state or {}

    if echoauth_ok is not None:

        ss["echoauth_ok"] = echoauth_ok

    if echoauth_token is not None:

        ss["echoauth_token"] = echoauth_token

    return _kernel.run_once(message, ss)
