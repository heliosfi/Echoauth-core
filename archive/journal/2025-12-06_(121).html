class Decision(Enum):

    ALLOW = auto()

    BLOCK = auto()

    ESCALATE = auto()

    REQUIRE_HUMAN = auto()

class RuleResult:

def apply_governance_spine(ctx: SystemContext) -> GovernanceDecision:

    """

    Runs S1–S23 in sequence and returns a single decision.

    First hard BLOCK wins. Then REQUIRE_HUMAN. Then ALLOW.

    """

    path: List[RuleResult] = []


    # Run core rules in the correct order

    for rule_fn in (

        apply_S1_identity_binding,

        apply_S2_caregiver_auth,

        apply_S3_location_gate,

        apply_S4_time_gate,

        apply_S5_red_state_lock,

        apply_S6_override_policy,

        apply_S7_telemetry_logging,

    ):

        res = rule_fn(ctx)

        path.append(res)

        if res.triggered and res.decision in (Decision.BLOCK, Decision.REQUIRE_HUMAN):

            return GovernanceDecision(res.decision, path)


    # Run remaining spine (S8–S23)

    for res in apply_S8_S9_etc(ctx):

        path.append(res)

        if res.triggered and res.decision in (Decision.BLOCK, Decision.REQUIRE_HUMAN):

            return GovernanceDecision(res.decision, path)


    # If nothing blocked, default ALLOW

    return GovernanceDecision(Decision.ALLOW, path)
