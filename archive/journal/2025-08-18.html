# hello_echoauth.py

import json, math, time

import numpy as np

import torch

import torch.nn as nn

import torch.nn.functional as F


# ---------------- Device ----------------

DEVICE = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

IS_CUDA = DEVICE.type == 'cuda'

if IS_CUDA:

    torch.backends.cudnn.benchmark = True


# --------------- Dummy Models ---------------

class MotherAffectNet(nn.Module):

    def __init__(self, out_dim=128):

        super().__init__()

        self.conv = nn.Sequential(

            nn.Conv2d(3, 16, 3, 2, 1), nn.ReLU(),

            nn.Conv2d(16, 32, 3, 2, 1), nn.ReLU(),

            nn.Conv2d(32, 64, 3, 2, 1), nn.ReLU()

        )

        self.fc = nn.Linear(64*28*28, out_dim)  # for 224x224

    def forward(self, x):

        x = self.conv(x).flatten(1)

        return self.fc(x)                       # [B,128]


class FatherVerifyNet(nn.Module):

    def __init__(self, out_dim=256):

        super().__init__()

        self.img = nn.Sequential(

            nn.Conv2d(3, 16, 3, 2, 1), nn.ReLU(),

            nn.Conv2d(16, 32, 3, 2, 1), nn.ReLU()

        )

        self.img_fc = nn.Linear(32*56*56, 128)

        self.aud_fc = nn.Linear(16000, 128)     # 1s @ 16kHz

        self.fuse   = nn.Linear(256, out_dim)

    def forward(self, img, audio):

        vi = self.img(img).flatten(1)

        vi = self.img_fc(vi)                    # [B,128]

        if audio.dim() == 3: audio = audio.squeeze(1)

        T = audio.size(1)

        if T > 16000: audio = audio[:, :16000]

        elif T < 16000: audio = F.pad(audio, (0, 16000 - T))

        va = self.aud_fc(audio)                 # [B,128]

        return self.fuse(torch.cat([vi, va], 1))# [B,256]


class FusionMLP(nn.Module):

    def __init__(self, in_dim=128+256):

        super().__init__()

        self.mlp = nn.Sequential(

            nn.Linear(in_dim, 128), nn.ReLU(),

            nn.Linear(128, 1), nn.Sigmoid()     # 0..1 score for demo

        )

    def forward(self, x): return self.mlp(x)


mother = MotherAffectNet().to(DEVICE).eval()

father = FatherVerifyNet().to(DEVICE).eval()

fuser  = FusionMLP().to(DEVICE).eval()


# --------------- Thresholds ---------------

THRESHOLD_ACCEPT = 0.80

THRESHOLD_RETRY  = 0.55


# --------------- Preprocess ---------------

MEAN = torch.tensor([0.485, 0.456, 0.406], device=DEVICE).view(1,3,1,1)

STD  = torch.tensor([0.229, 0.224, 0.225], device=DEVICE).view(1,3,1,1)


def preprocess_frame_bgr(frame_bgr_u8: np.ndarray):

    """HxWx3 uint8 BGR -> 1x3x224x224 float32 on DEVICE"""

    t = torch.from_numpy(frame_bgr_u8).to(DEVICE, non_blocking=IS_CUDA)  # H W C u8

    t = t.float().permute(2,0,1).unsqueeze(0)                            # 1 C H W

    t = t[:, [2,1,0], :, :]                                              # BGR->RGB

    t = F.interpolate(t, size=(224,224), mode='bilinear', align_corners=False)

    t = (t/255.0 - MEAN)/STD

    if IS_CUDA: t = t.contiguous(memory_format=torch.channels_last)

    return t


def preprocess_audio(audio_f32: np.ndarray, sr=16000):

    """1D float32 numpy @ 16kHz -> 1xT float32 on DEVICE"""

    wav = torch.from_numpy(audio_f32.astype(np.float32))

    peak = wav.abs().max()

    if peak > 0: wav = wav / peak

    return wav.unsqueeze(0).to(DEVICE, non_blocking=IS_CUDA)             # [1,T]


# --------------- Pipeline ---------------

@torch.inference_mode()

def echoauth(frame_tensor: torch.Tensor, audio_tensor: torch.Tensor) -> float:

    # simple sequential path (no streams) for hello-world clarity

    m_feat = mother(frame_tensor)                      # [1,128]

    f_feat = father(frame_tensor, audio_tensor)        # [1,256]

    fused  = fuser(torch.cat([m_feat, f_feat], 1))     # [1,1] (0..1)

    return float(fused.squeeze().item())


def verify_user(frame_bgr_u8: np.ndarray, audio_f32: np.ndarray):

    t0 = time.time()

    frame = preprocess_frame_bgr(frame_bgr_u8)

    audio = preprocess_audio(audio_f32, sr=16000)

    score = echoauth(frame, audio)

    decision = "accept" if score >= THRESHOLD_ACCEPT else \

               "retry"  if score >= THRESHOLD_RETRY  else "deny"

    return {

        "device": DEVICE.type,

        "score": round(score, 4),

        "decision": decision,

        "latency_ms": int((time.time() - t0)*1000)

    }


# --------------- Demo main ---------------

if __name__ == "__main__":

    # Synthetic inputs (no extra deps)

    # Random image (BGR uint8)

    H, W = 480, 640

    frame_bgr = (np.random.rand(H, W, 3) * 255).astype(np.uint8)


    # 1s sine wave @ 16kHz (440 Hz) + small noise

    sr = 16000

    t  = np.arange(sr) / sr

    audio = 0.5*np.sin(2*math.pi*440*t) + 0.02*np.random.randn(sr)


    out = verify_user(frame_bgr, audio)

    print(json.dumps(out, indent=2))
