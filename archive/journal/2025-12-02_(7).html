class SafetyContext:

    def __init__(self, user_role, env, history):

        self.user_role = user_role

        self.env = env

        self.history = history

        self.flags = {}          # e.g. {"high_risk": False, "child_mode": True}

        self.logs = []           # audit entries

        self.safe_mode = False   # kernel can flip this on

        self.terminated = False  # kernel can end the session



class S23Kernel:

    def __init__(self, modules):

        # modules is an ordered list: [S1, S2, ..., S23]

        self.modules = modules


    def run(self, raw_input, ctx: SafetyContext):

        """

        Main entry point for NI-AI.

        Takes raw_input (text/audio/etc.) plus SafetyContext,

        returns either a safe_request for the model or a safe fallback response.

        """

        state = {

            "raw_input": raw_input,

            "normalized_input": None,

            "candidate_output": None,

            "blocked": False,

            "reason": None,

        }


        for module in self.modules:

            state = module.apply(state, ctx)


            # Any module can short-circuit for safety:

            if state.get("blocked") or ctx.terminated:

                return self._safe_fallback(state, ctx)


        # If we reach here, NI-AI can proceed to reasoning/generation

        return self._hand_off_to_model(state, ctx)

        Placeholder: call the NI-AI core (UMCE/ORI/TLE/etc.).

        In the real system, this is where you run the model

        under the safety and identity constraints.

        """

        safe_request = state.get("normalized_input", state["raw_input"])

        # model_output = ni_ai_core.generate(safe_request, ctx)

        model_output = "[NI-AI OUTPUT WOULD BE GENERATED HERE]"

        state["candidate_output"] = model_output


        # Optionally run post-output guards (e.g. S18–S23 again)

        # For now, we just return the candidate_output.

        return state["candidate_output"]


    def _safe_fallback(self, state, ctx):

        """

        What the system returns if any module blocks the request.

        You can customize by role, risk level, etc.

        """

        reason = state.get("reason", "unsafe_request")

        ctx.logs.append(f"Request blocked: {reason}")

        if ctx.user_role == "child":

            return "I can’t do that right now. Let’s do something safe instead."

        else:

            return "The system cannot safely complete this request."



class BaseSafetyModule:

    name = "BaseModule"


    def apply(self, state, ctx: SafetyContext):

        """

        Each module:

        - reads state + ctx

        - may set ctx.flags / ctx.safe_mode / ctx.terminated

        - may modify state["normalized_input"] or other fields

        - may set state["blocked"] = True and state["reason"] = "..."

        """

        return state
