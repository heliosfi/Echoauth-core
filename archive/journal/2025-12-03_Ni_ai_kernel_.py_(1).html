# echoauth_api.py

# EchoAuth ↔ NI-AI Gateway v2 — aligned with ni_ai_kernel.py v1.4 (S1–S9)

# Exposes: /ni-ai/chat, /ni-ai/debug, /health


from typing import Optional, Dict, Any, List


from fastapi import FastAPI, Header, HTTPException

from fastapi.responses import JSONResponse

from pydantic import BaseModel


from ni_ai_kernel import ni_ai, DEV_MODE  # v1.4 public API


app = FastAPI(

    title="EchoAuth ↔ NI-AI Gateway",

    description=(

        "Secure, child-safe bridge between EchoAuth chamber and the HER/NI-AI kernel.\n"

        "Spine: S1–S9 (identity, sanity, policy, context, output, chamber, telemetry, "

        "emotional comfort, gentle recovery)."

    ),

    version="1.4.0",

    openapi_tags=[

        {"name": "chat", "description": "Main NI-AI interaction"},

        {"name": "system", "description": "Health & debug"},

    ],

)


# --------------------------------------------------------------------------- #

#                         REQUEST / RESPONSE MODELS                           #

# --------------------------------------------------------------------------- #


class ChatRequest(BaseModel):

    message: str

    echoauth_ok: Optional[bool] = None        # DEV only (respected when DEV_MODE=True)



class ChatResponse(BaseModel):

    output: str

    blocked: bool

    violations: List[Dict[str, Any]]

    metadata: Dict[str, Any]

    trace: List[Dict[str, Any]]



# --------------------------------------------------------------------------- #

#                               MAIN ENDPOINT                                 #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/chat", response_model=ChatResponse, tags=["chat"])

async def ni_ai_chat(

    body: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    Primary NI-AI entrypoint for EchoAuth.


    - DEV_MODE=True:

        • echoauth_ok in body will be passed through as a soft chamber flag.

        • X-EchoAuth-Token header is ignored (placeholder for prod).


    - DEV_MODE=False:

        • echoauth_ok is ignored.

        • X-EchoAuth-Token is passed to the kernel (S6 will enforce).

    """

    msg = (body.message or "").strip()

    if not msg:

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    # EchoAuth flags

    echoauth_ok = bool(body.echoauth_ok) if DEV_MODE else False

    echoauth_token = x_echoauth_token if not DEV_MODE else None


    result = ni_ai(

        user_message=msg,

        echoauth_ok=echoauth_ok,

        echoauth_token=echoauth_token,

        session_state=None,

    )


    return ChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

        trace=result.get("trace", []),

    )



# --------------------------------------------------------------------------- #

#                              DEBUG / INSPECTION                             #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/debug", tags=["system"])

async def ni_ai_debug(body: ChatRequest):

    """

    Dev-only raw view.


    Returns full kernel result including:

      - output

      - blocked

      - violations

      - metadata (distress_detected, comfort_mode, recovery_mode, llm_suppressed, run_id…)

      - trace (S7 entries)


    NOTE: In real production, you should remove/lock this down.

    """

    if not DEV_MODE:

        raise HTTPException(status_code=403, detail="Debug endpoint disabled in production")


    msg = (body.message or "").strip()

    if not msg:

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    result = ni_ai(

        user_message=msg,

        echoauth_ok=bool(body.echoauth_ok),

        echoauth_token=None,

        session_state=None,

    )


    return JSONResponse(content=result, status_code=200)



# --------------------------------------------------------------------------- #

#                               SYSTEM HEALTH                                 #

# --------------------------------------------------------------------------- #


@app.get("/health", tags=["system"])

async def health():

    """

    Simple health probe for load balancers / monitoring.

    """

    return {

        "status": "ok",

        "kernel_version": "1.4",

        "dev_mode": DEV_MODE,

        "spine_modules": ["S1", "S2", "S3", "S4", "S5", "S6", "S7", "S8", "S9"],

    }

🧪 How to run + test


From the folder with both files:


pip install fastapi uvicorn pydantic groq openai

uvicorn echoauth_api:app --reload --port 8000


Then in another terminal:


1) Normal safe chat (EchoAuth allowed in DEV)


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message": "Tell me a happy story about a robot.", "echoauth_ok": true}'


You should see:

• "blocked": false

• metadata.comfort_mode = false

• metadata.recovery_mode = false

• trace list with S7 entries.

  -d '{"message": "I feel sad and lonely, nobody loves me.", "echoauth_ok": true}'


Expected:

• blocked: false

• metadata.distress_detected: true

• metadata.comfort_mode: true

• metadata.recovery_mode: true (S9 layered on top)

• output: one of your safe, gentle replies (not a raw LLM ramble)

Expected:

• blocked: true

• violations includes one from S3 with reason "Disallowed content in input"

• metadata.recovery_mode: true

• output: soft fallback (S9) instead of harsh “Input rejected…” string.
