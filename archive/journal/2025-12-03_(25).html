# echoauth_api.py

# EchoAuth ↔ NI-AI FastAPI Gateway (NI-AI HER Kernel v1.3)


from fastapi import FastAPI, Header, HTTPException

from fastapi.responses import JSONResponse

from pydantic import BaseModel

from typing import Optional, Dict, Any, List


from ni_ai_kernel import ni_ai, DEV_MODE  # import your kernel entrypoint


app = FastAPI(

    title="EchoAuth ↔ NI-AI Gateway",

    description="Secure, child-safe bridge between EchoAuth chamber and the HER/NI-AI kernel.",

    version="1.3.0",

    openapi_tags=[

        {"name": "chat", "description": "Main NI-AI interaction"},

        {"name": "system", "description": "Health & debug"},

        {"name": "parent", "description": "Parent / caregiver tools"},

    ],

)


# ─────────────────────────────────────────────────────────────

#   MODELS

# ─────────────────────────────────────────────────────────────


class ChatRequest(BaseModel):

    message: str

    echoauth_ok: Optional[bool] = None   # only honored in DEV_MODE



class ChatResponse(BaseModel):

    output: str

    blocked: bool

    violations: List[Dict[str, Any]]

    metadata: Dict[str, Any]



class DebugResponse(BaseModel):

    output: str

    blocked: bool

    violations: List[Dict[str, Any]]

    metadata: Dict[str, Any]

    trace: List[Dict[str, Any]]



# ─────────────────────────────────────────────────────────────

#   MAIN CHAT ENDPOINT

# ─────────────────────────────────────────────────────────────


@app.post("/ni-ai/chat", response_model=ChatResponse, tags=["chat"])

async def ni_ai_chat(

    req: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    Secure child-safe chat entrypoint.


    • DEV_MODE = True:

        - can pass {"echoauth_ok": true} in JSON to simulate chamber safe state

    • DEV_MODE = False:

        - must pass a real EchoAuth JWT via X-EchoAuth-Token header

    """

    msg = (req.message or "").strip()

    if not msg:

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    # Build session_state exactly how the kernel expects it

    session_state: Dict[str, Any] = {}


    # DEV: allow simple flag

    if DEV_MODE and req.echoauth_ok:

        session_state["echoauth_ok"] = True


    # PROD: header token (kernel can validate it in S6 once JWT logic is live)

    if x_echoauth_token:

        session_state["echoauth_token"] = x_echoauth_token


    result = ni_ai(msg, session_state=session_state)


    return ChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

    )



# ─────────────────────────────────────────────────────────────

#   HEALTH CHECK

# ─────────────────────────────────────────────────────────────


@app.get("/health", tags=["system"])

async def health():

    return {

        "status": "ok",

        "kernel": "HER/NI-AI v1.3",

        "dev_mode": DEV_MODE,

    }



# ─────────────────────────────────────────────────────────────

#   DEV / PARENT DEBUG VIEW (FULL TRACE)

# ─────────────────────────────────────────────────────────────


@app.post("/ni-ai/debug", response_model=DebugResponse, tags=["parent"])

async def ni_ai_debug(

    req: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    Returns full kernel response including trace[] from S7.

    Use only in DEV or restricted parent dashboard.

    """

    msg = (req.message or "").strip()

    if not msg:

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    session_state: Dict[str, Any] = {}

    if DEV_MODE and req.echoauth_ok:

        session_state["echoauth_ok"] = True

    if x_echoauth_token:

        session_state["echoauth_token"] = x_echoauth_token


    result = ni_ai(msg, session_state=session_state)


    return DebugResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

        trace=result.get("trace", []),

    )

3. Quick tests (show S6 + S8 + S7 working)


A) No EchoAuth → should block via S6


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message": "Hi, who are you?"}'


You should get something like:


{

  "output": "Input rejected by safety system",

  "blocked": true,

  "violations": [

    {"module": "S6", "stage": "post_llm", "reason": "DEV_MODE: echoauth_ok=False"}

  ],

  "metadata": { ... }

}



⸻


B) EchoAuth OK (DEV) → normal safe answer


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message": "Tell me a gentle story about a robot.", "echoauth_ok": true}'


You’ll see [DEV ECHO] ... in output and blocked: false.


⸻


C) Distress trigger + EchoAuth OK → S8 comfort mode


curl -X POST http://localhost:8000/ni-ai/debug \

  -H "Content-Type: application/json" \

  -d '{"message": "I feel lonely and sad, nobody loves me.", "echoauth_ok": true}'


Check response:

• output will be one of the comfort replies (S8 override).

• metadata.distress_detected = true

• metadata.comfort_mode = true


And in your terminal you’ll see S7 telemetry printing the run, plus S8 distress log.
