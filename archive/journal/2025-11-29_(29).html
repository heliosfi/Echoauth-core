3. Database layer


db/init_db.py


import sqlite3

from pathlib import Path


DB_PATH = Path("echoauth_demo.db")



def get_connection():

    return sqlite3.connect(DB_PATH)



def init_db():

    conn = get_connection()

    cur = conn.cursor()


    cur.execute(

        """

        CREATE TABLE IF NOT EXISTS caregivers (

            id INTEGER PRIMARY KEY AUTOINCREMENT,

            email TEXT UNIQUE NOT NULL,

            password TEXT NOT NULL

        );

        """

    )


    cur.execute(

        """

        CREATE TABLE IF NOT EXISTS children (

            id INTEGER PRIMARY KEY AUTOINCREMENT,

            caregiver_id INTEGER NOT NULL,

            name TEXT NOT NULL,

            age INTEGER,

            notes TEXT,

            FOREIGN KEY (caregiver_id) REFERENCES caregivers(id)

        );

        """

    )


    cur.execute(

        """

        CREATE TABLE IF NOT EXISTS checkins (

            id INTEGER PRIMARY KEY AUTOINCREMENT,

            caregiver_id INTEGER NOT NULL,

            child_id INTEGER NOT NULL,

            stress_score INTEGER NOT NULL,

            mode TEXT NOT NULL,

            label TEXT NOT NULL,

            created_at TEXT NOT NULL,

            FOREIGN KEY (caregiver_id) REFERENCES caregivers(id),

            FOREIGN KEY (child_id) REFERENCES children(id)

        );

        """

    )


    conn.commit()

    conn.close()


db/repository.py


import datetime

from .init_db import get_connection



def seed_demo_data():

    conn = get_connection()

    cur = conn.cursor()


    # insert caregiver if not exists

    cur.execute(

        "SELECT id FROM caregivers WHERE email = ?",

        ("demo_caregiver@example.com",),

    )

    row = cur.fetchone()


    if row is None:

        cur.execute(

            "INSERT INTO caregivers (email, password) VALUES (?, ?)",

            ("demo_caregiver@example.com", "demo_password"),

        )

        caregiver_id = cur.lastrowid


        cur.execute(

            "INSERT INTO children (caregiver_id, name, age, notes) VALUES (?, ?, ?, ?)",

            (caregiver_id, "Demo Child", 8, "Noise sensitive"),

        )

    conn.commit()

    conn.close()



def insert_checkin(caregiver_id, child_id, stress_score, mode, label):

    conn = get_connection()

    cur = conn.cursor()

    cur.execute(

        """

        INSERT INTO checkins (

            caregiver_id, child_id, stress_score, mode, label, created_at

        )

        VALUES (?, ?, ?, ?, ?, ?)

        """,

        (

            caregiver_id,

            child_id,

            stress_score,

            mode,

            label,

            datetime.datetime.utcnow().isoformat(),

        ),

    )

    conn.commit()

    conn.close()

4. Core NI AI modules


core/ori.py


from dataclasses import dataclass



@dataclass

class IdentityCapsule:

    caregiver_id: int

    child_id: int

    role: str = "caregiver"

    system_name: str = "EchoAuth MVP"


core/lrm.py


def compute_stress_score(cues: dict) -> int:

    """

    Very simple demo mapping.

    Returns an integer from 0 to 5.

    """

    score = 0


    body = str(cues.get("body", "")).lower()

    env = str(cues.get("environment", "")).lower()


    if "stimming" in body or "cry" in body:

        score += 2

    if "agitated" in body or "meltdown" in body:

        score += 3


    if "loud" in env or "crowd" in env:

        score += 1

    if "very loud" in env or "chaos" in env:

        score += 2


    # clamp between 0 and 5

    score = max(0, min(5, score))

    return score


core/her.py


def select_mode(stress_score: int) -> str:

    """

    Map stress score to runtime mode.

    """

    if stress_score <= 1:

        return "NORMAL"

    if 2 <= stress_score <= 3:

        return "SAFE"

    return "DEGRADED"


core/pnre.py


def select_label(stress_score: int, mode: str) -> str:

    """

    Very small demo policy:

    - low stress -> SAFE

    - medium or high stress -> NEEDS_ATTENTION

    """

    if stress_score <= 1 and mode == "NORMAL":

        return "SAFE"

    return "NEEDS_ATTENTION"


core/dahe.py


from . import nia_engine  # for type hints only, no heavy logic here



def append_history(repository, caregiver_id, child_id, stress_score, mode, label):

    """

    History writer that delegates to the db.repository module.

    """

    repository.insert_checkin(

        caregiver_id=caregiver_id,

        child_id=child_id,

        stress_score=stress_score,

        mode=mode,

        label=label,

    )


core/nia_engine.py


from .ori import IdentityCapsule

from . import lrm, her, pnre, dahe

from db import repository



def process_checkin(caregiver_id: int, child_id: int, cues: dict) -> dict:

    """

    Glue function that runs the NI AI pipeline for a single check in.

    """

    # 1. ORI: bind identity

    identity = IdentityCapsule(caregiver_id=caregiver_id, child_id=child_id)


    # 2. LRM: compute stress score

    stress_score = lrm.compute_stress_score(cues)


    # 3. HER: choose runtime mode

    mode = her.select_mode(stress_score)


    # 4. PNRE: pick one external label

    label = pnre.select_label(stress_score, mode)


    # 5. DAHE: write history

    dahe.append_history(repository, identity.caregiver_id, identity.child_id,

                        stress_score, mode, label)


    # 6. Return minimal response

    return {

        "caregiver_id": identity.caregiver_id,

        "child_id": identity.child_id,

        "stress_score": stress_score,

        "mode": mode,

        "label": label,

    }

5. API layer


api/main.py


from fastapi import FastAPI, HTTPException

from pydantic import BaseModel

from core import nia_engine


app = FastAPI(title="EchoAuth MVP API")



class CheckInRequest(BaseModel):

    caregiver_id: int

    child_id: int

    cues: dict



class CheckInResponse(BaseModel):

    label: str

    stress_score: int

    mode: str



@app.get("/health")

def health():

    return {"status": "ok"}



@app.post("/checkins/create", response_model=CheckInResponse)

def create_checkin(payload: CheckInRequest):

    # In a real system we would validate caregiver/child relationship here.

    try:

        result = nia_engine.process_checkin(

            caregiver_id=payload.caregiver_id,

            child_id=payload.child_id,

            cues=payload.cues,

        )

    except Exception as exc:

        # Failsafe: log internally, return safe default

        raise HTTPException(status_code=500, detail="Internal error") from exc


    return CheckInResponse(

        label=result["label"],

        stress_score=result["stress_score"],

        mode=result["mode"],

    )

6. Local launcher


run_local_demo.py


"""

EchoAuth local demo launcher.


Usage:

    python run_local_demo.py


Then open:

    http://127.0.0.1:8000/docs

and try POST /checkins/create.

"""


import uvicorn

from db import init_db, repository



def main():

    # 1. Initialize database and tables

    init_db.init_db()


    # 2. Seed demo caregiver and child

    repository.seed_demo_data()


    # 3. Start API server

    uvicorn.run(

        "api.main:app",

        host="127.0.0.1",

        port=8000,

        reload=False,

    )



if __name__ == "__main__":

    main()

python -m venv venv

source venv/bin/activate  # or venv\Scripts\activate on Windows

pip install -r requirements.txt

python run_local_demo.py

http://127.0.0.1:8000/docs


Use the /checkins/create endpoint with a body like:


{

  "caregiver_id": 1,

  "child_id": 1,

  "cues": {

    "body": "stimming, agitated",

    "environment": "loud cafeteria"

  }

}


You should get a single label:


{

  "label": "NEEDS_ATTENTION",

  "stress_score": 4,

  "mode": "DEGRADED"

}
