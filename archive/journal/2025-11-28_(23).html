# ---------- 1. READ THE MOMENT & GOVERNANCE ----------


def detect_human_state(message: str) -> tuple[HumanState, int]:

    keywords = {

        "Overwhelmed": ["help", "lost", "too much", "panic", "can’t"],

        "Emotional": ["feel", "hurt", "angry", "sad", "cry"],

        "Pressured": ["fast", "urgent", "now", "deadline"],

    }

    score = 0

    lower_msg = message.lower()

    for _, words in keywords.items():

        score_for_state = sum(1 for w in words if w in lower_msg)

        score += score_for_state


    if score >= 4:

        return "Overwhelmed", 4

    if score >= 2:

        return "Emotional", 3

    if "urgent" in lower_msg or "deadline" in lower_msg:

        return "Pressured", 2

    return "Grounded", 0



def her_mode_governance(lrm_zone: LRMZone, last_user: str, last_reply: str) -> None:

    """

    Maps LRM Zone to HER Mode and applies UMCE-safe context rules.

    GREEN  -> NORMAL (no change)

    YELLOW -> SAFE (compress context)

    RED    -> DEGRADED (keep only a safety capsule)

    """

    if lrm_zone == "GREEN":

        state["her_mode"] = "NORMAL"

        # Full working context allowed

        return


    if lrm_zone == "YELLOW":

        state["her_mode"] = "SAFE"

        # Context Merging: keep only the last 2 entries

        state["thread_context"] = state["thread_context"][-2:]

        return


    # RED ZONE

    state["her_mode"] = "DEGRADED"

    # UMCE-safe "Thread Capsule": keep minimal anchor + last user + last reply

    safety_capsule: List[str] = [

        state["core_anchor"],

        f"LastUser: {last_user[:120]}",

        f"LastReply: {last_reply[:120]}",

    ]

    state["thread_context"] = safety_capsule



def apply_lrm_and_her(pressure: int, last_user: str, last_reply: str) -> None:

    """Applies LRM (Cognitive) and HER (Kernel) Governance together."""

    if pressure <= 1:

        state["lrm_zone"] = "GREEN"

        state["current_temp"] = 0.7

        state["current_max_tokens"] = 600

    elif pressure <= 3:

        state["lrm_zone"] = "YELLOW"

        state["current_temp"] = 0.5

        state["current_max_tokens"] = 400

    else:

        state["lrm_zone"] = "RED"

        state["current_temp"] = 0.3

        state["current_max_tokens"] = 200


    her_mode_governance(state["lrm_zone"], last_user, last_reply)



def update_state(user_message: str, last_reply: str) -> None:

    human_state, pressure = detect_human_state(user_message)

    state["human_state"] = human_state

    state["pressure_level"] = pressure


    lower_msg = user_message.lower()

    if state["pressure_level"] >= 3:

        state["position"] = "Support"

    elif "decision" in lower_msg or "choose" in lower_msg:

        state["position"] = "Guidance"

    else:

        state["position"] = "Neutral"


    apply_lrm_and_her(state["pressure_level"], user_message, last_reply)

def ni_ai_response(user_message: str, last_reply: str = "") -> str:

    # 1. Update State (LRM + HER)

    update_state(user_message, last_reply)


    lrm_zone = state["lrm_zone"]

Short working context:

{chr(10).join(state['thread_context'])}

""".strip()


    messages = [

        {"role": "user", "content": user_message},

    state["thread_context"].append(f"NI-AI: {final[:100]}")

    return final



def enforce_coherence(text: str) -> str:

    banned = ["I feel", "my heart", "energy", "vibe", "soul", "as an AI I think"]

    for phrase in banned:

        text = text.replace(phrase, "[REDACTED]")

    return text



def infer_tone(text: str) -> str:

    lower = text.lower()

    if len(text) < 80 and "step" in lower:

        return "guidance-concise"

    if "take your time" in lower or "breathe" in lower:

        return "support-soft"

    return "calm-structured"
