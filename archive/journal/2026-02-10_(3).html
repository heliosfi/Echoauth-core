import pandas as pd

from polygon import RESTClient


# ----------------------------

# Config

# ----------------------------

SYMBOL = "TSLA"

TF_MINUTES = 30


SUPPORT = 420.23

RESIST  = 425.50


VOLUME_LOOKBACK = 3

MIN_BODY_RATIO = 0.55


USE_MACD = True

MACD_FAST = 12

MACD_SLOW = 26

MACD_SIGNAL = 9


# Volatility + risk gates (recommended)

ATR_LEN = 14

STOP_ATR_MULT = 0.50

MAX_CANDLE_MULT = 1.50

MAX_RISK_DOLLARS = 3.50


# Time filter (ET)

ENABLE_TIME_FILTER = True

WINDOWS = [

    ("09:35", "10:45"),  # Open drive

    ("14:30", "15:55"),  # Power hour

]

MIDDAY_BLOCK = True

MIDDAY = ("11:15", "14:15")


# Entry mode

ENTRY_MODE = "CLOSE"  # "CLOSE" or "RETEST"

RETEST_TOL = 0.20

RETEST_TIMEOUT_BARS = 6


# Targets

LONG_TPS  = [430.00, 435.75, 440.66]

SHORT_TPS = [414.01, 412.00, 410.00]



# ----------------------------

# Helpers

# ----------------------------

def body_size(c): return abs(c["close"] - c["open"])

def range_size(c): return max(c["high"] - c["low"], 1e-9)

def body_ratio(c): return body_size(c) / range_size(c)

def is_doji_like(c): return body_ratio(c) < MIN_BODY_RATIO


def bearish_body_engulf(curr, prev):

    return (curr["close"] < curr["open"]) and (curr["open"] >= prev["close"]) and (curr["close"] <= prev["open"])


def bullish_body_expands(curr, prev):

    return (curr["close"] > curr["open"]) and (body_size(curr) > body_size(prev))


def calculate_macd_hist(df: pd.DataFrame) -> pd.Series:

    ema_fast = df["close"].ewm(span=MACD_FAST, adjust=False).mean()

    ema_slow = df["close"].ewm(span=MACD_SLOW, adjust=False).mean()

    macd_line = ema_fast - ema_slow

    signal_line = macd_line.ewm(span=MACD_SIGNAL, adjust=False).mean()

    return macd_line - signal_line  # histogram


def macd_green_expanding(hist: pd.Series, i: int) -> bool:

    return hist.iloc[i] > 0 and hist.iloc[i] > hist.iloc[i-1]


def macd_red_expanding(hist: pd.Series, i: int) -> bool:

    return hist.iloc[i] < 0 and hist.iloc[i] < hist.iloc[i-1]


def volume_expanding(curr, last_n: pd.DataFrame) -> bool:

    return curr["volume"] >= last_n["volume"].mean()


def calc_atr(df: pd.DataFrame, length: int = 14) -> pd.Series:

    prev_close = df["close"].shift(1)

    tr = pd.concat([

        (df["high"] - df["low"]),

        (df["high"] - prev_close).abs(),

        (df["low"] - prev_close).abs(),

    ], axis=1).max(axis=1)

    return tr.rolling(length).mean()


def in_time_window(dt, start_hhmm: str, end_hhmm: str) -> bool:

    t = dt.strftime("%H:%M")

    return start_hhmm <= t <= end_hhmm


def entry_allowed(dt) -> bool:

    if not ENABLE_TIME_FILTER:

        return True


    # Midday block

    if MIDDAY_BLOCK and in_time_window(dt, MIDDAY[0], MIDDAY[1]):

        return False


    return any(in_time_window(dt, s, e) for s, e in WINDOWS)


def confirm_candle_not_too_big(curr, atr_val) -> bool:

    if pd.isna(atr_val) or atr_val <= 0:

        return False

    return range_size(curr) <= atr_val * MAX_CANDLE_MULT


def build_long(entry_px, confirm_candle, atr_val):

    buf = atr_val * STOP_ATR_MULT

    raw_stop = min(RESIST, confirm_candle["low"])  # <-- correct direction

    stop = raw_stop - buf

    if abs(entry_px - stop) > MAX_RISK_DOLLARS:

        return None

    return {"dir": "LONG", "entry": entry_px, "stop": stop, "tps": LONG_TPS}


def build_short(entry_px, confirm_candle, atr_val):

    buf = atr_val * STOP_ATR_MULT

    raw_stop = max(SUPPORT, confirm_candle["high"])  # <-- correct direction

    stop = raw_stop + buf

    if abs(entry_px - stop) > MAX_RISK_DOLLARS:

        return None

    return {"dir": "SHORT", "entry": entry_px, "stop": stop, "tps": SHORT_TPS}


def bullish_confirm(curr, prev, last3, hist, i, atr_val):

    return (curr["close"] > RESIST

            and not is_doji_like(curr)

            and bullish_body_expands(curr, prev)

            and volume_expanding(curr, last3)

            and confirm_candle_not_too_big(curr, atr_val)

            and (not USE_MACD or macd_green_expanding(hist, i)))


def bearish_confirm(curr, prev, last3, hist, i, atr_val):

    return (curr["close"] < SUPPORT

            and not is_doji_like(curr)

            and bearish_body_engulf(curr, prev)

            and volume_expanding(curr, last3)

            and confirm_candle_not_too_big(curr, atr_val)

            and (not USE_MACD or macd_red_expanding(hist, i)))



# ----------------------------

# Fetch Data

# ----------------------------

client = RESTClient()  # assumes API key is set in env or config


# IMPORTANT: use a period where TSLA is around your levels (~420–430).

# Update these dates to Feb 2026 (example window; adjust as needed):

from_ = "2026-02-01"

to    = "2026-02-28"


aggs = client.get_aggs(SYMBOL, TF_MINUTES, "minute", from_, to, adjusted=False, sort="asc", limit=50000)


rows = []

for a in aggs:

    rows.append({

        "open": a.open, "high": a.high, "low": a.low, "close": a.close, "volume": a.volume,

        "datetime": pd.to_datetime(a.timestamp, unit="ms", utc=True).tz_convert("US/Eastern")

    })


df = pd.DataFrame(rows).sort_values("datetime").reset_index(drop=True)

df["atr"] = calc_atr(df, ATR_LEN)

df["hist"] = calculate_macd_hist(df)


# ----------------------------

# Backtest Loop

# ----------------------------

trades = []


direction = "NONE"

plan = None

pending = None  # for RETEST mode: {"dir", "level", "expiry_i", "confirm_i"}


warmup = max(MACD_SLOW * 2, ATR_LEN + 5, VOLUME_LOOKBACK + 2)


for i in range(warmup, len(df)):

    curr = df.iloc[i]

    prev = df.iloc[i - 1]

    last3 = df.iloc[i - VOLUME_LOOKBACK:i]

    atr_val = curr["atr"]

    hist = df["hist"]


    dt = curr["datetime"]


    # --- Manage open trade (bar-based fill model) ---

    if direction != "NONE" and plan is not None:

        if plan["dir"] == "LONG":

            # stop hit

            if curr["low"] <= plan["stop"]:

                trades.append({**plan, "exit": plan["stop"], "exit_i": i, "reason": "STOP"})

                direction, plan = "NONE", None

                continue

            # targets

            if curr["high"] >= plan["tps"][0]:

                # simple: take TP1 and mark as partially realized (log event)

                plan.setdefault("events", []).append(("TP1", plan["tps"][0], i))

            if curr["high"] >= plan["tps"][1]:

                plan.setdefault("events", []).append(("TP2", plan["tps"][1], i))

            if curr["high"] >= plan["tps"][2]:

                trades.append({**plan, "exit": plan["tps"][2], "exit_i": i, "reason": "TP3"})

                direction, plan = "NONE", None

                continue


        else:  # SHORT

            if curr["high"] >= plan["stop"]:

                trades.append({**plan, "exit": plan["stop"], "exit_i": i, "reason": "STOP"})

                direction, plan = "NONE", None

                continue

            if curr["low"] <= plan["tps"][0]:

                plan.setdefault("events", []).append(("TP1", plan["tps"][0], i))

            if curr["low"] <= plan["tps"][1]:

                plan.setdefault("events", []).append(("TP2", plan["tps"][1], i))

            if curr["low"] <= plan["tps"][2]:

                trades.append({**plan, "exit": plan["tps"][2], "exit_i": i, "reason": "TP3"})

                direction, plan = "NONE", None

                continue


        continue  # no new entries while in trade


    # --- RETEST mode: wait for retest fill ---

    if ENTRY_MODE == "RETEST" and pending is not None:

        if i > pending["expiry_i"]:

            pending = None

        else:

            if pending["dir"] == "LONG" and abs(curr["close"] - pending["level"]) <= RETEST_TOL:

                confirm_candle = df.iloc[pending["confirm_i"]]

                p = build_long(curr["close"], confirm_candle, atr_val)

                if p:

                    direction, plan = "LONG", p

                    plan["entry_i"] = i

                    pending = None

                else:

                    pending = None


            if pending and pending["dir"] == "SHORT" and abs(curr["close"] - pending["level"]) <= RETEST_TOL:

                confirm_candle = df.iloc[pending["confirm_i"]]

                p = build_short(curr["close"], confirm_candle, atr_val)

                if p:

                    direction, plan = "SHORT", p

                    plan["entry_i"] = i

                    pending = None

                else:

                    pending = None


        continue


    # --- Time filter for new signals ---

    if not entry_allowed(dt):

        continue


    # --- Signal checks ---

    if bullish_confirm(curr, prev, last3, hist, i, atr_val):

        if ENTRY_MODE == "CLOSE":

            p = build_long(curr["close"], curr, atr_val)

            if p:

                direction, plan = "LONG", p

                plan["entry_i"] = i

        else:

            pending = {"dir": "LONG", "level": RESIST, "expiry_i": i + RETEST_TIMEOUT_BARS, "confirm_i": i}


    elif bearish_confirm(curr, prev, last3, hist, i, atr_val):

        if ENTRY_MODE == "CLOSE":

            p = build_short(curr["close"], curr, atr_val)

            if p:

                direction, plan = "SHORT", p

                plan["entry_i"] = i

        else:

            pending = {"dir": "SHORT", "level": SUPPORT, "expiry_i": i + RETEST_TIMEOUT_BARS, "confirm_i": i}


# ----------------------------

# Results summary

# ----------------------------

results = []

for t in trades:

    entry = t["entry"]

    exit_ = t["exit"]

    pnl = (exit_ - entry) if t["dir"] == "LONG" else (entry - exit_)

    results.append({

        "dir": t["dir"],

        "entry_time": df.iloc[t["entry_i"]]["datetime"],

        "exit_time": df.iloc[t["exit_i"]]["datetime"],

        "entry": entry,

        "stop": t["stop"],

        "exit": exit_,

        "reason": t["reason"],

        "pnl_per_share": pnl,

        "events": t.get("events", [])

    })


res_df = pd.DataFrame(results)

print(res_df)

if not res_df.empty:

    print("\nSummary:")

    print(res_df["pnl_per_share"].describe())
