CONFIG:

  SYMBOL = "TSLA"

  TF = "30m"

  SUPPORT = 420.23

  RESIST  = 425.50

  VOLUME_LOOKBACK = 3              # avg of last 3 closed candles

  MIN_BODY_RATIO = 0.55            # avoid dojis: body must be >= 55% of candle range

  USE_MACD = true

  MACD_FAST = 12

  MACD_SLOW = 26

  MACD_SIGNAL = 9


STATE:

  direction = "NONE"               # NONE | LONG | SHORT

  entry_price = null

  stop_price = null

  targets = []


# -----------------------------

# Helpers

# -----------------------------

function body_size(c): return abs(c.close - c.open)

function range_size(c): return max(c.high - c.low, 0.0001)

function body_ratio(c): return body_size(c) / range_size(c)


function avg_volume(last_n_closed):

  return mean([c.volume for c in last_n_closed])


function is_doji_like(c):

  return body_ratio(c) < MIN_BODY_RATIO


# Engulf definition (body engulf):

# current red candle body fully covers prior candle body.

function bearish_body_engulf(curr, prev):

  return (curr.close < curr.open) AND

         (curr.open >= prev.close) AND

         (curr.close <= prev.open)


# MACD histogram expansion checks (use CLOSED candles only)

# hist = macd_line - signal_line

function macd_hist(candle_index_closed): return MACD_HIST_AT(candle_index_closed)


function macd_green_expanding(i):

  # i = index of just-closed candle

  return macd_hist(i) > 0 AND macd_hist(i) > macd_hist(i-1)


function macd_red_expanding(i):

  return macd_hist(i) < 0 AND macd_hist(i) < macd_hist(i-1)


# Volume expansion vs last 3 closed candles (excluding current)

function volume_expanding(curr, last3):

  return curr.volume >= avg_volume(last3)


# “Body > prior body” expansion check (bull)

function bullish_body_expands(curr, prev):

  return (curr.close > curr.open) AND (body_size(curr) > body_size(prev))


# -----------------------------

# Core: Confirm Candle Detection

# -----------------------------

function bullish_confirm(curr, prev, last3, i):

  return  (curr.close > RESIST) AND                 # must CLOSE above resist

          (NOT is_doji_like(curr)) AND

          bullish_body_expands(curr, prev) AND

          volume_expanding(curr, last3) AND

          (NOT USE_MACD OR macd_green_expanding(i))


function bearish_confirm(curr, prev, last3, i):

  return  (curr.close < SUPPORT) AND                # must CLOSE below support

          (NOT is_doji_like(curr)) AND

          bearish_body_engulf(curr, prev) AND

          volume_expanding(curr, last3) AND

          (NOT USE_MACD OR macd_red_expanding(i))


# -----------------------------

# Entry / Risk Model

# -----------------------------

function set_long_plan(curr):

  direction = "LONG"

  entry_price = curr.close


  # Stop: below broken level or below confirm candle low (whichever is tighter but valid)

  stop_price = min(RESIST, curr.low) - 0.15         # add small buffer; tune per TSLA spread/ATR


  targets = [

    430.00,

    435.75,

    440.66

  ]


function set_short_plan(curr):

  direction = "SHORT"

  entry_price = curr.close


  # Stop: above broken level or above confirm candle high (whichever is tighter but valid)

  stop_price = max(SUPPORT, curr.high) + 0.15       # buffer


  targets = [

    414.01,

    412.00,

    410.00

  ]


# -----------------------------

# Trade Management (simple)

# -----------------------------

function manage_trade(latest_price):

  if direction == "LONG":

    # hard stop

    if latest_price <= stop_price:

      EXIT_ALL("STOP_LONG")

      reset_state()


    # scale targets

    if latest_price >= targets[0]:

      TAKE_PARTIAL(0.33, "TP1_LONG")

      MOVE_STOP_TO(entry_price)                     # optional: break-even

    if latest_price >= targets[1]:

      TAKE_PARTIAL(0.33, "TP2_LONG")

      TRAIL_STOP("swing_low_or_vwap")               # optional

    if latest_price >= targets[2]:

      EXIT_ALL("TP3_LONG")

      reset_state()


  if direction == "SHORT":

    if latest_price >= stop_price:

      EXIT_ALL("STOP_SHORT")

      reset_state()


    if latest_price <= targets[0]:

      TAKE_PARTIAL(0.33, "TP1_SHORT")

      MOVE_STOP_TO(entry_price)

    if latest_price <= targets[1]:

      TAKE_PARTIAL(0.33, "TP2_SHORT")

      TRAIL_STOP("swing_high_or_vwap")

    if latest_price <= targets[2]:

      EXIT_ALL("TP3_SHORT")

      reset_state()


function reset_state():

  direction = "NONE"

  entry_price = null

  stop_price = null

  targets = []


# -----------------------------

# Main Loop — runs on candle close

# -----------------------------

ON_CANDLE_CLOSE(symbol=SYMBOL, tf=TF):

  curr = GET_CANDLE(0)              # just closed

  prev = GET_CANDLE(1)

  last3 = [GET_CANDLE(1), GET_CANDLE(2), GET_CANDLE(3)]

  i = CURRENT_CANDLE_INDEX()        # index of curr in series


  # If already in trade, do not re-signal; manage separately

  if direction != "NONE":

    return


  # Confirm & enter

  if bullish_confirm(curr, prev, last3, i):

    set_long_plan(curr)

    ALERT("BULL CONFIRM: 30m close > 425.50 w/ vol + MACD expansion. LONG enabled.")

    # OPTIONAL: Place order

    # PLACE_ORDER("BUY", qty=..., price="MKT")


  else if bearish_confirm(curr, prev, last3, i):

    set_short_plan(curr)

    ALERT("BEAR CONFIRM: 30m close < 420.23 w/ vol + MACD expansion + engulf. SHORT enabled.")

    # PLACE_ORDER("SELL_SHORT", qty=..., price="MKT")


  else:

    ALERT("NO CONFIRM: still inside 420.23–425.50 box. Stand down.")


# -----------------------------

# Tick / Price updates for management

# -----------------------------

ON_PRICE_UPDATE(symbol=SYMBOL):

  latest_price = LAST_PRICE()

  if direction != "NONE":

    manage_trade(latest_price)
