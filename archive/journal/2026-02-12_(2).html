Yes. Below is a minimal, sealed InvariantRegistry that:

• loads the canonical YAML once (startup)

• enforces SEALED + artifact name (fails hard if wrong)

• exposes get(id) and required_log_code(id) helpers

• returns copies to prevent mutation drift

• keeps the gate “flags-only” (registry does not evaluate conditions)


governance/invariant_registry.py


from __future__ import annotations


from dataclasses import dataclass

from pathlib import Path

from typing import Any, Dict, Optional

import copy


import yaml



class InvariantRegistryError(RuntimeError):

    pass



@dataclass(frozen=True)

class InvariantSpec:

    """Immutable view of an invariant."""

    raw: Dict[str, Any]


    @property

    def id(self) -> str:

        return self.raw["id"]


    @property

    def target_state(self) -> str:

        return self.raw["required_outcomes"]["state_transition"]["to"]


    @property

    def required_log_code(self) -> str:

        return self.raw["required_log_artifacts"][0]["code"]


    @property

    def required_log_fields(self) -> list[str]:

        return self.raw["required_log_artifacts"][0]["required_fields"]


    @property

    def escalation_required(self) -> bool:

        return bool(self.raw["required_outcomes"].get("escalation_required", False))


    @property

    def escalation_target(self) -> Optional[str]:

        return self.raw["required_outcomes"].get("escalation_target")



class InvariantRegistry:

    """

    Loads SEALED invariants once, serves immutable specs to runtime.


    Design constraints:

    - No condition parsing / eval

    - No runtime mutation of spec

    - Hard fail if schema is not SEALED or wrong artifact

    """


    _loaded: bool = False

    _schema: Dict[str, Any] = {}

    _by_id: Dict[str, Dict[str, Any]] = {}


    @classmethod

    def load(cls, schema_path: str | Path) -> None:

        if cls._loaded:

            return


        p = Path(schema_path)

        if not p.exists():

            raise InvariantRegistryError(f"Invariant schema not found: {p}")


        data = yaml.safe_load(p.read_text(encoding="utf-8"))


        # Hard seals

        if data.get("status") != "SEALED":

            raise InvariantRegistryError("Invariant schema must be SEALED")

        if data.get("artifact") != "S23_Envelope_Governance_Invariants":

            raise InvariantRegistryError("Unexpected artifact name (expected S23_Envelope_Governance_Invariants)")


        invs = data.get("invariants")

        if not isinstance(invs, list) or not invs:

            raise InvariantRegistryError("No invariants in schema")


        by_id: Dict[str, Dict[str, Any]] = {}

        for inv in invs:

            inv_id = inv.get("id")

            if not inv_id:

                raise InvariantRegistryError("Invariant missing id")

            if inv_id in by_id:

                raise InvariantRegistryError(f"Duplicate invariant id: {inv_id}")

            by_id[inv_id] = inv


        cls._schema = data

        cls._by_id = by_id

        cls._loaded = True


    @classmethod

    def get(cls, invariant_id: str) -> InvariantSpec:

        if not cls._loaded:

            raise InvariantRegistryError("InvariantRegistry not loaded. Call InvariantRegistry.load(...) at startup.")

        inv = cls._by_id.get(invariant_id)

        if inv is None:

            raise InvariantRegistryError(f"Unknown invariant id: {invariant_id}")


        # Return deep copy wrapped in immutable view to prevent accidental mutation drift.

        return InvariantSpec(raw=copy.deepcopy(inv))


    @classmethod

    def is_loaded(cls) -> bool:

        return cls._loaded

One strictness rule to keep it sealed


Load only from the canonical YAML (never from JSON) for runtime.

JSON stays as a derived artifact for tooling/other consumers.
