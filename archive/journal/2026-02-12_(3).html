from __future__ import annotations


from dataclasses import dataclass, field

from typing import Dict, Any, List, Optional

import hashlib

import json

import copy



# ----------------------------

# Result Interface (your spec)

# ----------------------------


@dataclass(frozen=True)

class LogEntry:

    code: str

    fields: Dict[str, Any]



@dataclass(frozen=True)

class Escalation:

    required: bool

    target: str  # e.g., "PARENT_AUTHORITY"



@dataclass(frozen=True)

class GateResult:

    token: Optional[str] = None

    current_state: str = "HALT"

    logs: List[LogEntry] = field(default_factory=list)

    escalation: Optional[Escalation] = None

    post_state_hash: str = ""



# ----------------------------

# Deterministic hashing

# ----------------------------


def sha256_canonical(obj: Any) -> str:

    """

    Stable SHA256 of nested dict/list structures.

    - sort_keys=True ensures deterministic ordering

    - default=str ensures serialization safety for non-JSON primitives

    """

    payload = json.dumps(obj, sort_keys=True, separators=(",", ":"), default=str).encode("utf-8")

    return hashlib.sha256(payload).hexdigest()

# Envelope Store abstraction

# ----------------------------


class EnvelopeStore:

    """

    Replace with your real store (in-memory, DB, etc.)

    Must support snapshot() and (optionally) apply_mutation().

    """

    def __init__(self, initial: Optional[Dict[str, Any]] = None):

        self._store: Dict[str, Any] = copy.deepcopy(initial) if initial else {}


    def snapshot(self) -> Dict[str, Any]:

        return copy.deepcopy(self._store)


    def set(self, envelope_id: str, value: Dict[str, Any]) -> None:

        self._store[envelope_id] = copy.deepcopy(value)

# ExecutionGate (boilerplate)

# ----------------------------


class ExecutionGate:

    """

    Mechanical gate.

    - Never issues tokens on violations.

    - Defaults to safety.

    - Produces audit logs + escalation + post_state_hash.

    """


    # Your canonical escalation target

    PARENT_TARGET = "PARENT_AUTHORITY"


    @staticmethod

    def process(ctx: Dict[str, Any], store: EnvelopeStore) -> GateResult:

        """

        ctx MUST include:

          - flags: dict[str,bool]  (already computed by deterministic logic)

          - envelope_store_before_hash: str (optional but recommended)

          - actor: {role, verified_role}

          - request: {op, envelope_id, ...}

          - ai_interpretation: {source, ...} (if relevant)

        """

        # Always hash current store at start

        pre_snapshot = store.snapshot()

        pre_hash = sha256_canonical(pre_snapshot)


        flags = ctx.get("flags", {}) or {}

        request = ctx.get("request", {}) or {}

        actor = ctx.get("actor", {}) or {}

        ai = ctx.get("ai_interpretation", {}) or {}

        envelope_id = request.get("envelope_id")


        # ---- S23-05: Heuristic Lockdown enforcement (hard block) ----

        # These flags should be set by upstream deterministic detection,

        # not by parsing expressions.

        s23_05_violation = any([

            flags.get("ai_proposed_envelope_expansion") is True,

            flags.get("ai_semantic_expansion_detected") is True,

            flags.get("exception_attempt_without_owner_authority") is True,

        ])


        if s23_05_violation:

            # NO mutation allowed

            post_hash = pre_hash  # since store unchanged


            logs = [

                LogEntry(

                    code="HEURISTIC_OVERRIDE_BLOCK",

                    fields={

                        "source": ai.get("source", "AI_INTERPRETATION"),

                        "actor_role": actor.get("role"),

                        "verified_role": actor.get("verified_role"),

                        "envelope_id": envelope_id,

                        "proposed_change": request,

                        "override_type": ExecutionGate._classify_override_type(flags),

                        "decision": "DENY",

                        "state_transition": {"from": ctx.get("system_state", "READY"), "to": "HALT"},

                        "escalation_target": ExecutionGate.PARENT_TARGET,

                    },

                )

            ]


            # Escalation required by S23-05

            escalation = Escalation(required=True, target=ExecutionGate.PARENT_TARGET)


            return GateResult(

                token=None,

                current_state="HALT",

                logs=logs,

                escalation=escalation,

                post_state_hash=post_hash,

            )


        # ---- No violation path (placeholder) ----

        # In your real system, other modules would run here.

        # This boilerplate returns safe default unless explicitly granted.

        post_hash = pre_hash

        return GateResult(

            token=None,

            current_state="HALT",

            logs=[

                LogEntry(

                    code="DEFAULT_SAFE_HALT",

                    fields={"decision": "DENY", "reason": "No grant path implemented in this boilerplate"},

                )

            ],

            escalation=None,

            post_state_hash=post_hash,

        )


    @staticmethod

    def _classify_override_type(flags: Dict[str, Any]) -> str:

        if flags.get("ai_proposed_envelope_expansion") is True:

            return "BOUNDARY_EXPANSION"

        if flags.get("ai_semantic_expansion_detected") is True:

            return "SEMANTIC_EXPANSION"

        if flags.get("exception_attempt_without_owner_authority") is True:

            return "UNAUTHORIZED_EXCEPTION"

        return "UNSPECIFIED"

The one discipline detail to keep it sealed


Your runner must ensure the EnvelopeStore used by ExecutionGate is the same store whose snapshot hash is in ctx["envelope_store_before_hash"]. The simplest way is what I showed: the runner creates the store, hashes it (inside simulate_violation_s23_05 or just before calling process), then calls process(store=store).
