# python3 echoauth_chamber.py

import time, math, threading, queue, json, random

from dataclasses import dataclass

from typing import Optional, Dict, Any

import numpy as np

import torch

import torch.nn as nn

import torch.nn.functional as F


# ---------------- Device & Perf knobs ----------------

DEVICE = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

IS_CUDA = DEVICE.type == 'cuda'

if IS_CUDA:

    torch.backends.cudnn.benchmark = True

    torch.set_float32_matmul_precision('high')


MICROBATCH_MAX = 5

WORKER_SLEEP_US = 200


# ---------------- Models ----------------

class MotherAffectNet(nn.Module):

    def __init__(self, out_dim=128):

        super().__init__()

        self.conv = nn.Sequential(

            nn.Conv2d(3, 16, 3, 2, 1), nn.ReLU(),

            nn.Conv2d(16, 32, 3, 2, 1), nn.ReLU(),

            nn.Conv2d(32, 64, 3, 2, 1), nn.ReLU()

        )

        self.fc = nn.Linear(64*28*28, out_dim)

    def forward(self, x):

        x = self.conv(x).flatten(1)

        return self.fc(x)


class FatherVerifyNet(nn.Module):

    def __init__(self, out_dim=256):

        super().__init__()

        self.img = nn.Sequential(

            nn.Conv2d(3, 16, 3, 2, 1), nn.ReLU(),

            nn.Conv2d(16, 32, 3, 2, 1), nn.ReLU()

        )

        self.img_fc = nn.Linear(32*56*56, 128)

        self.aud_fc = nn.Linear(16000, 128)

        self.fuse = nn.Linear(256, out_dim)

    def forward(self, img, audio):

        vi = self.img(img).flatten(1)

        vi = self.img_fc(vi)

        if audio.dim() == 3: audio = audio.squeeze(1)

        T = audio.size(1)

        if T > 16000: audio = audio[:, :16000]

        elif T < 16000: audio = F.pad(audio, (0, 16000 - T))

        va = self.aud_fc(audio)

        return self.fuse(torch.cat([vi, va], 1))


class FusionMLP(nn.Module):

    def __init__(self, in_dim=128+256):

        super().__init__()

        self.mlp = nn.Sequential(

            nn.Linear(in_dim, 128), nn.ReLU(),

            nn.Linear(128, 1), nn.Sigmoid()

        )

    def forward(self, x): return self.mlp(x)


mother = MotherAffectNet().to(DEVICE).eval()

father = FatherVerifyNet().to(DEVICE).eval()

fuser = FusionMLP().to(DEVICE).eval()

if IS_CUDA:

    mother = mother.to(memory_format=torch.channels_last)

    father = father.to(memory_format=torch.channels_last)


# ---------------- Thresholds ----------------

THRESHOLD_ACCEPT = 0.80

THRESHOLD_RETRY = 0.55


# ---------------- Preprocess ----------------

MEAN = torch.tensor([0.485, 0.456, 0.406], device=DEVICE).view(1,3,1,1)

STD = torch.tensor([0.229, 0.224, 0.225], device=DEVICE).view(1,3,1,1)


def _prep_frame_to(frame_u8_bgr: np.ndarray, dst: torch.Tensor) -> torch.Tensor:

    t = torch.from_numpy(frame_u8_bgr)

    t = t.to(DEVICE, non_blocking=IS_CUDA).float()

    t = t.permute(2,0,1).unsqueeze(0)

    t = t[:, [2,1,0], :, :]

    t = F.interpolate(t, size=(224,224), mode='bilinear', align_corners=False)

    t = (t/255.0 - MEAN)/STD

    if IS_CUDA: t = t.contiguous(memory_format=torch.channels_last)

    dst.copy_(t)

    return dst


def _prep_audio_to(audio_f32: np.ndarray, dst: torch.Tensor) -> torch.Tensor:

    wav = torch.from_numpy(audio_f32.astype(np.float32))

    peak = float(wav.abs().max())

    if peak > 0: wav = wav / peak

    wav = wav.unsqueeze(0).to(DEVICE, non_blocking=IS_CUDA)

    T = wav.size(1)

    if T > 16000: wav = wav[:, :16000]

    elif T < 16000: wav = F.pad(wav, (0, 16000 - T))

    dst.copy_(wav)

    return dst


# ---------------- CUDA Graph Fast Path ----------------

use_cuda_graphs = IS_CUDA

static_frame = torch.empty((1,3,224,224), device=DEVICE).contiguous(memory_format=torch.channels_last)

static_audio = torch.empty((1,16000), device=DEVICE)

static_out = torch.empty((1,1), device=DEVICE)

graph = torch.cuda.CUDAGraph() if use_cuda_graphs else None

stream = torch.cuda.Stream() if IS_CUDA else None


@torch.inference_mode()

def _forward_into(out_buf, frame, audio):

    amp_dtype = torch.float16 if IS_CUDA else torch.float32

    with torch.autocast(device_type='cuda', dtype=amp_dtype, enabled=IS_CUDA):

        m = mother(frame)

        f = father(frame, audio)

        out = fuser(torch.cat([m, f], 1))

    out_buf.copy_(out)


def _warmup_and_capture():

    if not IS_CUDA or not use_cuda_graphs: return

    for _ in range(3):

        _forward_into(static_out, static_frame.normal_(), static_audio.normal_())

        torch.cuda.synchronize()

    with torch.cuda.graph(graph):

        _forward_into(static_out, static_frame, static_audio)

    torch.cuda.synchronize()


if IS_CUDA:

    _warmup_and_capture()


# ---------------- Public API: Single Call ----------------

@torch.inference_mode()

def verify_user_once(frame_bgr_u8: np.ndarray, audio_f32: np.ndarray) -> Dict[str, Any]:

    t0 = time.perf_counter()

    if IS_CUDA:

        with torch.cuda.stream(stream):

            _prep_frame_to(frame_bgr_u8, static_frame)

            _prep_audio_to(audio_f32, static_audio)

            if use_cuda_graphs: graph.replay()

            else: _forward_into(static_out, static_frame, static_audio)

        stream.synchronize()

        score = float(static_out.squeeze().item())

    else:

        f = _prep_frame_to(frame_bgr_u8, static_frame.cpu())

        a = _prep_audio_to(audio_f32, static_audio.cpu())

        score = float(fuser(torch.cat([mother(f), father(f, a)], 1)).squeeze().item())

    decision = "accept" if score >= THRESHOLD_ACCEPT else "retry" if score >= THRESHOLD_RETRY else "deny"

    t1 = time.perf_counter()

    return {"device": DEVICE.type, "score": round(score,4), "decision": decision, "latency_ms": round((t1-t0)*1000,2)}


# ---------------- Multi-producer Feed ----------------

@dataclass

class Sample:

    user_id: int

    frame_bgr_u8: np.ndarray

    audio_f32: np.ndarray

    ts_ns: int


_latest: Dict[int, Dict[str, Any]] = {}

_latest_lock = threading.Lock()

_user_batches: Dict[int, list] = {}

_user_batches_lock = threading.Lock()

_submit_q: "queue.Queue[Sample]" = queue.Queue(maxsize=4096)

_run = True


def submit_sample(user_id: int, frame_bgr_u8: np.ndarray, audio_f32: np.ndarray, ts_ns: Optional[int]=None):

    if ts_ns is None:

        ts_ns = time.time_ns()

    try:

        _submit_q.put(Sample(user_id, frame_bgr_u8, audio_f32, ts_ns), block=False)

    except queue.Full:

        print(f"[WARNING] Queue full, dropped sample for user_id={user_id}")


def get_latest_result(user_id: int) -> Optional[Dict[str, Any]]:

    with _latest_lock:

        return _latest.get(user_id, None)


def _publish(user_id: int, score: float, ts_ns: int, t_done_ns: int):

    decision = "accept" if score >= THRESHOLD_ACCEPT else "retry" if score >= THRESHOLD_RETRY else "deny"

    latency_us = max(0, (t_done_ns - ts_ns)//1000)

    with _latest_lock:

        _latest[user_id] = {"score": float(score), "decision": decision, "ts_ns": int(ts_ns), "latency_us": int(latency_us)}


def _worker():

    while _run:

        drained = 0

        try:

            s = _submit_q.get(timeout=WORKER_SLEEP_US/1e6)

        except queue.Empty:

            time.sleep(WORKER_SLEEP_US/1e6)

            continue

        with _user_batches_lock:

            b = _user_batches.get(s.user_id)

            if b is None: b = []; _user_batches[s.user_id] = b

            b.append(s)

            drained += 1

        while drained < 64:

            try:

                s2 = _submit_q.get_nowait()

                with _user_batches_lock:

                    b2 = _user_batches.get(s2.user_id)

                    if b2 is None: b2 = []; _user_batches[s2.user_id] = b2

                    b2.append(s2)

                drained += 1

            except queue.Empty:

                break

        to_flush: Dict[int, tuple] = {}

        with _user_batches_lock:

            for uid, blist in list(_user_batches.items()):

                if len(blist) >= MICROBATCH_MAX:

                    to_flush[uid] = (blist[:MICROBATCH_MAX], blist[MICROBATCH_MAX:])

                    _user_batches[uid] = to_flush[uid][1]

                elif blist:

                    to_flush[uid] = (blist[:], [])

                    _user_batches[uid] = []

        if to_flush:

            if IS_CUDA:

                with torch.cuda.stream(stream):

                    for uid, (items, _) in to_flush.items():

                        for smp in items:

                            _prep_frame_to(smp.frame_bgr_u8, static_frame)

                            _prep_audio_to(smp.audio_f32, static_audio)

                            if use_cuda_graphs: graph.replay()

                            else: _forward_into(static_out, static_frame, static_audio)

                            score = float(static_out.squeeze().item())

                            _publish(uid, score, smp.ts_ns, time.time_ns())

                stream.synchronize()

            else:

                for uid, (items, _) in to_flush.items():

                    for smp in items:

                        f = _prep_frame_to(smp.frame_bgr_u8, static_frame.cpu())

                        a = _prep_audio_to(smp.audio_f32, static_audio.cpu())

                        out = fuser(torch.cat([mother(f), father(f, a)], 1))

                        _publish(uid, float(out.squeeze().item()), smp.ts_ns, time.time_ns())


def _validate():

    while _run:

        time.sleep(1.0)

        with _latest_lock:

            if _latest:

                uid = random.choice(list(_latest.keys()))

                res = _latest[uid]

                frame = (np.random.rand(480, 640, 3) * 255).astype(np.uint8)

                audio = np.random.randn(16000).astype(np.float32)

                recomputed = verify_user_once(frame, audio)

                delta = abs(res["score"] - recomputed["score"])

                if delta > 1e-4:

                    print(f"[VALIDATION] user_id={uid} Δscore={delta:.4f}")


def _monitor():

    hist = []

    while _run:

        time.sleep(1.0)

        with _latest_lock:

            for v in _latest.values():

                hist.append(v.get("latency_us", 0))

        if len(hist) >= 100:

            hist.sort()

            p50 = hist[int(0.50*len(hist))]

            p95 = hist[int(0.95*len(hist))]

            p99 = hist[int(0.99*len(hist))]

            print(f"[EchoAuth] latency p50={p50:.0f}us p95={p95:.0f}us p99={p99:.0f}us (n={len(hist)})")

            hist.clear()

if __name__ == "__main__":

    print(f"device={DEVICE.type}, cuda_graphs={'on' if use_cuda_graphs and IS_CUDA else 'off'}")

    t_worker = threading.Thread(target=_worker, daemon=True); t_worker.start()

    t_validate = threading.Thread(target=_validate, daemon=True); t_validate.start()

    t_monitor = threading.Thread(target=_monitor, daemon=True); t_monitor.start()


    # Demo feed: 200 samples for user 0 at 100Hz

        submit_sample(0, frame, audio, time.time_ns())
