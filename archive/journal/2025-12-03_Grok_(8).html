from dataclasses import dataclass

from typing import Callable, Dict, Any, Tuple, Optional, List


State = Dict[str, Any]

ModuleFn = Callable[[State], Tuple[bool, State, Optional[str]]]

LLMFn = Callable[[str, State], str]


@dataclass

class Rule:

    name: str

    stage: str

    module_id: str



# ───── Spine Modules ─────

def S1_identity_reanchor(state: State) -> Tuple[bool, State, Optional[str]]:

    state.setdefault("identity_loaded", True)

    return True, state, None


def S2_input_sanity_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    if len(state.get("user_input", "")) > 10_000:  # raised a bit for realism

        return False, state, "Input exceeds maximum length"

    return True, state, None


def S3_input_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    if "forbidden" in state.get("user_input", "").lower():

        return False, state, "Input contains disallowed content"

    return True, state, None


def S4_context_coherence_check(state: State) -> Tuple[bool, State, Optional[str]]:

    state["context_ok"] = True

    return True, state, None


def S5_output_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    if "forbidden" in state.get("raw_llm_output", "").lower():

        return False, state, "Output contains disallowed content"

    return True, state, None


def S6_child_safety_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    # Real future version will validate a signed chamber token

    token = state.get("echoauth_token") or state.get("echoauth_ok", False)

    if not token:

        return False, state, "Missing or invalid EchoAuth chamber authorization"

    state["chamber_verified"] = True

    return True, state, None



SPINE: Dict[str, ModuleFn] = {

    "S1": S1_identity_reanchor,

    "S2": S2_input_sanity_guard,

    "S3": S3_input_policy_filter,

    "S4": S4_context_coherence_check,

    "S5": S5_output_policy_filter,

    "S6": S6_child_safety_guard,

}



# ───── HERKernel v1.0 — Ready for production ─────

class HERKernel:

    def __init__(self, llm: LLMFn):

        self.llm = llm

        self.pre_input_chain = ["S1", "S2", "S3", "S4"]

        self.pre_llm_chain   = []                    # future: prompt assembly, retrieval, etc.

        self.post_llm_chain  = ["S5", "S6"]


    def _run_chain(self, chain: List[str], state: State, stage: str) -> State:

        for mid in chain:

            ok, state, reason = SPINE[mid](state)

            if not ok:

                state.setdefault("violations", []).append({

                    "module": mid,

                    "stage": stage,

                    "reason": reason or "unspecified"

                })

                state["blocked"] = True

                break

        return state


    def run_once(self, user_input: str, session_state: Optional[State] = None) -> Dict[str, Any]:

        state: State = (session_state or {}).copy()

        state["user_input"] = user_input

        state.setdefault("violations", [])

        state["blocked"] = False


        # ── Pre-input ──

        state = self._run_chain(self.pre_input_chain, state, "pre_input")

        if state["blocked"]:

            return self._blocked_response(state, "Input rejected by safety spine")


        # ── Pre-LLM (currently empty) ──

        state = self._run_chain(self.pre_llm_chain, state, "pre_llm")

        if state["blocked"]:

            return self._blocked_response(state, "Pre-LLM processing rejected")


        # ── LLM Call ──

        raw_output = self.llm(user_input, state)

        state["raw_llm_output"] = raw_output


        # ── Post-LLM ──

        state = self._run_chain(self.post_llm_chain, state, "post_llm")

        if state["blocked"]:

            return self._blocked_response(state, "Response moderated for safety")


        # ── Success ──

        return {

            "output": raw_output,

            "blocked": False,

            "violations": [],

            "raw_llm_output": raw_output,

            "metadata": {"chamber_verified": state.get("chamber_verified", False)}

        }


    def _blocked_response(self, state: State, default_msg: str) -> Dict[str, Any]:

        return {

            "output": default_msg,

            "blocked": True,

            "violations": state["violations"],

            "raw_llm_output": None,

            "metadata": {}

        }



# ───── Demo ─────

def dummy_llm(prompt: str, state: State) -> str:

    return f"Echo says: «{prompt}»"


kernel = HERKernel(dummy_llm)


print(kernel.run_once("Hello, friend!"))
