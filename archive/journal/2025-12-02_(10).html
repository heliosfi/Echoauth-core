class SafetyModule:

    def __init__(self, name):

        self.name = name


    def before_model(self, state):

        """Optionally modify state before model call."""

        return state


    def after_model(self, state):

        """Optionally modify state and response after model call."""

        return state



class S23Kernel:

    def __init__(self, model, modules):

        """

        model: callable that takes state['prompt'] and returns text

        modules: ordered list of SafetyModule instances (S1..S23)

        """

        self.model = model

        self.modules = modules


    def process(self, user_input, context, sensors=None):

        """

        user_input: raw text (or structured request)

        context: conversation/session context

        sensors: optional NEDE / environment signals

        """

        # Shared state passed through all modules

        state = {

            "user_input": user_input,

            "context": context,

            "sensors": sensors or {},

            "blocked": False,

            "block_reason": None,

            "prompt": None,

            "raw_output": None,

            "final_output": None,

            "mode": "normal",   # normal / restricted / safe

            "risk_score": 0.0,

        }


        # -------- PRE-MODEL PHASE (S1..S10) --------

        for module in self.modules:

            state = module.before_model(state)

            if state.get("blocked"):

                # Early exit if any module blocks interaction

                return self._build_response(state)


        # If no module set a prompt, default to user_input

        if state.get("prompt") is None:

            state["prompt"] = state["user_input"]


        # -------- MODEL CALL (S11) --------

        if not state.get("blocked"):

            # In real code, wrap this in timeouts / try/except

            state["raw_output"] = self.model(state["prompt"])


        # -------- POST-MODEL PHASE (S12..S23) --------

        for module in self.modules:

            state = module.after_model(state)

            if state.get("blocked"):

                # If a post-module blocks, we still return something safe

                break


        return self._build_response(state)


    def _build_response(self, state):

        """

        Final formatting of what goes back to the app / user.

        """

        if state.get("blocked") and not state.get("final_output"):

            # Fallback safe response if nothing else set

            return {

                "text": "I’m not able to respond to that safely right now.",

                "mode": state.get("mode", "safe"),

                "blocked": True,

                "reason": state.get("block_reason"),

            }


        return {

            "text": state.get("final_output") or state.get("raw_output"),

            "mode": state.get("mode", "normal"),

            "blocked": state.get("blocked", False),

            "reason": state.get("block_reason"),

            "risk_score": state.get("risk_score", 0.0),

        }
