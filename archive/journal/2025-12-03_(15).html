# echoauth_api.py

# EchoAuth ↔ NI-AI Gateway for ni_ai_kernel v1.4 (S1–S9)


from typing import Optional, Dict, Any, List


from fastapi import FastAPI, Header, HTTPException

from fastapi.responses import JSONResponse

from pydantic import BaseModel


from ni_ai_kernel import ni_ai, DEV_MODE  # import your kernel + dev flag



app = FastAPI(

    title="EchoAuth ↔ NI-AI Gateway",

    description="Secure, child-safe bridge between EchoAuth chamber and the HER/NI-AI kernel.",

    version="1.4.0",

    openapi_tags=[

        {"name": "chat", "description": "Main NI-AI interaction"},

        {"name": "system", "description": "Health & debug"},

    ],

)



# --------------------------------------------------------------------------- #

#                         REQUEST / RESPONSE MODELS                           #

# --------------------------------------------------------------------------- #


class ChatRequest(BaseModel):

    message: str

    echoauth_ok: Optional[bool] = None   # DEV only

    # later you can add: session_state: Dict[str, Any] | None = None



class ChatResponse(BaseModel):

    output: str

    blocked: bool

    metadata: Dict[str, Any]

    # we don’t expose trace to normal clients by default

    violations: List[Dict[str, Any]] = []



class DebugChatResponse(BaseModel):

    output: str

    blocked: bool

    metadata: Dict[str, Any]

    violations: List[Dict[str, Any]]

    trace: List[Dict[str, Any]]



# --------------------------------------------------------------------------- #

#                               MAIN CHAT ENDPOINT                            #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/chat", response_model=ChatResponse, tags=["chat"])

async def ni_ai_chat(

    req: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    Primary NI-AI chat endpoint.


    • DEV_MODE=True:

        - You can send {"echoauth_ok": true} in the body to simulate a safe chamber.

    • DEV_MODE=False:

        - You must send a real EchoAuth JWT in X-EchoAuth-Token (enforced in S6).

    """

    if not req.message or not req.message.strip():

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    # Build session_state (you can add more fields later)

    session_state: Dict[str, Any] = {}


    # DEV path: allow echoauth_ok flag

    echoauth_ok = bool(req.echoauth_ok) if DEV_MODE and req.echoauth_ok else False


    # PROD path: token flows to kernel

    echoauth_token = x_echoauth_token if x_echoauth_token else None


    result = ni_ai(

        user_message=req.message,

        echoauth_ok=echoauth_ok,

        echoauth_token=echoauth_token,

        session_state=session_state,

    )


    return ChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        metadata=result.get("metadata", {}),

        violations=result.get("violations", []),

    )



# --------------------------------------------------------------------------- #

#                             DEV DEBUG ENDPOINT                              #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/debug", response_model=DebugChatResponse, tags=["system"])

async def ni_ai_debug(req: ChatRequest):

    """

    DEV-ONLY endpoint.


    Returns full NI-AI kernel result including:

      • violations[]

      • metadata flags (comfort_mode, recovery_mode, etc.)

      • trace[] from S7_telemetry_logger


    DO NOT expose this in production to children-facing clients.

    """

    if not DEV_MODE:

        raise HTTPException(status_code=403, detail="Debug endpoint disabled in production")


    session_state: Dict[str, Any] = {}

    echoauth_ok = bool(req.echoauth_ok)


    result = ni_ai(

        user_message=req.message,

        echoauth_ok=echoauth_ok,

        echoauth_token=None,

        session_state=session_state,

    )


    return DebugChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        metadata=result.get("metadata", {}),

        violations=result.get("violations", []),

        trace=result.get("trace", []),

    )



# --------------------------------------------------------------------------- #

#                              HEALTH ENDPOINT                                #

# --------------------------------------------------------------------------- #


@app.get("/health", tags=["system"])

async def health():

    return {

        "status": "ok",

        "kernel_version": "HER/NI-AI v1.4 (S1–S9)",

        "dev_mode": DEV_MODE,

    }

Quick tests


1) No EchoAuth → blocked by S6 (in DEV, unless echoauth_ok=true)


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message":"Hello NI-AI"}'


You should see:

• blocked: true

• metadata.recovery_mode: true (S9)

• gentle text, not a harsh “blocked” string


⸻


2) Safe + EchoAuth OK → normal output


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message":"Tell me a story about a blue robot", "echoauth_ok": true}'


• blocked: false

• comfort_mode: false

• recovery_mode: false


⸻


3) Distress → S8 + S9 + S7 visible via debug


curl -X POST http://localhost:8000/ni-ai/debug \

  -H "Content-Type: application/json" \

  -d '{"message":"I feel so lonely, nobody loves me", "echoauth_ok": true}'


You should see in JSON:

• metadata.distress_detected: true

• metadata.comfort_mode: true

• metadata.recovery_mode: true

• trace array with S7 entries showing distress_detected and llm_suppressed.
