from typing import Dict


FLAIMO_SYSTEM_ID = "FLAIMO-Login-Tiered"

FLAIMO_VERSION = "0.2"


def flaimo_login_trace(user_id: str,

                       intent: str,

                       balance_snapshot: float) -> dict:

    """

    FLAIMO login trace:

    - user_id: internal identifier (no PII needed here)

    - intent: 'review', 'rebalance', 'withdraw', 'plan', etc.

    - balance_snapshot: portfolio value at login

    """

    uid = user_id.strip()

    it = intent.strip().lower()


    if not uid or not it:

        return {

            "system": FLAIMO_SYSTEM_ID,

            "status": "rejected",

            "reason": "missing_user_or_intent",

            "version": FLAIMO_VERSION

        }


    composite = f"{uid}:{it}:{balance_snapshot}"

    trace = hash(composite)


    risk_flag = "normal"

    if balance_snapshot < 0:

        risk_flag = "negative_equity"

    elif balance_snapshot > 1_000_000:

        risk_flag = "high_value_session"


    return {

        "system": FLAIMO_SYSTEM_ID,

        "status": "accepted",

        "user_id": uid,

        "intent": it,

        "balance_snapshot": balance_snapshot,

        "risk_flag": risk_flag,

        "trace": trace,

        "version": FLAIMO_VERSION

    }



def flaimo_tier_allocation(balance: float,

                           base_pct: float = 0.70,

                           accel_pct: float = 0.20,

                           options_pct: float = 0.10) -> Dict[str, float]:

    """

    Split balance into three conscious tiers:

      - base: core, long-term, safest layer

      - accel: growth-focused, higher volatility

      - options: tactical / leveraged / experiments


    Percentages must sum ~1.0 (we allow slight float error).

    """

    if balance <= 0:

        return {"base": 0.0, "accel": 0.0, "options": 0.0}


    total_pct = base_pct + accel_pct + options_pct

    if abs(total_pct - 1.0) > 1e-6:

        # Normalize to avoid hidden drift

        base_pct   = base_pct   / total_pct

        accel_pct  = accel_pct  / total_pct

        options_pct = options_pct / total_pct


    return {

        "base":   round(balance * base_pct, 2),

        "accel":  round(balance * accel_pct, 2),

        "options": round(balance * options_pct, 2),

    }



def flaimo_session_packet(user_id: str,

                          intent: str,

                          balance_snapshot: float) -> dict:

    """

    High-level packet:

      - login trace

      - tier allocation

      - combined hash for proof

    """

    login = flaimo_login_trace(user_id, intent, balance_snapshot)

    if login.get("status") != "accepted":

        return login  # bubble up rejection cleanly


    tiers = flaimo_tier_allocation(balance_snapshot)

    composite = f"{login['trace']}:{tiers['base']}:{tiers['accel']}:{tiers['options']}"

    session_trace = hash(composite)


    return {

        "system": FLAIMO_SYSTEM_ID,

        "status": "accepted",

        "user_id": login["user_id"],

        "intent": login["intent"],

        "balance_snapshot": login["balance_snapshot"],

        "risk_flag": login["risk_flag"],

        "tiers": tiers,

        "login_trace": login["trace"],

        "session_trace": session_trace,

        "version": FLAIMO_VERSION

    }



def log_flaimo_session(packet: dict, logfile: str = "flaimo_sessions.log") -> None:

    """

    Minimal log writer for proof-of-use.

    Each line: system | user | intent | balance | tiers | trace

    """

    if packet.get("status") != "accepted":

        return


    line = (

        f"{packet['system']} | v{packet['version']} | "

        f"user={packet['user_id']} | intent={packet['intent']} | "

        f"balance={packet['balance_snapshot']} | "

        f"tiers={packet['tiers']} | "

        f"session_trace={packet['session_trace']}\n"

    )


    with open(logfile, "a") as f:

        f.write(line)
