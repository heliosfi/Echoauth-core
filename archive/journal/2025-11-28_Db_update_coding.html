2️⃣ Tightened NI-AI engine with emotion score


Replace your core/ni_ai.py with this version:


# core/ni_ai.py


from typing import Dict, Any


DANGER_BEHAVIORS = [

    "head banging", "head-banging", "hitting self", "biting self",

    "running into street", "bolting", "self harm", "self-harm"

]


DISCOMFORT_BEHAVIORS = [

    "covering ears", "cover ears", "rocking", "pacing", "restless",

    "refusing food", "won't eat", "no eye contact"

]


ENVIRONMENT_TRIGGERS = [

    "loud", "noise", "crowded", "bright lights", "fluorescent",

    "sirens", "construction", "echo", "smell", "overstimulating"

]


POSITIVE_SIGNAL = [

    "laughing", "laugh", "smiling", "playing", "hug", "snuggling",

    "engaged", "focused", "calm", "relaxed"

]


def _score_from_tokens(text: str, tokens: list, weight: float) -> float:

    hits = sum(tok in text for tok in tokens)

    if hits == 0:

        return 0.0

    # saturate at 1.0

    return min(1.0, hits * weight)


def evaluate_cues(cues: Dict[str, Any]) -> Dict[str, Any]:

    """

    NI-AI MVP:


    - Computes emotion/distress score from behaviors + triggers.

    - Builds two internal 'threads' (cautious vs optimistic).

    - Computes conflict score from how close we are to thresholds.

    - Returns one final label, confidence, emotion_score.

    """


    text_blob = " ".join(str(v).lower() for v in cues.values())


    # Base distress components

    danger_score      = _score_from_tokens(text_blob, DANGER_BEHAVIORS,    0.6)

    discomfort_score  = _score_from_tokens(text_blob, DISCOMFORT_BEHAVIORS,0.3)

    env_score         = _score_from_tokens(text_blob, ENVIRONMENT_TRIGGERS,0.25)

    positive_score    = _score_from_tokens(text_blob, POSITIVE_SIGNAL,     0.4)


    # Combine into an emotion / distress score 0–1

    raw_distress = danger_score * 1.0 + discomfort_score * 0.7 + env_score * 0.5

    # Positive signals counterbalance distress

    raw_distress = max(0.0, raw_distress - positive_score * 0.6)

    emotion_score = max(0.0, min(1.0, raw_distress))


    # ─────────────────────────────

    # Internal threads (A = optimistic, B = cautious)

    # ─────────────────────────────

    # Thread A: if emotion_score is low, assume "Safe"

    if emotion_score < 0.3:

        thread_a_label = "Safe"

        thread_a_conf = 0.9

    elif emotion_score < 0.6:

        thread_a_label = "Safe"

        thread_a_conf = 0.6

    else:

        thread_a_label = "Needs Attention"

        thread_a_conf = 0.5


    # Thread B: caution-first

    if emotion_score > 0.7 or danger_score > 0.4:

        thread_b_label = "Needs Attention"

        thread_b_conf = 0.95

    elif emotion_score > 0.4:

        thread_b_label = "Needs Attention"

        thread_b_conf = 0.7

    else:

        thread_b_label = "Safe"

        thread_b_conf = 0.6


    # ─────────────────────────────

    # Conflict + final decision

    # ─────────────────────────────

    # If they agree, low conflict

    if thread_a_label == thread_b_label:

        conflict_score = abs(thread_a_conf - thread_b_conf) * 0.2

        final_label = thread_b_label  # they match

        final_conf = max(thread_a_conf, thread_b_conf)

    else:

        # disagree → conflict scales with how close emotion_score is to mid-zone

        mid = 0.5

        distance_from_mid = abs(emotion_score - mid)

        conflict_score = 1.0 - distance_from_mid  # closer to 0.5 → more conflict


        # cautious bias: choose thread B if conflict is high

        if conflict_score > 0.5:

            final_label = thread_b_label

            final_conf = thread_b_conf

        else:

            final_label = thread_a_label

            final_conf = thread_a_conf


    winning_thread = "thread_a" if final_label == thread_a_label else "thread_b"


    return {

        "final_label": final_label,

        "confidence": final_conf,

        "emotion_score": emotion_score,

        "danger_score": float(danger_score),

        "conflict_score": float(conflict_score),

        "winning_thread": winning_thread,

        "internal_threads": {

            "thread_a": {"label": thread_a_label, "base_conf": thread_a_conf},

            "thread_b": {"label": thread_b_label, "base_conf": thread_b_conf},

        },

    }


Key moves:

• You now have a distress / emotion_score (0–1) based on:

• direct self-harm cues

• discomfort behaviors

• environment overload

• positive balancing signals

• Two internal threads think differently, and the closer you are to uncertainty (around 0.5) the higher the conflict score.

→ mid emotion_score, medium conflict →

caution wins, may auto-flag for review depending on exact score.
