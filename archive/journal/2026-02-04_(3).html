from dataclasses import dataclass

from enum import Enum, auto

from typing import Optional


class State(Enum):

    NEUTRAL = auto()

    RISK_OFF_TREND_DOWN = auto()

    FLUSH_WAIT = auto()

    BOUNCE_ARMED = auto()

    PUT_ENTRY_READY = auto()

    IN_PUT = auto()

    ABORT = auto()


@dataclass

class Levels:

    bounce_zone_low: float = 176.0

    bounce_zone_high: float = 178.0

    hard_invalidate: float = 179.0

    trim_1: float = 173.5

    target_main: float = 169.5

    target_runner: float = 167.8


@dataclass

class Signals:

    price: float

    vwap: float

    vwap_slope_down: bool           # True if VWAP slope is negative

    ema_falling: bool               # True if fast EMA slope is negative

    structure_lower_highs_30m: bool # True if 30m market structure is LH/LL

    rsi_1m: float

    rsi_5m: float

    rsi_1h: float

    macd_5m: float                  # MACD value (or line) on 5m

    macd_hist_5m: float             # Histogram value on 5m

    macd_hist_turning_down: bool    # True if hist just flipped/rolled down

    vol_red_gt_green: bool          # Rejection volume confirmation

    wick_up: bool                   # Current candle shows upper wick at test

    next_candle_closed_red: bool    # The candle AFTER test closes red

    close_5m_above_vwap: bool       # 5m close above VWAP (trend repair)


@dataclass

class Position:

    in_put: bool = False

    entry_price: Optional[float] = None

    stop_price: Optional[float] = None

    size: float = 0.0


class SniperBot:

    def __init__(self, levels: Levels):

        self.levels = levels

        self.state = State.NEUTRAL

        self.pos = Position()


    def update_state(self, s: Signals):

        # ----- Master Trend Gate -----

        trend_down = (

            s.price < s.vwap

            and s.vwap_slope_down

            and s.ema_falling

            and s.structure_lower_highs_30m

            and s.rsi_1h < 35

        )


        if self.state != State.IN_PUT and not trend_down:

            self.state = State.NEUTRAL

            return


        if trend_down and self.state in [State.NEUTRAL, State.ABORT]:

            self.state = State.RISK_OFF_TREND_DOWN


        # ----- Hard Invalidation / Abort -----

        # VWAP reclaim + hold above 179 with volume = no puts

        if s.close_5m_above_vwap and s.price > self.levels.hard_invalidate:

            self.state = State.ABORT

            return


        # ----- Flush detection: RSI extreme means WAIT (do not add at lows) -----

        if self.state == State.RISK_OFF_TREND_DOWN:

            if s.rsi_1m <= 25 and s.price < self.levels.trim_1:

                # "new lows / flush zone" heuristic: oversold + below trim level

                self.state = State.FLUSH_WAIT


        # ----- Arm bounce only when RSI curls and selling pressure cools -----

        if self.state == State.FLUSH_WAIT:

            # RSI curl permission: from <25 into 30–40 band on 1m/5m

            if (s.rsi_1m >= 28 and s.rsi_1m <= 45) and (s.rsi_5m >= 30):

                self.state = State.BOUNCE_ARMED


        # ----- Entry readiness: bounce into zone + VWAP/EMA test -----

        if self.state == State.BOUNCE_ARMED:

            in_bounce_zone = self.levels.bounce_zone_low <= s.price <= self.levels.bounce_zone_high

            tested_vwap_or_ema = abs(s.price - s.vwap) <= 0.25 or True  # replace True with actual EMA test logic


            if in_bounce_zone and tested_vwap_or_ema:

                self.state = State.PUT_ENTRY_READY


        # ----- Confirm rejection and execute -----

        if self.state == State.PUT_ENTRY_READY:

            rejection = (

                s.wick_up

                and s.next_candle_closed_red

                and s.vol_red_gt_green

                and s.macd_5m < 0

                and s.macd_hist_turning_down

            )


            if rejection:

                self.enter_put(s)

                self.state = State.IN_PUT

            # If bounce becomes trend repair, abort

            elif s.close_5m_above_vwap and s.price > self.levels.hard_invalidate:

                self.state = State.ABORT


        # ----- Manage open position -----

        if self.state == State.IN_PUT and self.pos.in_put:

            self.manage_put(s)


    def enter_put(self, s: Signals):

        # Entry logic — size and fill handled elsewhere

        self.pos.in_put = True

        self.pos.entry_price = s.price

        self.pos.size = 1.0


        # Stop: above rejection high OR VWAP reclaim zone

        # Here we approximate: stop just above hard invalidate line

        self.pos.stop_price = max(self.levels.hard_invalidate + 0.15, s.vwap + 0.15)


        print(f"[ENTER PUT] price={s.price:.2f} stop={self.pos.stop_price:.2f}")


    def manage_put(self, s: Signals):

        # 1) Stop condition

        if s.price >= (self.pos.stop_price or 10**9) or (s.close_5m_above_vwap and s.price > self.levels.hard_invalidate):

            self.exit_put(reason="STOP / VWAP RECLAIM")

            self.state = State.ABORT

            return


        # 2) Profit taking ladder

        if s.price <= self.levels.trim_1:

            self.take_partial("TRIM_1")


        if s.price <= self.levels.target_main:

            self.take_majority("TARGET_MAIN")


        # Runner allowed only if momentum expands (proxy: hist more negative / strong red)

        if s.price <= self.levels.target_runner and s.macd_hist_5m < 0:

            self.hold_runner("RUNNER_ALLOWED")


    def take_partial(self, tag: str):

        # Example: reduce position by 30%

        if self.pos.size > 0.7:

            self.pos.size *= 0.7

            print(f"[TAKE PARTIAL] {tag} new_size={self.pos.size:.2f}")


    def take_majority(self, tag: str):

        # Example: reduce to a runner

        if self.pos.size > 0.25:

            self.pos.size = 0.25

            print(f"[TAKE MAJORITY] {tag} runner_size={self.pos.size:.2f}")


    def hold_runner(self, tag: str):

        print(f"[HOLD RUNNER] {tag}")


    def exit_put(self, reason: str):

        print(f"[EXIT PUT] reason={reason} entry={self.pos.entry_price} remaining_size={self.pos.size}")

        self.pos = Position()
