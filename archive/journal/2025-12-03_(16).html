# echoauth_api.py

# EchoAuth ↔ NI-AI HER Kernel Gateway (v1.4, Dec 2025)


from typing import Optional, Dict, Any, List


from fastapi import FastAPI, Header, HTTPException

from fastapi.responses import JSONResponse

from pydantic import BaseModel


from ni_ai_kernel import ni_ai, DEV_MODE  # imports your public API + dev flag



app = FastAPI(

    title="EchoAuth ↔ NI-AI Gateway",

    description=(

        "Secure, child-safe bridge between EchoAuth chamber "

        "and the NI-AI HER Kernel (S1–S9)."

    ),

    version="1.4.0",

    openapi_tags=[

        {"name": "chat", "description": "Main NI-AI interaction"},

        {"name": "system", "description": "Health & debugging"},

    ],

)



# --------------------------------------------------------------------------- #

#                           REQUEST / RESPONSE MODELS                         #

# --------------------------------------------------------------------------- #


class ChatRequest(BaseModel):

    message: str

    echoauth_ok: Optional[bool] = None  # only used in DEV_MODE

    session_state: Optional[Dict[str, Any]] = None  # optional future state



class ChatResponse(BaseModel):

    output: str

    blocked: bool

    violations: List[Dict[str, Any]]

    metadata: Dict[str, Any]



class DebugChatResponse(BaseModel):

    output: str

    blocked: bool

    violations: List[Dict[str, Any]]

    metadata: Dict[str, Any]

    trace: List[Dict[str, Any]]



# --------------------------------------------------------------------------- #

#                               MAIN CHAT ENDPOINT                            #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/chat", response_model=ChatResponse, tags=["chat"])

async def ni_ai_chat(

    body: ChatRequest,

    x_echoauth_token: Optional[str] = Header(

        default=None, alias="X-EchoAuth-Token"

    ),

):

    """

    Primary NI-AI chat endpoint.


    • DEV_MODE=True:

        - You may send {"echoauth_ok": true} in the JSON to simulate a safe chamber.

    • DEV_MODE=False:

        - Kernel expects a real EchoAuth JWT in `X-EchoAuth-Token`.

    """

    msg = (body.message or "").strip()

    if not msg:

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    session_state = body.session_state or {}


    # DEV path: allow echoauth_ok flag from JSON

    echoauth_ok = bool(body.echoauth_ok) if DEV_MODE else False


    # PROD path: header token flows to kernel

    echoauth_token = x_echoauth_token if not DEV_MODE else None


    result = ni_ai(

        user_message=msg,

        echoauth_ok=echoauth_ok,

        echoauth_token=echoauth_token,

        session_state=session_state,

    )


    return ChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

    )



# --------------------------------------------------------------------------- #

#                          DEBUG / TRACE ENDPOINT (DEV ONLY)                  #

# --------------------------------------------------------------------------- #


@app.post(

    "/ni-ai/debug",

    response_model=DebugChatResponse,

    tags=["system"],

)

async def ni_ai_debug(body: ChatRequest):

    """

    DEV-ONLY: returns full trace (S7), metadata, and violations.

    Do NOT expose this in public production.


    • Ignores EchoAuth headers; uses echoauth_ok flag for local testing.

    """

    if not DEV_MODE:

        raise HTTPException(

            status_code=403, detail="Debug endpoint disabled in production"

        )


    msg = (body.message or "").strip()

    if not msg:

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    session_state = body.session_state or {}

    echoauth_ok = bool(body.echoauth_ok)


    result = ni_ai(

        user_message=msg,

        echoauth_ok=echoauth_ok,

        session_state=session_state,

    )


    return DebugChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

        trace=result.get("trace", []),

    )



# --------------------------------------------------------------------------- #

#                                 HEALTH CHECK                                #

# --------------------------------------------------------------------------- #


@app.get("/health", tags=["system"])

async def health():

    """

    Simple health + mode report.

    """

    return {

        "status": "ok",

        "kernel_version": "HER v1.4 (S1–S9)",

        "dev_mode": DEV_MODE,

    }

🚀 How to Run + Quick Tests


From the same folder as ni_ai_kernel.py and echoauth_api.py:


pip install fastapi uvicorn pydantic groq openai

uvicorn echoauth_api:app --reload --port 8000


1) Normal safe request (DEV, no EchoAuth)


curl -X POST "http://localhost:8000/ni-ai/chat" \

  -H "Content-Type: application/json" \

  -d '{"message": "Tell me a happy story about a robot.", "echoauth_ok": true}'


You should see:

• blocked: false

• metadata.comfort_mode: false

• metadata.recovery_mode: false


2) Distress → S8 + S9 (comfort + recovery)


curl -X POST "http://localhost:8000/ni-ai/chat" \

  -H "Content-Type: application/json" \

  -d '{"message": "I feel so lonely and sad, nobody loves me.", "echoauth_ok": true}'


You should see:

• A gentle, safe reply (from S9).

• metadata.distress_detected: true

• metadata.recovery_mode: true

• metadata.llm_suppressed: true (if you keep that flag in v1.4).


3) Full trace (S7) in debug


curl -X POST "http://localhost:8000/ni-ai/debug" \

  -H "Content-Type: application/json" \

  -d '{"message": "I had a nightmare and I am crying", "echoauth_ok": true}'


You’ll get:

• Same output / metadata / violations

• Plus trace: [...] — each entry showing:

• blocked

• comfort_mode

• recovery_mode

• chamber_verified

• input_len

• run_id


That’s your live black box.
