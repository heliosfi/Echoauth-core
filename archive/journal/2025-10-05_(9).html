# --- add near top with other imports ---

import threading

import simpleaudio as sa  # pip install simpleaudio


# --- after your AAC_SYMBOLS dict, add tone presets (feel free to edit) ---

MUSIC_TONES = {

    "relax_528": {"frequency": 528.0, "mode": "sine"},      # commonly used relaxing tone

    "focus_432": {"frequency": 432.0, "mode": "sine"},      # softer, mellow feel

    "ground_396": {"frequency": 396.0, "mode": "sine"},     # low grounding tone

    "binaural_10": {"frequency": 440.0, "mode": "binaural", "beat_hz": 10.0},  # 440/450 Hz ears

}


# --- global player state ---

_play_lock = threading.Lock()

_play_obj = None


def _make_tone(

    freq_hz: float,

    duration_sec: float = 5.0,

    sample_rate: int = 44100,

    volume: float = 0.5

) -> np.ndarray:

    """Mono sine wave, int16 for simpleaudio."""

    t = np.linspace(0, duration_sec, int(sample_rate*duration_sec), endpoint=False)

    wave = np.sin(2*np.pi*freq_hz*t)

    wave = (volume * wave).clip(-1.0, 1.0)

    return (wave * 32767).astype(np.int16)


def _make_binaural(

    carrier_hz: float,

    beat_hz: float = 10.0,

    duration_sec: float = 5.0,

    sample_rate: int = 44100,

    volume: float = 0.5

) -> np.ndarray:

    """

    Return stereo int16: left=carrier, right=carrier+beat.

    Note: Binaural beats rely on headphones; individual comfort varies.

    """

    n = int(sample_rate*duration_sec)

    t = np.arange(n) / sample_rate

    left = np.sin(2*np.pi*carrier_hz*t)

    right = np.sin(2*np.pi*(carrier_hz + beat_hz)*t)

    left = (volume * left).clip(-1.0, 1.0)

    right = (volume * right).clip(-1.0, 1.0)

    stereo = np.stack([left, right], axis=1)  # (N,2), float

    return (stereo * 32767).astype(np.int16)


def _stop_audio_locked():

    global _play_obj

    if _play_obj is not None:

        try:

            _play_obj.stop()

        except Exception:

            pass

        _play_obj = None


@app.route("/music", methods=["POST"])

def music():

    """

    JSON:

      {

        "action": "start" | "stop",

        "tone": "relax_528" | "focus_432" | "ground_396" | "binaural_10",

        "duration": 5.0,          # seconds (only for action=start)

        "volume": 0.5             # 0.1..1.0

      }

    """

    try:

        data = request.get_json(force=True)

        action = data.get("action", "start")

        volume = float(data.get("volume", 0.5))

        volume = max(0.1, min(1.0, volume))


        with _play_lock:

            if action == "stop":

                _stop_audio_locked()

                return jsonify({"status": "stopped"})


            # start

            tone_key = data.get("tone", "relax_528")

            cfg = MUSIC_TONES.get(tone_key, MUSIC_TONES["relax_528"])

            duration = float(data.get("duration", 5.0))

            duration = max(1.0, min(60.0, duration))  # cap at 60s for safety


            _stop_audio_locked()  # stop any playing sound first


            mode = cfg.get("mode", "sine")

            if mode == "binaural":

                carrier = float(cfg.get("frequency", 440.0))

                beat = float(cfg.get("beat_hz", 10.0))

                stereo_i16 = _make_binaural(carrier, beat, duration, 44100, volume)

                _play_obj = sa.play_buffer(stereo_i16, num_channels=2, bytes_per_sample=2, sample_rate=44100)

            else:

                freq = float(cfg.get("frequency", 528.0))

                mono_i16 = _make_tone(freq, duration, 44100, volume)

                _play_obj = sa.play_buffer(mono_i16, num_channels=1, bytes_per_sample=2, sample_rate=44100)


            # (Optional) log encrypted note to records

            try:

                user_id = int(data.get("user_id", 1))

                desc = f"Played tone '{tone_key}' vol={volume} dur={duration}s"

                ts = time.strftime("%Y-%m-%d %H:%M:%S")

                enc = cipher.encrypt(desc.encode("utf-8"))

                conn.execute("INSERT INTO records VALUES (?, ?, ?)", (user_id, enc, ts))

                conn.commit()

            except Exception as e:

                app.logger.warning(f"Music log insert skipped: {e}")


            return jsonify({"status": "playing", "tone": tone_key, "volume": volume, "duration": duration})

    except Exception as e:

        app.logger.exception("Music endpoint error")

        return jsonify({"status": "error", "message": str(e)}), 400


Notes:

• Tones are generated locally (no internet), converted to int16 for simpleaudio.

• Binaural beats require headphones (we keep volume capped and optional).

• We log a short encrypted note to your records DB—handy for caregivers.
