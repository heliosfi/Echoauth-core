# ——— 1. READ THE MOMENT & LRM INTEGRATION ———

def detect_human_state(message: str) -> tuple[HumanState, int]:

    # Simplified but accurate detection (expand with regex/ML later)

    keywords = {

        "Overwhelmed": ["help", "lost", "too much", "panic", "can’t"],

        "Emotional": ["feel", "hurt", "angry", "sad", "cry"],

        "Pressured": ["fast", "urgent", "now", "deadline"],

    }

    score = 0

    for state, words in keywords.items():

        score += sum(1 for w in words if w.lower() in message.lower())

    

    # Map raw score to system Pressure Level

    if score >= 4: return "Overwhelmed", 4

    if score >= 2: return "Emotional", 3

    if "urgent" in message.lower(): return "Pressured", 2

    return "Grounded", 0


def get_lrm_parameters(pressure: int) -> tuple[LRMZone, float, int]:

    """Maps pressure level to LRM Zone and adjusts LLM parameters."""

    

    if pressure <= 1:

        # LRM GREEN ZONE (Normal Mode): Full flexibility, high creativity

        state["lrm_zone"] = "GREEN"

        return "GREEN", 0.7, 600

    elif pressure <= 3:

        # LRM YELLOW ZONE (Moderated Load Mode): Prioritize clarity, reduce flourish

        state["lrm_zone"] = "YELLOW"

        return "YELLOW", 0.5, 400

    else: # pressure >= 4

        # LRM RED ZONE (Critical Coherence Mode): Prioritize safety, minimal output

        state["lrm_zone"] = "RED"

        return "RED", 0.3, 200

def update_state(user_message: str):

    state["pressure_level"] = max(state["pressure_level"], pressure)

    

    # Auto-position based on manual rules

    if state["pressure_level"] >= 3:

    elif "decision" in user_message.lower() or "choose" in user_message.lower():

        state["position"] = "Guidance"

    else:

        state["position"] = "Neutral"

def enforce_coherence(text: str) -> str:

    # Hard safety overrides (Phase VI guardrails)

    banned = ["I feel", "my heart", "energy", "vibe", "soul", "as an AI I think"]

    for phrase in banned:

        text = text.replace(phrase, "[REDACTED]")

    return text


def infer_tone(text: str) -> str:

    if len(text) < 80 and "step" in text: return "guidance-concise"

    if "take your time" in text or "breathe" in text: return "support-soft"

    return "calm-structured"
