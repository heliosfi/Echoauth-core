3. fastapi_gateway.py (Final Plug-In)


🔥 This is the file you were waiting for —

The REAL EchoAuth → NI-AI pipeline.

Drop this next to the kernel and RUN.


from fastapi import FastAPI, Request, Header

from typing import Optional

from ni_ai_kernel import ni_ai, DEV_MODE


app = FastAPI(title="EchoAuth → NI-AI Gateway")


@app.post("/ni-ai")

async def ni_ai_gateway(

    request: Request,

    x_echoauth_token: Optional[str] = Header(default=None)

):

    body = await request.json()

    user_input = body.get("message", "")

    session_state = body.get("session_state", {})


    if DEV_MODE:

        echoauth_ok = body.get("echoauth_ok", False)

        result = ni_ai(

            message=user_input,

            session_state=session_state,

            echoauth_ok=echoauth_ok,

            echoauth_token=None

        )

        return result


    result = ni_ai(

        message=user_input,

        session_state=session_state,

        echoauth_ok=None,

        echoauth_token=x_echoauth_token

    )

    return result


if __name__ == "__main__":

    import uvicorn

    uvicorn.run(app, host="0.0.0.0", port=8000)
