• Prototype Code: Here’s a simple Flask app integrating EchoAuth for auth and basic record sharing (run with python app.py). It uses EchoAuth for login verification before allowing record access.

from flask import Flask, request, jsonify

import numpy as np

from echoauth_chamber import verify_user_once  # From echoauth_chamber.py

import sqlite3

from cryptography.fernet import Fernet  # For encryption


app = Flask(__name__)


# Dummy encryption key (generate once with Fernet.generate_key())

key = Fernet.generate_key()

cipher = Fernet(key)


# SQLite setup for records

conn = sqlite3.connect('records.db', check_same_thread=False)

conn.execute('CREATE TABLE IF NOT EXISTS records (user_id INTEGER, note BLOB, date TEXT)')

conn.commit()


@app.route('/auth', methods=['POST'])

def auth():

    frame = np.frombuffer(request.files['frame'].read(), dtype=np.uint8).reshape((480, 640, 3))  # Adjust shape

    audio = np.frombuffer(request.files['audio'].read(), dtype=np.float32)

    result = verify_user_once(frame, audio)

    if result['decision'] == 'accept':

        return jsonify({'status': 'authenticated', 'score': result['score']})

    return jsonify({'status': 'denied', 'score': result['score']}), 401


@app.route('/add_record', methods=['POST'])

def add_record():

    encrypted_note = cipher.encrypt(note.encode())

    conn.execute('INSERT INTO records VALUES (?, ?, ?)', (user_id, encrypted_note, date))

    conn.commit()

    return jsonify({'status': 'success'})


@app.route('/get_records', methods=['GET'])

def get_records():

    user_id = request.args.get('user_id')

    cur = conn.execute('SELECT note, date FROM records WHERE user_id = ?', (user_id,))

    rows = cur.fetchall()

    return jsonify(decrypted)


if __name__ == '__main__':

    app.run(debug=True)
