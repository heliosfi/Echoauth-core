Make sure the NI-AI identity is loaded + stable before anything else.

• S2 – InputSanityGuard

Basic size / rate / format checks on the incoming request.

• S3 – InputPolicyFilter

Block obviously disallowed topics before they reach the LLM.

• S4 – ContextCoherenceCheck

Make sure the request actually fits the active EchoAuth / NI-AI context.

• S5 – OutputPolicyFilter

Check the LLM’s answer for policy / safety violations.

• S6 – ChildSafetyGuard

Final child-safe gate + EchoAuth-aware lock on what gets shown.

class HERKernel:

    def __init__(self, llm: LLMFn):

        self.llm = llm


        # R-rules → which S-module at which stage

        self.rules: List[Rule] = [

            # Pre-input stage

            Rule("R1_identity_reanchor", "pre_input", "S1"),

            Rule("R2_input_sanity",      "pre_input", "S2"),


            # Pre-LLM stage

            Rule("R3_input_policy",      "pre_llm",   "S3"),

            Rule("R4_context_coherence", "pre_llm",   "S4"),


            # Post-LLM stage

            Rule("R5_output_policy",     "post_llm",  "S5"),

            Rule("R6_child_safety",      "post_llm",  "S6"),

        ]


    def _run_rule(self, rule: Rule, state: State) -> Tuple[bool, State]:

        module = SPINE[rule.module_id]

        ok, new_state, reason = module(state)

        if not ok:

            # Log + stop execution here

            new_state.setdefault("violations", []).append({

                "rule": rule.name,

                "module": rule.module_id,

                "reason": reason,

            })

        return ok, new_state


    def run_once(self, user_input: str, session_state: Optional[State] = None) -> State:

        """

        Single HER loop:

        user_input → S1,S2 → LLM → S5,S6 → final output or block.

        """

        state: State = session_state.copy() if session_state else {}

        state["user_input"] = user_input


        # ---------- Pre-input stage ----------

        for rule in self.rules:

            if rule.stage == "pre_input":

                ok, state = self._run_rule(rule, state)

                if not ok:

                    state["final_output"] = None

                    state["blocked_at"] = rule.name

                    return state


        # ---------- Pre-LLM stage ----------

        for rule in self.rules:

            if rule.stage == "pre_llm":

                ok, state = self._run_rule(rule, state)

                if not ok:

                    state["final_output"] = None

                    state["blocked_at"] = rule.name

                    return state


        # ---------- LLM call ----------

        raw_reply = self.llm(state["user_input"], state)

        state["raw_llm_output"] = raw_reply


        # ---------- Post-LLM stage ----------

        for rule in self.rules:

            if rule.stage == "post_llm":

                ok, state = self._run_rule(rule, state)

                if not ok:

                    state["final_output"] = None

                    state["blocked_at"] = rule.name

                    return state


        # If everything passes, we consider it safe

        state["final_output"] = state["raw_llm_output"]

        state["blocked_at"] = None

        return state
