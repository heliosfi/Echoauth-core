def preprocess_audio_gpu(audio_f32, orig_sr=44100):

    if not torch.is_tensor(audio_f32):

    else:

        wav = audio_f32.float()

        wav = wav / peak

# ---------- Core EchoAuth step ----------

@torch.inference_mode()

def echoauth(frame_gpu, audio_gpu):

    m_feat = mother(frame_gpu)

    f_feat = father(frame_gpu, audio_gpu)

    fused = fuser(torch.cat([m_feat, f_feat], dim=1))

    score = float(fused.squeeze().item())

    return score


def verify_user(frame_bgr_u8, audio_f32):

    frame = preprocess_gpu(frame_bgr_u8)

    audio = preprocess_audio_gpu(audio_f32)

    score = echoauth(frame, audio)

    return score
