# ============================================================

# SNIPER BOT — TSLA Breakout Confirm + Risk Gates (30m)

# Adds:

#   - ATR-based buffers (instead of fixed $0.15)

#   - MAX_CANDLE_MULT gate (don’t chase oversized confirm candles)

#   - MAX_RISK_DOLLARS gate (don’t take trades with too-wide stop)

#   - RETEST_ENTRY mode (optional): enter on retest instead of breakout close

# ============================================================


CONFIG:

  SYMBOL = "TSLA"

  TF = "30m"


  SUPPORT = 420.23

  RESIST  = 425.50


  VOLUME_LOOKBACK = 3

  MIN_BODY_RATIO = 0.55


  USE_MACD = true


  # --- Risk gates ---

  ATR_LEN = 14

  STOP_ATR_MULT = 0.35                 # buffer size = ATR * 0.35 (tune 0.25–0.60)

  MAX_CANDLE_MULT = 1.50               # reject confirm candle if range > ATR * 1.50

  MAX_RISK_DOLLARS = 2.50              # max distance entry->stop (shares); options use underlying proxy

  MIN_RR_TP1 = 1.0                     # require at least 1R to TP1


  # --- Entry mode ---

  ENTRY_MODE = "RETEST"                # "CLOSE" or "RETEST"

  RETEST_TOL = 0.20                    # acceptable proximity to level on retest (dollars)

  RETEST_TIMEOUT_BARS = 6              # how many 30m bars we wait for a retest


STATE:

  direction = "NONE"                   # NONE | LONG | SHORT

  entry_price = null

  stop_price = null

  targets = []

  pending_signal = "NONE"              # NONE | LONG | SHORT

  pending_level = null

  pending_expiry_bar = null


# -----------------------------

# Helpers

# -----------------------------

function body_size(c): return abs(c.close - c.open)

function range_size(c): return max(c.high - c.low, 0.0001)

function body_ratio(c): return body_size(c) / range_size(c)

function is_doji_like(c): return body_ratio(c) < MIN_BODY_RATIO

function mean_volume(candles): return mean([c.volume for c in candles])


function volume_expanding(curr, last3):

  return curr.volume >= mean_volume(last3)


function bullish_body_expands(curr, prev):

  return (curr.close > curr.open) AND (body_size(curr) > body_size(prev))


function bearish_body_engulf(curr, prev):

  return (curr.close < curr.open) AND

         (curr.open >= prev.close) AND

         (curr.close <= prev.open)


# MACD histogram expansion checks (closed candles)

function macd_hist(i): return MACD_HIST_AT(i)

function macd_green_expanding(i): return macd_hist(i) > 0 AND macd_hist(i) > macd_hist(i-1)

function macd_red_expanding(i): return macd_hist(i) < 0 AND macd_hist(i) < macd_hist(i-1)


function ATR(i, len): return ATR_AT(i, len)   # ATR value at candle i (closed)


# Size / chase gates

function confirm_candle_not_too_big(curr, atr):

  return range_size(curr) <= atr * MAX_CANDLE_MULT


function calc_stop_buffer(atr):

  return atr * STOP_ATR_MULT


function risk_dollars(entry, stop):

  return abs(entry - stop)


function rr_ok(entry, stop, tp1):

  r = risk_dollars(entry, stop)

  if r <= 0: return false

  return abs(tp1 - entry) / r >= MIN_RR_TP1


# -----------------------------

# Confirm Candle Detection (unchanged logic + size gate)

# -----------------------------

function bullish_confirm(curr, prev, last3, i, atr):

  return (curr.close > RESIST) AND

         (NOT is_doji_like(curr)) AND

         bullish_body_expands(curr, prev) AND

         volume_expanding(curr, last3) AND

         confirm_candle_not_too_big(curr, atr) AND

         (NOT USE_MACD OR macd_green_expanding(i))


function bearish_confirm(curr, prev, last3, i, atr):

  return (curr.close < SUPPORT) AND

         (NOT is_doji_like(curr)) AND

         bearish_body_engulf(curr, prev) AND

         volume_expanding(curr, last3) AND

         confirm_candle_not_too_big(curr, atr) AND

         (NOT USE_MACD OR macd_red_expanding(i))


# -----------------------------

# Plan Builders (ATR stops + risk gates)

# -----------------------------

function build_long_plan(entry_basis_price, confirm_candle, atr):

  buf = calc_stop_buffer(atr)


  # stop: below broken level OR below confirm low, choose the lower (safer) then subtract buffer

  raw_stop = min(RESIST, confirm_candle.low)

  stop = raw_stop - buf


  # targets

  tp1 = 430.00

  tp2 = 435.75

  tp3 = 440.66


  # max risk gate

  if risk_dollars(entry_basis_price, stop) > MAX_RISK_DOLLARS:

    return null


  # R:R gate

  if NOT rr_ok(entry_basis_price, stop, tp1):

    return null


  return {dir:"LONG", entry:entry_basis_price, stop:stop, targets:[tp1,tp2,tp3]}


function build_short_plan(entry_basis_price, confirm_candle, atr):

  buf = calc_stop_buffer(atr)


  raw_stop = max(SUPPORT, confirm_candle.high)

  stop = raw_stop + buf


  tp1 = 414.01

  tp2 = 412.00

  tp3 = 410.00


  if risk_dollars(entry_basis_price, stop) > MAX_RISK_DOLLARS:

    return null

  if NOT rr_ok(entry_basis_price, stop, tp1):

    return null


  return {dir:"SHORT", entry:entry_basis_price, stop:stop, targets:[tp1,tp2,tp3]}


# -----------------------------

# Entry Modes

# -----------------------------

function arm_retest(dir, level, current_bar_index):

  pending_signal = dir

  pending_level = level

  pending_expiry_bar = current_bar_index + RETEST_TIMEOUT_BARS


function retest_hit(price, level):

  return abs(price - level) <= RETEST_TOL


# -----------------------------

# Main — runs on 30m candle close

# -----------------------------

ON_CANDLE_CLOSE(symbol=SYMBOL, tf=TF):

  if direction != "NONE":

    return


  curr = GET_CANDLE(0)                 # closed

  prev = GET_CANDLE(1)

  last3 = [GET_CANDLE(1), GET_CANDLE(2), GET_CANDLE(3)]

  i = CURRENT_CANDLE_INDEX()


  atr = ATR(i, ATR_LEN)


  # 1) If waiting on a retest, keep waiting or expire

  if pending_signal != "NONE":

    if i > pending_expiry_bar:

      pending_signal = "NONE"

      pending_level = null

      pending_expiry_bar = null

      ALERT("RETEST EXPIRED: no entry.")

    return


  # 2) Check for confirm candle

  if bullish_confirm(curr, prev, last3, i, atr):

    if ENTRY_MODE == "CLOSE":

      plan = build_long_plan(curr.close, curr, atr)

      if plan != null:

        APPLY_PLAN(plan)

        ALERT("LONG CONFIRM + RISK OK: entered on close.")

      else:

        ALERT("LONG CONFIRM but FAILED risk gates (too big / too wide / RR).")

    else:

      arm_retest("LONG", RESIST, i)

      ALERT("LONG CONFIRM: armed RETEST entry at ~425.50.")


  else if bearish_confirm(curr, prev, last3, i, atr):

    if ENTRY_MODE == "CLOSE":

      plan = build_short_plan(curr.close, curr, atr)

      if plan != null:

        APPLY_PLAN(plan)

        ALERT("SHORT CONFIRM + RISK OK: entered on close.")

      else:

        ALERT("SHORT CONFIRM but FAILED risk gates (too big / too wide / RR).")

    else:

      arm_retest("SHORT", SUPPORT, i)

      ALERT("SHORT CONFIRM: armed RETEST entry at ~420.23.")


  else:

    ALERT("NO CONFIRM: inside 420.23–425.50 box. Stand down.")


# -----------------------------

# Live Price — handles RETEST entries + trade management

# -----------------------------

ON_PRICE_UPDATE(symbol=SYMBOL):

  px = LAST_PRICE()

  i = CURRENT_CANDLE_INDEX()

  atr = ATR(i-1, ATR_LEN)              # last closed ATR (safer for live)


  # RETEST entry logic

  if direction == "NONE" AND pending_signal != "NONE":

    if pending_signal == "LONG" AND retest_hit(px, RESIST):

      confirm_candle = GET_CANDLE(1)   # last closed confirm candle

      plan = build_long_plan(px, confirm_candle, atr)

      if plan != null:

        APPLY_PLAN(plan)

        pending_signal = "NONE"

        pending_level = null

        pending_expiry_bar = null

        ALERT("LONG RETEST FILL: entered near 425.50.")

      else:

        ALERT("RETEST HIT but FAILED risk gates. No trade.")

        pending_signal = "NONE"


    if pending_signal == "SHORT" AND retest_hit(px, SUPPORT):

      confirm_candle = GET_CANDLE(1)

      plan = build_short_plan(px, confirm_candle, atr)

      if plan != null:

        APPLY_PLAN(plan)

        pending_signal = "NONE"

        pending_level = null

        pending_expiry_bar = null

        ALERT("SHORT RETEST FILL: entered near 420.23.")

      else:

        ALERT("RETEST HIT but FAILED risk gates. No trade.")

        pending_signal = "NONE"


  # trade management (same as before; recommend stop orders server-side)

  if direction != "NONE":

    manage_trade(px)
