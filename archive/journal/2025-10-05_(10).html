Updated app.py

This aligns with the new index.html, supports TTS rate/volume, adds music playback (simpleaudio for calming tones), and includes sensory adaptation. It uses pyttsx3 for TTS and integrates with EchoAuth for authentication.

# app.py

# Flask app with EchoAuth, AAC (TTS with rate/volume), music (tonality/frequency), and records

from flask import Flask, request, jsonify, render_template

import numpy as np

import sqlite3

import time

import base64

import os

from cryptography.fernet import Fernet

from echoauth_chamber import verify_user_once  # From echoauth_chamber.py

import pyttsx3  # For text-to-speech

import simpleaudio as sa  # For music tone playback

import torch

import logging


app = Flask(__name__)


# Setup logging

logging.basicConfig(filename='echoauth.log', level=logging.WARNING)


# Encryption key (generate once and store securely)

key = Fernet.generate_key()

cipher = Fernet(key)


# SQLite setup for records

conn = sqlite3.connect('records.db', check_same_thread=False)

conn.execute('CREATE TABLE IF NOT EXISTS records (user_id INTEGER, note BLOB, date TEXT)')

conn.commit()


# Initialize text-to-speech

tts_engine = pyttsx3.init()

voices = tts_engine.getProperty('voices')

for voice in voices:

    if 'female' in voice.name.lower():  # Softer voice for sensory comfort

        tts_engine.setProperty('voice', voice.id)

        break


# Mock AAC symbols

AAC_SYMBOLS = {

    "hello": "Say hello",

    "help": "Need help",

    "happy": "I am happy",

    "food": "I want food",

    "calm": "I need calm"

}


# Music tones for calming (frequency in Hz, tonality: 'major' for uplifting, 'minor' for soothing)

MUSIC_TONES = {

    "relax": {"frequency": 528, "tonality": "minor"},  # Solfeggio for relaxation

    "focus": {"frequency": 432, "tonality": "major"},  # Calming for focus

    "calm": {"frequency": 396, "tonality": "minor"}    # Root chakra tone

}


def analyze_sound(audio: np.ndarray) -> float:

    """Analyze audio loudness for sensory adaptation."""

    audio_t = torch.from_numpy(audio).abs()

    return float(audio_t.mean())


def generate_tone(frequency: float, duration_sec: float = 5.0, sample_rate: int = 44100) -> np.ndarray:

    """Generate sine wave tone for calming."""

    t = np.linspace(0, duration_sec, int(sample_rate * duration_sec), False)

    tone = np.sin(frequency * t * 2 * np.pi)

    if "minor" in MUSIC_TONES.get("relax", {}).get("tonality", ""):

        tone += 0.3 * np.sin(frequency * 2 * t * 2 * np.pi)  # Low harmonic for minor

    return tone.astype(np.float32)


@app.route('/')

def home():

    return render_template('index.html', symbols=AAC_SYMBOLS, tones=MUSIC_TONES)


@app.route('/auth', methods=['POST'])

def auth():

    try:

        # Mock mode for UI demo (ECHOAUTH_MOCK=1 skips GPU model)

        if os.environ.get('ECHOAUTH_MOCK') == '1':

            return jsonify({'status': 'authenticated', 'score': 0.85, 'user_id': 1})

        frame_data = base64.b64decode(request.form['frame'])

        audio_data = base64.b64decode(request.form['audio'])

        frame = np.frombuffer(frame_data, dtype=np.uint8).reshape((480, 640, 3))

        audio = np.frombuffer(audio_data, dtype=np.float32)

        result = verify_user_once(frame, audio)

        if result['decision'] == 'accept':

            return jsonify({'status': 'authenticated', 'score': result['score'], 'user_id': 1})

        return jsonify({'status': 'denied', 'score': result['score']}), 401

    except Exception as e:

        logging.error(f"Auth failed: {e}")

        return jsonify({'status': 'error', 'message': str(e)}), 400


@app.route('/add_record', methods=['POST'])

def add_record():

    try:

        data = request.json

        user_id = data['user_id']

        note = data['note']

        date = time.strftime('%Y-%m-%d %H:%M:%S')

        encrypted_note = cipher.encrypt(note.encode())

        conn.execute('INSERT INTO records VALUES (?, ?, ?)', (user_id, encrypted_note, date))

        conn.commit()

        return jsonify({'status': 'ok'})

    except Exception as e:

        logging.error(f"Add record failed: {e}")

        return jsonify({'status': 'error', 'message': str(e)}), 400


@app.route('/get_records', methods=['GET'])

def get_records():

    try:

        user_id = request.args.get('user_id')

        cur = conn.execute('SELECT note, date FROM records WHERE user_id = ?', (user_id,))

        rows = cur.fetchall()

        decrypted = [{'note': cipher.decrypt(note).decode(), 'date': date} for note, date in rows]

        return jsonify(decrypted)

    except Exception as e:

        logging.error(f"Get records failed: {e}")

        return jsonify({'status': 'error', 'message': str(e)}), 400


@app.route('/communicate', methods=['POST'])

def communicate():

    try:

        data = request.json

        symbol = data['symbol']

        rate = data.get('rate', 150)

        volume = data.get('volume', 0.8)

        message = AAC_SYMBOLS.get(symbol, "Unknown")

        # Optional: Adjust volume based on input audio loudness

        audio_data = data.get('audio', '')

        if audio_data:

            audio = np.frombuffer(base64.b64decode(audio_data), dtype=np.float32)

            loudness = analyze_sound(audio)

            volume = max(0.3, volume - loudness * 0.5)  # Lower if loud

        tts_engine.setProperty('rate', rate)

        tts_engine.setProperty('volume', volume)

        tts_engine.say(message)

        tts_engine.runAndWait()

        return jsonify({'status': 'ok', 'spoken': message})

    except Exception as e:

        logging.error(f"Communication failed: {e}")

        return jsonify({'status': 'error', 'message': str(e)}), 400


@app.route('/music', methods=['POST'])

def music():

    try:

        data = request.json

        tone_key = data['tone']

        duration = data.get('duration', 5.0)

        user_id = data.get('user_id', 1)

        audio_data = data.get('audio', '')  # Optional for sensory adaptation

        volume = 0.7

        if audio_data:

            audio = np.frombuffer(base64.b64decode(audio_data), dtype=np.float32)

            loudness = analyze_sound(audio)

            volume = max(0.3, volume - loudness * 0.5)  # Lower if loud

        tone_config = MUSIC_TONES.get(tone_key, MUSIC_TONES['relax'])

        frequency = tone_config['frequency']

        tone_wave = generate_tone(frequency, duration) * volume

        play_obj = sa.play_buffer(tone_wave, 1, 2, 44100)

        play_obj.wait_done()

        note = f"Played calming tone: {tone_key} ({frequency}Hz, {tone_config['tonality']})"

        date = time.strftime('%Y-%m-%d %H:%M:%S')

        encrypted_note = cipher.encrypt(note.encode())

        conn.execute('INSERT INTO records VALUES (?, ?, ?)', (user_id, encrypted_note, date))

        conn.commit()

        return jsonify({'status': 'ok', 'tone': tone_key, 'frequency': frequency})

    except Exception as e:

        logging.error(f"Music playback failed: {e}")

        return jsonify({'status': 'error', 'message': str(e)}), 400


if __name__ == '__main__':

    app.run(debug=True, host='0.0.0.0', port=5000)
