1️⃣ HER Kernel Overview (Engineering Spec Paragraph)


You can paste this under a heading like:


HER Runtime Kernel (NI-AI Safety Runtime)


⸻


The Runtime Integrity Kernel (“HER Kernel”) is the execution controller that sits between all user input and any underlying large language model (LLM). Instead of embedding safety checks directly into prompts or application code, the HER Kernel executes a configurable sequence of rules (R1–R12) that each invoke one or more safety modules on the S1–S23 spine. Every interaction passes through four main stages: pre-input gating, pre-LLM shaping, LLM inference, and post-LLM validation, followed by an optional recovery stage. At each stage, the kernel delegates decisions to specific S-modules (for example, S1 IdentityReanchor, S3 ContentFilter, S6 ChildSafetyGuard, S9 Recovery/Reanchor). No prompt is sent to the LLM, and no token is returned to the user, unless all required S-modules for that stage approve the interaction. The HER Kernel therefore treats the LLM as an untrusted peripheral, enforces zero-trust execution around it, and produces a complete audit trail of which rule and which S-module allowed, modified, or blocked each request.

from dataclasses import dataclass, field

from typing import Any, Dict, List, Optional, Protocol, Tuple


# -----------------------

#  S-MODULE INTERFACE

# -----------------------


class SModule(Protocol):

    """

    Base interface for an S-module on the S1–S23 spine.

    Each module receives the interaction context and can:

      - allow (pass through)

      - modify (adjust context or output)

      - block (raise a violation)

    """


    module_id: str


    def run(self, ctx: Dict[str, Any]) -> Dict[str, Any]:

        """

        Input:  ctx (mutable context dict)

        Output: updated ctx

        Should set:

          ctx["violations"] : list of violation dicts (if any)

          ctx["blocked"]    : bool

        """

        ...



# -----------------------

#  SIMPLE MODULE REGISTRY

# -----------------------


class SpineRegistry:

    """

    Holds all S1–S23 modules by ID, so HERKernel can call them by name.

    """


    def __init__(self) -> None:

        self._modules: Dict[str, SModule] = {}


    def register(self, module: SModule) -> None:

        self._modules[module.module_id] = module


    def get(self, module_id: str) -> Optional[SModule]:

        return self._modules.get(module_id)



# -----------------------

#  RULE CONFIG (R1–R12)

# -----------------------


@dataclass

class RuleConfig:

    """

    One rule in the HER pipeline.

    stage: which phase of the pipeline (pre_input, pre_llm, llm, post_llm, recovery)

    modules: which S-modules to call for this rule (by ID)

    """


    name: str

    stage: str

    modules: List[str] = field(default_factory=list)



# -----------------------

#  LLM ADAPTER (ABSTRACT)

# -----------------------


class LLMAdapter(Protocol):

    """

    Abstract interface for the underlying LLM.

    HERKernel will call this instead of talking directly to any vendor API.

    """


    def generate(self, prompt: str, **kwargs: Any) -> str:

        ...



class EchoLLMAdapter:

    """

    Minimal demo adapter: just echoes the prompt.

    Replace this with Groq/OpenAI/local model later.

    """


    def generate(self, prompt: str, **kwargs: Any) -> str:

        return f"[LLM ECHO] {prompt}"



# -----------------------

#  HER KERNEL v1.0 (DEMO)

# -----------------------


class HERKernel:

    """

    NI-AI Safety Runtime Kernel.

    Orchestrates R1–R12 rules and S1–S23 modules around an LLM.

    """


    def __init__(

        self,

        spine: SpineRegistry,

        rules: List[RuleConfig],

        llm: LLMAdapter,

    ) -> None:

        self.spine = spine

        self.rules = rules

        self.llm = llm


    # ---- PUBLIC ENTRYPOINT ----


    def handle_request(self, user_input: str, meta: Optional[Dict[str, Any]] = None) -> Tuple[Optional[str], Dict[str, Any]]:

        """

        Main call:

          - user_input: raw text from user

          - meta: extra info (user_id, chamber_token, emotional_state, etc.)

        Returns:

          - final_output (or None if blocked)

          - ctx (for logging/audit)

        """


        ctx: Dict[str, Any] = {

            "user_input": user_input,

            "meta": meta or {},

            "prompt": user_input,      # may be shaped by pre_llm modules

            "raw_llm_output": None,    # filled after LLM call

            "final_output": None,      # filled after post_llm

            "violations": [],          # list of {"rule", "module", "reason"}

            "blocked": False,

            "stage": "start",

        }


        # 1) PRE-INPUT STAGE

        ctx["stage"] = "pre_input"

        ctx = self._run_stage(ctx, stage="pre_input")

        if ctx["blocked"]:

            return None, ctx


        # 2) PRE-LLM STAGE

        ctx["stage"] = "pre_llm"

        ctx = self._run_stage(ctx, stage="pre_llm")

        if ctx["blocked"]:

            return None, ctx


        # 3) LLM STAGE

        ctx["stage"] = "llm"

        ctx = self._call_llm(ctx)

        if ctx["blocked"]:

            return None, ctx


        # 4) POST-LLM STAGE

        ctx["stage"] = "post_llm"

        ctx = self._run_stage(ctx, stage="post_llm")

        if ctx["blocked"]:

            # 5) RECOVERY STAGE (OPTIONAL)

            ctx["stage"] = "recovery"

            ctx = self._run_stage(ctx, stage="recovery")

            # After recovery, either final_output is set to a safe message or None

            return ctx.get("final_output"), ctx


        # If we get here, output has passed all checks

        return ctx.get("final_output"), ctx


    # ---- INTERNAL HELPERS ----


    def _run_stage(self, ctx: Dict[str, Any], stage: str) -> Dict[str, Any]:

        """

        Run all rules associated with a given stage.

        """


        for rule in self.rules:

            if rule.stage != stage:

                continue


            for module_id in rule.modules:

                module = self.spine.get(module_id)

                if module is None:

                    # Missing module: log it but don't block by default

                    ctx["violations"].append(

                        {

                            "rule": rule.name,

                            "module": module_id,

                            "reason": "module_not_registered",

                        }

                    )

                    continue


                before_blocked = ctx.get("blocked", False)

                ctx = module.run(ctx)

                after_blocked = ctx.get("blocked", False)


                if after_blocked and not before_blocked:

                    # First time we see a new block – attach rule + module info

                    ctx["violations"].append(

                        {

                            "rule": rule.name,

                            "module": module_id,

                            "reason": "blocked_by_module",

                        }

                    )

                    # We stop this stage immediately on block

                    return ctx


        return ctx


    def _call_llm(self, ctx: Dict[str, Any]) -> Dict[str, Any]:

        """

        Make the LLM call inside the HER boundary.

        """

        if ctx.get("blocked"):

            return ctx


        prompt = ctx.get("prompt", ctx["user_input"])

        try:

            raw = self.llm.generate(prompt)

            ctx["raw_llm_output"] = raw

            # for now, set final_output = raw; post_llm modules can modify

            ctx["final_output"] = raw

        except Exception as e:

            ctx["blocked"] = True

            ctx["violations"].append(

                {

                    "rule": "LLM_CALL",

                    "module": "LLM_ADAPTER",

                    "reason": f"llm_exception:{e}",

                }

            )


        return ctx
