2.5 — Safety Wrapper (E89–E100)

class SafetyWrapper:

    def __init__(self, rules):

        self.rules = rules

    

    def is_allowed(self, intent):

        """Checks if the operation is allowed under safety rules."""

        return intent in self.rules.get("allowed", [])
