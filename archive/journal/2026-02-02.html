2️⃣ SNIPER BOT — EXECUTION ENGINE (PSEUDO-CODE)


This is drop-in logic. No narrative.


def sniper_bot_execute(market):


    # --- INPUTS ---

    price = market.price

    vwap = market.vwap

    macd_state = market.macd   # expanding | flat | contracting

    rsi = market.rsi

    theta = market.theta

    delta = market.delta

    iv_trend = market.iv_trend # rising | falling | flat


    # --- DERIVED ---

    delta_theta_ratio = delta / abs(theta)


    # --- LAYER 1: BIAS ---

    if price > vwap:

        bias = "BULLISH"

    elif price < vwap:

        bias = "BEARISH"

    else:

        bias = "CHOP"


    # --- LAYER 2: MOMENTUM ---

    if macd_state != "expanding":

        momentum_ok = False

    else:

        momentum_ok = True


    # --- LAYER 3: ENERGY ---

    if bias == "BULLISH" and rsi > 60:

        energy_ok = True

    elif bias == "BEARISH" and rsi < 40:

        energy_ok = True

    else:

        energy_ok = False


    # --- LAYER 4: OPTIONS PERMISSION ---

    options_allowed = True


    if abs(theta) > 0.10:

        options_allowed = False


    if delta_theta_ratio < 3:

        options_allowed = False


    if iv_trend != "rising":

        options_allowed = False


    # --- STRATEGY ROUTER ---

    if bias == "BULLISH":

        if options_allowed and momentum_ok and energy_ok:

            return "BUY_CALLS"

        elif price > vwap:

            return "CALL_SPREAD"

        else:

            return "SELL_PUT_SPREAD"


    if bias == "BEARISH":

        if options_allowed and momentum_ok and energy_ok:

            return "BUY_PUTS"

        elif price < vwap:

            return "PUT_SPREAD"

        else:

            return "SELL_CALL_SPREAD"


    return "NO_TRADE"
