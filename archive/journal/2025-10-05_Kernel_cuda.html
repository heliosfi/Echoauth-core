static constexpr int   M  = 100;     // symbols

static constexpr int   Q  = 1024;    // ingress queue depth / symbol

static constexpr int   W  = 200;     // rolling window

static constexpr int   KVAL = 10;    // validation subset size

static constexpr int   NUM_STREAM_BUCKETS = 10; // stream buckets

static constexpr int   BACKOFF_NS_INIT = 60;    // kernel backoff

static constexpr float EPS_VWAP = 1e-4f;

static constexpr float EPS_RSI  = 1e-3f;

static constexpr int   ERR_CAP = 1 << 16;       // overflow log capacity

static constexpr int   MAX_THREAD_BATCH = 5;    // per-thread micro-batch flush

// 32B aligned tick: keep price, volume, and feed timestamp

struct __align__(32) Tick {

  float     p;

  float     v;

  uint64_t  ts_ns; // from feed

  uint64_t  pad;   // padding for alignment (unused)

};


struct OverflowEvent {

  uint64_t clock;

  int      sym;

  uint64_t delta;

  uint64_t produced;

  uint64_t next_seq;

};


//========================

// Managed device controls

//========================

__device__ __managed__ int   d_backoff_ns = BACKOFF_NS_INIT;

__device__ __managed__ float d_avg_delta  = 0.f;  // EMA of produced-next_seq for auto-backoff

// Kernels

//========================

__global__ void rolling_persistent_safe_log(

  const Tick*  __restrict__ d_ring,     // [M*Q]

  const uint64_t* __restrict__ d_head,  // [M] produced count per symbol

  float* __restrict__ d_rb_p,           // [M*W]

  float* __restrict__ d_rb_v,           // [M*W]

  float* __restrict__ d_rb_d,           // [M*W]

  float* __restrict__ out_vwap,         // mapped host [M]

  float* __restrict__ out_rsi,          // mapped host [M]

  float* __restrict__ d_last_px,        // [M] last price (for decisions)

  uint64_t* __restrict__ out_ts_ns,     // mapped host [M] last tick ts

  OverflowEvent* __restrict__ d_err_buf,// [ERR_CAP]

  unsigned int* __restrict__ d_err_tail,

  int Mx, int Wx, int Qx, int err_cap)

{

  const int m = blockIdx.x; if (m >= Mx) return;

  const int lane = threadIdx.x & 31;


  // numeric stability

  double sum_pv=0.0, sum_v=0.0;

  float  sum_up=0.f, sum_dn=0.f;

  int    wpos=0; bool filled=false, have_prev=false;

  float  prev_p=0.f;

  uint64_t next_seq=0;


  while (true) {

    int backoff = d_backoff_ns;


    uint64_t produced = 0;

    if (lane==0) produced = d_head[m];

    produced = __shfl_sync(0xffffffff, produced, 0);


    if (produced == next_seq) { __nanosleep(backoff); continue; }


    uint64_t delta = produced - next_seq;

    if (lane==0) {

      float old = d_avg_delta;

      d_avg_delta = 0.9f*old + 0.1f*(float)delta;

    }


    // overflow: resync from indicator rings and log

    if (delta > (uint64_t)Qx) {

      if (lane==0) {

        unsigned int idx = atomicAdd(d_err_tail, 1u);

        d_err_buf[idx % err_cap] = OverflowEvent{clock64(), m, delta, produced, next_seq};

        sum_pv=0.0; sum_v=0.0; sum_up=0.f; sum_dn=0.f;

        for (int k=0;k<Wx;++k) {

          float p=d_rb_p[m*Wx+k], v=d_rb_v[m*Wx+k], d=d_rb_d[m*Wx+k];

          sum_pv += (double)p*v; sum_v += (double)v;

          sum_up += (d>0.f)?d:0.f; sum_dn += (d<0.f)?-d:0.f;

        }

        next_seq = produced;

      }

      continue;

    }


    // drain burst

    uint64_t last_ts = 0;

    while (next_seq < produced) {

      const size_t pos = next_seq % Qx;

      Tick tk = (lane==0) ? d_ring[m*Qx + pos] : Tick{0,0,0,0};

      float p = __shfl_sync(0xffffffff, tk.p, 0);

      float v = __shfl_sync(0xffffffff, tk.v, 0);

      uint64_t ts = __shfl_sync(0xffffffff, tk.ts_ns, 0);


      if (lane==0) {

        float d=0.f;

        if (have_prev) d = p - prev_p; else have_prev=true;

        prev_p = p;


        float p_old=0.f, v_old=0.f, d_old=0.f;

        if (filled) {

          p_old=d_rb_p[m*Wx + wpos];

          v_old=d_rb_v[m*Wx + wpos];

          d_old=d_rb_d[m*Wx + wpos];

        }

        d_rb_p[m*Wx + wpos] = p;

        d_rb_v[m*Wx + wpos] = v;

        d_rb_d[m*Wx + wpos] = d;

        wpos = (wpos + 1) % Wx;

        if (!filled && wpos==0) filled=true;


        sum_pv += (double)p*v - (filled ? (double)p_old*v_old : 0.0);

        sum_v  += (double)v   - (filled ? (double)v_old      : 0.0);


        float up=(d>0.f)?d:0.f, dn=(d<0.f)?-d:0.f;

        float up_old=(d_old>0.f)?d_old:0.f, dn_old=(d_old<0.f)?-d_old:0.f;

        sum_up += up - (filled?up_old:0.f);

        sum_dn += dn - (filled?dn_old:0.f);


        d_last_px[m] = p;   // for decision kernel

        last_ts = ts;       // publish once after burst

        ++next_seq;

      }

    }


    // publish (lane0)

    if (lane==0) {

      float vwap = (sum_v>0.0)?(float)(sum_pv/sum_v):NAN;

      float rs   = sum_up / fmaxf(sum_dn, 1e-6f);

      float rsi  = 100.f - 100.f/(1.f + rs);

      out_vwap[m] = vwap;

      out_rsi [m] = rsi;

      if (last_ts) out_ts_ns[m] = last_ts;

      __threadfence_system();

    }

  }

}


// decision kernel: simple rule per provided symbol list

// signal: -1 sell, 0 hold, +1 buy

__global__ void decision_kernel_for_syms(

  const int* __restrict__ syms, int K,

  const float* __restrict__ out_vwap,

  const float* __restrict__ out_rsi,

  const float* __restrict__ d_last_px,

  int32_t* __restrict__ out_signals)

{

  int i = blockIdx.x*blockDim.x + threadIdx.x;

  if (i>=K) return;

  int m = syms[i];

  float vwap = out_vwap[m];

  float rsi  = out_rsi[m];

  float px   = d_last_px[m];


  int32_t sig = 0;

  if (!isnan(vwap) && !isnan(rsi)) {

    if (rsi < 30.f && px < vwap) sig = +1;

    else if (rsi > 70.f && px > vwap) sig = -1;

  }

  out_signals[m] = sig; // single-writer per m if you don't supply dupes in 'syms'

}

inline cudaStream_t stream_for_sym(int m) { return g_buckets[m % NUM_STREAM_BUCKETS]; }

// Single tick enqueue (ordered in stream: payload -> head)

inline void enqueue_tick_symbol(

  int m, const Tick& t,

  Tick* d_ring, uint64_t* d_head,

  uint64_t start_seq, // assigned sequence for this tick

  int Q, cudaStream_t s)

{

  size_t pos = start_seq % Q;

  CHECK_CUDA(cudaMemcpyAsync(&d_ring[m*Q + pos], &t, sizeof(Tick),

                             cudaMemcpyHostToDevice, s));

  uint64_t head_val = start_seq + 1;

  CHECK_CUDA(cudaMemcpyAsync(&d_head[m], &head_val, sizeof(uint64_t),

                             cudaMemcpyHostToDevice, s));

}


// Batched enqueue with explicit start sequence (no overlap across threads)

inline void enqueue_tick_batch_symbol(

  int m, const Tick* host_batch, int count,

  Tick* d_ring, uint64_t* d_head,

  uint64_t start_seq, int Q, cudaStream_t s)

{

  size_t start = start_seq % Q;

  if (start + (size_t)count <= (size_t)Q) {

    CHECK_CUDA(cudaMemcpyAsync(&d_ring[m*Q + start], host_batch,

                               count*sizeof(Tick), cudaMemcpyHostToDevice, s));

  } else {

    size_t first = Q - start;

    size_t second = count - first;

    CHECK_CUDA(cudaMemcpyAsync(&d_ring[m*Q + start], host_batch,

                               first*sizeof(Tick), cudaMemcpyHostToDevice, s));

    CHECK_CUDA(cudaMemcpyAsync(&d_ring[m*Q + 0], host_batch + first,

                               second*sizeof(Tick), cudaMemcpyHostToDevice, s));

  }

  uint64_t head_val = start_seq + (uint64_t)count;

  CHECK_CUDA(cudaMemcpyAsync(&d_head[m], &head_val, sizeof(uint64_t),

                             cudaMemcpyHostToDevice, s));

}


//========================

// Polygon-like feed adapter (thread-safe)

//========================


// Thread-local micro-batches to reduce PCIe calls under bursts.

// Each feed thread keeps a small per-symbol batch (up to MAX_THREAD_BATCH).

struct ThreadBatch {

  // compact per-thread staging; avoid huge maps: store last (sym,count) runs

  Tick buf[MAX_THREAD_BATCH];

  int  cnt = 0;

  int  sym = -1;

};


thread_local ThreadBatch tls_batch;


// Map symbol to id (read-only map protected by shared_mutex)

inline bool map_symbol_to_id(const char* symbol, int& out_id) {

  std::shared_lock lk(g_sym_mutex);

  auto it = g_symbol_to_id.find(symbol);

  if (it == g_symbol_to_id.end()) return false;

  out_id = it->second;

  return true;

}


// Called by the WebSocket library (may be from up to 4 threads)

extern "C" void on_polygon_tick(const char* symbol, float price, float volume, uint64_t timestamp_ns)

{

  int m = -1;

  if (!map_symbol_to_id(symbol, m)) return; // unknown symbol, drop or log


  Tick t{price, volume, timestamp_ns, 0};


  // simple per-thread, per-symbol micro-batching

  if (tls_batch.sym == -1) tls_batch.sym = m;


  if (tls_batch.sym != m || tls_batch.cnt == MAX_THREAD_BATCH) {

    // flush previous

    if (tls_batch.cnt > 0) {

      uint64_t start_seq = g_seq[tls_batch.sym].fetch_add(tls_batch.cnt, std::memory_order_relaxed);

      cudaStream_t s = stream_for_sym(tls_batch.sym);

      enqueue_tick_batch_symbol(tls_batch.sym, tls_batch.buf, tls_batch.cnt,

                                g_d_ring, g_d_head, start_seq, Q, s);

      // (Optional) trigger decision kernel for that symbol set (see below)

    }

    tls_batch.sym = m;

    tls_batch.cnt = 0;

  }


  tls_batch.buf[tls_batch.cnt++] = t;


  // If full, flush now

  if (tls_batch.cnt == MAX_THREAD_BATCH) {

    uint64_t start_seq = g_seq[m].fetch_add(tls_batch.cnt, std::memory_order_relaxed);

    cudaStream_t s = stream_for_sym(m);

    enqueue_tick_batch_symbol(m, tls_batch.buf, tls_batch.cnt, g_d_ring, g_d_head, start_seq, Q, s);

    tls_batch.cnt = 0;

    // (Optional) trigger decision kernel for that symbol

  }

}


// Optionally flush at thread end or periodically (not strictly required for continuous feeds)

inline void flush_tls_batch()

{

  if (tls_batch.cnt > 0 && tls_batch.sym >= 0) {

    int m = tls_batch.sym;

    uint64_t start_seq = g_seq[m].fetch_add(tls_batch.cnt, std::memory_order_relaxed);

    cudaStream_t s = stream_for_sym(m);

    enqueue_tick_batch_symbol(m, tls_batch.buf, tls_batch.cnt, g_d_ring, g_d_head, start_seq, Q, s);

    tls_batch.cnt = 0;

    tls_batch.sym = -1;

  }

}

int main() {

  CHECK_CUDA(cudaSetDevice(0));

  int prio_low, prio_high;

  CHECK_CUDA(cudaDeviceGetStreamPriorityRange(&prio_low, &prio_high));


  // --- Symbol map (fixed mapping) ---

  {

    std::unique_lock lk(g_sym_mutex);

    // Fill with your 100 symbols; example stub:

    // g_symbol_to_id["AAPL"]=0; g_symbol_to_id["MSFT"]=1; ...

    for (int i=0;i<M;++i) {

      g_symbol_to_id[std::string("SYM")+std::to_string(i)] = i;

    }

  }


  // --- Device buffers ---

  Tick* d_ring=nullptr;        CHECK_CUDA(cudaMalloc(&d_ring, (size_t)M*Q*sizeof(Tick)));

  uint64_t* d_head=nullptr;    CHECK_CUDA(cudaMalloc(&d_head, (size_t)M*sizeof(uint64_t)));

                               CHECK_CUDA(cudaMemset(d_head, 0, (size_t)M*sizeof(uint64_t)));

  float *d_rb_p=nullptr, *d_rb_v=nullptr, *d_rb_d=nullptr;

  CHECK_CUDA(cudaMalloc(&d_rb_p, (size_t)M*W*sizeof(float)));

  CHECK_CUDA(cudaMalloc(&d_rb_v, (size_t)M*W*sizeof(float)));

  CHECK_CUDA(cudaMalloc(&d_rb_d, (size_t)M*W*sizeof(float)));

  CHECK_CUDA(cudaMemset(d_rb_p, 0, (size_t)M*W*sizeof(float)));

  CHECK_CUDA(cudaMemset(d_rb_v, 0, (size_t)M*W*sizeof(float)));

  CHECK_CUDA(cudaMemset(d_rb_d, 0, (size_t)M*W*sizeof(float)));


  // last price (device)

  float* d_last_px=nullptr;    CHECK_CUDA(cudaMalloc(&d_last_px, (size_t)M*sizeof(float)));

                               CHECK_CUDA(cudaMemset(d_last_px, 0, (size_t)M*sizeof(float)));


  // overflow log

  OverflowEvent* d_err_buf=nullptr; unsigned int* d_err_tail=nullptr;

  CHECK_CUDA(cudaMalloc(&d_err_buf, ERR_CAP * sizeof(OverflowEvent)));

  CHECK_CUDA(cudaMalloc(&d_err_tail, sizeof(unsigned int)));

  CHECK_CUDA(cudaMemset(d_err_tail, 0, sizeof(unsigned int)));


  // mapped host outputs (zero-copy)

  float    *h_out_vwap=nullptr, *h_out_rsi=nullptr; float *d_out_vwap=nullptr, *d_out_rsi=nullptr;

  CHECK_CUDA(cudaHostAlloc(&h_out_vwap, (size_t)M*sizeof(float), cudaHostAllocMapped));

  CHECK_CUDA(cudaHostAlloc(&h_out_rsi,  (size_t)M*sizeof(float), cudaHostAllocMapped));

  CHECK_CUDA(cudaHostGetDevicePointer(&d_out_vwap, h_out_vwap, 0));

  CHECK_CUDA(cudaHostGetDevicePointer(&d_out_rsi,  h_out_rsi,  0));


  // mapped host timestamps (from kernel)

  uint64_t *h_out_ts_ns=nullptr; uint64_t* d_out_ts_ns=nullptr;

  CHECK_CUDA(cudaHostAlloc(&h_out_ts_ns, (size_t)M*sizeof(uint64_t), cudaHostAllocMapped));

  CHECK_CUDA(cudaHostGetDevicePointer(&d_out_ts_ns, h_out_ts_ns, 0));

  std::fill(h_out_ts_ns, h_out_ts_ns+M, 0);


  // mapped host signals (decision kernel output)

  int32_t *h_out_signals=nullptr; int32_t* d_out_signals=nullptr;

  CHECK_CUDA(cudaHostAlloc(&h_out_signals, (size_t)M*sizeof(int32_t), cudaHostAllocMapped));

  CHECK_CUDA(cudaHostGetDevicePointer(&d_out_signals, h_out_signals, 0));

  std::fill(h_out_signals, h_out_signals+M, 0);


  // stream buckets

  g_buckets.resize(NUM_STREAM_BUCKETS);

  for (int b=0;b<NUM_STREAM_BUCKETS;++b)

    CHECK_CUDA(cudaStreamCreateWithFlags(&g_buckets[b], cudaStreamNonBlocking));


  // hot and validation streams

  cudaStream_t hot, val_stream, dec_stream;

  CHECK_CUDA(cudaStreamCreateWithPriority(&hot,        cudaStreamNonBlocking, prio_high));

  CHECK_CUDA(cudaStreamCreateWithPriority(&val_stream, cudaStreamNonBlocking, prio_low));

  CHECK_CUDA(cudaStreamCreateWithPriority(&dec_stream, cudaStreamNonBlocking, prio_high));


  // zero host counters

  for (int i=0;i<M;++i) g_seq[i].store(0, std::memory_order_relaxed);


  // publish device pointers for adapter

  g_d_ring = d_ring; g_d_head = d_head;


  // launch persistent kernel (warp-per-symbol)

  rolling_persistent_safe_log<<<M, 32, 0, hot>>>(

    d_ring, d_head, d_rb_p, d_rb_v, d_rb_d,

    d_out_vwap, d_out_rsi, d_last_px, d_out_ts_ns,

    d_err_buf, d_err_tail, M, W, Q, ERR_CAP);

  CHECK_CUDA(cudaGetLastError());


  // --- Decision trigger helper: run for a set of symbols (unique) ---

  auto run_decision_for = [&](const std::vector<int>& syms){

    if (syms.empty()) return;

    // copy syms to device temp

    static int* d_syms = nullptr; static size_t cap = 0;

    if (syms.size() > cap) {

      if (d_syms) cudaFree(d_syms);

      cap = syms.size();

      CHECK_CUDA(cudaMalloc(&d_syms, cap*sizeof(int)));

    }

    CHECK_CUDA(cudaMemcpyAsync(d_syms, syms.data(), syms.size()*sizeof(int),

                               cudaMemcpyHostToDevice, dec_stream));

    int block=64, grid=(int)((syms.size()+block-1)/block);

    decision_kernel_for_syms<<<grid, block, 0, dec_stream>>>(

      d_syms, (int)syms.size(), d_out_vwap, d_out_rsi, d_last_px, d_out_signals);

    CHECK_CUDA(cudaGetLastError());

  };


  // --- Simulated Polygon feed driver (replace with real WS pumps) ---

  // Here, we simulate multiple feeder threads calling on_polygon_tick()

  std::atomic<bool> run{true};


  auto feeder_thread = [&](int tid){

    std::mt19937 rng{std::random_device{}()};

    std::uniform_int_distribution<int> sym_pick(0, M-1);

    std::uniform_real_distribution<float> px_j(-0.5f, 0.5f);

    std::uniform_real_distribution<float> vj(0.f, 1000.f);

    // aim 10k ticks/s across threads -> sleep ~100us per tick per thread

    auto next_tick = std::chrono::high_resolution_clock::now();

    while (run.load(std::memory_order_relaxed)) {

      // microburst up to MAX_THREAD_BATCH ticks (possibly same symbol)

      int n = 1 + (rng() % MAX_THREAD_BATCH);

      for (int i=0;i<n;++i) {

        int m = sym_pick(rng);

        // Map id back to a symbol string to exercise the adapter

        std::string sym = "SYM"+std::to_string(m);

        float base_px = 100.f + 0.01f*m;

        float price = base_px + px_j(rng);

        float volume = 1.0e6f + vj(rng);

        uint64_t ts_ns = (uint64_t)std::chrono::duration_cast<std::chrono::nanoseconds>(

          std::chrono::high_resolution_clock::now().time_since_epoch()).count();

        // Call the adapter (thread-safe)

        on_polygon_tick(sym.c_str(), price, volume, ts_ns);

      }

      next_tick += std::chrono::microseconds(100);

      std::this_thread::sleep_until(next_tick);

      // Periodically flush TLS batch

      if ((rng() & 0x3F)==0) flush_tls_batch();


      // Optionally trigger decision for some set recently touched (demo: all)

      // For real feed, collect the symbols you just enqueued and pass unique set.

      if ((rng() & 0x1F)==0) {

        std::vector<int> syms; syms.reserve(16);

        for (int j=0;j<16;++j) syms.push_back(rng()%M);

        std::sort(syms.begin(), syms.end());

        syms.erase(std::unique(syms.begin(), syms.end()), syms.end());

        run_decision_for(syms);

      }

    }

    flush_tls_batch();

  };


  // Launch up to 4 feeder threads (like multiple WS connections)

  std::thread feeders[4];

  for (int i=0;i<4;++i) feeders[i] = std::thread(feeder_thread, i);


  // --- Validation + latency monitor every 50ms ---

  int *d_subset=nullptr; float *d_dv=nullptr, *d_dr=nullptr;

  CHECK_CUDA(cudaMalloc(&d_subset, KVAL*sizeof(int)));

  CHECK_CUDA(cudaMalloc(&d_dv,     KVAL*sizeof(float)));

  CHECK_CUDA(cudaMalloc(&d_dr,     KVAL*sizeof(float)));


  // latency stats

  std::vector<double> lat_samples; lat_samples.reserve(4096);

  std::mutex lat_mtx;


  auto validator_loop = [&](){

    std::mt19937 rng{std::random_device{}()};

    std::vector<int> universe(M); std::iota(universe.begin(), universe.end(), 0);


    while (run.load(std::memory_order_relaxed)) {

      std::shuffle(universe.begin(), universe.end(), rng);

      int h_subset[KVAL]; for (int i=0;i<KVAL;++i) h_subset[i]=universe[i];


      CHECK_CUDA(cudaMemcpyAsync(d_subset, h_subset, KVAL*sizeof(int),

                                 cudaMemcpyHostToDevice, val_stream));

      int block=64, grid=(KVAL+block-1)/block;

      // reuse d_rb_*, d_out_vwap/rsi

      validate_subset_K<<<grid, block, 0, val_stream>>>(

        d_subset, KVAL, d_rb_p, d_rb_v, d_rb_d, d_out_vwap, d_out_rsi, M, W, d_dv, d_dr);

      CHECK_CUDA(cudaGetLastError());


      float h_dv[KVAL], h_dr[KVAL];

      CHECK_CUDA(cudaMemcpyAsync(h_dv, d_dv, KVAL*sizeof(float), cudaMemcpyDeviceToHost, val_stream));

      CHECK_CUDA(cudaMemcpyAsync(h_dr, d_dr, KVAL*sizeof(float), cudaMemcpyDeviceToHost, val_stream));

      CHECK_CUDA(cudaStreamSynchronize(val_stream));


      for (int i=0;i<KVAL;++i) {

        if (h_dv[i] > EPS_VWAP || h_dr[i] > EPS_RSI) {

          std::cerr << "[VALIDATION] sym=" << h_subset[i]

                    << " Δvwap=" << h_dv[i] << " Δrsi=" << h_dr[i] << "\n";

        }

      }


      // overflow log snapshot + reset if large

      unsigned int tail=0;

      CHECK_CUDA(cudaMemcpy(&tail, d_err_tail, sizeof(unsigned int), cudaMemcpyDeviceToHost));

      if (tail) {

        unsigned int nread = std::min(tail, 8u);

        unsigned int start = (tail >= nread) ? (tail - nread) : 0;

        std::vector<OverflowEvent> tmp(nread);

        for (unsigned int i=0;i<nread;++i) {

          CHECK_CUDA(cudaMemcpy(&tmp[i], &d_err_buf[(start+i)%ERR_CAP],

                                sizeof(OverflowEvent), cudaMemcpyDeviceToHost));

        }

        for (auto& ev: tmp) {

          std::cerr << "[OVERFLOW] sym="<<ev.sym<<" delta="<<(unsigned long long)ev.delta

                    <<" produced="<<(unsigned long long)ev.produced

                    <<" next_seq="<<(unsigned long long)ev.next_seq<<"\n";

        }

        if (tail >= ERR_CAP/2) CHECK_CUDA(cudaMemsetAsync(d_err_tail, 0, sizeof(unsigned int), val_stream));

      }


      // latency monitor: compare wall-clock now to last out_ts_ns per symbol

      auto now_ns = (uint64_t)std::chrono::duration_cast<std::chrono::nanoseconds>(

                      std::chrono::high_resolution_clock::now().time_since_epoch()).count();

      {

        std::lock_guard<std::mutex> lk(lat_mtx);

        for (int m=0;m<M;++m) {

          uint64_t ts = h_out_ts_ns[m];

          if (ts) {

            double us = (double)(now_ns - ts)/1000.0;

            if (us > 0 && us < 1e6) lat_samples.push_back(us);

          }

        }

        if (lat_samples.size() >= 1000) {

          std::sort(lat_samples.begin(), lat_samples.end());

          double p50=lat_samples[(size_t)(0.50*lat_samples.size())];

          double p95=lat_samples[(size_t)(0.95*lat_samples.size())];

          double p99=lat_samples[(size_t)(0.99*lat_samples.size())];

          std::cout << "[LAT] p50="<<p50<<"us p95="<<p95<<"us p99="<<p99<<"us\n";

          if (p99 > 100.0) {

            std::cerr << "[WARN] p99 latency exceeds 100us: "<<p99<<"us\n";

          }

          lat_samples.clear();

        }

      }


      // backoff auto-tune: pull EMA and adjust

      float avg_delta=0.f;

      CHECK_CUDA(cudaMemcpy(&avg_delta, &d_avg_delta, sizeof(float), cudaMemcpyDeviceToHost));

      int new_backoff = d_backoff_ns; // read current

      if (avg_delta > 2.f) new_backoff = std::max(20, d_backoff_ns - 10);

      else if (avg_delta < 0.5f) new_backoff = std::min(100, d_backoff_ns + 10);

      if (new_backoff != d_backoff_ns)

        CHECK_CUDA(cudaMemcpyAsync(&d_backoff_ns, &new_backoff, sizeof(int), cudaMemcpyHostToDevice, val_stream));


      std::this_thread::sleep_for(std::chrono::milliseconds(50));

    }

  };


  std::thread validator(validator_loop);


  // Let it run for a while (demo). Replace with your WS run loop and proper shutdown.

  std::this_thread::sleep_for(std::chrono::seconds(5));

  run.store(false);

  for (auto& t: feeders) t.join();

  validator.join();


  // Cleanup

  for (auto& s: g_buckets) cudaStreamDestroy(s);

  cudaStreamDestroy(hot);

  cudaStreamDestroy(val_stream);

  cudaStreamDestroy(dec_stream);


  cudaFree(d_ring); cudaFree(d_head);

  cudaFree(d_rb_p); cudaFree(d_rb_v); cudaFree(d_rb_d);

  cudaFree(d_last_px);

  cudaFree(d_err_buf); cudaFree(d_err_tail);

  cudaFree(d_subset);  cudaFree(d_dv); cudaFree(d_dr);

  cudaFreeHost(h_out_vwap); cudaFreeHost(h_out_rsi);

  cudaFreeHost(h_out_ts_ns); cudaFreeHost(h_out_signals);


  CHECK_CUDA(cudaDeviceReset());

  std::cout << "Done.\n";

  return 0;

}
