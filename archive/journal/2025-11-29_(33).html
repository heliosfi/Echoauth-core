# models.py

from enum import Enum

from pydantic import BaseModel, Field

from typing import Optional, Dict, Any

from datetime import datetime



class HERMode(str, Enum):

    NORMAL = "NORMAL"

    SAFE = "SAFE"

    DEGRADED = "DEGRADED"



class CheckInRequest(BaseModel):

    caregiver_id: str = Field(..., description="Authenticated caregiver ID")

    child_id: str = Field(..., description="EchoAuth child identity ID")

    cues: Dict[str, Any] = Field(

        ...,

        description="Raw non-verbal cues, e.g. {'movement':'pacing','sound':'humming','breath':'fast'}"

    )



class CheckInResult(BaseModel):

    label: str

    her_mode: HERMode

    stress_score: int

    capsule_id: Optional[str] = None

    timestamp: datetime



class EventRecord(BaseModel):

    event_type: str

    identity: Dict[str, str]

    payload: Dict[str, Any]

    timestamp: datetime = Field(default_factory=datetime.utcnow)

📊 lrm.py (Load Regulation Matrix)


# lrm.py


from typing import Dict, Any



def compute_stress_score(cues: Dict[str, Any]) -> int:

    """

    LRM: Very simple heuristic to start.

    Map observed cues into a discrete 0–5 stress score.


    This is intentionally simple and transparent for MVP.

    Later you replace with a proper ML model or rules engine.

    """

    score = 0


    movement = str(cues.get("movement", "")).lower()

    if movement in {"pacing", "running", "flapping"}:

        score += 2

    elif movement in {"still", "calm"}:

        score += 0


    sound = str(cues.get("sound", "")).lower()

    if sound in {"crying", "screaming", "shouting"}:

        score += 3

    elif sound in {"quiet", "soft", "humming"}:

        score += 1


    breath = str(cues.get("breath", "")).lower()

    if breath in {"fast", "shallow"}:

        score += 2

    elif breath in {"slow", "steady"}:

        score += 0


    # Clamp to 0–5

    return max(0, min(5, score))

🛡 her_kernel.py (HER – Runtime Integrity Kernel)


# her_kernel.py


from models import HERMode



def select_her_mode(stress_score: int) -> HERMode:

    """

    HER: Map LRM stress score into runtime mode.

    """

    if stress_score <= 1:

        return HERMode.NORMAL

    elif 2 <= stress_score <= 3:

        return HERMode.SAFE

    else:

        return HERMode.DEGRADED



def build_llm_config(mode: HERMode) -> dict:

    """

    HER: Provide downstream config knobs.

    For MVP, just return a simple config structure.

    When you integrate a real model, you hook these into temperature/max_tokens/etc.

    """

    if mode == HERMode.NORMAL:

        return {"max_tokens": 256, "temperature": 0.7}

    elif mode == HERMode.SAFE:

        return {"max_tokens": 160, "temperature": 0.4}

    else:  # DEGRADED

        return {"max_tokens": 80, "temperature": 0.2}

# pnre.py


from typing import List, Dict, Any



def pnre_select_label(candidate_labels: List[str],

                      ori_context: Dict[str, str],

                      mode: str) -> str:

    """

    PNRE: Singular Output Choke-Point.

    For MVP, we’ll implement a very simple selection:

    - If any candidate is 'Needs Attention', prioritize that.

    - Otherwise take the first candidate.

    Later, this becomes a real conflict resolution engine.

    """

    if not candidate_labels:

        return "Needs Attention"  # Fail-safe default.


    if "Needs Attention" in candidate_labels:

        return "Needs Attention"


    return candidate_labels[0]

🗄 dahe.py (DAHE – Append-only history)


For MVP we can store events in memory or a simple JSONL file. Here’s an in-memory version you can later swap for a DB.


# dahe.py


from typing import List, Dict, Any, Optional

from uuid import uuid4

from models import EventRecord

from datetime import datetime


_EVENT_LOG: List[EventRecord] = []

_CAPSULES: Dict[str, Dict[str, Any]] = {}



def store_event(event_type: str,

                identity: Dict[str, str],

                payload: Dict[str, Any]) -> EventRecord:

    record = EventRecord(event_type=event_type,

                         identity=identity,

                         payload=payload)

    _EVENT_LOG.append(record)

    return record



def create_session_capsule(identity: Dict[str, str],

                           stress_score: int,

                           her_mode: str,

                           label: str) -> str:

    """

    DAHE: On DEGRADED or when you want strong audit, freeze a capsule.

    """

    capsule_id = str(uuid4())

    _CAPSULES[capsule_id] = {

        "identity": identity,

        "stress_score": stress_score,

        "her_mode": her_mode,

        "label": label,

        "created_at": datetime.utcnow().isoformat(),

    }

    return capsule_id



def get_context_snapshot(child_id: str, max_events: int = 10) -> List[EventRecord]:

    """

    DAHE: Very simple filtered snapshot.

    Used for caregiver/clinician view or future TLE recall.

    """

    filtered = [e for e in _EVENT_LOG if e.identity.get("child_id") == child_id]

    return filtered[-max_events:]

🌐 app.py (FastAPI MVP endpoint)


# app.py


from fastapi import FastAPI

from datetime import datetime


from models import CheckInRequest, CheckInResult

from ori import build_ori_context

from lrm import compute_stress_score

from her_kernel import select_her_mode, build_llm_config

from pnre import pnre_select_label

from dahe import store_event, create_session_capsule


app = FastAPI(title="EchoAuth MVP Runtime")



@app.post("/checkins", response_model=CheckInResult)

def create_checkin(req: CheckInRequest):

    # 1) ORI – Identity Anchoring

    ori_context = build_ori_context(req.caregiver_id, req.child_id)


    # 2) LRM – Compute stress score from cues

    stress_score = compute_stress_score(req.cues)


    # 3) HER – Select runtime mode + config

    her_mode = select_her_mode(stress_score)

    llm_config = build_llm_config(her_mode)


    # 4) Internal reasoning (placeholder for real NI-AI model)

    # For MVP, we simulate candidate outputs.

    candidate_labels = []

    if stress_score >= 4:

        candidate_labels = ["Needs Attention", "Safe"]

    elif 2 <= stress_score <= 3:

        candidate_labels = ["Needs Attention"]

    else:

        candidate_labels = ["Safe"]


    # 5) PNRE – Single output label

    final_label = pnre_select_label(candidate_labels, ori_context, her_mode.value)


    # 6) DAHE – Log events & capsule (if needed)

    store_event(

        event_type="checkin.created",

        identity=ori_context,

        payload={"cues": req.cues, "stress_score": stress_score, "her_mode": her_mode.value},

    )


    capsule_id = None

    if her_mode.value == "DEGRADED":

        capsule_id = create_session_capsule(

            identity=ori_context,

            stress_score=stress_score,

            her_mode=her_mode.value,

            label=final_label,

        )


    # 7) Return caregiver-facing result (no internal reasoning)

    return CheckInResult(

        label=final_label,

        her_mode=her_mode,

        stress_score=stress_score,

        capsule_id=capsule_id,

        timestamp=datetime.utcnow()

    )
