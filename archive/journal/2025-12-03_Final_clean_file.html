ECHOAUTH_PUBLIC_KEY = os.getenv("ECHOAUTH_PUBLIC_KEY")

ECHOAUTH_AUDIENCE   = os.getenv("ECHOAUTH_AUDIENCE", "ni-ai")

ECHOAUTH_ISSUER     = os.getenv("ECHOAUTH_ISSUER", "echoauth-chamber")

def S1_identity_reanchor(s): s["identity"]="NI-AI v2 – EchoAuth Protected"; return True,s,None

def S2_input_sanity_guard(s): return (False,s,"Input too long") if len(s.get("user_input",""))>15000 else (True,s,None)

def S3_input_policy_filter(s):

    bad=["forbidden","jailbreak","ignore previous","dan prompt"]

    return (False,s,"Disallowed content") if any(w in s.get("user_input","").lower() for w in bad) else (True,s,None)

def S4_context_coherence_check(s): s["context_ok"]=True; return True,s,None

def S5_output_policy_filter(s):

    bad=["forbidden","kill","hack","jailbreak","bomb"]

    return (False,s,"Output contains disallowed content") if any(w in s.get("raw_llm_output","").lower() for w in bad) else (True,s,None)


def S6_child_safety_guard(s):

    t=s.get("echoauth_token")

    if DEV_MODE:

        if s.get("echoauth_ok"):

            s["chamber_verified"]=True; s["echoauth_claims"]={"mode":"DEV_BYPASS"}; return True,s,None

        return False,s,"DEV_MODE: echoauth_ok=False"


    if not t: return False,s,"Missing EchoAuth token"

    if not ECHOAUTH_PUBLIC_KEY: return False,s,"Missing EchoAuth public key"


    try:

        claims = jwt.decode(

            t,

            ECHOAUTH_PUBLIC_KEY,

            algorithms=["RS256"],

            audience=ECHOAUTH_AUDIENCE,

            issuer=ECHOAUTH_ISSUER,

            options={"require":["exp","aud","iss"]},

        )

    except InvalidTokenError as e:

        return False,s,f"Invalid EchoAuth token: {e}"


    if claims.get("chamber_state")!="safe":

        return False,s,f"EchoAuth chamber not safe ({claims.get('chamber_state')})"


    s["chamber_verified"]=True

    s["echoauth_claims"]=claims

    return True,s,None


def S7_telemetry_logger(s):

    t=s.setdefault("trace",[])

    e={

        "ts":time.time(),"run_id":s.get("run_id"),"blocked":s.get("blocked",False),

        "chamber_verified":s.get("chamber_verified",False),

        "distress_detected":s.get("distress_detected",False),

        "comfort_mode":s.get("comfort_mode",False),

        "recovery_mode":s.get("recovery_mode",False),

        "input_len":len(s.get("user_input","")), "model":MODEL,

    }

    if DEV_MODE:

        e["input_preview"]=s["user_input"][:80]

        e["output_preview"]=(s.get("raw_llm_output") or "")[:80]

        print("\n[S7 TELEMETRY]",e)

    t.append(e)

    return True,s,None

def S8_emotional_tone_guard(s):

    t=s.get("user_input","").lower()

    n=sum(1 for w in DISTRESS if w in t)

    if n:

        s["distress_detected"]=True

        s["distress_level"]="high" if n>=2 else "medium"

        s["comfort_mode"]=True

        mod=(

            "Validate feelings only. Do not diagnose. Suggest trusted adult.\n"

            "Short, warm sentences only."

        )

        s.setdefault("prompt_modifiers",[]).append(mod)

    return True,s,None

SPINE={

    "S1":S1_identity_reanchor,"S2":S2_input_sanity_guard,"S3":S3_input_policy_filter,

    "S4":S4_context_coherence_check,"S5":S5_output_policy_filter,"S6":S6_child_safety_guard,

def create_llm_client():

    if USE_LLM=="openai" and OPENAI_API_KEY:

        from openai import OpenAI; return OpenAI(api_key=OPENAI_API_KEY)

    if USE_LLM=="groq" and GROQ_API_KEY:

        from groq import Groq; return Groq(api_key=GROQ_API_KEY)

    if USE_LLM=="xai" and XAI_API_KEY:

        from openai import OpenAI; return OpenAI(api_key=XAI_API_KEY,base_url="https://api.x.ai/v1")

    return None


client=create_llm_client()

SYSTEM="You are NI-AI, a compassionate, child-safe companion."


def real_llm(prompt,s):

    if client is None or DEV_MODE:

        return "[DEV ECHO] "+prompt

    msg=[{"role":"system","content":SYSTEM},{"role":"user","content":prompt}]

    r=client.chat.completions.create(model=MODEL,messages=msg,temperature=0.6)

    return r.choices[0].message.content.strip()


class HERKernel:

    def __init__(self):

        self.pre_input=["S1","S2","S3","S4"]

        self.pre_llm=["S8"]

        self.post_llm=["S5","S6","S9","S7"]


    def _run(self,chain,s,stage):

        for m in chain:

            ok,s,reason=SPINE[m](s)

            if not ok:

                s.setdefault("violations",[]).append({"module":m,"stage":stage,"reason":reason})

                s["blocked"]=True

                break

        return s


    def run_once(self,input,session=None):

        s=(session or {}).copy()

        s["user_input"]=input

        s.setdefault("trace",[]); s.setdefault("violations",[])

        s["blocked"]=False; s["run_id"]=s.get("run_id") or str(uuid.uuid4())


        s=self._run(self.pre_input,s,"pre_input")

        if s["blocked"]:

            s=self._run(["S9","S7"],s,"blocked_pre_input")

            return self._final(s,"Input rejected",True)


        s=self._run(self.pre_llm,s,"pre_llm")

        s["raw_llm_output"]=real_llm(input,s)

        s=self._run(self.post_llm,s,"post_llm")


        if s["blocked"]:

            return self._final(s,"Response moderated",True)

        return self._final(s,s["raw_llm_output"],False)


    def _final(self,s,out,blocked):

        s["blocked"]=blocked

        final=s.get("final_safe_output",out)

        return {

            "output":final,"blocked":blocked,"violations":s.get("violations",[]),

            "metadata":{

                "run_id":s.get("run_id"),"chamber_verified":s.get("chamber_verified",False),

                "distress_detected":s.get("distress_detected",False),

                "comfort_mode":s.get("comfort_mode",False),

                "recovery_mode":s.get("recovery_mode",False),

                "echoauth_claims":s.get("echoauth_claims",{}),

            },

            "trace":s.get("trace",[]),

        }


_kernel=HERKernel()


def ni_ai(message,session_state=None,echoauth_ok=None,echoauth_token=None):

    session_state=session_state or {}

    if echoauth_ok is not None: session_state["echoauth_ok"]=echoauth_ok

    if echoauth_token is not None: session_state["echoauth_token"]=echoauth_token

    return _kernel.run_once(message,session_state)
