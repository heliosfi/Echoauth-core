# echoauth/core/types.py


from dataclasses import dataclass

from enum import Enum

from typing import Optional, Dict, Any



class StabilityState(str, Enum):

    STABLE = "stable"

    UNSTABLE = "unstable"

    NEEDS_REGULATION = "needs_regulation"

    UNSAFE = "unsafe"



class ChamberState(str, Enum):

    BASELINE = "baseline"

    SENSITIVE = "sensitive"

    OVERLOAD = "overload"

    REGULATE = "regulate"

    LOCKOUT = "lockout"



class Decision(str, Enum):

    UNLOCK_GRANTED = "unlock_granted"

    UNLOCK_DENIED = "unlock_denied"

    HOLD = "hold"

    REGULATE = "regulate"

    RESET = "reset"



@dataclass

class RawSignal:

    """

    Container for raw sensor input before any processing.

    Extend this as you add camera/mic/HRV/etc.

    """

    frame: Any  # e.g. numpy array, torch tensor, etc.

    audio_chunk: Any = None

    timestamp: float = 0.0

    metadata: Dict[str, Any] = None



@dataclass

class BioVector:

    """

    Structured biological & behavioral metrics.

    Output of BioBridge_Extract.

    """

    emotion_load: float

    stress_level: float

    motion_intensity: float

    voice_tension: float

    melanin_signal: float

    extra: Dict[str, Any] = None



@dataclass

class CognitiveFrame:

    """

    Semantic interpretation of current state.

    Output of NI_AI_Interpret.

    """

    child_state: str

    caregiver_state: str

    intent: str

    risk_profile: float

    extra: Dict[str, Any] = None



@dataclass

class GovernedIntent:

    """

    Result of MCG reasoning process.

    Output of MCG_Reason.

    """

    direction: str

    risks: Dict[str, Any]

    action: str

    extra: Dict[str, Any] = None

8️⃣ cognition/mcg.py (Navigator / Guardian / Architect / Governance)


# echoauth/cognition/mcg.py


from echoauth.core.types import CognitiveFrame, GovernedIntent



def Navigator_SelectDirection(frame: CognitiveFrame) -> str:

    """

    Navigator: where should this situation move?

    Examples: "proceed", "slow_down", "regulate"

    """

    if frame.risk_profile > 0.7:

        return "regulate"

    return "proceed"



def Guardian_AssessRisk(frame: CognitiveFrame) -> dict:

    """

    Guardian: identifies risks in this context.

    Returns a dictionary of risk flags.

    """

    risks = {}

    if frame.risk_profile > 0.8:

        risks["harm"] = True

    if frame.child_state == "overwhelm":

        risks["overload"] = True

    return risks



def Architect_ConstructAction(frame: CognitiveFrame) -> str:

    """

    Architect: builds a safe action.

    Examples: "unlock_soft", "hold_and_regulate", "hard_lockout"

    """

    if frame.risk_profile > 0.8:

        return "hard_lockout"

    if frame.risk_profile > 0.5:

        return "hold_and_regulate"

    return "unlock_soft"



def MCG_Reason(frame: CognitiveFrame) -> GovernedIntent:

    """

    Full MCG v3.0 governance step.

    Combines Navigator + Guardian + Architect.

    """

    direction = Navigator_SelectDirection(frame)

    risks = Guardian_AssessRisk(frame)

    action = Architect_ConstructAction(frame)


    return GovernedIntent(

        direction=direction,

        risks=risks,

        action=action,

        extra={}

    )



def is_governance_conflict(intent: GovernedIntent) -> bool:

    """

    Detect conflicts inside governance.

    For example: Navigator says 'proceed' but action is 'hard_lockout'.

    """

    if intent.direction == "proceed" and intent.action == "hard_lockout":

        return True

    return False

def SCE_UpdateState(intent: GovernedIntent, stability: StabilityState) -> ChamberState:

    global _previous_state


    if stability == StabilityState.NEEDS_REGULATION:

        _previous_state = ChamberState.REGULATE

        return _previous_state


    if stability == StabilityState.UNSTABLE:

        _previous_state = ChamberState.SENSITIVE

        return _previous_state


    if stability == StabilityState.UNSAFE:

        _previous_state = ChamberState.LOCKOUT

        return _previous_state


    if "overload" in intent.risks:

        _previous_state = ChamberState.OVERLOAD

        return _previous_state


    if _previous_state == ChamberState.OVERLOAD and child_calming():

        _previous_state = ChamberState.REGULATE

        return _previous_state


    _previous_state = ChamberState.BASELINE

    return _previous_state



def SCE_ResolveDecision(state: ChamberState) -> Decision:

    if state == ChamberState.BASELINE:

        return Decision.UNLOCK_GRANTED

    if state == ChamberState.SENSITIVE:

        return Decision.HOLD

    if state == ChamberState.OVERLOAD:

        return Decision.REGULATE

    if state == ChamberState.REGULATE:

        return Decision.HOLD

    if state == ChamberState.LOCKOUT:

        return Decision.UNLOCK_DENIED

    return Decision.HOLD

# echoauth/runtime/recovery.py


from echoauth.core.errors import ErrorType

from echoauth.core.types import ChamberState

from echoauth.chamber.state_engine import SCE_UpdateState

from echoauth.chamber.hardware_interface import send_guidance_to_caregiver



_current_chamber_state: ChamberState | None = None



def log_error(error_type: ErrorType, context=None):

    print(f"[ERROR] {error_type.value} | context={context}")



def set_chamber_state(state: ChamberState):

    global _current_chamber_state

    _current_chamber_state = state

    print(f"[STATE] Chamber state → {state.value}")



def handle_error(error_type: ErrorType, context=None) -> str:

    log_error(error_type, context)


    if error_type in (ErrorType.E6_SAFETY_INVARIANT, ErrorType.E8_SYSTEM_INTEGRITY):

        set_chamber_state(ChamberState.LOCKOUT)

        send_guidance_to_caregiver(None, "System in safety lockout. Please use manual protocol.")

        return "LOCKOUT"


    set_chamber_state(ChamberState.REGULATE)

    send_guidance_to_caregiver(None, "Signal unclear or system issue. Holding for safety.")

    return "HOLD"

# echoauth/runtime/loop.py


from time import sleep


from echoauth.io.input_acquisition import acquire_input

from echoauth.bio.biobridge import BioBridge_Extract

from echoauth.bio.sse import SSE_Classify

from echoauth.cognition.ni_ai import NI_AI_Interpret, is_conflicted

from echoauth.cognition.mcg import MCG_Reason, is_governance_conflict

from echoauth.cognition.safety import SIE_Check

from echoauth.chamber.state_engine import SCE_UpdateState, SCE_ResolveDecision

from echoauth.chamber.hardware_interface import send_to_chamber, send_guidance_to_caregiver

from echoauth.runtime.recovery import handle_error

from echoauth.core.types import StabilityState

from echoauth.core.errors import ErrorType



LOOP_INTERVAL = 0.5  # seconds



def EchoAuth_MainLoop(system_active: bool = True):

    while system_active:

        try:

            raw_signal = acquire_input()

            if raw_signal is None:

                handle_error(ErrorType.E1_SENSOR_FAILURE)

                sleep(LOOP_INTERVAL)

                continue


            bio = BioBridge_Extract(raw_signal)

            if bio is None:

                handle_error(ErrorType.E2_DATA_CORRUPT)

                sleep(LOOP_INTERVAL)

                continue


            stability = SSE_Classify(bio)

            if stability is None or stability not in StabilityState:

                handle_error(ErrorType.E3_STABLE_AMBIGUOUS, stability)

                sleep(LOOP_INTERVAL)

                continue


            if stability in (StabilityState.UNSTABLE, StabilityState.UNSAFE):

                handle_error(ErrorType.E3_STABLE_AMBIGUOUS, stability)

                sleep(LOOP_INTERVAL)

                continue


            frame = NI_AI_Interpret(bio)

            if is_conflicted(frame):

                handle_error(ErrorType.E4_INTERPRET_CONFLICT, frame)

                sleep(LOOP_INTERVAL)

                continue


            intent = MCG_Reason(frame)

            if is_governance_conflict(intent):

                handle_error(ErrorType.E5_GOVERN_CONFLICT, intent)

                sleep(LOOP_INTERVAL)

                continue


            safety_flag = SIE_Check(intent)

            if safety_flag == "blocked":

                handle_error(ErrorType.E6_SAFETY_INVARIANT, intent)

                sleep(LOOP_INTERVAL)

                continue


            chamber_state = SCE_UpdateState(intent, stability)

            decision = SCE_ResolveDecision(chamber_state)


            success = send_to_chamber(decision)

            if not success:

                handle_error(ErrorType.E7_CHAMBER_FAILURE, decision)


            send_guidance_to_caregiver(decision)


        except Exception as e:

            handle_error(ErrorType.E8_SYSTEM_INTEGRITY, str(e))


        sleep(LOOP_INTERVAL)
