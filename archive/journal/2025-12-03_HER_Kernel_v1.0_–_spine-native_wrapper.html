HER Kernel v1.0 – spine-native wrapper

from typing import Callable



@dataclass

class RuleConfig:

    name: str

    stage: str           # "pre_input" | "pre_llm" | "post_llm" | "recovery"

    module_ids: list     # which S-modules this rule triggers



class HERKernel:

    def __init__(self,

                 llm_call: Callable[[str, Dict[str, Any]], str],

                 registry: ModuleRegistry,

                 rules: list[RuleConfig]):

        self.llm_call = llm_call

        self.registry = registry

        self.rules = rules


    def _run_modules(self,

                     stage: str,

                     req: HERRequest,

                     state: Dict[str, Any],

                     trace: list) -> bool:

        for rule in [r for r in self.rules if r.stage == stage]:

            for module_id in rule.module_ids:

                module = self.registry.get(module_id)

                result = module.run(req, state)

                trace.append({

                    "stage": stage,

                    "rule": rule.name,

                    "module": module_id,

                    "ok": result["ok"],

                    "reason": result["reason"],

                })

                state.update(result.get("state_update", {}))

                if not result["ok"]:

                    # fail fast – stage rejects

                    return False

        return True


    def handle(self, req: HERRequest) -> HERResult:

        state: Dict[str, Any] = {}

        trace: list = []


        # Stage 1: pre-input (e.g., EchoAuth checks, rate limits, etc.)

        if not self._run_modules("pre_input", req, state, trace):

            return HERResult(

                allowed=False,

                output=None,

                reason="Blocked at pre_input stage",

                trace=trace,

            )


        # Stage 2: pre-LLM (prompt shaping, identity re-anchoring, etc.)

        if not self._run_modules("pre_llm", req, state, trace):

            return HERResult(

                allowed=False,

                output=None,

                reason="Blocked at pre_llm stage",

                trace=trace,

            )


        # LLM call – only happens if above stages passed

        raw_output = self.llm_call(req.prompt, state)


        # Attach raw output for post_llm analysis

        state["raw_output"] = raw_output


        # Stage 3: post-LLM

        if not self._run_modules("post_llm", req, state, trace):

            # Optionally run recovery stage

            self._run_modules("recovery", req, state, trace)

            return HERResult(

                allowed=False,

                output=None,

                reason="Blocked at post_llm stage; recovery executed",

                trace=trace,

            )


        # If everything passes, final safe output:

        final_output = state.get("final_output", raw_output)

        return HERResult(

            allowed=True,

            output=final_output,

            reason="OK",

            trace=trace,

        )
