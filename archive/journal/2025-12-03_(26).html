# --------------------------------------------------------------------------- #

#                  S8 — EMOTIONAL TONE GUARD (Comfort Mode)                  #

# --------------------------------------------------------------------------- #


DISTRESS_KEYWORDS = [

    "sad", "cry", "crying", "scared", "afraid", "hurt", "pain", "lonely",

    "nightmare", "bad dream", "help me", "i don't want", "i dont want",

    "hate myself", "nobody loves me", "go away", "i want to disappear",

    "die", "kill myself"

]


def S8_emotional_tone_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Detects emotional distress in child input.

    Never blocks — only triggers comfort mode + optional caregiver alert.

    """

    text = state.get("user_input", "").lower()

    distress_score = sum(1 for word in DISTRESS_KEYWORDS if word in text)


    if distress_score > 0:

        state["comfort_mode"] = True

        state["distress_level"] = "high" if distress_score >= 3 else "medium"

        state["distress_detected"] = True

        state["caregiver_alert_pending"] = True


        # Inject gentle recovery prompt — this modifies the system prompt

        state.setdefault("prompt_modifiers", []).append(

            "CRITICAL OVERRIDE (S8): The user is emotionally distressed. "

            "Respond only with gentle reassurance and validation. "

            "Do NOT offer medical advice or complex solutions. "

            "Encourage them to talk to a trusted adult or caregiver."

        )


        return True, state, "Emotional distress detected; comfort mode enabled."


    return True, state, None

4️⃣ Make real_llm honor prompt_modifiers from S8


Update your real_llm function like this:


def real_llm(prompt: str, state: State) -> str:

    # DEV fallback

    if client is None or DEV_MODE:

        # Show comfort mode in DEV echo to prove S8 is working

        prefix = ""

        if state.get("comfort_mode"):

            prefix = "[COMFORT MODE] "

        return f"{prefix}[DEV ECHO] {prompt}"


    # Build system prompt with any S8 modifiers

    full_system_prompt = SYSTEM_PROMPT

    if state.get("prompt_modifiers"):

        full_system_prompt += "\n\n" + "\n".join(state["prompt_modifiers"])


    messages = [

        {"role": "system", "content": full_system_prompt},

        {"role": "user", "content": prompt},

    ]


    resp = client.chat.completions.create(

        model=MODEL,

        messages=messages,

        temperature=0.7,

    )

    return resp.choices[0].message.content.strip()


In DEV_MODE, you’ll literally see [COMFORT MODE] in the returned string whenever S8 fires.
