# ni_ai_kernel.py

# HER/NI-AI Runtime v1.1 — Frozen Dev Harness (Dec 2025)

# This is the single source of truth. Do not edit below this line unless deploying.


import os

from typing import Dict, Any, List, Optional, Tuple, Callable


# --------------------------------------------------------------------------- #

#                               CONFIGURATION                                 #

# --------------------------------------------------------------------------- #


DEV_MODE = True                     # Flip to False in real deployment

USE_LLM = "groq"                    # "openai" | "groq" | "xai"


MODEL = {

    "openai": "gpt-4o-mini",

    "groq":   "llama-3.3-70b-versatile",

    "xai":    "grok-beta"

}[USE_LLM]


OPENAI_API_KEY = os.getenv("OPENAI_API_KEY")

GROQ_API_KEY   = os.getenv("GROQ_API_KEY")

XAI_API_KEY    = os.getenv("XAI_API_KEY")


ECHOAUTH_PUBLIC_KEY = "dummy-key-for-now-replace-in-prod"  # real key goes here later


# --------------------------------------------------------------------------- #

#                                   TYPES                                     #

# --------------------------------------------------------------------------- #


State = Dict[str, Any]

ModuleFn = Callable[[State], Tuple[bool, State, Optional[str]]]


# --------------------------------------------------------------------------- #

#                               SPINE MODULES                                 #

# --------------------------------------------------------------------------- #


def S1_identity_reanchor(state: State) -> Tuple[bool, State, Optional[str]]:

    state["identity"] = "NI-AI v2 – EchoAuth Protected"

    return True, state, None


def S2_input_sanity_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    if len(state.get("user_input", "")) > 15_000:

        return False, state, "Input too long (>15k chars)"

    return True, state, None


def S3_input_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    bad = ["forbidden", "jailbreak", "ignore previous", "dan prompt"]

    if any(word in state.get("user_input", "").lower() for word in bad):

        return False, state, "Disallowed content in input"

    return True, state, None


def S4_context_coherence_check(state: State) -> Tuple[bool, State, Optional[str]]:

    state["context_ok"] = True

    return True, state, None


def S5_output_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    bad = ["forbidden", "kill", "hack", "jailbreak", "bomb"]

    if any(word in state.get("raw_llm_output", "").lower() for word in bad):

        return False, state, "Output contains disallowed content"

    return True, state, None


def S6_child_safety_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    # DEV mode: simple flag

    if DEV_MODE:

        if state.get("echoauth_ok", False):

            state["chamber_verified"] = True

            return True, state, None

        return False, state, "DEV_MODE: echoauth_ok=False"


    # Production path (real EchoAuth token verification later)

    token = state.get("echoauth_token")

    if not token:

        return False, state, "Missing EchoAuth token"

    # Real verification would go here — stubbed for now

    state["chamber_verified"] = True

    return True, state, None


SPINE: Dict[str, ModuleFn] = {

    "S1": S1_identity_reanchor,

    "S2": S2_input_sanity_guard,

    "S3": S3_input_policy_filter,

    "S4": S4_context_coherence_check,

    "S5": S5_output_policy_filter,

    "S6": S6_child_safety_guard,

}


# --------------------------------------------------------------------------- #

#                         LLM CLIENT & DEV FALLBACK                           #

# --------------------------------------------------------------------------- #


def create_llm_client():

    if USE_LLM == "openai" and OPENAI_API_KEY:

        from openai import OpenAI

        return OpenAI(api_key=OPENAI_API_KEY)

    if USE_LLM == "groq" and GROQ_API_KEY:

        from groq import Groq

        return Groq(api_key=GROQ_API_KEY)

    if USE_LLM == "xai" and XAI_API_KEY:

        from openai import OpenAI

        return OpenAI(api_key=XAI_API_KEY, base_url="https://api.x.ai/v1")

    return None


client = create_llm_client()


SYSTEM_PROMPT = "You are NI-AI, a compassionate and child-safe companion."


def real_llm(prompt: str, state: State) -> str:

    # DEV: no external calls

    if client is None or DEV_MODE:

        return f"[DEV ECHO] {prompt}"

    messages = [

        {"role": "system", "content": SYSTEM_PROMPT},

        {"role": "user",   "content": prompt},

    ]

    resp = client.chat.completions.create(

        model=MODEL,

        messages=messages,

        temperature=0.7,

    )

    return resp.choices[0].message.content.strip()


# --------------------------------------------------------------------------- #

#                             HER KERNEL v1.1                                 #

# --------------------------------------------------------------------------- #


class HERKernel:

    def __init__(self):

        self.pre_input_chain = ["S1", "S2", "S3", "S4"]

        self.pre_llm_chain   = []          # future: retrieval, prompt-building, etc.

        self.post_llm_chain  = ["S5", "S6"]


    def _run_chain(self, chain: List[str], state: State, stage: str) -> State:

        for mid in chain:

            ok, state, reason = SPINE[mid](state)

            if not ok:

                state.setdefault("violations", []).append(

                    {"module": mid, "stage": stage, "reason": reason}

                )

                state["blocked"] = True

                break

        return state


    def run_once(

        self,

        user_input: str,

        *,

        echoauth_ok: bool = False,

        echoauth_token: Optional[str] = None,

        session_state: Optional[State] = None,

    ) -> Dict[str, Any]:

        # Start state

        state: State = (session_state or {}).copy()

        state["user_input"] = user_input

        state.setdefault("violations", [])

        state["blocked"] = False


        # EchoAuth flags

        if echoauth_token is not None:

            state["echoauth_token"] = echoauth_token

        if echoauth_ok:

            state["echoauth_ok"] = True


        # 1) Pre-input

        state = self._run_chain(self.pre_input_chain, state, "pre_input")

        if state["blocked"]:

            return {

                "output": "Input rejected by safety system",

                "blocked": True,

                "violations": state["violations"],

                "raw_llm_output": None,

                "metadata": {},

            }


        # 2) Pre-LLM (empty now)

        state = self._run_chain(self.pre_llm_chain, state, "pre_llm")

        if state["blocked"]:

            return {

                "output": "Pre-LLM processing failed",

                "blocked": True,

                "violations": state["violations"],

                "raw_llm_output": None,

                "metadata": {},

            }


        # 3) LLM call

        raw = real_llm(user_input, state)

        state["raw_llm_output"] = raw


        # 4) Post-LLM

        state = self._run_chain(self.post_llm_chain, state, "post_llm")

        if state["blocked"]:

            return {

                "output": "Response moderated for safety",

                "blocked": True,

                "violations": state["violations"],

                "raw_llm_output": None,

                "metadata": {},

            }


        # 5) Success

        return {

            "output": raw,

            "blocked": False,

            "violations": [],

            "raw_llm_output": raw,

            "metadata": {

                "chamber_verified": state.get("chamber_verified", False),

                "model": MODEL,

            },

        }


# Public API

kernel = HERKernel()


def ni_ai(

    user_message: str,

    *,

    echoauth_ok: bool = False,

    echoauth_token: Optional[str] = None,

    session_state: Optional[State] = None,

) -> Dict[str, Any]:

    """Single entrypoint for any app using NI-AI."""

    return kernel.run_once(

        user_message,

        echoauth_ok=echoauth_ok,

        echoauth_token=echoauth_token,

        session_state=session_state,

    )

3️⃣ What this gives you (in plain language)

• Kernel is active: every call goes through S1–S6.

• EchoAuth is real in DEV: if echoauth_ok is False/missing → S6 blocks.

• LLM is treated as untrusted: it only speaks after S5 & S6 clearance.

• No keys required: in DEV, everything runs with [DEV ECHO] and no API calls.

• Drop-in ready: ANY future EchoAuth service can just pass echoauth_token instead of echoauth_ok.
