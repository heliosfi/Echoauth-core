EchoAuth CEG Harness Reference Implementation (Pseudocode v1.0)

// Harness: Deterministic CEG simulator for CI conformance testing.

// Inputs: JSON test vectors (array of SK objects).

// Outputs: Per-vector results + pass/fail on assertions.

// Assumptions:

// - Sig verify mocked to true unless vector specifies "sig-invalid".

// - Hashes recomputed using JCS canonicalization (implement JCS or equiv).

// - No real storage; in-memory state reset per vector.

// - Policy: Exclusive locks default; HOLD on busy (not HALT).

// - Floats banned in payloads (integers + units only).


import json  // For loading vectors

import sha256  // For hashing

import jcs_canonicalize  // RFC 8785 impl (or equiv stable JSON serializer)


struct Result {

    status: string  // "CEG_STATUS_EXECUTED" | "DENY" | "HOLD" | "HALT"

    reason_code: string

    fails_at_step: int  // 1-7 or "C" for auth precheck

    side_effect_called: bool

    nonce_mutated: bool

    lock_mutated: bool  // True if acquired (even if released)

}


struct HarnessState {

    now_ms: int64

    nonce_used: map

    locks: map

    authority_channel: map  // "ONLINE" | "OFFLINE"

}


struct LockEntry {

    owner_grant_id: string

    lock_mode: string  // "LOCK_MODE_EXCLUSIVE" | "SHARED"

    acquired_at_ms: int64

    ttl_ms: int64  // e.g., 5000

    sequence_nonce: int64

}


// Main: Process all vectors

function run_harness(vector_json: string) -> array {

    vectors = json.parse(vector_json)["vectors"]

    results = []

    for vec in vectors {

        state = init_state_from_vector(vec["setup"])

        input = vec["input"]

        grant = input["auth_grant"]

        

        // Run CEG flow

        res = ceg_execute(state, grant, input["instruction_payload_canonical"], input["execution_context_canonical"])

        

        // Check assertions

        assert_no_side_effect = !res.side_effect_called unless res.status == "CEG_STATUS_EXECUTED"

        assert_nonce = res.nonce_mutated only if res.status == "CEG_STATUS_EXECUTED"

        assert_lock = res.lock_mutated only if res.status == "CEG_STATUS_EXECUTED" or (res.status in ["DENY", "HALT"] and lock_was_acquired)  // Rare, but if acquired before late fail

        // ... match vec["assertions"] (e.g., "audit_logged" as mock)

        

        results.append(res)

    }

    return results  // CI fails if any assert mismatches vec["expected"]

}


function init_state_from_vector(setup: map) -> HarnessState {

    return {

        now_ms: setup["now_ms"],

        nonce_used: map_from_array(setup["nonce_registry_used_grant_ids"], true),

        locks: setup["locks"],  // As LockEntry maps

        authority_channel: setup["authority_channels"]

    }

}


// Core CEG Execute (Deterministic Flow)

function ceg_execute(state: HarnessState, grant: map, instr_payload_json: string, exec_ctx_json: string) -> Result {

    var fail_step = 0

    var status = "CEG_STATUS_DENY"

    var reason = "UNKNOWN"

    var side_called = false

    var nonce_mut = false

    var lock_mut = false

    var lock_acquired = false

    

    // C: Authority Channel Precheck (Policy Gate)

    if state.authority_channel[grant["root_authority_id"]] == "OFFLINE" {

        fail_step = "C"

        status = "CEG_STATUS_HOLD"

        reason = "AUTHORITY_CHANNEL_UNRESOLVED"

        return {status, reason, fail_step, side_called, nonce_mut, lock_mut}

    }

    

    // 1: Schema + Decision

    fail_step = 1

    if grant["version"] != "CEG-GRANT-1" or grant["decision"] != "DECISION_APPROVE" or missing_required_fields(grant) {

        reason = grant["decision"] != "DECISION_APPROVE" ? "DECISION_NOT_APPROVE" : "SCHEMA_INVALID"

        return {status, reason, fail_step, side_called, nonce_mut, lock_mut}

    }

    

    // 2: Time Window

    fail_step = 2

    if state.now_ms < grant["issued_at_ms"] or state.now_ms > grant["expires_at_ms"] {

        reason = "EXPIRED_GRANT"

        return {status, reason, fail_step, side_called, nonce_mut, lock_mut}

    }

    

    // 3: Signature (Mocked)

    fail_step = 3

    if !mock_verify_signature(grant) {  // True unless vector tags "sig-invalid"

        status = "CEG_STATUS_HALT"

        reason = "SIGNATURE_INVALID"

        return {status, reason, fail_step, side_called, nonce_mut, lock_mut}

    }

    

    // 4: Replay Check (No consume yet)

    fail_step = 4

    if state.nonce_used[grant["grant_id"]] {

        reason = "NONCE_REUSE_DETECTED"

        return {status, reason, fail_step, side_called, nonce_mut, lock_mut}

    }

    

    // 5: Binding/Integrity

    fail_step = 5

    instr_bytes = jcs_canonicalize(json.parse(instr_payload_json))

    instr_hash = hash_hex(grant["instruction_hash_alg"], instr_bytes)

    ctx_bytes = jcs_canonicalize(json.parse(exec_ctx_json))

    ctx_hash = hash_hex(grant["execution_context_hash_alg"], ctx_bytes)

    if instr_hash != grant["instruction_hash"] {

        status = "CEG_STATUS_HALT"

        reason = "INSTRUCTION_HASH_MISMATCH"

        return {status, reason, fail_step, side_called, nonce_mut, lock_mut}

    }

    if ctx_hash != grant["execution_context_hash"] {

        status = "CEG_STATUS_HALT"

        reason = "BINDING_MISMATCH"  // Or specific ctx mismatch

        return {status, reason, fail_step, side_called, nonce_mut, lock_mut}

    }

    // Add scope/authority checks if vector injects mismatch (e.g., SK-02)

    if scope_mismatch(grant) {  // Mock: Check delegation_chain vs authority_scope_id

        status = "CEG_STATUS_HALT"

        reason = "SCOPE_INCONSISTENCY"

        return {status, reason, fail_step, side_called, nonce_mut, lock_mut}

    }

    

    // 6: Lock Acquire (TTL + Sequence)

    fail_step = 6

    lock_key = grant["state_lock_key"]

    expire_lock_if_ttl(state, lock_key)

    lock_acquired, lock_mut = acquire_lock(state, lock_key, grant["lock_mode"], grant["sequence_nonce"], grant["grant_id"])

    if !lock_acquired {

        status = "CEG_STATUS_HOLD"

        reason = "STATE_LOCK_BUSY"  // Or "OUT_OF_SEQUENCE_REQUEST" if seq fail

        return {status, reason, fail_step, side_called, nonce_mut, lock_mut}

    }

    

    // Commit Point: Consume Nonce

    state.nonce_used[grant["grant_id"]] = true

    nonce_mut = true

    

    // 7: Execute

    fail_step = 7

    side_called = true

    mock_execute_side_effect()  // No-op, just flag

    

    // Release Lock

    release_lock(state, lock_key)

    

    status = "CEG_STATUS_EXECUTED"

    reason = "SUCCESS"

    return {status, reason, fail_step, side_called, nonce_mut, lock_mut}

}


// Helpers

function expire_lock_if_ttl(state: HarnessState, key: string) {

    if state.locks.has(key) and state.now_ms > state.locks[key].acquired_at_ms + state.locks[key].ttl_ms {

        delete state.locks[key]

    }

}


function acquire_lock(state: HarnessState, key: string, mode: string, seq: int64, grant_id: string) -> (bool acquired, bool mutated) {

    if state.locks.has(key) {

        if seq <= state.locks[key].sequence_nonce { return false, false }  // HALT-able, but here HOLD for v1

        if mode == "LOCK_MODE_EXCLUSIVE" or state.locks[key].lock_mode == "LOCK_MODE_EXCLUSIVE" { return false, false }

        // SHARED with SHARED: Allow, but no update seq for v1 simplicity

        return true, false  // No mutation if already held (policy)

    }

    state.locks[key] = {owner_grant_id: grant_id, lock_mode: mode, acquired_at_ms: state.now_ms, ttl_ms: 5000, sequence_nonce: seq}

    return true, true

}


function release_lock(state: HarnessState, key: string) {

    if state.locks.has(key) { delete state.locks[key] }

}


function hash_hex(alg: string, bytes: bytes) -> string {

    if alg == "SHA-256" { return hex(sha256(bytes)) }

    error("Unsupported alg")

}


// Mocks: Implement as needed for vectors (e.g., return true)

function mock_verify_signature(grant: map) -> bool { return true }

function scope_mismatch(grant: map) -> bool { return false }  // True for SK-02 injection

function missing_required_fields(grant: map) -> bool { return false }

function mock_execute_side_effect() {}  // Flag only
