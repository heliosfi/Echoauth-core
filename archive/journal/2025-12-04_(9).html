# echoauth_token_generator.py

# Utility to generate an RS256 key pair and issue a valid EchoAuth JWT for testing.

# This simulates the EchoAuth Chamber backend.


import os

import jwt

import datetime

from datetime import timezone, timedelta

from typing import Tuple


from cryptography.hazmat.primitives import serialization as crypto_serialization

from cryptography.hazmat.primitives.asymmetric import rsa

from cryptography.hazmat.backends import default_backend


# --------------------------------------------------------------------------- #

#                             CONFIGURATION & CLAIMS                          #

# --------------------------------------------------------------------------- #


ECHOAUTH_AUDIENCE = "ni-ai"

ECHOAUTH_ISSUER   = "echoauth-chamber"


TEST_CLAIMS_DATA = {

    "child_id": "ni-ai-test-child-001",

    "caregiver_id": "ni-ai-test-parent-42",

    "session_id": "sess-" + os.urandom(4).hex(),

    "chamber_state": "safe",

    "mode": "production_test",

}


def generate_key_pair() -> Tuple[str, str]:

    """Generates a new private/public key pair (RS256 compatible)."""

    private_key = rsa.generate_private_key(

        public_exponent=65537,

        key_size=2048,

        backend=default_backend()

    )


    private_pem = private_key.private_bytes(

        crypto_serialization.Encoding.PEM,

        crypto_serialization.PrivateFormat.PKCS8,

        crypto_serialization.NoEncryption()

    ).decode("utf-8")


    public_key = private_key.public_key()

    public_pem = public_key.public_bytes(

        crypto_serialization.Encoding.PEM,

        crypto_serialization.PublicFormat.SubjectPublicKeyInfo

    ).decode("utf-8")


    return private_pem, public_pem


def generate_jwt(private_key_pem: str) -> str:

    """Issues the JWT using the private key and required claims."""

    now = datetime.datetime.now(timezone.utc)


    claims = {

        "aud": ECHOAUTH_AUDIENCE,

        "iss": ECHOAUTH_ISSUER,

        "exp": now + timedelta(minutes=60),

        "iat": now,

        "nbf": now,

    }

    claims.update(TEST_CLAIMS_DATA)


    encoded_jwt = jwt.encode(

        claims,

        private_key_pem,

        algorithm="RS256",

    )

    return encoded_jwt


# --------------------------------------------------------------------------- #

#                                 EXECUTION                                   #

# --------------------------------------------------------------------------- #


if __name__ == "__main__":

    print("--- EchoAuth Test Token Generator ---")

    print("Generating temporary RS256 key pair and signing JWT...")


    private_key, public_key = generate_key_pair()

    test_token = generate_jwt(private_key)


    print("\n=======================================================")

    print("✅ TEST CONFIGURATION OUTPUT (EchoAuth Backend Simulation)")

    print("=======================================================\n")


    # 1. Public key → env for ni_ai_kernel server

    sanitized_public_key = public_key.replace('\n', '\\n')

    print("1. Set this env var on the NI-AI server:\n")

    print(f"ECHOAUTH_PUBLIC_KEY='{sanitized_public_key}'\n")


    # 2. Token → X-EchoAuth-Token header

    print("2. Use this JWT as 'X-EchoAuth-Token' header when DEV_MODE=False:\n")

    print(test_token, "\n")


    print("3. Steps:")

    print("   a) In ni_ai_kernel.py, set DEV_MODE = False.")

    print("   b) Export ECHOAUTH_PUBLIC_KEY as above (plus any audience/issuer overrides).")

    print("   c) Run: uvicorn echoauth_api:app --reload")

    print("   d) Call POST /ni-ai/chat with header X-EchoAuth-Token: <token>.")

    print("=======================================================")
