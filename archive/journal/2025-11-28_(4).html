"""

EchoAuth™ + NI-AI Local MVP Demo

Attorney/Engineer proof-of-use terminal run.

One output enforcement. Internal logs only.

"""


import uuid

import hashlib

import json

from datetime import datetime

from pathlib import Path


from sqlalchemy.orm import Session

from db.session import SessionLocal, engine, Base

from db import models

from core.ni_ai import evaluate_cues


# ────────────────────────────────

# Local audit log file (private)

# ────────────────────────────────

LOG_PATH = Path("./audit_logs.json")

if not LOG_PATH.exists():

    LOG_PATH.write_text("[]", encoding="utf-8")



def init_db():

    """Create DB tables if missing."""

    Base.metadata.create_all(bind=engine)



def log_internal(checkin_id: str, record: dict):

    """Append private audit log (never UI-surfaced)."""

    current = json.loads(LOG_PATH.read_text(encoding="utf-8"))

    record["checkin_id"] = checkin_id

    record["logged_at"] = datetime.utcnow().isoformat()

    current.append(record)

    LOG_PATH.write_text(json.dumps(current, indent=2), encoding="utf-8")



def get_or_create_caregiver(db: Session) -> models.Caregiver:

    email = "demo.cg@echoauth.local"

    cg = db.query(models.Caregiver).filter(models.Caregiver.email == email).first()

    if cg:

        return cg

    cg = models.Caregiver(

        name="Demo CG",

        email=email,

        hashed_password=hashlib.sha256("demo123".encode()).hexdigest(),

        created_at=datetime.utcnow(),

    )

    db.add(cg)

    db.commit()

    db.refresh(cg)

    return cg



def get_or_create_child(db: Session, cg: models.Caregiver) -> models.Child:

    ch = db.query(models.Child).filter(

        models.Child.caregiver_id == cg.id, models.Child.name == "Demo Child"

    ).first()

    if ch:

        return ch

    ch = models.Child(

        caregiver_id=cg.id,

        name="Demo Child",

        age_range="6-8",

        notes="Local MVP record.",

        created_at=datetime.utcnow(),

    )

    db.add(ch)

    db.commit()

    db.refresh(ch)

    return ch



def caregiver_intake_encode(cues: dict, child_id: int):

    """

    Cue packet → Identity token encode BEFORE inference.

    (Your core patent doctrine in software.)

    """

    token = {

        "token_id": str(uuid.uuid4()),

        "child_id": child_id,

        "cue": cues,

        "encoded_at": datetime.utcnow().isoformat(),

        "origin": "caregiver_intake",

    }

    return token



def simulate_checkin(db: Session, cg: models.Caregiver, ch: models.Child):

    # 1. Caregiver cue packet

    cue_packet = {

        "behavior": "head banging, pacing, covering ears",

        "environment": "loud crowded store, bright lights",

        "tone": "agitated sounds, eyes uneasy",

    }


    # 2. Encode cue into identity token before NI-AI inference

    identity_token = caregiver_intake_encode(cue_packet, ch.id)


    # 3. Send token cues to NI-AI evaluator (internal threads compete)

    reasoning_out = evaluate_cues(identity_token)


    # 4. Extract final result + scores

    emotion = reasoning_out["emotion_score"]

    danger = reasoning_out["danger_score"]

    conflict = reasoning_out["conflict_score"]

    conf = reasoning_out["confidence"]


    # 5. Auto-flag rule (logical spine)

    needs_review = False

    review_status = "none"

    if emotion > 0.7 or conflict > 0.6 or (danger > 0.0 and conf < 0.85):

        needs_review = True

        review_status = "pending"


    # 6. Store Check-In row

    ci = models.CheckIn(

        caregiver_id=cg.id,

        child_id=ch.id,

        observed_cues=json.dumps(identity_token),

        result_label=reasoning_out["final_label"],

        confidence=conf,

        status="success",

        emotion_score=emotion,

        needs_review=needs_review,

        review_status=review_status,

        created_at=datetime.utcnow(),

    )

    db.add(ci)

    db.commit()

    db.refresh(ci)


    # 7. Store NI-AI private log

    ai_log = models.NIAILog(

        checkin_id=ci.id,

        winning_thread=reasoning_out["winning_thread"],

        conflict_score=conflict,

        internal_notes=json.dumps(reasoning_out["internal_threads"]),

        created_at=datetime.utcnow(),

    )

    db.add(ai_log)

    db.commit()


    # 8. Private audit logging (legal/clinical use)

    log_internal(ci.id, {

        "inventor_access": True,

        "caregiver": cg.name,

        "child": ch.name,

        "emotion_score": emotion,

        "danger_score": danger,

        "conflict_score": conflict,

        "confidence_score": conf,

        "decision": reasoning_out["final_label"],

    })


    # 9. Terminal: one unambiguous caregiver output

    print("──────────── EchoAuth NI-AI MVP Demo ────────────")

    print(f"Caregiver:       {cg.name}")

    print(f"Child:           {ch.name}")

    print(f"Cue:             {cue_packet}")

    print("──────────────── Evaluation Result ──────────────")

    print(f"Status:          {reasoning_out['final_label']}")

    print(f"Confidence:      {conf:.2f}")

    print(f"Emotion/Stress:  {emotion:.2f} (0=Calm, 1=High)")

    print(f"Danger score:    {danger:.2f}")

    print(f"Conflict score:  {conflict:.2f}")

    print("──────────────── Review Routing ─────────────────")

    print(f"Needs Review:    {needs_review}")

    print(f"Review Status:   {review_status}")

    print("────────────────────────────────────────────────")

    print("Note: Internal reasoning kept in private audit logs.")



def main():

    init_db()

    db = SessionLocal()

    try:

        cg = get_or_create_caregiver(db)

        ch = get_or_create_child(db, cg)

        simulate_checkin(db, cg, ch)

    finally:

        db.close()



if __name__ == "__main__":

    main()
