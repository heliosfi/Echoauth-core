# capture_feed.py

# Requires: pip install opencv-python pyaudio

import cv2, pyaudio, numpy as np, threading, time

from echoauth_chamber import submit_sample, get_latest_result, _run


def capture_loop(user_id=0, rate_hz=10.0):

    cap = cv2.VideoCapture(0)

    pa = pyaudio.PyAudio()

    stream = pa.open(format=pyaudio.paFloat32, channels=1, rate=16000, input=True, frames_per_buffer=16000)

    period = 1.0 / rate_hz

    try:

        while _run:

            ret, frame = cap.read()

            if not ret:

                time.sleep(period); continue

            audio = np.frombuffer(stream.read(16000, exception_on_overflow=False), dtype=np.float32)

            submit_sample(user_id, frame, audio, time.time_ns())

            time.sleep(period)

    finally:

        cap.release()

        stream.stop_stream(); stream.close(); pa.terminate()


if __name__ == "__main__":

    threading.Thread(target=capture_loop, args=(0,10.0), daemon=True).start()

    for _ in range(50):

        res = get_latest_result(0)

        if res: print(res)

        time.sleep(0.1)


# build_trt_module.py

import torch

import torch_tensorrt as trt

from echoauth_chamber import MotherAffectNet, FatherVerifyNet, FusionMLP, DEVICE


class EchoAuthEnd2End(torch.nn.Module):

    def __init__(self):

        super().__init__()

        self.mother = MotherAffectNet()

        self.father = FatherVerifyNet()

        self.fuser  = FusionMLP()

    def forward(self, frame, audio):

        m = self.mother(frame)

        f = self.father(frame, audio)

        return self.fuser(torch.cat([m, f], 1))


if __name__ == "__main__":

    m = EchoAuthEnd2End().eval().to(DEVICE)

    # optionally load weights

    # m.mother.load_state_dict(torch.load("mother.pt", map_location=DEVICE))

    # ...

    scripted = torch.jit.script(m)

    trt_mod = trt.compile(

        scripted,

        inputs=[

            trt.Input((1,3,224,224), dtype=torch.float16, device=trt.Device("cuda:0")),

            trt.Input((1,16000),     dtype=torch.float16, device=trt.Device("cuda:0")),

        ],

        enabled_precisions={torch.float16},

        truncate_long_and_double=True

    )

    trt_mod.save("echoauth_trt.ts")

    print("Wrote echoauth_trt.ts")
