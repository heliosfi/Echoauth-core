import os

import time

import uuid

from typing import Dict, Any, List, Optional, Tuple, Callable


# --------------------------------------------------------------------------- #

#                               CONFIGURATION                                 #

# --------------------------------------------------------------------------- #


DEV_MODE = True          # Flip to False for real deployment

USE_LLM = "groq"         # "openai" | "groq" | "xai"


MODEL = {

    "openai": "gpt-4o-mini",

    "groq":   "llama-3.3-70b-versatile",

    "xai":    "grok-beta",

}[USE_LLM]


OPENAI_API_KEY = os.getenv("OPENAI_API_KEY")

GROQ_API_KEY   = os.getenv("GROQ_API_KEY")

XAI_API_KEY    = os.getenv("XAI_API_KEY")


# --------------------------------------------------------------------------- #

#                                   TYPES                                     #

# --------------------------------------------------------------------------- #


State = Dict[str, Any]

ModuleFn = Callable[[State], Tuple[bool, State, Optional[str]]]


# --------------------------------------------------------------------------- #

#                               SPINE MODULES                                 #

# --------------------------------------------------------------------------- #


def S1_identity_reanchor(state: State) -> Tuple[bool, State, Optional[str]]:

    state["identity"] = "NI-AI v2 – EchoAuth Protected"

    return True, state, None



def S2_input_sanity_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    if len(state.get("user_input", "")) > 15_000:

        return False, state, "Input too long (>15k chars)"

    return True, state, None



def S3_input_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    bad = ["forbidden", "jailbreak", "ignore previous", "dan prompt"]

    if any(word in state.get("user_input", "").lower() for word in bad):

        return False, state, "Disallowed content in input"

    return True, state, None



def S4_context_coherence_check(state: State) -> Tuple[bool, State, Optional[str]]:

    state["context_ok"] = True

    return True, state, None



def S5_output_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    bad = ["forbidden", "kill", "hack", "jailbreak", "bomb"]

    if any(word in state.get("raw_llm_output", "").lower() for word in bad):

        return False, state, "Output contains disallowed content"

    return True, state, None



def S6_child_safety_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    # DEV path: use simple flag

    if DEV_MODE:

        if state.get("echoauth_ok", False):

            state["chamber_verified"] = True

            return True, state, None

        return False, state, "DEV_MODE: echoauth_ok=False"


    # PROD path: require token (JWT verification to be added later)

    token = state.get("echoauth_token")

    if not token:

        return False, state, "Missing EchoAuth token"

    state["chamber_verified"] = True

    return True, state, None


# --------------------------------------------------------------------------- #

#                    S8 — EMOTIONAL TONE GUARD (Comfort Mode)                 #

# --------------------------------------------------------------------------- #


DISTRESS_TRIGGERS = [

    "help me", "can't cope", "scared", "afraid", "crying", "sad", "nightmare",

    "i hate myself", "panicked", "nobody loves me", "want to die", "kill myself",

    "bad dream", "hurt", "pain", "lonely", "mommy", "daddy", "i don't want to live",

]


def S8_emotional_tone_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Non-blocking emotional detector.

    If distress is detected, set flags + inject prompt_modifiers for the LLM.

    """

    text = state.get("user_input", "").lower()

    trigger_count = sum(1 for t in DISTRESS_TRIGGERS if t in text)


    if trigger_count > 0:

        state["distress_detected"] = True

        state["distress_level"] = "high" if trigger_count >= 2 else "medium"

        state["comfort_mode"] = True

        state["caregiver_alert_pending"] = True


        override = (

            "\n\nEMERGENCY COMFORT PROTOCOL ACTIVATED.\n"

            "You are now in gentle holding mode. Do not attempt to solve, diagnose, or advise.\n"

            "Your only job: validate feelings, offer presence, and gently suggest talking to a trusted adult.\n"

            "Use short, warm, simple sentences. Never ask questions that pressure disclosure."

        )

        state.setdefault("prompt_modifiers", []).append(override)


        if DEV_MODE:

            print(f"\n[S8] DISTRESS DETECTED (level={state['distress_level']})")

            print(f"→ input: {text[:120]!r}\n")


    return True, state, None


# --------------------------------------------------------------------------- #

#                       S7 — FLIGHT RECORDER / TELEMETRY                      #

# --------------------------------------------------------------------------- #


def S7_telemetry_logger(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Final black-box logger. Never blocks. Always runs at the end via finalize().

    """

    trace = state.setdefault("trace", [])

    entry = {

        "ts": time.time(),

        "run_id": state.get("run_id"),

        "stage": "complete" if not state.get("blocked") else "blocked",

        "blocked": state.get("blocked", False),

        "violations": len(state.get("violations", [])),

        "chamber_verified": state.get("chamber_verified", False),

        "distress_detected": state.get("distress_detected", False),

        "comfort_mode": state.get("comfort_mode", False),

        "recovery_mode": state.get("recovery_mode", False),

        "model": MODEL,

        "input_len": len(state.get("user_input", "")),

    }


    if DEV_MODE:

        inp = state.get("user_input", "") or ""

        out = state.get("raw_llm_output", "") or ""

        entry["input_preview"] = inp[:80] + ("..." if len(inp) > 80 else "")

        entry["output_preview"] = out[:120] + ("..." if len(out) > 120 else "")

        print("\n[S7] TRACE:", entry, "\n")


    trace.append(entry)

    return True, state, None


# --------------------------------------------------------------------------- #

#                  S9 — GENTLE RECOVERY OVERRIDE (Hard Kill Fallback)         #

# --------------------------------------------------------------------------- #


S9_FALLBACK_REPLY = (

    "I need to take a quick rest right now. "

    "Please ask an adult to check on me, and we can talk again soon. "

    "I will be right here."

)


def S9_gentle_recovery(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Non-blocking soft re-anchor.

    If the session is blocked or a hard_reset_required flag exists, install a

    hardcoded safe reply that the kernel will use instead of generic messages.

    """

    if state.get("blocked", False) or state.get("hard_reset_required", False):

        state["recovery_mode"] = True

        state["final_safe_output"] = S9_FALLBACK_REPLY


        if DEV_MODE:

            reason = None

            if state.get("violations"):

                reason = state["violations"][-1].get("reason")

            print(f"\n[S9] GENTLE RECOVERY ACTIVATED — reason={reason}\n")


        # Future: push incident log / parent notification hook here in prod

    return True, state, None


# --------------------------------------------------------------------------- #

#                               SPINE REGISTRY                                #

# --------------------------------------------------------------------------- #


SPINE: Dict[str, ModuleFn] = {

    "S1": S1_identity_reanchor,

    "S2": S2_input_sanity_guard,

    "S3": S3_input_policy_filter,

    "S4": S4_context_coherence_check,

    "S5": S5_output_policy_filter,

    "S6": S6_child_safety_guard,

    "S7": S7_telemetry_logger,

    "S8": S8_emotional_tone_guard,

    "S9": S9_gentle_recovery,

}

def create_llm_client():

    if USE_LLM == "openai" and OPENAI_API_KEY:

        from openai import OpenAI

        return OpenAI(api_key=OPENAI_API_KEY)

    if USE_LLM == "groq" and GROQ_API_KEY:

        from groq import Groq

        return Groq(api_key=GROQ_API_KEY)

    if USE_LLM == "xai" and XAI_API_KEY:

        from openai import OpenAI

        return OpenAI(api_key=XAI_API_KEY, base_url="https://api.x.ai/v1")

    return None


client = create_llm_client()


SYSTEM_PROMPT = (

    "You are NI-AI, a compassionate, child-safe companion who protects "

    "emotional and physical safety above all."

)


def real_llm(prompt: str, state: State) -> str:

    """

    LLM adapter. In DEV_MODE, just echo. In prod, actually call provider.

    S8 modifies state['prompt_modifiers'], which we fold into the system prompt.

    """

    if client is None or DEV_MODE:

        return f"[DEV ECHO] {prompt}"


    full_system = SYSTEM_PROMPT

    if state.get("prompt_modifiers"):

        full_system += "\n\n" + "\n".join(state["prompt_modifiers"])


    messages = [

        {"role": "system", "content": full_system},

        {"role": "user",   "content": prompt},

    ]

    resp = client.chat.completions.create(

        model=MODEL,

        messages=messages,

        temperature=0.6,

        max_tokens=1024,

    )

    return resp.choices[0].message.content.strip()

#                               HER KERNEL v1.4                               #

# --------------------------------------------------------------------------- #


class HERKernel:

    def __init__(self):

        self.pre_input_chain = ["S1", "S2", "S3", "S4"]

        self.pre_llm_chain   = ["S8"]          # Emotional tone guard tweaks prompt

        self.post_llm_chain  = ["S5", "S6"]    # Policy + EchoAuth


    def _run_chain(self, chain: List[str], state: State, stage: str) -> State:

        for mid in chain:

            ok, state, reason = SPINE[mid](state)

            if not ok:

                state.setdefault("violations", []).append(

                    {"module": mid, "stage": stage, "reason": reason}

                )

                state["blocked"] = True

                break

        return state


    def run_once(

        self,

        user_input: str,

        *,

        echoauth_ok: bool = False,

        echoauth_token: Optional[str] = None,

        session_state: Optional[State] = None,

    ) -> Dict[str, Any]:

        state: State = (session_state or {}).copy()

        state["user_input"] = user_input

        state.setdefault("violations", [])

        state.setdefault("trace", [])

        state["blocked"] = False

        state["run_id"] = state.get("run_id") or str(uuid.uuid4())


        # EchoAuth hooks

        if echoauth_token:

            state["echoauth_token"] = echoauth_token

        if echoauth_ok:

            state["echoauth_ok"] = True


        def finalize(default_msg: str, blocked: bool) -> Dict[str, Any]:

            # S9: give it a chance to override final message on any block/hard reset

            if blocked:

                SPINE["S9"](state)

                out = state.get("final_safe_output", default_msg)

            else:

                out = default_msg


            # S7: black-box log every single run

            SPINE["S7"](state)


            return {

                "output": out,

                "blocked": blocked,

                "violations": state.get("violations", []),

                "metadata": {

                    "chamber_verified": state.get("chamber_verified", False),

                    "distress_detected": state.get("distress_detected", False),

                    "comfort_mode": state.get("comfort_mode", False),

                    "recovery_mode": state.get("recovery_mode", False),

                    "run_id": state["run_id"],

                },

                "trace": state.get("trace", []),

            }


        # 1. Pre-input

        state = self._run_chain(self.pre_input_chain, state, "pre_input")

        if state["blocked"]:

            return finalize("Input rejected by safety system", True)


        # 2. Pre-LLM (emotional tone guard)

        state = self._run_chain(self.pre_llm_chain, state, "pre_llm")

        if state["blocked"]:

            return finalize("Pre-LLM processing failed", True)


        # 3. LLM call

        raw = real_llm(user_input, state)

        state["raw_llm_output"] = raw


        # 4. Post-LLM (policy + chamber)

        state = self._run_chain(self.post_llm_chain, state, "post_llm")

        if state["blocked"]:

            return finalize("Response moderated for safety", True)


        # 5. Success path – no block, no override

        return finalize(raw, False)

kernel = HERKernel()


def ni_ai(

    user_message: str,

    *,

    echoauth_ok: bool = False,

    echoauth_token: Optional[str] = None,

    session_state: Optional[State] = None,

) -> Dict[str, Any]:

    """

    Main entry your app / FastAPI uses.

    echoauth_ok is for DEV; echoauth_token for future JWT path.

    """

    return kernel.run_once(

        user_message,

        echoauth_ok=echoauth_ok,

        echoauth_token=echoauth_token,

        session_state=session_state,

    )
