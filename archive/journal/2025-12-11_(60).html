"""

MelaninCognitionOS v3.0 (Scaffold)

----------------------------------

A simple console-based helper to run your Level 3 OS:

- Temporal Integration Engine (TIE)

- Multi-Layer State Bus (MLSB)

- Action Synthesis Engine (ASE)

- Hybrid Roles: Navigator, Guardian, Architect


Run this in your terminal with: python melanin_cognition_os.py

"""


from dataclasses import dataclass, field

from typing import Optional, Dict



@dataclass

class TimeAxis:

    past: str       # T-

    present: str    # T0

    future: str     # T_plus



@dataclass

class LayerAxis:

    physical: str   # L1

    emotional: str  # L2

    mental: str     # L3

    vortex: str     # L4 (intuition/pattern sense)



@dataclass

class RoleOutput:

    navigator_direction: str

    guardian_warnings: str

    architect_expression: str

    architect_next_steps: str



@dataclass

class MelaninCognitionOS:

    """Core OS object representing your Level 3 decision frame."""

    context_title: str

    time_axis: Optional[TimeAxis] = None

    layer_axis: Optional[LayerAxis] = None

    role_output: Optional[RoleOutput] = None

    metadata: Dict[str, str] = field(default_factory=dict)


    def set_time_axis(self, past: str, present: str, future: str):

        self.time_axis = TimeAxis(past=past, present=present, future=future)


    def set_layer_axis(self, physical: str, emotional: str, mental: str, vortex: str):

        self.layer_axis = LayerAxis(

            physical=physical,

            emotional=emotional,

            mental=mental,

            vortex=vortex

        )


    def run_role_synthesis(self):

        """

        Very simple first-pass synthesis.

        You can customize this logic to better match your natural style.

        """


        if self.time_axis is None or self.layer_axis is None:

            raise ValueError("TimeAxis and LayerAxis must be set before synthesis.")


        # Navigator: choose direction based on future + vortex + mental

        nav_lines = []

        if self.time_axis.future.strip():

            nav_lines.append(f"Target future state: {self.time_axis.future}")

        if self.layer_axis.vortex.strip():

            nav_lines.append(f"Intuitive guidance: {self.layer_axis.vortex}")

        if self.layer_axis.mental.strip():

            nav_lines.append(f"Mental framing: {self.layer_axis.mental}")


        navigator_direction = "\n".join(nav_lines) or "Navigator is awaiting clearer inputs."


        # Guardian: highlight risks & safety points

        guardian_lines = []

        if self.layer_axis.emotional.strip():

            guardian_lines.append(f"Emotional check: {self.layer_axis.emotional}")

        if self.layer_axis.physical.strip():

            guardian_lines.append(f"Physical/real-world constraints: {self.layer_axis.physical}")


        # You can expand this with any risk rules you want

        if "tired" in self.layer_axis.physical.lower():

            guardian_lines.append("⚠ You may want rest before making a final call.")

        if "angry" in self.layer_axis.emotional.lower():

            guardian_lines.append("⚠ Avoid making irreversible decisions while angry.")


        guardian_warnings = "\n".join(guardian_lines) or "Guardian sees no immediate emotional/physical flags."


        # Architect: turn everything into an expression + next steps

        # Here we just stitch things together in a clean, simple way.

        context = self.context_title

        architect_expression = (

            f"For the context: '{context}',\n"

            f"- Past: {self.time_axis.past}\n"

            f"- Present: {self.time_axis.present}\n"

            f"- Future: {self.time_axis.future}\n\n"

            "Short explanation you can say out loud:\n"

        )


        # Craft a one-liner summary

        one_liner = self._build_one_liner()

        architect_expression += f"➤ \"{one_liner}\"\n"


        # Next steps: simple actionable suggestions

        architect_next_steps = (

            "Suggested next steps:\n"

            "- Clarify your future outcome in 1–2 bullet points.\n"

            "- Decide on ONE concrete action you can take in the next 24 hours.\n"

            "- Decide on ONE longer-term move (1–3 months).\n"

        )


        self.role_output = RoleOutput(

            navigator_direction=navigator_direction,

            guardian_warnings=guardian_warnings,

            architect_expression=architect_expression,

            architect_next_steps=architect_next_steps

        )


    def _build_one_liner(self) -> str:

        """

        Builds a one-liner based on present, future and mental framing.

        You can edit this to reflect your speaking style.

        """

        present = self.time_axis.present.strip()

        future = self.time_axis.future.strip()

        mental = self.layer_axis.mental.strip()


        if future and mental:

            return f"{present} – I’m moving this toward {future}, by focusing on {mental}."

        elif future:

            return f"{present} – and the direction is toward {future}."

        elif mental:

            return f"{present} – framed through {mental}."

        else:

            return present or "I am still clarifying this in my mind."


    def pretty_print(self):

        """Prints the OS state + role outputs in a clean way."""

        print("\n" + "=" * 60)

        print(f" Melanin Cognition OS – Context: {self.context_title}")

        print("=" * 60)


        print("\n[Time Axis]")

        print(f" Past (T-):     {self.time_axis.past}")

        print(f" Present (T0):  {self.time_axis.present}")

        print(f" Future (T+):   {self.time_axis.future}")


        print("\n[Layer Axis]")

        print(f" Physical (L1): {self.layer_axis.physical}")

        print(f" Emotional(L2): {self.layer_axis.emotional}")

        print(f" Mental  (L3):  {self.layer_axis.mental}")

        print(f" Vortex  (L4):  {self.layer_axis.vortex}")


        print("\n[Navigator Direction]")

        print(self.role_output.navigator_direction)


        print("\n[Guardian Warnings]")

        print(self.role_output.guardian_warnings)


        print("\n[Architect Expression]")

        print(self.role_output.architect_expression)


        print("\n[Architect Next Steps]")

        print(self.role_output.architect_next_steps)


        print("\n" + "=" * 60 + "\n")



def interactive_run():

    """

    Simple CLI runner.

    You can comment this out if you want to import the class in other files.

    """

    print("Melanin Cognition OS v3.0 – Room Mode Helper")

    print("-------------------------------------------\n")


    context = input("Context / situation (e.g. 'Investor asks about EchoAuth'): ").strip()

    os_obj = MelaninCognitionOS(context_title=context)


    print("\n--- Time Axis (Past / Present / Future) ---")

    past = input("T- (Past pattern / backstory): ")

    present = input("T0 (What is happening right now?): ")

    future = input("T+ (Desired / likely future if you act well): ")

    os_obj.set_time_axis(past=past, present=present, future=future)


    print("\n--- Layer Axis (Physical / Emotional / Mental / Vortex) ---")

    physical = input("L1 Physical (body, money, logistics, environment): ")

    emotional = input("L2 Emotional (your state, their state): ")

    mental = input("L3 Mental (facts, logic, structure you see): ")

    vortex = input("L4 Vortex (intuitive read / deeper pattern sense): ")

    os_obj.set_layer_axis(physical=physical, emotional=emotional, mental=mental, vortex=vortex)


    os_obj.run_role_synthesis()

    os_obj.pretty_print()



if __name__ == "__main__":

    interactive_run()
