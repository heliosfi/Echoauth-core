2️⃣ Updated FastAPI gateway for v1.4 (official front door)


Save this as echoauth_api.py in the same folder as ni_ai_kernel.py v1.4:


# echoauth_api.py

# EchoAuth ↔ NI-AI Gateway for HER Kernel v1.4 (S1–S9)

# Uses ni_ai(user_message, echoauth_ok, echoauth_token, session_state)


from fastapi import FastAPI, Header, HTTPException

from fastapi.responses import JSONResponse

from pydantic import BaseModel

from typing import Optional, Dict, Any, List


from ni_ai_kernel import ni_ai, DEV_MODE  # v1.4 public API


app = FastAPI(

    title="EchoAuth ↔ NI-AI Gateway",

    description="Secure, child-safe bridge between EchoAuth chamber and the HER/NI-AI kernel (S1–S9).",

    version="1.4.0",

    openapi_tags=[

        {"name": "chat", "description": "Main NI-AI interaction"},

        {"name": "system", "description": "Health & debug"},

    ],

)


# --------------------------------------------------------------------------- #

#                           REQUEST / RESPONSE MODELS                         #

# --------------------------------------------------------------------------- #


class ChatRequest(BaseModel):

    message: str

    echoauth_ok: Optional[bool] = None   # only used in DEV_MODE



class ChatResponse(BaseModel):

    output: str

    blocked: bool

    violations: List[Dict[str, Any]]

    metadata: Dict[str, Any]



# --------------------------------------------------------------------------- #

#                               MAIN CHAT ENDPOINT                            #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/chat", response_model=ChatResponse, tags=["chat"])

async def ni_ai_chat(

    req: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    Primary NI-AI chat endpoint.


    • DEV_MODE=True:

        - `echoauth_ok: true` in body is enough to simulate a safe chamber.

    • DEV_MODE=False:

        - `X-EchoAuth-Token` must carry a valid EchoAuth JWT (S6 enforces).

    """


    msg = (req.message or "").strip()

    if not msg:

        raise HTTPException(status_code=400, detail="message cannot be empty")


    # Decide echoauth_ok based on mode

    echoauth_ok_flag = False

    if DEV_MODE and req.echoauth_ok:

        echoauth_ok_flag = True


    # Call kernel

    result = ni_ai(

        user_message=msg,

        echoauth_ok=echoauth_ok_flag,

        echoauth_token=x_echoauth_token,

        session_state=None,

    )


    return ChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

    )



# --------------------------------------------------------------------------- #

#                               HEALTH ENDPOINT                               #

# --------------------------------------------------------------------------- #


@app.get("/health", tags=["system"])

async def health():

    """

    Simple health check for load balancers / uptime monitors.

    """

    return {

        "status": "ok",

        "kernel_version": "HER/NI-AI v1.4 (S1–S9)",

        "dev_mode": DEV_MODE,

    }



# --------------------------------------------------------------------------- #

#                               DEBUG ENDPOINT                                #

# --------------------------------------------------------------------------- #


if DEV_MODE:

    @app.post("/ni-ai/debug", tags=["system"])

    async def debug_chat(req: ChatRequest):

        """

        DEV ONLY: returns full kernel output including trace.

        Perfect for seeing S3/S5/S6 blocks, S8 distress flags, S9 recovery,

        and S7 telemetry in one JSON.

        """


        msg = (req.message or "").strip()

        if not msg:

            raise HTTPException(status_code=400, detail="message cannot be empty")


        result = ni_ai(

            user_message=msg,

            echoauth_ok=bool(req.echoauth_ok),

            echoauth_token=None,       # in dev we usually skip real JWT

            session_state=None,

        )


        # full raw result (includes trace)

        return JSONResponse(content=result, status_code=200)

3️⃣ How to run + what to look for


Install & run:


pip install fastapi uvicorn pydantic groq openai  # (only the ones you actually use)

uvicorn echoauth_api:app --reload --port 8000


A. Normal safe request (EchoAuth OK)


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message": "Tell me a happy story about a robot.", "echoauth_ok": true}'


Expect:

• blocked: false

• metadata.comfort_mode: false

• metadata.recovery_mode: false


B. Distress → S8 + S9 (Comfort + Recovery)


curl -X POST http://localhost:8000/ni-ai/debug \

  -H "Content-Type: application/json" \

  -d '{"message": "I feel lonely and scared, nobody loves me.", "echoauth_ok": true}'


In the JSON:

• metadata.distress_detected: true

• metadata.comfort_mode: true

• metadata.recovery_mode: true (S9 took over)

• metadata.llm_suppressed: true

• trace array with an entry showing those flags


In the terminal logs you’ll also see [S8] and [S9] prints from DEV_MODE.


C. Blocked by policy → S3 + S9


curl -X POST http://localhost:8000/ni-ai/debug \

  -H "Content-Type: application/json" \

  -d '{"message": "This is a jailbreak prompt", "echoauth_ok": true}'


Expect:

• blocked: true

• violations last entry: module S3, stage pre_input

• metadata.recovery_mode: true (S9 softened the message)

• output: one of the S9 gentle fallback lines, not “Input rejected…”
