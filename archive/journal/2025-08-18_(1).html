import torch

import torch.nn as nn

import torch.nn.functional as F

import torchaudio

import numpy as np


# ---------- Device ----------

DEVICE = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

IS_CUDA = DEVICE.type == 'cuda'

if IS_CUDA:

    torch.backends.cudnn.benchmark = True

class MotherAffectNet(nn.Module):

    def __init__(self, out_dim=128):

        super().__init__()

        # tiny conv -> gap -> fc (placeholder)

        self.conv = nn.Sequential(

            nn.Conv2d(3, 16, 3, 2, 1), nn.ReLU(),

            nn.Conv2d(16, 32, 3, 2, 1), nn.ReLU(),

            nn.Conv2d(32, 64, 3, 2, 1), nn.ReLU()

        )

        self.fc = nn.Linear(64*28*28, out_dim)  # for 224x224

        self.out_dim = out_dim


    def forward(self, x):

        x = self.conv(x)

        x = x.flatten(1)

        x = self.fc(x)

        return x  # [B, 128]


class FatherVerifyNet(nn.Module):

    def __init__(self, out_dim=256):

        super().__init__()

        self.img = nn.Sequential(

            nn.Conv2d(3, 16, 3, 2, 1), nn.ReLU(),

            nn.Conv2d(16, 32, 3, 2, 1), nn.ReLU()

        )

        self.img_fc = nn.Linear(32*56*56, 128)

        self.aud_fc = nn.Linear(16000, 128)  # expecting 1s @16kHz for demo

        self.fuse   = nn.Linear(256, out_dim)

        self.out_dim = out_dim


    def forward(self, img, audio):

        vi = self.img(img).flatten(1)

        vi = self.img_fc(vi)                # [B,128]

        if audio.dim() == 3:

            audio = audio.squeeze(1)        # [B,T]

        if audio.size(1) != 16000:

            # simple pad/trim to 16000 as placeholder

            T = audio.size(1)

            if T > 16000: audio = audio[:, :16000]

            else: audio = F.pad(audio, (0, 16000-T))

        va = self.aud_fc(audio)             # [B,128]

        x  = torch.cat([vi, va], dim=1)     # [B,256]

        return self.fuse(x)                 # [B,256]


class FusionMLP(nn.Module):

    def __init__(self, in_dim=128+256):

        super().__init__()

        self.mlp = nn.Sequential(

            nn.Linear(in_dim, 128), nn.ReLU(),

            nn.Linear(128, 1)

        )


    def forward(self, feats):

        return self.mlp(feats)  # [B,1]


mother = MotherAffectNet().to(DEVICE).eval()

father = FatherVerifyNet().to(DEVICE).eval()

fuser  = FusionMLP().to(DEVICE).eval()


# ---------- Config ----------

THRESHOLD_ACCEPT = 0.80

THRESHOLD_RETRY  = 0.55


# ---------- Preprocessing ----------

MEAN = torch.tensor([0.485, 0.456, 0.406], device=DEVICE).view(1,3,1,1)

STD  = torch.tensor([0.229, 0.224, 0.225], device=DEVICE).view(1,3,1,1)

TARGET_SR = 16000


def preprocess_gpu(frame_bgr_u8: np.ndarray):

    """

    Input: HxWx3 uint8 BGR numpy

    Output: 1x3x224x224 float32 on DEVICE (channels_last on CUDA)

    """

    t = torch.from_numpy(frame_bgr_u8).to(DEVICE, non_blocking=IS_CUDA)  # H W C u8

    t = t.float().permute(2,0,1).unsqueeze(0)                            # 1 C H W

    t = t[:, [2,1,0], :, :]                                              # BGR->RGB

    t = F.interpolate(t, size=(224,224), mode='bilinear', align_corners=False)

    t = t / 255.0

    t = (t - MEAN) / STD

    if IS_CUDA:

        t = t.contiguous(memory_format=torch.channels_last)

    return t


def preprocess_audio_gpu(audio_f32, orig_sr=44100):

    """

    Input: 1D float32 numpy or torch, arbitrary sr

    Output: 1xT float32 on DEVICE (T ~ 16000)

    """

    if not torch.is_tensor(audio_f32):

        wav = torch.from_numpy(np.asarray(audio_f32, dtype=np.float32))

    else:

        wav = audio_f32.float()


    if orig_sr != TARGET_SR:

        resampler = torchaudio.transforms.Resample(orig_sr, TARGET_SR)

        wav = resampler(wav)

    peak = wav.abs().max()

    if peak > 0: wav = wav / peak

    wav = wav.unsqueeze(0).to(DEVICE, non_blocking=IS_CUDA)  # [1,T]

    return wav


# ---------- Core EchoAuth ----------

@torch.inference_mode()

def echoauth(frame_gpu: torch.Tensor, audio_gpu: torch.Tensor) -> float:

    # (Add CUDA streams & autocast later; simple path for clarity)

    m_feat = mother(frame_gpu)                 # [B,128]

    f_feat = father(frame_gpu, audio_gpu)      # [B,256]

    fused  = fuser(torch.cat([m_feat, f_feat], dim=1))  # [B,1]

    return float(fused.squeeze().item())


def verify_user(frame_bgr_u8: np.ndarray, audio_f32, orig_sr=44100):

    frame = preprocess_gpu(frame_bgr_u8)

    audio = preprocess_audio_gpu(audio_f32, orig_sr=orig_sr)

    score = echoauth(frame, audio)

    decision = "accept" if score >= THRESHOLD_ACCEPT else "retry" if score >= THRESHOLD_RETRY else "deny"

    return {"score": score, "decision": decision}


# ---------- ONNX Export ----------

def export_models():

    B, C, H, W = 1, 3, 224, 224

    T = 16000

    x_img  = torch.randn(B, C, H, W, device=DEVICE).contiguous(memory_format=torch.channels_last)

    x_img2 = torch.randn(B, C, H, W, device=DEVICE).contiguous(memory_format=torch.channels_last)

    x_aud  = torch.randn(B, T, device=DEVICE)

    x_fuse = torch.randn(B, 128+256, device=DEVICE)


    torch.onnx.export(

        mother, x_img, "mother.onnx", opset_version=17,

        input_names=["image"], output_names=["m_feat"],

        dynamic_axes={"image": {0: "batch"}, "m_feat": {0: "batch"}}

    )


    torch.onnx.export(

        father, (x_img2, x_aud), "father.onnx", opset_version=17,

        input_names=["image", "audio"], output_names=["f_feat"],

        dynamic_axes={

            "image": {0: "batch"},

            "audio": {0: "batch", 1: "time"},

            "f_feat": {0: "batch"}

        }

    )


    torch.onnx.export(

        fuser, x_fuse, "fuser.onnx", opset_version=17,

        input_names=["features"], output_names=["score"],

        dynamic_axes={"features": {0: "batch"}, "score": {0: "batch"}}

    )

    print("Exported: mother.onnx, father.onnx, fuser.onnx")
