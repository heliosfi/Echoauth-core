1. What F.I Actually Does (in practice)


For NI-AI, Field Intelligence is the module that:

1. Ingests signals

• Market data (prices, volume, news sentiment)

• User/emotional signals (stress, overload, pacing)

• System signals (latency, errors, load)

2. Normalizes them

• Cleans, scales, time-aligns streams

• Converts chaos → structured features

3. Hands them to NI-AI Coherence / SRK

• “Given this environment, what’s the coherent response?”

• Avoid overreaction, fear, FOMO, system panic

4. Returns an action

• e.g. “NO TRADE – environment unstable”

• or “LOW-RISK BUY ZONE – within guardrails”

• or “SOFTEN TONE – user overloaded”

• or “RATE-LIMIT QUERIES – infra at 80%”


So coding-wise, F.I is a pipeline:


raw_signals → features → NI-AI coherence → decision → log

3. Core Data Model (Signals + Context)


# fi_engine/signals.py

from dataclasses import dataclass

from typing import Dict, Any

from datetime import datetime


@dataclass

class RawSignal:

    source: str             # "market", "user", "system"

    timestamp: datetime

    payload: Dict[str, Any] # raw dict (price, rsi, cpu, mood, etc.)


@dataclass

class FeatureVector:

    domain: str             # "market", "neuro", "infra"

    timestamp: datetime

    features: Dict[str, float]  # cleaned numeric features

    meta: Dict[str, Any] = None

4. Field Intelligence Engine – Orchestrator


This is the heart: it runs cycles, calls NI-AI, and returns a stable decision.


# fi_engine/engine.py

from datetime import datetime

from typing import Dict, Any, List


from .signals import RawSignal, FeatureVector

from . import coherence_adapter, decisions

from .collectors import markets, user_state, system_health

from .features import market_features, user_features



class FieldIntelligenceEngine:

    """

    F.I = Field Intelligence for NI-AI.

    Listens to the field, stabilizes it through NI-AI, and outputs decisions.

    """


    def __init__(self, mode: str = "market"):

        self.mode = mode  # "market", "hybrid", "neuro", etc.


    def collect_signals(self) -> List[RawSignal]:

        signals = []


        if self.mode in ("market", "hybrid"):

            m = markets.collect_market_snapshot()

            signals.append(m)


        if self.mode in ("neuro", "hybrid"):

            u = user_state.collect_user_state()

            signals.append(u)


        if self.mode in ("infra", "hybrid"):

            s = system_health.collect_system_metrics()

            signals.append(s)


        return signals


    def featurize(self, raw: List[RawSignal]) -> List[FeatureVector]:

        vectors: List[FeatureVector] = []


        for sig in raw:

            if sig.source == "market":

                fv = market_features.build_features(sig)

            elif sig.source == "user":

                fv = user_features.build_features(sig)

            else:

                # system infra features can be added later

                continue


            if fv:

                vectors.append(fv)


        return vectors


    def run_cycle(self) -> Dict[str, Any]:

        """

        One full F.I cycle:

        1) collect signals

        2) make features

        3) call NI-AI Coherence

        4) make decision

        """

        raw = self.collect_signals()

        feats = self.featurize(raw)


        coherence_state = coherence_adapter.evaluate(feats)

        decision = decisions.make_decision(self.mode, feats, coherence_state)


        return {

            "timestamp": datetime.utcnow().isoformat(),

            "mode": self.mode,

            "coherence_state": coherence_state,

            "decision": decision,

        }

# fi_engine/features/market_features.py

from .signals import RawSignal, FeatureVector


def build_features(sig: RawSignal) -> FeatureVector:

    p = sig.payload

    price = float(p["price"])

    vwap = float(p.get("vwap", price))

    rsi = float(p.get("rsi_14", 50))

    vol = float(p.get("volume", 0))


    features = {

        "price": price,

        "vwap_diff": price - vwap,

        "rsi_14": rsi,

        "volume_millions": vol / 1_000_000.0,

        "overbought_flag": 1.0 if rsi > 70 else 0.0,

        "oversold_flag": 1.0 if rsi < 30 else 0.0,

    }


    return FeatureVector(

        domain="market",

        timestamp=sig.timestamp,

        features=features,

        meta={"symbol": p.get("symbol", "UNKNOWN")},

    )

6. Hook into NI-AI Coherence / SRK


This is where F.I calls your Coherence Engine + SRK logic:


# fi_engine/coherence_adapter.py

from typing import List, Dict, Any

from .signals import FeatureVector


def evaluate(vectors: List[FeatureVector]) -> Dict[str, Any]:

    """

    Bridges Field Intelligence into NI-AI Coherence Engine (C1–C5) + SRK.

    Here we mock the logic; later this calls your real NI-AI core.

    """

    if not vectors:

        return {"coherence_score": 0.0, "state": "NO_DATA"}


    # Simple example: aggregate risk/stress signals

    risk_score = 0.0


    for v in vectors:

        f = v.features

        # Market risk: distance from VWAP + oversold/overbought tension

        risk_score += abs(f.get("vwap_diff", 0)) * 0.1

        if f.get("overbought_flag", 0) == 1.0:

            risk_score += 1.0

        if f.get("oversold_flag", 0) == 1.0:

            risk_score += 0.5


    # Clamp & interpret

    risk_score = min(max(risk_score, 0.0), 10.0)


    if risk_score < 3:

        state = "STABLE"

    elif risk_score < 6:

        state = "ELEVATED"

    else:

        state = "UNSTABLE"


    return {

        "coherence_score": max(0.0, 10.0 - risk_score),

        "risk_score": risk_score,

        "state": state,

    }

# fi_engine/decisions.py

from typing import Dict, Any, List

from .signals import FeatureVector


def make_decision(mode: str,

                  feats: List[FeatureVector],

                  coherence_state: Dict[str, Any]) -> Dict[str, Any]:

    state = coherence_state.get("state", "UNKNOWN")

    score = coherence_state.get("coherence_score", 0.0)


    if mode == "market":

        return _market_decision(state, score, feats)

    elif mode == "neuro":

        return _neuro_decision(state, score, feats)

    else:

        return {"action": "NO_ACTION", "reason": "UNKNOWN_MODE"}



def _market_decision(state: str,

                     score: float,

                     feats: List[FeatureVector]) -> Dict[str, Any]:

    if state == "UNSTABLE" or score < 4.0:

        return {

            "action": "NO_TRADE",

            "risk": "HIGH",

            "note": "Environment unstable – NI-AI SRK veto.",

        }

    if state == "ELEVATED":

        return {

            "action": "REDUCED_SIZE",

            "risk": "MEDIUM",

            "note": "Proceed with caution – tighten stops, smaller size.",

        }

    return {

        "action": "OK_TO_SCAN",

        "risk": "LOW",

        "note": "Environment stable – OK to search for setups.",

    }



def _neuro_decision(state: str,

                    score: float,

                    feats: List[FeatureVector]) -> Dict[str, Any]:

    if state == "UNSTABLE":

        return {

            "action": "SOFTEN_ENVIRONMENT",

            "note": "Reduce stimuli, slow pacing, simpler language.",

        }

    return {

        "action": "NORMAL_FLOW",

        "note": "User environment acceptable.",

    }
