"""

EchoAuth™ + NI-AI Local MVP Demo (Final Launcher)


Purpose:

  - Initialize local DB

  - Create demo caregiver + child (if missing)

  - Encode caregiver cue packet into identity token BEFORE inference

  - Run NI-AI evaluate_cues() on tokenized cues

  - Apply danger / emotion / conflict flag rules

  - Store CheckIn + NIAILog in DB

  - Append private audit log (JSON file)

  - Print ONE clear caregiver-facing result to terminal


No UI. No dark boxes. No multi-output.

"""


import uuid

import hashlib

import json

from datetime import datetime

from pathlib import Path


from sqlalchemy.orm import Session


from db.session import SessionLocal, engine

from db import models

from core.ni_ai import evaluate_cues



# ─────────────────────────────────────────

# DB INIT

# ─────────────────────────────────────────


def init_db():

    """Create DB tables if they do not exist."""

    models.Base.metadata.create_all(bind=engine)



# ─────────────────────────────────────────

# PRIVATE AUDIT LOG (FILE)

# ─────────────────────────────────────────


AUDIT_LOG_PATH = Path("./audit_logs.json")



def _init_audit_file():

    if not AUDIT_LOG_PATH.exists():

        AUDIT_LOG_PATH.write_text("[]", encoding="utf-8")



def log_internal_audit(checkin_id: int, payload: dict):

    """Append a private audit log record (never exposed to child UI)."""

    _init_audit_file()

    current = json.loads(AUDIT_LOG_PATH.read_text(encoding="utf-8"))

    record = {

        "checkin_id": checkin_id,

        "logged_at": datetime.utcnow().isoformat(),

        **payload,

    }

    current.append(record)

    AUDIT_LOG_PATH.write_text(json.dumps(current, indent=2), encoding="utf-8")



# ─────────────────────────────────────────

# HELPERS: CAREGIVER + CHILD

# ─────────────────────────────────────────


def get_or_create_caregiver(db: Session) -> models.Caregiver:

    """Create a demo caregiver if one does not already exist."""

    email = "demo.caregiver@echoauth.local"

    cg = db.query(models.Caregiver).filter(

        models.Caregiver.email == email

    ).first()

    if cg:

        return cg


    cg = models.Caregiver(

        name="Demo Caregiver",

        email=email,

        hashed_password=hashlib.sha256("demo123".encode()).hexdigest(),

        created_at=datetime.utcnow(),

    )

    db.add(cg)

    db.commit()

    db.refresh(cg)

    return cg



def get_or_create_child(db: Session, caregiver: models.Caregiver) -> models.Child:

    """Create a demo child linked to caregiver if not present."""

    ch = db.query(models.Child).filter(

        models.Child.caregiver_id == caregiver.id,

        models.Child.name == "Demo Child"

    ).first()

    if ch:

        return ch


    ch = models.Child(

        caregiver_id=caregiver.id,

        name="Demo Child",

        age_range="6-8",

        notes="Created by local EchoAuth MVP demo.",

        created_at=datetime.utcnow(),

    )

    db.add(ch)

    db.commit()

    db.refresh(ch)

    return ch



# ─────────────────────────────────────────

# IDENTITY TOKEN ENCODING (BEFORE INFERENCE)

# ─────────────────────────────────────────


def encode_cue_to_identity_token(cues: dict, child_id: int) -> dict:

    """

    Encode caregiver cues into an identity token BEFORE NI-AI inference.

    This is your core doctrine in code: identity anchoring precedes reasoning.

    """

    return {

        "token_id": str(uuid.uuid4()),

        "child_id": child_id,

        "cue": cues,

        "encoded_at": datetime.utcnow().isoformat(),

        "origin": "caregiver_intake",

    }



# ─────────────────────────────────────────

# MAIN DEMO CHECK-IN FLOW

# ─────────────────────────────────────────


def simulate_checkin(db: Session, caregiver: models.Caregiver, child: models.Child):

    """

    Run a single, high-signal check-in through the NI-AI engine,

    store DB records, log internal details, and print ONE final result.

    """


    # 1. Caregiver describes real-world cues (high-distress example)

    cue_packet = {

        "behavior": "head banging on wall, pacing, covering ears",

        "environment": "very loud crowded store with bright fluorescent lights",

        "caregiver_observation": "looks overwhelmed, close to meltdown",

    }


    # 2. Identity token encoding (before inference)

    identity_token = encode_cue_to_identity_token(cue_packet, child.id)


    # 3. NI-AI evaluation on tokenized cues

    #    evaluate_cues() should:

    #       - compute final_label

    #       - confidence

    #       - emotion_score

    #       - danger_score

    #       - conflict_score

    #       - winning_thread

    #       - internal_threads (for logging)

    eval_result = evaluate_cues(identity_token)


    final_label = eval_result["final_label"]

    confidence = eval_result["confidence"]

    emotion_score = eval_result["emotion_score"]

    danger_score = eval_result.get("danger_score", 0.0)

    conflict_score = eval_result["conflict_score"]

    winning_thread = eval_result["winning_thread"]

    internal_threads = eval_result["internal_threads"]


    # 4. Auto-flag logic (same as your API rules)

    needs_review = False

    review_status = "none"


    if (

        emotion_score > 0.7

        or conflict_score > 0.6

        or (danger_score > 0.0 and confidence < 0.85)

    ):

        needs_review = True

        review_status = "pending"


    # 5. Store CheckIn in DB (one result only)

    checkin = models.CheckIn(

        caregiver_id=caregiver.id,

        child_id=child.id,

        observed_cues=json.dumps(identity_token),

        result_label=final_label,

        confidence=confidence,

        status="success",

        emotion_score=emotion_score,

        needs_review=needs_review,

        review_status=review_status,

        created_at=datetime.utcnow(),

    )

    db.add(checkin)

    db.commit()

    db.refresh(checkin)


    # 6. Store NI-AI internal log (private)

    ni_log = models.NIAILog(

        checkin_id=checkin.id,

        winning_thread=winning_thread,

        conflict_score=conflict_score,

        internal_notes=json.dumps(internal_threads),

        created_at=datetime.utcnow(),

    )

    db.add(ni_log)

    db.commit()


    # 7. File-based private audit record

    log_internal_audit(checkin.id, {

        "caregiver_name": caregiver.name,

        "child_name": child.name,

        "emotion_score": emotion_score,

        "danger_score": danger_score,

        "conflict_score": conflict_score,

        "confidence_score": confidence,

        "final_label": final_label,

        "needs_review": needs_review,

        "review_status": review_status,

    })


    # 8. Terminal output: ONE clear caregiver result, no multi-output

    print("──────────── EchoAuth™ NI-AI Local MVP Demo ────────────")

    print(f"Caregiver:       {caregiver.name} (id={caregiver.id})")

    print(f"Child:           {child.name} (id={child.id})")

    print("Cue Packet:")

    print(json.dumps(cue_packet, indent=2))

    print("──────────────── Evaluation Result ──────────────────────")

    print(f"Status:          {final_label}")        # 'Safe' or 'Needs Attention'

    print(f"Confidence:      {confidence:.2f}")

    print(f"Emotion/Stress:  {emotion_score:.2f}  (0=Calm, 1=High Distress)")

    print(f"Danger score:    {danger_score:.2f}")

    print(f"Conflict score:  {conflict_score:.2f}")

    print("──────────────── Review Routing ────────────────────────")

    print(f"Needs Review:    {needs_review}")

    print(f"Review Status:   {review_status}")

    print("──────────────── Notes ─────────────────────────────────")

    print("Internal reasoning threads are logged privately for audit,")

    print("and are never exposed to the child-facing layer.")

    print("──────────────────────── End Demo ─────────────────────")



# ─────────────────────────────────────────

# ENTRYPOINT

# ─────────────────────────────────────────


def main():

    init_db()

    db = SessionLocal()

    try:

        caregiver = get_or_create_caregiver(db)

        child = get_or_create_child(db, caregiver)

        simulate_checkin(db, caregiver, child)

    finally:

        db.close()



if __name__ == "__main__":

    main()
