1) app.py


from flask import Flask, render_template, request, jsonify

from uuid import uuid4

from datetime import datetime, timedelta, timezone

from threading import Lock


app = Flask(__name__)


# In-memory store for prototype only

REQUESTS = {}

TOKENS = {}

AUDIT_LOG = []


store_lock = Lock()


TOKEN_TTL_SECONDS = 30



def now_utc():

    return datetime.now(timezone.utc)



def ts():

    return now_utc().isoformat()



def log_event(request_id, event, detail=None):

    entry = {

        "timestamp": ts(),

        "request_id": request_id,

        "event": event,

        "detail": detail or {}

    }

    AUDIT_LOG.append(entry)



@app.route("/")

def home():

    return jsonify({

        "message": "EchoAuth demo running",

        "routes": {

            "child_ui": "/child",

            "parent_ui": "/parent",

            "logs": "/logs"

        }

    })



@app.route("/child")

def child_ui():

    return render_template("child.html")



@app.route("/parent")

def parent_ui():

    return render_template("parent.html")



@app.route("/request_action", methods=["POST"])

def request_action():

    data = request.get_json(force=True)

    action = data.get("action", "unlock_door")


    request_id = str(uuid4())


    with store_lock:

        REQUESTS[request_id] = {

            "id": request_id,

            "action": action,

            "created_at": ts(),

            "status": "pending",

            "executed": False

        }


    log_event(request_id, "REQUEST_CREATED", {"action": action})

    log_event(request_id, "AWAITING_AUTHORITY")


    return jsonify({

        "ok": True,

        "request": REQUESTS[request_id]

    })



@app.route("/pending_requests", methods=["GET"])

def pending_requests():

    with store_lock:

        pending = [

            r for r in REQUESTS.values()

            if r["status"] == "pending"

        ]

    return jsonify({"pending_requests": pending})



@app.route("/approve/<request_id>", methods=["POST"])

def approve(request_id):

    with store_lock:

        req = REQUESTS.get(request_id)

        if not req:

            return jsonify({"ok": False, "error": "Request not found"}), 404


        if req["status"] != "pending":

            return jsonify({"ok": False, "error": f"Request already {req['status']}"}), 400


        issued_at = now_utc()

        expires_at = issued_at + timedelta(seconds=TOKEN_TTL_SECONDS)


        token = {

            "request_id": request_id,

            "approved_by": "parent_demo",

            "issued_at": issued_at.isoformat(),

            "expires_at": expires_at.isoformat(),

            "used": False

        }


        TOKENS[request_id] = token

        req["status"] = "approved"


    log_event(request_id, "APPROVED", {"approved_by": "parent_demo"})

    return jsonify({"ok": True, "token": token})



@app.route("/deny/<request_id>", methods=["POST"])

def deny(request_id):

    with store_lock:

        req = REQUESTS.get(request_id)

        if not req:

            return jsonify({"ok": False, "error": "Request not found"}), 404


        if req["status"] != "pending":

            return jsonify({"ok": False, "error": f"Request already {req['status']}"}), 400


        req["status"] = "denied"


    log_event(request_id, "DENIED", {"denied_by": "parent_demo"})

    log_event(request_id, "EXECUTION_BLOCKED", {"reason": "Denied by parent"})

    return jsonify({"ok": True})



@app.route("/execute/<request_id>", methods=["POST"])

def execute(request_id):

    with store_lock:

        req = REQUESTS.get(request_id)

        if not req:

            return jsonify({"ok": False, "result": "DENY", "reason": "Request not found"}), 404


        token = TOKENS.get(request_id)


        # No token = HOLD

        if token is None:

            log_event(request_id, "EXECUTION_HELD", {"reason": "No valid approval token"})

            return jsonify({

                "ok": False,

                "result": "HOLD",

                "reason": "Awaiting authority"

            }), 200


        # Already used = DENY

        if token["used"]:

            log_event(request_id, "EXECUTION_BLOCKED", {"reason": "Token already used"})

            return jsonify({

                "ok": False,

                "result": "DENY",

                "reason": "Token already used"

            }), 400


        # Request mismatch = DENY

        if token["request_id"] != request_id:

            log_event(request_id, "EXECUTION_BLOCKED", {"reason": "Token request mismatch"})

            return jsonify({

                "ok": False,

                "result": "DENY",

                "reason": "Token mismatch"

            }), 400


        # Expired = DENY

        if now_utc() > datetime.fromisoformat(token["expires_at"]):

            req["status"] = "expired"

            log_event(request_id, "EXPIRED", {"reason": "Approval token expired"})

            log_event(request_id, "EXECUTION_BLOCKED", {"reason": "Expired token"})

            return jsonify({

                "ok": False,

                "result": "DENY",

                "reason": "Approval expired"

            }), 400


        # Single-use enforcement

        token["used"] = True

        req["executed"] = True

        req["status"] = "executed"


    log_event(request_id, "EXECUTION_ALLOWED", {"action": req["action"]})

    return jsonify({

        "ok": True,

        "result": "EXECUTE",

        "message": f"Action executed: {req['action']}"

    })



@app.route("/status/<request_id>", methods=["GET"])

def status(request_id):

    req = REQUESTS.get(request_id)

    if not req:

        return jsonify({"ok": False, "error": "Request not found"}), 404


    related_logs = [x for x in AUDIT_LOG if x["request_id"] == request_id]

    token = TOKENS.get(request_id)


    return jsonify({

        "ok": True,

        "request": req,

        "token": token,

        "audit_log": related_logs

    })



@app.route("/logs", methods=["GET"])

def logs():

    return jsonify({"audit_log": AUDIT_LOG})



if __name__ == "__main__":

    app.run(debug=True)

2) templates/child.html


<!DOCTYPE html>

<html lang="en">

<head>

  <meta charset="UTF-8" />

  <title>EchoAuth Child Interface</title>

  <style>

    body { font-family: Arial, sans-serif; max-width: 700px; margin: 40px auto; }

    button { padding: 12px 20px; font-size: 16px; cursor: pointer; }

    .box { border: 1px solid #ccc; padding: 16px; margin-top: 16px; border-radius: 8px; }

    pre { white-space: pre-wrap; word-wrap: break-word; background: #f6f6f6; padding: 12px; border-radius: 8px; }

  </style>

</head>

<body>

  <h1>EchoAuth Demo — Child Side</h1>

  <p>Press the button to request a protected action.</p>


  <button onclick="requestAction()">Request Unlock Door</button>

  <button onclick="tryExecute()" id="executeBtn" disabled>Try Execute</button>

  <button onclick="refreshStatus()" id="statusBtn" disabled>Refresh Status</button>


  <div class="box">

    <strong>Current Request ID:</strong>

    <div id="requestId">None</div>

  </div>


  <div class="box">

    <strong>Status:</strong>

    <pre id="statusBox">No request yet.</pre>

  </div>


  <script>

    let currentRequestId = null;


    async function requestAction() {

      const res = await fetch("/request_action", {

        method: "POST",

        headers: {"Content-Type": "application/json"},

        body: JSON.stringify({ action: "unlock_door" })

      });


      const data = await res.json();

      currentRequestId = data.request.id;


      document.getElementById("requestId").textContent = currentRequestId;

      document.getElementById("statusBox").textContent =

        "Request created.\nSystem state: HOLD until parent acts.";

      document.getElementById("executeBtn").disabled = false;

      document.getElementById("statusBtn").disabled = false;

    }


    async function tryExecute() {

      if (!currentRequestId) return;


      const res = await fetch(`/execute/${currentRequestId}`, {

        method: "POST"

      });


      const data = await res.json();

      document.getElementById("statusBox").textContent =

        JSON.stringify(data, null, 2);

    }


    async function refreshStatus() {

      if (!currentRequestId) return;


      const res = await fetch(`/status/${currentRequestId}`);

      const data = await res.json();

      document.getElementById("statusBox").textContent =

        JSON.stringify(data, null, 2);

    }

  </script>

</body>

</html>

<!DOCTYPE html>

<html lang="en">

<head>

  <meta charset="UTF-8" />

  <title>EchoAuth Parent Interface</title>

  <style>

    body { font-family: Arial, sans-serif; max-width: 900px; margin: 40px auto; }

    .request-card {

      border: 1px solid #ccc;

      padding: 16px;

      margin-bottom: 12px;

      border-radius: 8px;

    }

    button {

      padding: 10px 16px;

      margin-right: 8px;

      cursor: pointer;

    }

    pre {

      white-space: pre-wrap;

      word-wrap: break-word;

      background: #f6f6f6;

      padding: 12px;

      border-radius: 8px;

    }

  </style>

</head>

<body>

  <h1>EchoAuth Demo — Parent Side</h1>

  <p>Approve or deny pending requests.</p>


  <button onclick="loadPending()">Refresh Pending Requests</button>


  <div id="pendingContainer" style="margin-top: 20px;"></div>


  <script>

    async function loadPending() {

      const res = await fetch("/pending_requests");

      const data = await res.json();


      const container = document.getElementById("pendingContainer");

      container.innerHTML = "";


      if (!data.pending_requests.length) {

        container.innerHTML = "<p>No pending requests.</p>";

        return;

      }


      data.pending_requests.forEach(req => {

        const div = document.createElement("div");

        div.className = "request-card";

        div.innerHTML = `

          <p><strong>Request ID:</strong> ${req.id}</p>

          <p><strong>Action:</strong> ${req.action}</p>

          <p><strong>Created:</strong> ${req.created_at}</p>

          <button onclick="approveRequest('${req.id}')">Approve</button>

          <button onclick="denyRequest('${req.id}')">Deny</button>

          <pre id="result-${req.id}">Awaiting action.</pre>

        `;

        container.appendChild(div);

      });

    }


    async function approveRequest(id) {

      const res = await fetch(`/approve/${id}`, { method: "POST" });

      const data = await res.json();

      document.getElementById(`result-${id}`).textContent =

        JSON.stringify(data, null, 2);

      loadPending();

    }


    async function denyRequest(id) {

      const res = await fetch(`/deny/${id}`, { method: "POST" });

      const data = await res.json();

      document.getElementById(`result-${id}`).textContent =

        JSON.stringify(data, null, 2);

      loadPending();

    }


    loadPending();

  </script>

</body>

</html>

5) how to test it


Test 1 — HOLD works

1. Open /child

2. Press Request Unlock Door

3. Press Try Execute before parent approves


Expected result:


{

  "ok": false,

  "result": "HOLD",

  "reason": "Awaiting authority"

}


That proves:

• no approval

• no execution

• safe hold state


⸻


Test 2 — DENY works

1. Create request from child side

2. Open /parent

3. Press Deny

4. Back on child side, press Try Execute


Expected:

• blocked

• no execution


⸻


Test 3 — APPROVE works

1. Create request

2. Parent presses Approve

3. Child presses Try Execute


Expected result:


{

  "ok": true,

  "result": "EXECUTE",

  "message": "Action executed: unlock_door"

}



⸻


Test 4 — single-use works


After successful execute, press Try Execute again.


Expected:

• denied

• token already used


That proves:

• no replay

• no duplicate authority use


⸻


Test 5 — expiration works

1. Request action

2. Parent approves

3. Wait over 30 seconds

4. Try execute


Expected:

• denied

• approval expired


That proves:

• no stale authority

• no zombie permission

6) what this prototype proves


This demo proves all of your core claims in small form:

• no execution without live authority

• hold is valid

• approval is time-bound

• approval is single-use

• execution is mechanically gated

• denial blocks action


That is enough for a first prototype.
