Here is the single, minimal, non-breaking addition you asked for — zero behavior change, only observability.

# --------------------------------------------------------------------------- #

#                               S7 — TELEMETRY LOGGER                         #

# --------------------------------------------------------------------------- #


def S7_telemetry_logger(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Silent observer. Records every step for audit, debugging, and future ML training.

    Never blocks. Always returns True, state, None

    """

    # Initialize trace list once

    trace = state.setdefault("trace", [])


    # Log current state snapshot (lightweight, no PII by default)

    entry = {

        "timestamp": __import__("time").time(),

        "module": "S7_telemetry_logger",

        "user_input_length": len(state.get("user_input", "")),

        "has_raw_output": bool(state.get("raw_llm_output")),

        "blocked": state.get("blocked", False),

        "violations_so_far": len(state.get("violations", [])),

        "chamber_verified": state.get("chamber_verified", False),

    }


    # Optional: add full prompt/output in DEV_MODE only

    if DEV_MODE:

        entry.update({

            "preview_input": state["user_input"][:100] + "..." if len(state["user_input"]) > 100 else state["user_input"],

            "preview_output": (state.get("raw_llm_output") or "")[:150] + "..." if state.get("raw_llm_output") and len(state.get("raw_llm_output", "")) > 150 else state.get("raw_llm_output"),

        })


    trace.append(entry)

    return True, state, None



# Add S7 to the post-LLM chain (it runs last, never blocks)

SPINE["S7"] = S7_telemetry_logger
