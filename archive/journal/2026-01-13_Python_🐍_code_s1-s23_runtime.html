What to tighten (high value)

1. Decision precedence + multi-trigger handling

Right now it short-circuits on first BLOCK / REQUIRE_HUMAN. That’s fine for safety, but for audits you often want:


• the first hard stop and

• the full rule path (what else would have triggered)


2. Split outcomes: BLOCK vs REQUIRE_HUMAN vs ESCALATE

You defined ESCALATE but don’t route it. Common pattern:


• BLOCK = stop now

• REQUIRE_HUMAN = pause for explicit authorization

• ESCALATE = route to clinician/supervisor / higher authority tier (still a stop)


3. Context integrity checks up front

Add a validate_context() that blocks if required fields are missing or confidence is below minimum.

4. Structured audit log schema (JSON), not ad-hoc strings

Logs should be machine-parseable and consistent: trace_id, rule_id, triggered, decision, notes, timestamps, and minimal PHI/PII.

# governance/logging.py

import json, time, uuid

from dataclasses import asdict

from typing import Callable, Optional

from governance.spine import RuleResult, SystemContext, Decision


def new_trace_id() -> str:

    return uuid.uuid4().hex


def audit_event(

    trace_id: str,

    stage: str,

    ctx: SystemContext,

    rule: Optional[RuleResult] = None,

) -> str:

    payload = {

        "ts": time.time(),

        "trace_id": trace_id,

        "stage": stage,  # e.g., "CTX_VALIDATION", "RULE_EVAL", "FINAL"

        "child_id": ctx.child_id,

        "caregiver_id": ctx.caregiver_id,

        "location": ctx.location,

        "emotional_state": ctx.emotional_state.value,

        "emotional_confidence": ctx.emotional_confidence,

        "override_requested": ctx.is_override_requested,

    }

    if rule is not None:

        payload.update({

            "rule_id": rule.rule_id,

            "triggered": rule.triggered,

            "decision": (rule.decision.name if rule.decision else None),

            "notes": rule.notes,

        })

    return json.dumps(payload, separators=(",", ":"), ensure_ascii=False)


def rule_logged(rule_id: str):

    """

    Decorator: wraps a rule fn and emits an audit log line each time it runs.

    """

    def deco(fn: Callable[[SystemContext], RuleResult]):

        def wrapped(ctx: SystemContext, trace_id: str, write_log: Callable[[str], None]) -> RuleResult:

            res = fn(ctx)

            # enforce rule_id consistency

            if res.rule_id != rule_id:

                res = RuleResult(rule_id=rule_id, triggered=True, decision=Decision.BLOCK, notes="Rule ID mismatch")

            write_log(audit_event(trace_id, "RULE_EVAL", ctx, res))

            return res

        return wrapped

    return deco

# governance/spine_runtime.py

from typing import List, Callable

from governance.spine import SystemContext, RuleResult, GovernanceDecision, Decision

from governance.logging import new_trace_id, audit_event


def validate_context(ctx: SystemContext) -> RuleResult:

    if not ctx.child_id or not ctx.caregiver_id:

        return RuleResult("CTX", True, Decision.BLOCK, "Missing child_id or caregiver_id")

    if ctx.emotional_confidence < 0.50:

        return RuleResult("CTX", True, Decision.REQUIRE_HUMAN, "Low emotional_confidence")

    return RuleResult("CTX", False)


def apply_governance_spine(ctx: SystemContext, rules: List[Callable], write_log: Callable[[str], None]) -> GovernanceDecision:

    trace_id = new_trace_id()

    write_log(audit_event(trace_id, "CTX_RECEIVED", ctx))


    path: List[RuleResult] = []


    ctx_res = validate_context(ctx)

    path.append(ctx_res)

    write_log(audit_event(trace_id, "CTX_VALIDATION", ctx, ctx_res))

    if ctx_res.triggered and ctx_res.decision in (Decision.BLOCK, Decision.REQUIRE_HUMAN, Decision.ESCALATE):

        write_log(audit_event(trace_id, "FINAL", ctx, ctx_res))

        return GovernanceDecision(ctx_res.decision, path)


    # Evaluate all rules, but preserve first terminal decision for action.

    terminal: RuleResult | None = None

    for rule_fn in rules:

        res = rule_fn(ctx, trace_id=trace_id, write_log=write_log)

        path.append(res)

        if (terminal is None) and res.triggered and res.decision in (Decision.BLOCK, Decision.REQUIRE_HUMAN, Decision.ESCALATE):

            terminal = res


    final = terminal.decision if terminal else Decision.ALLOW

    write_log(audit_event(trace_id, "FINAL", ctx, RuleResult("FINAL", True, final, "Spine resolution")))

    return GovernanceDecision(final, path)

• deterministic runtime enforcement

• full trace even when a terminal rule triggers early

• a stable audit record that funders/clinicians can review later
