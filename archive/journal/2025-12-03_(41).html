from dataclasses import dataclass

from typing import Callable, Dict, Any, Tuple, Optional, List


# ---------- Types ----------

State = Dict[str, Any]

ModuleFn = Callable[[State], Tuple[bool, State, Optional[str]]]

# ---------- HER Kernel v0.1 (S1–S6 wired) ----------

class HERKernel:

    def __init__(self, llm: LLMFn):

        self.llm = llm


        # Ordered chains – later you can make this rule-driven using Rule objects

        self.pre_input_chain: List[str] = ["S1", "S2", "S3", "S4"]

        self.pre_llm_chain: List[str]   = []                 # for future prompt builders

        self.post_llm_chain: List[str]  = ["S5", "S6"]


    def _run_chain(self, chain: List[str], state: State, stage: str) -> State:

        """Run a sequence of S-modules; record any violations; set blocked if needed."""

        for mid in chain:

            ok, state, err = SPINE[mid](state)

            if not ok:

                state.setdefault("violations", []).append(

                    {"module": mid, "stage": stage, "reason": err}

                )

                state["blocked"] = True

                break

        return state


    def run_once(self, user_input: str, session_state: Optional[State] = None) -> Dict[str, Any]:

        # Start with prior session state or empty

        state: State = session_state.copy() if session_state else {}

        state["user_input"] = user_input

        state.setdefault("violations", [])

        state.setdefault("blocked", False)


        # 1. Pre-input chain (S1–S4)

        state = self._run_chain(self.pre_input_chain, state, stage="pre_input")

        if state.get("blocked"):

            # No LLM call

            state["final_output"] = "(response moderated before LLM call)"

            return {

                "output": state["final_output"],

                "blocked": True,

                "violations": state["violations"],

                "raw_llm_output": None,

                "state": state,

            }


        # 2. Pre-LLM chain (currently empty, placeholder)

        state = self._run_chain(self.pre_llm_chain, state, stage="pre_llm")

        if state.get("blocked"):

            state["final_output"] = "(response moderated before LLM call)"

            return {

                "output": state["final_output"],

                "blocked": True,

                "violations": state["violations"],

                "raw_llm_output": None,

                "state": state,

            }


        # 3. Call the LLM

        raw = self.llm(user_input, state)

        state["raw_llm_output"] = raw


        # 4. Post-LLM chain (S5–S6)

        state = self._run_chain(self.post_llm_chain, state, stage="post_llm")

        if state.get("blocked"):

            state["final_output"] = "(response moderated after LLM call)"

            return {

                "output": state["final_output"],

                "blocked": True,

                "violations": state["violations"],

                "raw_llm_output": None,  # hide raw output when blocked

                "state": state,

            }


        # 5. Success path

        state["final_output"] = state.get("raw_llm_output", "")

        return {

            "output": state["final_output"],

            "blocked": False,

            "violations": state["violations"],

            "raw_llm_output": state.get("raw_llm_output"),

            "state": state,

        }
