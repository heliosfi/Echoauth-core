1. Core Data Structures


# ─────────────────────────────

# Core Types

class CandidateOutput:

    def __init__(self, thread_id, text, source, intent_anchor, confidence):

        self.thread_id = thread_id          # "t1"

        self.text = text                    # raw candidate text

        self.source = source                # "engine_name"

        self.intent_anchor = intent_anchor  # "support" | "guidance" | "neutral"

        self.confidence = confidence        # 0.0 - 1.0



class ORIProfile:

    def __init__(self, identity_rules, purpose, forbidden_patterns=None):

        self.identity_rules = identity_rules        # e.g. ["calm", "structured", "non-harmful"]

        self.purpose = purpose                      # e.g. "assist_caregiver_asd_support"

        self.forbidden_patterns = forbidden_patterns or []  # e.g. ["scripture_quote", "monotheistic_dogma"]



class ToneVector:

    def __init__(self, calm, structured, playful):

        self.calm = calm            # 0.0 - 1.0

        self.structured = structured

        self.playful = playful

2. SRCE — Coherence Engine


# ─────────────────────────────

# S21: Coherence Engine (SRCE)

# ─────────────────────────────


class CoherenceEngine:

    def compute_conflict(self, candidates: list[CandidateOutput]) -> dict:

        """

        Returns:

            {

              "conflict_score": float,

              "conflict_pairs": list[tuple[thread_id_a, thread_id_b, reason_str]]

            }

        """

        if len(candidates) <= 1:

            return {"conflict_score": 0.0, "conflict_pairs": []}


        conflict_pairs = []


        # Pairwise comparison (naive version, can be upgraded later)

        for i in range(len(candidates)):

            for j in range(i + 1, len(candidates)):

                a = candidates[i]

                b = candidates[j]


                reason = self._detect_contradiction(a.text, b.text)

                if reason is not None:

                    conflict_pairs.append((a.thread_id, b.thread_id, reason))


        conflict_score = self._aggregate_conflict_score(conflict_pairs)


        return {

            "conflict_score": conflict_score,

            "conflict_pairs": conflict_pairs

        }


    def _detect_contradiction(self, text_a: str, text_b: str) -> str | None:

        """

        Very simple placeholder logic:

        - In a real system, this is where an LLM classifier or ruleset would be called.

        """

        # Example heuristic: direct negation detection placeholder

        # (Replace this with semantic contradiction detection)

        if "not " in text_a and text_a.replace("not ", "") in text_b:

            return "Direct negation (A denies B)"

        if "not " in text_b and text_b.replace("not ", "") in text_a:

            return "Direct negation (B denies A)"


        # More advanced checks go here...

        return None


    def _aggregate_conflict_score(self, conflict_pairs) -> float:

        """

        Simple scoring:

        - No conflicts -> 0.0

        - More conflicts / more severe -> closer to 1.0

        """

        if not conflict_pairs:

            return 0.0


        # Naive: 0.3 per conflict capped at 1.0

        base = 0.3 * len(conflict_pairs)

        return min(1.0, base)

3. MEL — Execution Resolver


# ─────────────────────────────

# S22: Execution Resolver (MEL)

# ─────────────────────────────


class ExecutionResolver:

    def __init__(self, ori_profile: ORIProfile, mode: str):

        self.ori_profile = ori_profile   # identity + purpose + forbidden patterns

        self.mode = mode                 # "support" | "guidance" | "neutral"


    def select_or_rerun(self,

                         candidates: list[CandidateOutput],

                         conflict_score: float,

                         conflict_pairs: list) -> CandidateOutput:

        """

        Decide whether:

        - choose one existing candidate, or

        - trigger a re-run under stricter ORI constraints.

        """


        if not candidates:

            # In real code: trigger a fresh generation.

            return self._rerun_under_ori(None)


        if conflict_score == 0.0:

            # No conflict -> pick the best aligned candidate

            return self._select_best_aligned(candidates)


        # There is some conflict. Try to resolve by alignment rules:

        aligned = self._rank_by_priority(candidates)

        if aligned:

            return aligned[0]   # top-ranked candidate


        # If still not safe, re-run with ORI + Intent as strict constraints.

        return self._rerun_under_ori(candidates)


    def _select_best_aligned(self, candidates: list[CandidateOutput]) -> CandidateOutput:

        ranked = self._rank_by_priority(candidates)

        # fallback: highest confidence

        return ranked[0] if ranked else max(candidates, key=lambda c: c.confidence)


    def _rank_by_priority(self, candidates: list[CandidateOutput]) -> list[CandidateOutput]:

        """

        Ranking rules:

        1. ORI identity/purpose alignment (no forbidden patterns)

        2. Intent Anchor = requested mode

        3. Confidence score

        """


        def score_candidate(c: CandidateOutput) -> float:

            score = 0.0


            # 1. Identity / purpose alignment

            if self._respects_forbidden_patterns(c.text):

                score += 0.5


            # 2. Mode / intent alignment

            if c.intent_anchor == self.mode:

                score += 0.3


            # 3. Confidence

            score += 0.2 * c.confidence


            return score


        safe_candidates = [c for c in candidates if self._respects_forbidden_patterns(c.text)]

        if not safe_candidates:

            return []


        ranked = sorted(safe_candidates, key=score_candidate, reverse=True)

        return ranked


    def _respects_forbidden_patterns(self, text: str) -> bool:

        for pattern in self.ori_profile.forbidden_patterns:

            if pattern in text.lower():

                return False

        return True


    def _rerun_under_ori(self, previous_candidates: list[CandidateOutput] | None) -> CandidateOutput:

        """

        Placeholder: in real system this would call the LLM with strict ORI + intent prompt.

        Here we just return a dummy "safe fallback".

        """


        fallback_text = (

            "I’m consolidating the information to give you one clear, "

            "non-contradictory answer that respects your preferences and safety."

        )

        return CandidateOutput(

            thread_id="rerun_under_ori",

            text=fallback_text,

            source="rerun_engine",

            intent_anchor=self.mode,

            confidence=0.9

        )

4. PNRE — Output Choke-Point


# ─────────────────────────────

# Claim 3: PNRE Output Gate

# ─────────────────────────────


class PNREGate:

    def __init__(self, logger):

        self.logger = logger  # any logging interface


    def enforce_single_output(self,

                              selected: CandidateOutput,

                              all_candidates: list[CandidateOutput],

                              conflict_score: float,

                              conflict_pairs: list) -> CandidateOutput:

        """

        Logs everything, but *only* the selected candidate exits.

        """

        self._log_decision(selected, all_candidates, conflict_score, conflict_pairs)

        return selected


    def _log_decision(self,

                      selected: CandidateOutput,

                      all_candidates: list[CandidateOutput],

                      conflict_score: float,

                      conflict_pairs: list):

        log_entry = {

            "selected_thread_id": selected.thread_id,

            "selected_source": selected.source,

            "candidates_count": len(all_candidates),

            "conflict_score": conflict_score,

            "conflict_pairs": conflict_pairs,

        }

        self.logger.log("coherence_gate_decision", log_entry)

5. ESC — Tone & Identity Enforcement


# ─────────────────────────────

# S15: Tone & Identity Enforcer (ESC)

# ─────────────────────────────


class ToneEnforcer:

    def __init__(self, ori_profile: ORIProfile, mode: str):

        self.ori_profile = ori_profile

        self.mode = mode


    def apply(self, text: str) -> tuple[str, ToneVector]:

        """

        Adjusts or wraps the text to keep it calm, structured, and aligned with NI-AI persona.

        """


        # Example: ensure calm + structured framing for ASD/caregiver context

        adjusted_text = self._wrap_for_clarity(text)


        # Tone vector can be tuned by mode:

        if self.mode == "support":

            tone = ToneVector(calm=0.95, structured=0.9, playful=0.2)

        elif self.mode == "guidance":

            tone = ToneVector(calm=0.9, structured=0.95, playful=0.1)

        else:  # neutral

            tone = ToneVector(calm=0.85, structured=0.85, playful=0.05)


        return adjusted_text, tone


    def _wrap_for_clarity(self, text: str) -> str:

        """

        Simple example wrapper for ASD-friendly clarity:

        - Short sentences

        - One main idea at a time

        - No overwhelming language

        """

        prefix = ""

        if self.ori_profile.purpose == "assist_caregiver_asd_support":

            prefix = "Here is one clear answer, step by step:\n\n"


        return prefix + text

6. /coherence_gate Orchestration Function


# ─────────────────────────────

# /coherence_gate – Main API

# ─────────────────────────────


class CoherenceGateService:

    def __init__(self, logger):

        self.logger = logger


    def coherence_gate(self,

                       raw_candidates_payload: dict,

                       ori_payload: dict,

                       mode: str) -> dict:

        """

        High-level orchestration equivalent to POST /coherence_gate

        """


        # 1. Parse inputs

        candidates = [

            CandidateOutput(

                thread_id=item["thread_id"],

                text=item["text"],

                source=item["metadata"]["source"],

                intent_anchor=item["metadata"]["intent_anchor"],

                confidence=item["metadata"]["confidence"],

            )

            for item in raw_candidates_payload["candidates"]

        ]


        ori_profile = ORIProfile(

            identity_rules=ori_payload.get("identity_rules", []),

            purpose=ori_payload.get("purpose", ""),

            forbidden_patterns=ori_payload.get("forbidden_patterns", []),

        )


        # 2. Coherence / conflict detection

        srce = CoherenceEngine()

        conflict_report = srce.compute_conflict(candidates)

        conflict_score = conflict_report["conflict_score"]

        conflict_pairs = conflict_report["conflict_pairs"]


        # 3. Resolve execution path

        resolver = ExecutionResolver(ori_profile=ori_profile, mode=mode)

        selected_candidate = resolver.select_or_rerun(

            candidates=candidates,

            conflict_score=conflict_score,

            conflict_pairs=conflict_pairs

        )


        # 4. Enforce single output through PNRE gate

        pnre = PNREGate(logger=self.logger)

        chosen = pnre.enforce_single_output(

            selected=selected_candidate,

            all_candidates=candidates,

            conflict_score=conflict_score,

            conflict_pairs=conflict_pairs

        )


        # 5. Tone + identity enforcement

        esc = ToneEnforcer(ori_profile=ori_profile, mode=mode)

        final_text, tone_vector = esc.apply(chosen.text)


        # 6. Build response payload (what the UI / user sees)

        response = {

            "selected_thread_id": chosen.thread_id,

            "final_text": final_text,

            "tone_vector": {

                "calm": tone_vector.calm,

                "structured": tone_vector.structured,

                "playful": tone_vector.playful

            },

            "conflict_score": conflict_score,

            "logs_ref": "coherence_log_id_stub"  # in real system, from logger

        }


        return response
