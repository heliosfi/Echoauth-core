# scanner.py — mock token scanner / event producer (devnet-safe)

import time

import logging

import threading

import random

from queue import Queue, Empty

from dataclasses import dataclass


logger = logging.getLogger("scanner")

logger.setLevel(logging.INFO)

ch = logging.StreamHandler()

ch.setFormatter(logging.Formatter("%(asctime)s - %(name)s - %(levelname)s - %(message)s"))

logger.addHandler(ch)


@dataclass

class Candidate:

    mint: str

    name: str

    timestamp: float

    metadata: dict


class Scanner(threading.Thread):

    """

    Mock scanner that emits Candidate objects into a Queue.

    Replace the ._poll_real_network method for real RPC scanning (devnet only).

    """

    def __init__(self, out_q: Queue, poll_interval: float = 1.0, seed: int = 42):

        super().__init__(daemon=True)

        self.out_q = out_q

        self.poll_interval = poll_interval

        self._running = True

        random.seed(seed)


    def run(self):

        logger.info("Scanner started (mock mode).")

        counter = 0

        while self._running:

            try:

                # Mock new token event

                mint = f"MOCK{int(time.time())%100000}-{counter}"

                name = f"MockCoin{counter%100}"

                metadata = {

                    "liquidity_usd": random.uniform(5_000, 250_000),

                    "holders": random.randint(3, 5000),

                    "creator_score": random.random()

                }

                cand = Candidate(mint=mint, name=name, timestamp=time.time(), metadata=metadata)

                logger.info(f"Scanned candidate: {cand.mint} liquidity={metadata['liquidity_usd']:.0f} holders={metadata['holders']}")

                self.out_q.put(cand)

                counter += 1

                time.sleep(self.poll_interval)

            except Exception as e:

                logger.exception("Scanner error: %s", e)

                time.sleep(self.poll_interval)


    def stop(self):

        self._running = False

# features.py — compute simple, interpretable features for each candidate

import math

import logging

from typing import Dict


logger = logging.getLogger("features")

logger.setLevel(logging.INFO)


def compute_features(candidate) -> Dict[str, float]:

    """

    Input: Candidate object with metadata

    Output: dict of numeric features

    """

    m = candidate.metadata

    liquidity = float(m.get("liquidity_usd", 0.0))

    holders = float(m.get("holders", 0.0))

    creator = float(m.get("creator_score", 0.0))


    # Hand-crafted features (simple, transparent)

    f_liq = math.log1p(liquidity)

    f_holders = math.log1p(holders)

    f_concentration = max(0.0, 1.0 - (holders / (holders + 100.0)))  # surrogate

    f_creator = creator


    features = {

        "f_liq": f_liq,

        "f_holders": f_holders,

        "f_concentration": f_concentration,

        "f_creator": f_creator

    }

    logger.debug("Features for %s: %s", candidate.mint, features)

    return features

# policy.py — family-defined risk/policy rules (FLAIMO-compatible)

from dataclasses import dataclass

import logging


logger = logging.getLogger("policy")

logger.setLevel(logging.INFO)


@dataclass

class Policy:

    name: str

    min_liquidity_usd: float = 10_000.0

    min_holders: int = 10

    max_concentration: float = 0.75

    min_score: float = 0.5  # min model score to consider

    max_trade_pct: float = 0.01  # fraction of portfolio per trade


    def passes(self, candidate, features: dict, model_score: float) -> (bool, str):

        m = candidate.metadata

        liquidity = m.get("liquidity_usd", 0.0)

        holders = m.get("holders", 0)

        conc = features.get("f_concentration", 1.0)

        if liquidity < self.min_liquidity_usd:

            return False, f"liquidity {liquidity:.0f} < min {self.min_liquidity_usd:.0f}"

        if holders < self.min_holders:

            return False, f"holders {holders} < min {self.min_holders}"

        if conc > self.max_concentration:

            return False, f"concentration {conc:.2f} > max {self.max_concentration:.2f}"

        if model_score < self.min_score:

            return False, f"score {model_score:.3f} < min_score {self.min_score:.3f}"

        return True, "ok"

# model_stub.py — tiny heuristic "model" (replace with trained PyTorch later)

import math

import logging


logger = logging.getLogger("model_stub")

logger.setLevel(logging.INFO)


def score_from_features(features: dict) -> float:

    """

    Deterministic heuristic scoring in [0,1].

    Replaceable with a trained model (model_train.py).

    """

    # simple weighted sum + sigmoid

    w = {"f_liq": 0.35, "f_holders": 0.35, "f_concentration": -0.2, "f_creator": 0.5}

    s = 0.0

    for k,v in features.items():

        s += w.get(k, 0.0) * v

    # sigmoid

    score = 1 / (1 + math.exp(-0.08*(s - 5.0)))

    score = max(0.0, min(1.0, score))

    logger.debug("Model score: %.4f (s=%.3f)", score, s)

    return score

# simulator.py — orchestrates scanner -> features -> model -> policy -> executor

import logging

import time

from queue import Queue, Empty

from scanner import Scanner, Candidate

from features import compute_features

from model_stub import score_from_features

from policy import Policy

from executor_stub import Executor

from threading import Thread


logger = logging.getLogger("simulator")

logger.setLevel(logging.INFO)

ch = logging.StreamHandler()

ch.setFormatter(logging.Formatter("%(asctime)s - %(name)s - %(levelname)s - %(message)s"))

logger.addHandler(ch)


class Pipeline:

    def __init__(self, policy: Policy = None):

        self.q = Queue()

        self.scanner = Scanner(self.q, poll_interval=1.0)

        self.policy = policy or Policy(name="conservative")

        self.executor = Executor()

        self._running = False


    def start(self):

        logger.info("Starting pipeline with policy %s", self.policy)

        self._running = True

        self.scanner.start()

        self.worker_thread = Thread(target=self._worker, daemon=True)

        self.worker_thread.start()


    def stop(self):

        logger.info("Stopping pipeline")

        self.scanner.stop()

        self._running = False


    def _worker(self):

        while self._running:

            try:

                cand = self.q.get(timeout=1.0)

            except Empty:

                continue

            try:

                logger.info("Processing %s", cand.mint)

                feats = compute_features(cand)

                score = score_from_features(feats)

                ok, reason = self.policy.passes(cand, feats, score)

                if not ok:

                    logger.info("Policy rejected %s: %s (score=%.3f)", cand.mint, reason, score)

                    continue

                # Determine allocation (conservative pct of imaginary portfolio)

                portfolio_usd = 1000.0  # simulation placeholder

                amount = portfolio_usd * self.policy.max_trade_pct

                receipt = self.executor.execute_buy(cand, amount_usd=amount, slippage=0.03)

                logger.info("Executed: %s", receipt)

                # simple post-trade logic placeholder

                time.sleep(0.25)

            except Exception as e:

                logger.exception("Pipeline processing error: %s", e)


if __name__ == "__main__":

    p = Pipeline()

    try:

        p.start()

        # run for N seconds

        runtime = 20

        logger.info("Pipeline running for %d seconds (mock)", runtime)

        time.sleep(runtime)

    finally:

        p.stop()

        logger.info("Pipeline finished.")

# SniperBot — Prototype (Mock / Devnet-Safe)


**Purpose:** Starter prototype showing the end-to-end pipeline for scanning new token events, computing features, scoring via a heuristic model, applying family policies (FLAIMO), and simulated execution (no real funds).


> **IMPORTANT:** These files are simulation-only. **Do not run any real mainnet trades** with these stubs. Replace executor logic only after proper audits, secure key management, devnet testing, and compliance review.


## Files in this folder

- `scanner.py` — mock scanner that emits Candidate events

- `features.py` — computes simple numeric features

- `policy.py` — family policy object (conservative/balanced/growth)

- `model_stub.py` — deterministic heuristic scoring

- `executor_stub.py` — simulated executor (NO MAINNET)

- `simulator.py` — orchestrator that runs the pipeline
