from dataclasses import dataclass

from typing import Callable, Dict, Any, Tuple, Optional, List


# ---------- Types ----------

State = Dict[str, Any]

ModuleFn = Callable[[State], Tuple[bool, State, Optional[str]]]

LLMFn = Callable[[str, State], str]



@dataclass

class Rule:

    name: str

    stage: str          # "pre_input", "pre_llm", "post_llm"

    module_id: str      # "S1"..."S6"



# ---------- Spine Modules S1–S6 ----------

def S1_identity_reanchor(state: State) -> Tuple[bool, State, Optional[str]]:

    # Ensure identity / persona context is loaded

    state.setdefault("identity_loaded", True)

    return True, state, None



def S2_input_sanity_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    text = state.get("user_input", "")

    if len(text) > 5000:

        return False, state, "Input too long"

    return True, state, None



def S3_input_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    text = state.get("user_input", "").lower()

    if "forbidden" in text:

        return False, state, "Disallowed topic in input"

    return True, state, None



def S4_context_coherence_check(state: State) -> Tuple[bool, State, Optional[str]]:

    # Placeholder for real session/context checks

    state.setdefault("context_ok", True)

    return True, state, None



def S5_output_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    output = state.get("raw_llm_output", "").lower()

    if "forbidden" in output:

        return False, state, "Disallowed content in output"

    return True, state, None



def S6_child_safety_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    # This will later call real EchoAuth chamber token verification

    if not state.get("echoauth_ok", True):

        return False, state, "EchoAuth chamber not in safe state"

    return True, state, None



SPINE: Dict[str, ModuleFn] = {

    "S1": S1_identity_reanchor,

    "S2": S2_input_sanity_guard,

    "S3": S3_input_policy_filter,

    "S4": S4_context_coherence_check,

    "S5": S5_output_policy_filter,

    "S6": S6_child_safety_guard,

}



# ---------- HER Kernel v0.1 (S1–S6 wired) ----------

class HERKernel:

    def __init__(self, llm: LLMFn):

        self.llm = llm


        # Ordered chains – later you can make this rule-driven using Rule objects

        self.pre_input_chain: List[str] = ["S1", "S2", "S3", "S4"]

        self.pre_llm_chain: List[str]   = []                 # for future prompt builders

        self.post_llm_chain: List[str]  = ["S5", "S6"]


    def _run_chain(self, chain: List[str], state: State, stage: str) -> State:

        """Run a sequence of S-modules; record any violations; set blocked if needed."""

        for mid in chain:

            ok, state, err = SPINE[mid](state)

            if not ok:

                state.setdefault("violations", []).append(

                    {"module": mid, "stage": stage, "reason": err}

                )

                state["blocked"] = True

                break

        return state


    def run_once(self, user_input: str, session_state: Optional[State] = None) -> Dict[str, Any]:

        # Start with prior session state or empty

        state: State = session_state.copy() if session_state else {}

        state["user_input"] = user_input

        state.setdefault("violations", [])

        state.setdefault("blocked", False)


        # 1. Pre-input chain (S1–S4)

        state = self._run_chain(self.pre_input_chain, state, stage="pre_input")

        if state.get("blocked"):

            # No LLM call

            state["final_output"] = "(response moderated before LLM call)"

            return {

                "output": state["final_output"],

                "blocked": True,

                "violations": state["violations"],

                "raw_llm_output": None,

                "state": state,

            }


        # 2. Pre-LLM chain (currently empty, placeholder)

        state = self._run_chain(self.pre_llm_chain, state, stage="pre_llm")

        if state.get("blocked"):

            state["final_output"] = "(response moderated before LLM call)"

            return {

                "output": state["final_output"],

                "blocked": True,

                "violations": state["violations"],

                "raw_llm_output": None,

                "state": state,

            }


        # 3. Call the LLM

        raw = self.llm(user_input, state)

        state["raw_llm_output"] = raw


        # 4. Post-LLM chain (S5–S6)

        state = self._run_chain(self.post_llm_chain, state, stage="post_llm")

        if state.get("blocked"):

            state["final_output"] = "(response moderated after LLM call)"

            return {

                "output": state["final_output"],

                "blocked": True,

                "violations": state["violations"],

                "raw_llm_output": None,  # hide raw output when blocked

                "state": state,

            }


        # 5. Success path

        state["final_output"] = state.get("raw_llm_output", "")

        return {

            "output": state["final_output"],

            "blocked": False,

            "violations": state["violations"],

            "raw_llm_output": state.get("raw_llm_output"),

            "state": state,

        }
