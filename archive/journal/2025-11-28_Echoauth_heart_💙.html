Let’s give EchoAuth a local heartbeat — a single Python script that:

• boots the DB,

• simulates one caregiver + child,

• runs a high-distress cue packet through the NI-AI engine,

• logs it like the API would,

• and prints the final result + whether it’s flagged for review.


No web server required for this demo (you can still run FastAPI separately).

echoauth_mvp/

The script below goes in run_local_demo.py.

🧠 NI-AI demo launcher: run_local_demo.py


"""

EchoAuth™ Local MVP Demo Launcher


This script:

1. Initializes the local DB.

2. Creates one caregiver and one child (if they don't already exist).

3. Simulates a high-distress check-in using NI-AI evaluate_cues().

4. Stores CheckIn + NIAILog like the API would.

5. Prints the final result, emotion_score, and whether it was flagged for review.

"""


from datetime import datetime

import json


from db.session import SessionLocal, engine

from db import models

from core.ni_ai import evaluate_cues



def init_db():

    """Create tables if they do not exist."""

    models.Base.metadata.create_all(bind=engine)



def get_or_create_caregiver(db):

    """Return a demo caregiver, creating one if needed."""

    email = "demo_caregiver@echoauth.local"

    caregiver = db.query(models.Caregiver).filter(

        models.Caregiver.email == email

    ).first()


    if caregiver:

        return caregiver


    # Minimal password handling just for local demo

    caregiver = models.Caregiver(

        name="Demo Caregiver",

        email=email,

        hashed_password="demo",  # for demo only; real system uses hash

        created_at=datetime.utcnow(),

    )

    db.add(caregiver)

    db.commit()

    db.refresh(caregiver)

    return caregiver



def get_or_create_child(db, caregiver):

    """Return a demo child, creating one if needed."""

    child = (

        db.query(models.Child)

        .filter(

            models.Child.caregiver_id == caregiver.id,

            models.Child.name == "Demo Child",

        )

        .first()

    )


    if child:

        return child


    child = models.Child(

        caregiver_id=caregiver.id,

        name="Demo Child",

        age_range="6-8",

        notes="Created by local MVP demo script.",

        created_at=datetime.utcnow(),

    )

    db.add(child)

    db.commit()

    db.refresh(child)

    return child



def simulate_checkin(db, caregiver, child):

    """

    Run a single high-distress check-in through NI-AI and store results.


    This cue packet is intentionally tuned to trigger:

    - elevated emotion_score

    - 'Needs Attention' label

    - auto-flagged review

    """

    cue_packet = {

        "behavior": "head banging on wall, pacing, covering ears",

        "environment": "very loud crowded room with bright lights",

        "caregiver_observation": "looks overwhelmed and close to meltdown",

    }


    # 1. NI-AI evaluation

    eval_result = evaluate_cues(cue_packet)


    final_label = eval_result["final_label"]

    confidence = eval_result["confidence"]

    emotion_score = eval_result["emotion_score"]

    conflict_score = eval_result["conflict_score"]

    danger_score = eval_result.get("danger_score", 0.0)

    winning_thread = eval_result["winning_thread"]

    internal_threads = eval_result["internal_threads"]


    # 2. Apply the same auto-flag logic as the API

    needs_review = False

    review_status = "none"


    if (

        conflict_score > 0.6

        or emotion_score > 0.7

        or (danger_score > 0.0 and confidence < 0.85)

    ):

        needs_review = True

        review_status = "pending"


    # 3. Store CheckIn

    checkin = models.CheckIn(

        caregiver_id=caregiver.id,

        child_id=child.id,

        observed_cues=json.dumps(cue_packet),

        result_label=final_label,

        confidence=confidence,

        status="success",

        emotion_score=emotion_score,

        needs_review=needs_review,

        review_status=review_status,

        created_at=datetime.utcnow(),

    )

    db.add(checkin)

    db.commit()

    db.refresh(checkin)


    # 4. Store NI-AI private log

    log_entry = models.NIAILog(

        checkin_id=checkin.id,

        winning_thread=winning_thread,

        conflict_score=conflict_score,

        internal_notes=json.dumps(internal_threads),

        created_at=datetime.utcnow(),

    )

    db.add(log_entry)

    db.commit()


    # 5. Print result summary to terminal (no UI, no boxes)

    print("──────────────── EchoAuth Local MVP Demo ────────────────")

    print(f"Caregiver: {caregiver.name} (id={caregiver.id})")

    print(f"Child:     {child.name} (id={child.id})")

    print("Cue packet:")

    print(json.dumps(cue_packet, indent=2))

    print("──────────────── NI-AI Evaluation ──────────────────────")

    print(f"Final label:     {final_label}")

    print(f"Confidence:      {confidence:.2f}")

    print(f"Emotion score:   {emotion_score:.2f}")

    print(f"Danger score:    {danger_score:.2f}")

    print(f"Conflict score:  {conflict_score:.2f}")

    print(f"Winning thread:  {winning_thread}")

    print("──────────────── Review Status ─────────────────────────")

    print(f"Check-in ID:     {checkin.id}")

    print(f"Needs review:    {needs_review}")

    print(f"Review status:   {review_status}")

    print("──────────────── End of Demo ──────────────────────────")



def main():

    init_db()

    db = SessionLocal()

    try:

        caregiver = get_or_create_caregiver(db)

        child = get_or_create_child(db, caregiver)

        simulate_checkin(db, caregiver, child)

    finally:

        db.close()



if __name__ == "__main__":

    main()

🧪 How to use this

1. Make sure you have the dependencies installed:


pip install sqlalchemy fastapi uvicorn pydantic


2. Ensure db/session.py, db/models.py, and core/ni_ai.py are present and match the fields we wired (including emotion_score, needs_review, review_status, etc.)

3. From your project root (echoauth_mvp/), run:


python run_local_demo.py


You should see terminal output like:

• the cue packet,

• final label (Needs Attention in this high-distress example),

• confidence, emotion_score, danger_score, conflict_score,

• and Needs review: True, Review status: pending.


That’s your local, code-level proof-of-use:

the EchoAuth cortex evaluates, collapses, logs, flags, and prints.
