1. High Level MVP Architecture


Goal:

Small, clean backend that lets a caregiver:

• Register themself and their child

• Send a set of non-verbal cues for the child

• Get back one clear authentication result

• Log conflicts internally without exposing them


Components

• api/

• main.py – FastAPI app, routes

• routes_caregiver.py – caregiver + child endpoints

• routes_auth.py – authentication endpoint

• core/

• ni_ai.py – NI-AI interfaces: TLE, CoherenceEngine, PNRE, LRM

• logger.py – internal audit logging

• db/

• models.py – SQLAlchemy models

• session.py – DB session helper

3. NI-AI MVP Interfaces


We are not coding the full brain.

We are defining clear hooks that match your patent logic.


# core/ni_ai.py


from typing import List, Dict, Any

from dataclasses import dataclass


@dataclass

class NIContext:

    caregiver_id: int

    child_id: int

    cues: Dict[str, Any]  # e.g. {"gesture": "...", "sound": "..."}

    intent: str = "authenticate_child"


@dataclass

class NIResult:

    status: str            # "success" or "failure"

    result_label: str      # e.g. "authenticated"

    confidence: float      # 0.0 - 1.0

    winning_thread_id: str # internal label

    conflict_score: float  # 0.0 if no conflict

    notes: str = ""        # internal notes


class ThreadLinkingEngine:

    def build_threads(self, ctx: NIContext) -> List[Dict[str, Any]]:

        # MVP: just one thread; later can expand.

        return [{"id": "thread_1", "score": 0.9, "status": "candidate"}]


class CoherenceEngine:

    def evaluate_conflicts(self, threads: List[Dict[str, Any]]) -> float:

        # MVP: no conflict. Later: compare values.

        return 0.0


class LoadRegulationMonitor:

    def evaluate_load(self, ctx: NIContext) -> float:

        # MVP: fixed low load. Later: use cue complexity.

        return 0.2


class OutputCollapseEnforcer:

    def select_winner(self, threads: List[Dict[str, Any]]) -> Dict[str, Any]:

        # MVP: pick max score

        return max(threads, key=lambda t: t.get("score", 0.0))


class NIEngine:

    def __init__(self):

        self.tle = ThreadLinkingEngine()

        self.coherence = CoherenceEngine()

        self.lrm = LoadRegulationMonitor()

        self.pnre = OutputCollapseEnforcer()


    def authenticate(self, ctx: NIContext) -> NIResult:

        threads = self.tle.build_threads(ctx)

        conflict_score = self.coherence.evaluate_conflicts(threads)

        load_score = self.lrm.evaluate_load(ctx)

        winner = self.pnre.select_winner(threads)


        # MVP logic: if winner score and load are acceptable, success

        base_score = winner.get("score", 0.0)

        if base_score >= 0.7 and load_score <= 0.7:

            return NIResult(

                status="success",

                result_label="authenticated",

                confidence=base_score,

                winning_thread_id=winner["id"],

                conflict_score=conflict_score,

                notes="MVP authentication path."

            )

        else:

            return NIResult(

                status="failure",

                result_label="not_authenticated",

                confidence=base_score,

                winning_thread_id=winner["id"],

                conflict_score=conflict_score,

                notes="MVP authentication failure path."

            )


This enforces:

• One output per call

• Conflict score computed (even if zero now)

• Load considered, even in a simple way

• Notes kept internal

# db/session.py

from sqlalchemy import create_engine

from sqlalchemy.orm import sessionmaker, declarative_base


DATABASE_URL = "sqlite:///./echoauth.db"  # MVP simple DB


engine = create_engine(

    DATABASE_URL,

    connect_args={"check_same_thread": False}

)


SessionLocal = sessionmaker(bind=engine, autoflush=False, autocommit=False)

Base = declarative_base()


db/models.py


# db/models.py

from sqlalchemy import Column, Integer, String, Float, ForeignKey, DateTime, Text

from sqlalchemy.orm import relationship

from datetime import datetime

from .session import Base


class Caregiver(Base):

    __tablename__ = "caregivers"


    id = Column(Integer, primary_key=True, index=True)

    name = Column(String, nullable=False)

    email = Column(String, unique=True, index=True, nullable=False)

    hashed_password = Column(String, nullable=False)

    created_at = Column(DateTime, default=datetime.utcnow)


    children = relationship("Child", back_populates="caregiver")


class Child(Base):

    __tablename__ = "children"


    id = Column(Integer, primary_key=True, index=True)

    caregiver_id = Column(Integer, ForeignKey("caregivers.id"), nullable=False)

    name = Column(String, nullable=False)

    profile_notes = Column(Text, nullable=True)

    created_at = Column(DateTime, default=datetime.utcnow)


    caregiver = relationship("Caregiver", back_populates="children")


class AuthSession(Base):

    __tablename__ = "auth_sessions"


    id = Column(Integer, primary_key=True, index=True)

    caregiver_id = Column(Integer, ForeignKey("caregivers.id"), nullable=False)

    child_id = Column(Integer, ForeignKey("children.id"), nullable=False)

    status = Column(String, nullable=False)          # "success" / "failure"

    result_label = Column(String, nullable=False)    # short phrase

    confidence = Column(Float, nullable=False)

    created_at = Column(DateTime, default=datetime.utcnow)


class NIAILog(Base):

    __tablename__ = "ni_ai_logs"


    id = Column(Integer, primary_key=True, index=True)

    auth_session_id = Column(Integer, ForeignKey("auth_sessions.id"), nullable=False)

    winning_thread = Column(String, nullable=False)

    conflict_score = Column(Float, nullable=False)

    internal_notes = Column(Text, nullable=True)

    created_at = Column(DateTime, default=datetime.utcnow)

main.py (FastAPI entry)


# api/main.py

from fastapi import FastAPI, Depends, HTTPException

from sqlalchemy.orm import Session

from db.session import SessionLocal, engine, Base

from db import models

from core.ni_ai import NIEngine, NIContext

from pydantic import BaseModel

from typing import Dict, Any


Base.metadata.create_all(bind=engine)


app = FastAPI(title="EchoAuth MVP")


ni_engine = NIEngine()


def get_db():

    db = SessionLocal()

    try:

        yield db

    finally:

        db.close()


# Pydantic schemas


class CaregiverCreate(BaseModel):

    name: str

    email: str

    password: str


class ChildCreate(BaseModel):

    caregiver_id: int

    name: str

    profile_notes: str | None = None


class AuthRequest(BaseModel):

    caregiver_id: int

    child_id: int

    cues: Dict[str, Any]


# Simple password hash placeholder

import hashlib


def hash_password(pw: str) -> str:

    return hashlib.sha256(pw.encode("utf-8")).hexdigest()


@app.post("/caregivers/register")

def register_caregiver(payload: CaregiverCreate, db: Session = Depends(get_db)):

    existing = db.query(models.Caregiver).filter_by(email=payload.email).first()

    if existing:

        raise HTTPException(status_code=400, detail="Email already registered.")

    cg = models.Caregiver(

        name=payload.name,

        email=payload.email,

        hashed_password=hash_password(payload.password)

    )

    db.add(cg)

    db.commit()

    db.refresh(cg)

    return {"id": cg.id, "name": cg.name, "email": cg.email}


@app.post("/children/register")

def register_child(payload: ChildCreate, db: Session = Depends(get_db)):

    caregiver = db.query(models.Caregiver).get(payload.caregiver_id)

    if not caregiver:

        raise HTTPException(status_code=404, detail="Caregiver not found.")

    child = models.Child(

        caregiver_id=payload.caregiver_id,

        name=payload.name,

        profile_notes=payload.profile_notes

    )

    db.add(child)

    db.commit()

    db.refresh(child)

    return {"id": child.id, "caregiver_id": child.caregiver_id, "name": child.name}


@app.post("/auth/request")

def authenticate_child(payload: AuthRequest, db: Session = Depends(get_db)):

    # Basic existence checks

    caregiver = db.query(models.Caregiver).get(payload.caregiver_id)

    child = db.query(models.Child).get(payload.child_id)

    if not caregiver or not child:

        raise HTTPException(status_code=404, detail="Caregiver or child not found.")


    # Build NI context and run engine

    ctx = NIContext(

        caregiver_id=payload.caregiver_id,

        child_id=payload.child_id,

        cues=payload.cues

    )

    result = ni_engine.authenticate(ctx)


    # Store auth session

    session_record = models.AuthSession(

        caregiver_id=payload.caregiver_id,

        child_id=payload.child_id,

        status=result.status,

        result_label=result.result_label,

        confidence=result.confidence

    )

    db.add(session_record)

    db.commit()

    db.refresh(session_record)


    # Log NI-AI internal outcome

    log = models.NIAILog(

        auth_session_id=session_record.id,

        winning_thread=result.winning_thread_id,

        conflict_score=result.conflict_score,

        internal_notes=result.notes

    )

    db.add(log)

    db.commit()


    # Return single clear response (no internal details)

    return {

        "auth_session_id": session_record.id,

        "status": result.status,

        "result": result.result_label,

        "confidence": result.confidence

    }
