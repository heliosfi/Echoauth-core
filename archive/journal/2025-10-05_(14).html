Love this vision. Let’s lock in something you can actually run the day your CUDA laptop lands: a tiny, secure EchoAuth Portal that uses the EchoAuth chamber for sign-in and lets you store/share notes for your son with teachers/doctors.


Below is a single-file Flask app that plugs straight into echoauth_chamber.py. It:

• accepts a frame (BGR .npy or raw bytes) + audio (float32 .npy or raw bytes),

• calls verify_user_once(...),

• on accept, returns a short-lived JWT,

• provides /records CRUD (encrypted at rest with Fernet) gated by the JWT.


1) app.py (drop this beside echoauth_chamber.py)


# app.py

import os, time, base64, json, io, sqlite3, secrets

from typing import Tuple

from functools import wraps

from datetime import datetime, timedelta


from flask import Flask, request, jsonify

from werkzeug.middleware.proxy_fix import ProxyFix


import numpy as np

import jwt

from cryptography.fernet import Fernet


from echoauth_chamber import verify_user_once  # same folder


# ---------- Config ----------

JWT_SECRET = os.environ.get("ECHOAUTH_JWT_SECRET", secrets.token_hex(32))

JWT_TTL_MIN = int(os.environ.get("ECHOAUTH_JWT_TTL_MIN", "30"))  # token lifetime

FERNET_KEY = os.environ.get("ECHOAUTH_FERNET_KEY") or Fernet.generate_key().decode()

DB_PATH = os.environ.get("ECHOAUTH_DB_PATH", "records.db")


cipher = Fernet(FERNET_KEY.encode())


# ---------- App ----------

app = Flask(__name__)

app.wsgi_app = ProxyFix(app.wsgi_app, x_for=1, x_proto=1)


# ---------- DB ----------

conn = sqlite3.connect(DB_PATH, check_same_thread=False)

conn.execute("""

CREATE TABLE IF NOT EXISTS records (

  id INTEGER PRIMARY KEY AUTOINCREMENT,

  user_id INTEGER NOT NULL,

  author TEXT NOT NULL,

  note BLOB NOT NULL,

  created_at TEXT NOT NULL

)

""")

conn.commit()


# ---------- Helpers ----------

def make_jwt(user_id: int) -> str:

    now = datetime.utcnow()

    payload = {

        "sub": str(user_id),

        "iat": int(now.timestamp()),

        "exp": int((now + timedelta(minutes=JWT_TTL_MIN)).timestamp())

    }

    return jwt.encode(payload, JWT_SECRET, algorithm="HS256")


def require_jwt(fn):

    @wraps(fn)

    def inner(*args, **kwargs):

        h = request.headers.get("Authorization", "")

        if not h.startswith("Bearer "):

            return jsonify({"error": "missing token"}), 401

        token = h.split(" ", 1)[1].strip()

        try:

            claims = jwt.decode(token, JWT_SECRET, algorithms=["HS256"])

            request.user_id = int(claims["sub"])

        except Exception as e:

            return jsonify({"error": "invalid token"}), 401

        return fn(*args, **kwargs)

    return inner


def _decode_array(file_storage, expect_dtype=None) -> np.ndarray:

    """

    Accepts either:

      - raw bytes (frame: HxWx3 uint8 BGR; audio: float32 mono 16k samples)

      - .npy files (np.save format)

    """

    buf = file_storage.read()

    if file_storage.filename and file_storage.filename.endswith(".npy"):

        arr = np.load(io.BytesIO(buf))

    else:

        # raw bytes path needs shape info; for demo assume audio raw float32/16000 and frame raw H*W*3 u8 480x640

        if expect_dtype == "frame":

            arr = np.frombuffer(buf, dtype=np.uint8)

            arr = arr.reshape((480, 640, 3))

        elif expect_dtype == "audio":

            arr = np.frombuffer(buf, dtype=np.float32)

        else:

            raise ValueError("Unknown raw payload type")

    return arr


# ---------- Routes ----------

@app.route("/auth/echoauth", methods=["POST"])

def auth_echoauth():

    """

    POST multipart/form-data:

      - frame: .npy or raw BGR bytes (480x640x3 u8)  (for quick start)

      - audio: .npy or raw float32 mono @16kHz      (16000 samples)

      - user_id: optional int (who this session belongs to)

    Returns: { status, score, decision, token? }

    """

    if "frame" not in request.files or "audio" not in request.files:

        return jsonify({"error": "missing frame or audio"}), 400

    try:

        frame = _decode_array(request.files["frame"], expect_dtype="frame")

        audio = _decode_array(request.files["audio"], expect_dtype="audio")

    except Exception as e:

        return jsonify({"error": f"bad inputs: {e}"}), 400


    # who is being authenticated (e.g., guardian/teacher id linked to a child profile)

    user_id = int(request.form.get("user_id", "0"))


    res = verify_user_once(frame, audio)  # {device, score, decision, latency_ms}

    out = {"status": res["decision"], "score": res["score"], "latency_ms": res["latency_ms"]}

    if res["decision"] == "accept":

        out["token"] = make_jwt(user_id)

    return jsonify(out), (200 if res["decision"] == "accept" else 401)


@app.route("/records", methods=["GET"])

@require_jwt

def list_records():

    # ?limit=50

    lim = min(int(request.args.get("limit", "50")), 200)

    cur = conn.execute("SELECT id, author, note, created_at FROM records WHERE user_id=? ORDER BY id DESC LIMIT ?",

                       (request.user_id, lim))

    rows = cur.fetchall()

    items = []

    for rid, author, note_blob, ts in rows:

        try:

            note = cipher.decrypt(note_blob).decode("utf-8")

        except Exception:

            note = "[DECRYPT ERROR]"

        items.append({"id": rid, "author": author, "note": note, "created_at": ts})

    return jsonify(items)


@app.route("/records", methods=["POST"])

@require_jwt

def add_record():

    """

    JSON: { "author": "mom/teacher/doctor", "note": "free text" }

    """

    data = request.get_json(force=True, silent=True) or {}

    author = (data.get("author") or "unknown")[:64]

    note   = (data.get("note") or "").strip()

    if not note:

        return jsonify({"error": "empty note"}), 400

    enc = cipher.encrypt(note.encode("utf-8"))

    ts = datetime.utcnow().isoformat(timespec="seconds") + "Z"

    conn.execute("INSERT INTO records (user_id, author, note, created_at) VALUES (?, ?, ?, ?)",

                 (request.user_id, author, enc, ts))

    conn.commit()

    return jsonify({"ok": True})


@app.route("/healthz")

def healthz():

    return jsonify({"ok": True})


if __name__ == "__main__":

    # For local dev only. Use a real WSGI server in prod (gunicorn/uvicorn).

    app.run(host="0.0.0.0", port=8080, debug=False)

4) How this maps to your goals

• Non-verbal support: EchoAuth is the secure front door (face+voice = who’s accessing). You can embed any AAC tool in the same portal after auth.
