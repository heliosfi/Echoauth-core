import uuid, json


def create_reasoning_capsule(state: dict) -> str:

    cid = str(uuid.uuid4()).upper()

    capsule = {"capsule_id": cid, "sealed_state": state}

    with open(f"capsule_{cid}.json", "w") as f:

        json.dump(capsule, f, indent=2)

    return cid

🛠️ db/local_db.py


import json, datetime, uuid


DB_FILE = "checkin_history.json"


def load_history() -> list:

    try:

        with open(DB_FILE, "r") as f:

            return json.load(f)

    except FileNotFoundError:

        return []


def write_checkin(result: str, confidence: float, stress: float) -> str:

    entry = {

        "event_id": str(uuid.uuid4()).upper(),

        "timestamp": datetime.datetime.now().isoformat(),

        "stress_score": stress,

        "runtime_mode": None,

        "auth_result": result,

        "confidence": confidence

    }

    history = load_history()

    history.append(entry)

    with open(DB_FILE, "w") as f:

        json.dump(history, f, indent=2)

    return entry["event_id"]

🚀 run_local_demo.py (Main MVP Pilot Launch)


import datetime, json

from core.arbitration import map_stress_to_mode

from core.runtime_modes import adjust_llm_config

from core.reasoning_capsule import create_reasoning_capsule

from db.local_db import write_checkin


def score_cues(cues: list[str]) -> float:

    # placeholder minimal emotion/pressure scoring

    keywords = ["cry", "noise", "shake", "hide", "run", "panic"]

    hits = sum(1 for c in cues for k in keywords if k in c.lower())

    return min(hits * 0.2, 1.0)


def pilot_run():

    cues_today = ["quiet hum", "look away", "small cry", "hide sound"]

    stress = score_cues(cues_today)

    mode = map_stress_to_mode(stress)

    llm_cfg = adjust_llm_config(mode)

    # Seal reasoning only when DEGRADED

    capsule = None

    if mode == "DEGRADED":

        capsule = create_reasoning_capsule({"date": datetime.date.today().isoformat(), "mode": mode})

    # Record checkin

    event_id = write_checkin("SAFE" if stress < 0.7 else "NEEDS_ATTENTION", confidence=0.99, stress=stress)

    # Return minimal external result

    return {"label": "SAFE" if stress < 0.7 else "NEEDS_ATTENTION", "llm_config": llm_cfg, "capsule": capsule or "NONE", "event": event_id}


out = pilot_run()

print(json.dumps(out, indent=2))
