# --- CORE LOGIC ---

def calculate_indicators(df):

    """Adds MACD and ATR to the dataframe."""

    # MACD

    ema12 = df['close'].ewm(span=12, adjust=False).mean()

    ema26 = df['close'].ewm(span=26, adjust=False).mean()

    macd_line = ema12 - ema26

    signal_line = macd_line.ewm(span=9, adjust=False).mean()

    df['macd_hist'] = macd_line - signal_line

    

    # ATR

    high_low = df['high'] - df['low']

    high_cp = np.abs(df['high'] - df['close'].shift())

    low_cp = np.abs(df['low'] - df['close'].shift())

    df['tr'] = pd.concat([high_low, high_cp, low_cp], axis=1).max(axis=1)

    df['atr'] = df['tr'].rolling(window=ATR_LEN).mean()

    return df


def is_power_hour(dt):

    """Filters for high-probability windows (Morning & Power Hour)."""

    current_time = dt.time()

    return (time(9, 30) <= current_time <= time(11, 0)) or \

           (time(14, 30) <= current_time <= time(16, 0))


# --- RISK GATES ---

def validate_risk(entry, stop, tp1):

    risk = abs(entry - stop)

    reward = abs(tp1 - entry)

    if risk > MAX_RISK_DOLLARS or risk == 0:

        return False, "Risk exceeds max $ or zero risk"

    if (reward / risk) < 1.0:

        return False, "R:R < 1.0 to TP1"

    return True, "Risk OK"


def get_position_size(entry, stop):

    """Calculates shares to buy based on a fixed % risk."""

    risk_per_share = abs(entry - stop)

    total_dollar_risk = ACCOUNT_BALANCE * RISK_PER_TRADE_PERCENT

    return max(1, int(total_dollar_risk / risk_per_share))  # At least 1 share


# --- DYNAMIC TARGETS ---

def calculate_atr_targets(entry, atr, direction):

    """Sets TPs as multiples of ATR for volatility-adjusted exits."""

    if direction == "LONG":

        return [entry + (atr * 2), entry + (atr * 4), entry + (atr * 6)]

    return [entry - (atr * 2), entry - (atr * 4), entry - (atr * 6)]


# --- NEWS FILTER ---

def check_news_filter(client, current_dt):

    """Checks for scheduled events or recent high-impact news."""

    # Check scheduled events

    for event_dt in HIGH_IMPACT_EVENTS:

        if event_dt - timedelta(minutes=DISABLE_MINUTES_BEFORE_EVENT) <= current_dt < event_dt + timedelta(hours=DISABLE_AFTER_EVENT_HOURS):

            return True, "High-impact scheduled event detected. Bot disabled."


    # Fetch and check recent news for high-impact keywords

    from_date = (current_dt - timedelta(hours=NEWS_LOOKBACK_HOURS)).isoformat()

    news = client.get_ticker_news(ticker=SYMBOL, limit=10, order="desc", sort="published_utc", published_utc_gte=from_date)

    for article in news:

        title_lower = article.title.lower()

        if any(kw in title_lower for kw in HIGH_IMPACT_KEYWORDS):

            return True, f"High-impact news detected: {article.title}. Bot disabled."

    

    return False, "No high-impact news. Proceeding."


# --- TRADE MANAGEMENT ---

def manage_trade(df, i, state):

    curr = df.iloc[i]

    if state['direction'] == "LONG":

        if curr['low'] <= state['stop']:

            print(f"[{curr.name}] EXIT_ALL: STOP_LONG hit at {curr['low']}.")

            state = reset_state(state)

            return state

        hit_count = sum(1 for tp in state['targets'] if curr['high'] >= tp)

        if hit_count > 0:

            print(f"[{curr.name}] TAKE_PARTIAL: Hit {hit_count} TPs for LONG.")

        if curr['high'] >= state['targets'][-1]:

            print(f"[{curr.name}] EXIT_ALL: TP3_LONG hit.")

            state = reset_state(state)

    

    elif state['direction'] == "SHORT":

        if curr['high'] >= state['stop']:

            print(f"[{curr.name}] EXIT_ALL: STOP_SHORT hit at {curr['high']}.")

            state = reset_state(state)

            return state

        hit_count = sum(1 for tp in state['targets'] if curr['low'] <= tp)

        if hit_count > 0:

            print(f"[{curr.name}] TAKE_PARTIAL: Hit {hit_count} TPs for SHORT.")

        if curr['low'] <= state['targets'][-1]:

            print(f"[{curr.name}] EXIT_ALL: TP3_SHORT hit.")

            state = reset_state(state)

    return state


def reset_state(state):

    state['direction'] = "NONE"

    state['entry'] = None

    state['stop'] = None

    state['targets'] = []

    state['shares'] = 0

    state['pending_dir'] = None

    state['pending_level'] = None

    return state

def process_candle(df, i, state, client):

    curr = df.iloc[i]

    prev = df.iloc[i-1]

    last3_vol = df['volume'].iloc[max(0, i-3):i].mean()

    

    # Skip if outside high-probability time windows or NaN ATR/MACD

    if not is_power_hour(curr.name) or pd.isna(curr['atr']) or pd.isna(curr['macd_hist']):

        return state


    # News Filter: Check for high-impact events or news

    disable, msg = check_news_filter(client, curr.name)

    if disable:

        print(f"[{curr.name}] News Filter: {msg}")

        return state


    # Check for pending retest entry

    if state['pending_dir']:

        touched_level = False

        if state['pending_dir'] == "LONG":

            touched_level = (curr['low'] <= state['pending_level'] + RETEST_TOL) and (curr['close'] > state['pending_level'])

        elif state['pending_dir'] == "SHORT":

            touched_level = (curr['high'] >= state['pending_level'] - RETEST_TOL) and (curr['close'] < state['pending_level'])

        

        if touched_level:

            entry = curr['close']

            targets =_atr_targets(entry, curr['atr'], state['pending_dir'])

            tp1 = targets[0]

            valid, risk_msg = validate_risk(entry, state['stop'], tp1)

            if valid:

                state['direction'] = state['pending_dir']

                state['entry'] = entry

                state['targets'] = targets

                state['shares'] = get_position_size(entry, state['stop'])

                state['pending_dir'] = None

                state['pending_level'] = None

                print(f"[{curr.name}] 🎯 RETEST FILLED: {state['direction']} {state['shares']} shares at {entry}, stop {state['stop']}.")

            else:

                print(f"[{curr.name}] Retest rejected: {risk_msg}")

                state['pending_dir'] = None

                state['pending_level'] = None


    # If in trade, manage it

    if state['direction'] != "NONE":

        state = manage_trade(df, i, state)

        return state


    # New Break Detection with Volume Surge

    avg_vol = df['volume'].iloc[max(0, i-10):i].mean()  # Lookback 10 for better avg

    vol_surge = curr['volume'] > (avg_vol * 1.5)


    # Bullish Break

    is_bull_break = (curr['close'] > RESIST and

                     curr['macd_hist'] > prev['macd_hist'] and 

                     curr['macd_hist'] > 0)


    if is_bull_break:

        # ATR Gate for candle size

            print(f"[{curr.name}] Bull signal rejected: Candle too large.")
