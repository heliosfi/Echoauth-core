db/session.py


from sqlalchemy import create_engine

from sqlalchemy.orm import sessionmaker, declarative_base


DATABASE_URL = "sqlite:///./echoauth.db"  # MVP: local file DB


engine = create_engine(

    DATABASE_URL,

    connect_args={"check_same_thread": False}

)


SessionLocal = sessionmaker(bind=engine, autoflush=False, autocommit=False)

Base = declarative_base()


db/models.py


from sqlalchemy import Column, Integer, String, Text, Float, ForeignKey, DateTime

from sqlalchemy.orm import relationship

from datetime import datetime

from .session import Base


class Caregiver(Base):

    __tablename__ = "caregivers"


    id = Column(Integer, primary_key=True, index=True)

    name = Column(String, nullable=False)

    email = Column(String, unique=True, index=True, nullable=False)

    hashed_password = Column(String, nullable=False)

    created_at = Column(DateTime, default=datetime.utcnow)


    children = relationship("Child", back_populates="caregiver")



class Child(Base):

    __tablename__ = "children"


    id = Column(Integer, primary_key=True, index=True)

    caregiver_id = Column(Integer, ForeignKey("caregivers.id"), nullable=False)

    name = Column(String, nullable=False)

    age_range = Column(String, nullable=True)  # e.g. "6–8"

    notes = Column(Text, nullable=True)

    created_at = Column(DateTime, default=datetime.utcnow)


    caregiver = relationship("Caregiver", back_populates="children")

    cues = relationship("CueDefinition", back_populates="child")

    checkins = relationship("CheckIn", back_populates="child")



class CueDefinition(Base):

    """

    Caregiver’s dictionary of non-verbal cues for a child.

    """

    __tablename__ = "cue_definitions"


    id = Column(Integer, primary_key=True, index=True)

    child_id = Column(Integer, ForeignKey("children.id"), nullable=False)

    cue_type = Column(String, nullable=False)       # gesture / sound / behavior

    description = Column(Text, nullable=False)      # what it looks like

    meaning = Column(Text, nullable=False)          # what caregiver says it means

    created_at = Column(DateTime, default=datetime.utcnow)


    child = relationship("Child", back_populates="cues")



class CheckIn(Base):

    """

    Single caregiver check-in event.

    """

    __tablename__ = "checkins"


    id = Column(Integer, primary_key=True, index=True)

    caregiver_id = Column(Integer, ForeignKey("caregivers.id"), nullable=False)

    child_id = Column(Integer, ForeignKey("children.id"), nullable=False)

    observed_cues = Column(Text, nullable=False)    # JSON string or text summary

    result_label = Column(String, nullable=False)   # "Safe" / "Needs Attention"

    confidence = Column(Float, nullable=False)

    status = Column(String, nullable=False)         # "success" / "failure"

    created_at = Column(DateTime, default=datetime.utcnow)


    child = relationship("Child", back_populates="checkins")



class NIAILog(Base):

    """

    Internal NI-AI decision log – for caregivers/legal only, never UI-exposed.

    """

    __tablename__ = "ni_ai_logs"


    id = Column(Integer, primary_key=True, index=True)

    checkin_id = Column(Integer, ForeignKey("checkins.id"), nullable=False)

    winning_thread = Column(String, nullable=False)

    conflict_score = Column(Float, nullable=False)

    internal_notes = Column(Text, nullable=True)

    created_at = Column(DateTime, default=datetime.utcnow)


Hooking it into FastAPI (api/main.py minimal skeleton)


from fastapi import FastAPI, Depends, HTTPException

from sqlalchemy.orm import Session

from db.session import SessionLocal, engine

from db import models

import hashlib


app = FastAPI(title="EchoAuth MVP")


# create tables

models.Base.metadata.create_all(bind=engine)


def get_db():

    db = SessionLocal()

    try:

        yield db

    finally:

        db.close()


def hash_password(pw: str) -> str:

    return hashlib.sha256(pw.encode("utf-8")).hexdigest()

2️⃣ Simple Caregiver UI Form (HTML + fetch, no frameworks)


This is just a static HTML page that calls the FastAPI backend.

Save as frontend/index.html and open in a browser (or serve statically).


<!DOCTYPE html>

<html>

<head>

  <meta charset="UTF-8">

  <title>EchoAuth Caregiver MVP</title>

  <style>

    body { font-family: system-ui, sans-serif; margin: 20px; }

    h1 { font-size: 1.5rem; }

    .block { border: 1px solid #ddd; padding: 12px; margin-bottom: 16px; border-radius: 6px; }

    label { display: block; margin-top: 6px; }

    input, textarea { width: 100%; padding: 6px; margin-top: 2px; }

    button { margin-top: 8px; padding: 6px 12px; cursor: pointer; }

    #status { margin-top: 12px; white-space: pre-wrap; }

  </style>

</head>

<body>

  <h1>EchoAuth Caregiver MVP</h1>


  <!-- Simple config -->

  <div class="block">

    <label>Backend URL</label>

    <input id="backendUrl" value="http://localhost:8000">

  </div>


  <!-- Register caregiver -->

  <div class="block">

    <h2>1. Register Caregiver</h2>

    <label>Name</label>

    <input id="cgName">

    <label>Email</label>

    <input id="cgEmail">

    <label>Password</label>

    <input id="cgPassword" type="password">

    <button onclick="registerCaregiver()">Register</button>

  </div>


  <!-- Register child -->

  <div class="block">

    <h2>2. Register Child</h2>

    <label>Caregiver ID</label>

    <input id="childCaregiverId">

    <label>Child Name</label>

    <input id="childName">

    <label>Age Range</label>

    <input id="childAgeRange" placeholder="e.g. 6–8">

    <label>Notes</label>

    <textarea id="childNotes"></textarea>

    <button onclick="registerChild()">Register Child</button>

  </div>


  <!-- Check-in -->

  <div class="block">

    <h2>3. Check-In (Cues)</h2>

    <label>Caregiver ID</label>

    <input id="checkinCaregiverId">

    <label>Child ID</label>

    <input id="checkinChildId">

    <label>Cues (JSON)</label>

    <textarea id="checkinCues" placeholder='{"gesture": "hand flapping", "mood": "excited"}'></textarea>

    <button onclick="createCheckin()">Send Check-In</button>

  </div>


  <div id="status"></div>


  <script>

    async function post(path, data) {

      const base = document.getElementById('backendUrl').value;

      const res = await fetch(base + path, {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify(data)

      });

      if (!res.ok) {

        const err = await res.json().catch(() => ({}));

        throw new Error(err.detail || ('HTTP ' + res.status));

      }

      return res.json();

    }


    function showStatus(msg) {

      document.getElementById('status').textContent = msg;

    }


    async function registerCaregiver() {

      try {

        const data = await post('/caregivers/register', {

          name: document.getElementById('cgName').value,

          email: document.getElementById('cgEmail').value,

          password: document.getElementById('cgPassword').value

        });

        showStatus('Caregiver registered:\n' + JSON.stringify(data, null, 2));

        document.getElementById('childCaregiverId').value = data.id;

        document.getElementById('checkinCaregiverId').value = data.id;

      } catch (e) {

        showStatus('Error: ' + e.message);

      }

    }


    async function registerChild() {

      try {

        const data = await post('/children/register', {

          caregiver_id: parseInt(document.getElementById('childCaregiverId').value, 10),

          name: document.getElementById('childName').value,

          age_range: document.getElementById('childAgeRange').value,

          notes: document.getElementById('childNotes').value

        });

        showStatus('Child registered:\n' + JSON.stringify(data, null, 2));

        document.getElementById('checkinChildId').value = data.id;

      } catch (e) {

        showStatus('Error: ' + e.message);

      }

    }


    async function createCheckin() {

      try {

        const cuesText = document.getElementById('checkinCues').value || "{}";

        const cues = JSON.parse(cuesText);

        const data = await post('/checkins/create', {

          caregiver_id: parseInt(document.getElementById('checkinCaregiverId').value, 10),

          child_id: parseInt(document.getElementById('checkinChildId').value, 10),

          cues: cues

        });

        showStatus('Check-in result:\n' + JSON.stringify(data, null, 2));

      } catch (e) {

        showStatus('Error: ' + e.message);

      }

    }

  </script>

</body>

</html>

• Lets you send a check-in with cues (as JSON)

• Shows the single, coherent result returned by your backend.
