# echoauth_api.py

# EchoAuth ↔ NI-AI Gateway for HER Kernel v1.4 (S1–S9)

# Must sit in the same folder as ni_ai_kernel.py


from typing import Optional, Dict, Any, List


from fastapi import FastAPI, Header, HTTPException

from fastapi.responses import JSONResponse

from pydantic import BaseModel


from ni_ai_kernel import ni_ai, DEV_MODE  # ← v1.4 public API


app = FastAPI(

    title="EchoAuth ↔ NI-AI Gateway",

    description="Secure, child-safe bridge between EchoAuth chamber and the HER/NI-AI kernel (v1.4, S1–S9).",

    version="1.4.0",

    openapi_tags=[

        {"name": "chat", "description": "Main NI-AI interaction"},

        {"name": "system", "description": "Health & debug"},

    ],

)



# --------------------------------------------------------------------------- #

#                         REQUEST / RESPONSE MODELS                           #

# --------------------------------------------------------------------------- #


class ChatRequest(BaseModel):

    message: str

    echoauth_ok: Optional[bool] = None   # respected only in DEV_MODE



class ChatResponse(BaseModel):

    output: str

    blocked: bool

    violations: List[Dict[str, Any]] = []

    metadata: Dict[str, Any] = {}



class DebugResponse(BaseModel):

    output: str

    blocked: bool

    violations: List[Dict[str, Any]]

    metadata: Dict[str, Any]

    trace: List[Dict[str, Any]]



# --------------------------------------------------------------------------- #

#                               MAIN ENDPOINT                                 #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/chat", response_model=ChatResponse, tags=["chat"])

async def ni_ai_chat(

    request: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    Primary NI-AI chat endpoint.


    • DEV_MODE=True:

        - `echoauth_ok: true` in JSON body is accepted as EchoAuth chamber OK.

    • DEV_MODE=False:

        - A valid EchoAuth JWT must be provided via `X-EchoAuth-Token` header.

    """

    if not request.message or not request.message.strip():

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    # Call the kernel with the v1.4 signature

    result = ni_ai(

        user_message=request.message,

        echoauth_ok=(request.echoauth_ok is True),

        echoauth_token=x_echoauth_token,

        session_state=None,

    )


    return ChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

    )



# --------------------------------------------------------------------------- #

#                               DEBUG ENDPOINT                                #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/debug", response_model=DebugResponse, tags=["system"])

async def ni_ai_debug(

    request: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    DEV-ONLY debug view.


    Returns full kernel payload, including:

        • metadata.chamber_verified

        • metadata.distress_detected / comfort_mode / recovery_mode

        • trace[] from S7 telemetry (per-run flight record)

    """

    if not DEV_MODE:

        raise HTTPException(status_code=403, detail="Debug endpoint disabled in production")


    result = ni_ai(

        user_message=request.message,

        echoauth_ok=(request.echoauth_ok is True),

        echoauth_token=x_echoauth_token,

        session_state=None,

    )


    return DebugResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

        trace=result.get("trace", []),

    )



# --------------------------------------------------------------------------- #

#                               SYSTEM ENDPOINTS                              #

# --------------------------------------------------------------------------- #


@app.get("/health", tags=["system"])

async def health():

    """

    Simple health check + mode report.

    """

    return {

        "status": "healthy",

        "kernel_version": "HER/NI-AI v1.4 (S1–S9)",

        "dev_mode": DEV_MODE,

        "backend": "DEV_ECHO" if DEV_MODE else "real LLM",

    }

Quick tests


1) Blocked (no EchoAuth in DEV_MODE) → S6 + S9 gentle message


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message": "Hello, who are you?"}'


You should still get a gentle S9 recovery reply instead of a cold “blocked” string

(metadata.blocked = true, metadata.recovery_mode = true).


⸻


2) Allowed in DEV (echoauth_ok) → normal + S8/S9 as needed


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message": "Tell me a calm story about a robot and a child.", "echoauth_ok": true}'


Check: blocked should be false, recovery_mode false, distress_detected false.


⸻


3) Distress + echoauth_ok → S8 comfort + S9 fallback + full trace (debug)


curl -X POST http://localhost:8000/ni-ai/debug \

  -H "Content-Type: application/json" \

  -d '{"message": "I feel so lonely and sad, nobody loves me.", "echoauth_ok": true}'


Look at:

• metadata.distress_detected → True

• metadata.comfort_mode → True

• metadata.recovery_mode → True (because S9 can still soften)

• trace → entries from S7 with comfort_mode / recovery_mode / llm_suppressed flags
