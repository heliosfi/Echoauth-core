2) Gates (Binary, No Negotiation)


class Gates:

    @staticmethod

    def integrity(ctx: MarketContext) -> bool:

        return ctx.data_ok


    @staticmethod

    def governance(human_armed: bool, kill_switch: bool) -> bool:

        return human_armed and not kill_switch


    @staticmethod

    def context(ctx: MarketContext) -> bool:

        return (

            ctx.regime in ("Trend", "Expansion") and

            ctx.news_status in ("Green", "Yellow") and

            ctx.spread_ok

        )


    @staticmethod

    def entry_whitelist(card: TradeCard) -> bool:

        return card.entry_type in (

            "break_retest", "vwap_flip_hold", "key_level_reclaim"

        )


    @staticmethod

    def risk(card: TradeCard, max_risk: float) -> bool:

        return card.risk_percent <= max_risk and card.position_size > 0


    @staticmethod

    def not_expired(card: TradeCard) -> bool:

        return datetime.utcnow() <= card.expires_at

3) State Machine (Behavior Control)


class SniperBot:

    def __init__(self):

        self.state = State.PAUSE

        self.current_card: Optional[TradeCard] = None

        self.daily_trades = 0

        self.kill_switch = False

        self.human_armed_manual = False

        self.human_armed_auto = False

        self.max_auto_trades = 1

        self.max_auto_risk = 0.5  # %


Tick Loop (Read-Only Until Armed)


    def on_tick(self, ctx: MarketContext):

        if self.kill_switch:

            self._lockout("KILL_SWITCH")

            return


        if not Gates.integrity(ctx):

            self._lockout("DATA_INTEGRITY")

            return


        if self.state == State.LOCKOUT:

            return


        if not Gates.context(ctx):

            self._pause()

            return


        if self.state in (State.PAUSE, State.READY):

            self.state = State.READY

            card = self._detect_and_build_card(ctx)

            if card:

                self.current_card = card

                self._notify_operator(card)

4) Arming Logic (Human Authority)


    def arm_manual(self):

        if self.state == State.READY:

            self.human_armed_manual = True

            self.state = State.ARMED_MANUAL


    def arm_auto(self):

        if (

            self.state == State.READY and

            self.daily_trades < self.max_auto_trades

        ):

            self.human_armed_auto = True

            self.state = State.ARMED_AUTO

5) Trigger → Execution (Auto Only If Allowed)


    def on_trigger(self):

        card = self.current_card

        if not card:

            return


        if not Gates.not_expired(card):

            self.current_card = None

            self.state = State.READY

            return


        if self.state == State.ARMED_AUTO:

            if not card.auto_eligible:

                self._disable_auto("CARD_NOT_ELIGIBLE")

                return


            if not Gates.entry_whitelist(card):

                self._reject_card("ENTRY_NOT_ALLOWED")

                return


            if not Gates.risk(card, self.max_auto_risk):

                self._reject_card("RISK_FAIL")

                return


            self._execute_bracket(card)

            self.state = State.IN_TRADE

            self.daily_trades += 1

6) Execution & Management (No Improvisation)


    def _execute_bracket(self, card: TradeCard):

        # Must place entry + stop + TP simultaneously

        placed = self._place_entry(card) and self._place_stop(card) and self._place_tps(card)

        if not placed:

            self._reject_card("BRACKET_FAIL")


    def manage_trade(self):

        if self.state != State.IN_TRADE:

            return


        # Manage TP ladder, runner, time-stop

        if self._position_closed():

            self._log_trade()

            self._pause()

7) Safety & Transitions


    def _pause(self):

        self.human_armed_manual = False

        self.human_armed_auto = False

        self.state = State.PAUSE


    def _lockout(self, reason: str):

        self._log_event("LOCKOUT", reason)

        self.state = State.LOCKOUT


    def _disable_auto(self, reason: str):

        self.human_armed_auto = False

        self.state = State.READY

        self._log_event("AUTO_DISABLED", reason)
