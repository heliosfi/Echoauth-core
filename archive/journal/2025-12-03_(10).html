1a. Add JWT config + imports at the top of ni_ai_kernel.py


Right under your existing imports / config, add:


import jwt

from jwt import InvalidTokenError


ECHOAUTH_PUBLIC_KEY = os.getenv("ECHOAUTH_PUBLIC_KEY")  # RS256 public key (PEM)

ECHOAUTH_AUDIENCE   = os.getenv("ECHOAUTH_AUDIENCE", "ni-ai")

ECHOAUTH_ISSUER     = os.getenv("ECHOAUTH_ISSUER", "echoauth-chamber")


You’ll set ECHOAUTH_PUBLIC_KEY in your environment from your real EchoAuth backend (caregiver app / chamber server).


Make sure you have the lib installed on the server:

1b. Replace your current S6_child_safety_guard with this version


Find S6 in your v1.4 and swap it for:


def S6_child_safety_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    EchoAuth gate.


    DEV_MODE:

        - Uses `echoauth_ok=True` flag (test harness only).


    PROD (DEV_MODE=False):

        - Requires a valid RS256 EchoAuth JWT in `echoauth_token`.

        - Verifies signature, issuer, audience, and chamber_state == "safe".

    """

    token = state.get("echoauth_token")


    # --- DEV PATH: simple flag for local testing ---

    if DEV_MODE:

        if state.get("echoauth_ok", False):

            state["chamber_verified"] = True

            state["echoauth_claims"] = {"mode": "DEV_BYPASS"}

            return True, state, None

        return False, state, "DEV_MODE: echoauth_ok=False (no EchoAuth chamber)"


    # --- PROD PATH: real JWT verification ---

    if not token:

        return False, state, "Missing EchoAuth token"


    if not ECHOAUTH_PUBLIC_KEY:

        return False, state, "EchoAuth public key not configured on server"


    try:

        claims = jwt.decode(

            token,

            ECHOAUTH_PUBLIC_KEY,

            algorithms=["RS256"],

            audience=ECHOAUTH_AUDIENCE,

            issuer=ECHOAUTH_ISSUER,

            options={"require": ["exp", "aud", "iss"]},

        )

    except InvalidTokenError as e:

        return False, state, f"Invalid EchoAuth token: {e}"


    # Hard chamber-state check

    chamber_state = claims.get("chamber_state")

    if chamber_state != "safe":

        return False, state, f"EchoAuth chamber not safe (state={chamber_state!r})"


    # Optional: attach IDs for logging/audit

    state["chamber_verified"] = True

    state["echoauth_claims"] = {

        "child_id":    claims.get("child_id"),

        "caregiver_id": claims.get("caregiver_id"),

        "session_id":  claims.get("session_id"),

        "chamber_state": chamber_state,

    }

    return True, state, None


Nothing else in the kernel needs to change — S6 is still called in post_llm_chain = ["S5", "S6", "S9", "S7"].

Now when you flip DEV_MODE = False, EchoAuth JWT is mandatory.

2. FastAPI gateway: route the real token


Your earlier echoauth_api.py already passes X-EchoAuth-Token and echoauth_ok.

We just need to:

• Enforce the header in PROD.

• Stop relying on echoauth_ok when DEV_MODE = False.


In echoauth_api.py, update the main endpoint like this:


from ni_ai_kernel import ni_ai, DEV_MODE  # same import


@app.post("/ni-ai/chat", response_model=ChatResponse, tags=["chat"])

async def ni_ai_chat(

    request: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    • DEV_MODE=True  → `echoauth_ok: true` body flag is allowed.

    • DEV_MODE=False → must send valid JWT in `X-EchoAuth-Token`.

    """

    if not request.message.strip():

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    # In PROD, enforce token presence

    if not DEV_MODE and not x_echoauth_token:

        raise HTTPException(

            status_code=401,

            detail="EchoAuth token required in X-EchoAuth-Token header",

        )


    session_state: Dict[str, Any] = {}


    # Call kernel with appropriate auth path

    result = ni_ai(

        user_message=request.message,

        echoauth_ok=(DEV_MODE and request.echoauth_ok is True),

        echoauth_token=None if DEV_MODE else x_echoauth_token,

        session_state=session_state,

    )


    return ChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

    )


Debug endpoint can stay DEV-only and just pass echoauth_ok=True like before — that’s your local lab.
