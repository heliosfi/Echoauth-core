# ============================================================

# TIME-OF-DAY FILTER (ET) for TSLA 30m Breakout Confirm

# ============================================================


CONFIG:

  TIMEZONE = "America/New_York"


  ENABLE_TIME_FILTER = true


  # Entry windows (ET)

  WINDOW_OPEN_START  = "09:35"

  WINDOW_OPEN_END    = "10:45"


  WINDOW_POWER_START = "14:30"

  WINDOW_POWER_END   = "15:55"


  # Midday regime control

  MIDDAY_START = "11:15"

  MIDDAY_END   = "14:15"


  MIDDAY_MODE = "BLOCK"         # "BLOCK" or "STRICT"

  STRICT_MIN_BODY_RATIO = 0.65  # higher than normal

  STRICT_MAX_CANDLE_MULT = 1.20 # tighter anti-chase

  STRICT_VOL_MULT = 1.25        # require 25% more than last3 avg


# -----------------------------

# Time helpers

# -----------------------------

function now_et_hhmm():

  return CURRENT_TIME_HHMM(TIMEZONE)


function in_window(hhmm, start, end):

  return (hhmm >= start) AND (hhmm <= end)


function entry_allowed_now():

  if NOT ENABLE_TIME_FILTER: return true


  t = now_et_hhmm()


  # Midday handling

  if in_window(t, MIDDAY_START, MIDDAY_END):

    if MIDDAY_MODE == "BLOCK":

      return false

    # if STRICT, allow but will tighten filters elsewhere

    return true


  # Preferred entry windows

  if in_window(t, WINDOW_OPEN_START, WINDOW_OPEN_END): return true

  if in_window(t, WINDOW_POWER_START, WINDOW_POWER_END): return true


  # Otherwise: no new entries

  return false


function strict_mode_now():

  t = now_et_hhmm()

  return ENABLE_TIME_FILTER AND (MIDDAY_MODE == "STRICT") AND in_window(t, MIDDAY_START, MIDDAY_END)


# -----------------------------

# Override filters when midday strict mode is active

# -----------------------------

function effective_min_body_ratio():

  return strict_mode_now() ? STRICT_MIN_BODY_RATIO : MIN_BODY_RATIO


function effective_max_candle_mult():

  return strict_mode_now() ? STRICT_MAX_CANDLE_MULT : MAX_CANDLE_MULT


function effective_volume_ok(curr, last3):

  base = mean_volume(last3)

  if strict_mode_now():

    return curr.volume >= base * STRICT_VOL_MULT

  return curr.volume >= base


function confirm_candle_not_too_big(curr, atr):

  return range_size(curr) <= atr * effective_max_candle_mult()


function is_doji_like(c):

  return (body_size(c) / range_size(c)) < effective_min_body_ratio()

# MAIN (30m close) — Add the gate right at the top

# -----------------------------

ON_CANDLE_CLOSE(symbol=SYMBOL, tf=TF):

  if direction != "NONE": return


  # NEW: Time gate for *arming* signals

  if NOT entry_allowed_now():

    ALERT("TIME FILTER: no new entries in this window.")

    return


  curr = GET_CANDLE(0)

  prev = GET_CANDLE(1)

  last3 = [GET_CANDLE(1), GET_CANDLE(2), GET_CANDLE(3)]

  i = CURRENT_CANDLE_INDEX()

  atr = ATR(i, ATR_LEN)


  # Replace volume check calls with effective_volume_ok(...)

  # bullish_confirm / bearish_confirm should call:

  #   effective_volume_ok(curr, last3)

  # and use is_doji_like(curr) + confirm_candle_not_too_big(curr, atr)


  if bullish_confirm(curr, prev, last3, i, atr):

    if ENTRY_MODE == "CLOSE":

      plan = build_long_plan(curr.close, curr, atr)

      ...

    else:

      arm_retest("LONG", RESIST, i)

      ALERT("LONG CONFIRM: armed RETEST (time-ok).")


  else if bearish_confirm(curr, prev, last3, i, atr):

    if ENTRY_MODE == "CLOSE":

      plan = build_short_plan(curr.close, curr, atr)

      ...

    else:
