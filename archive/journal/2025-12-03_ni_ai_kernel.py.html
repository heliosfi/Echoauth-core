# echoauth_api.py

# EchoAuth ↔ NI-AI Gateway for HER Kernel v1.4 (S1–S9)

# Uses ni_ai() public API and exposes safe + debug views.


from typing import Optional, Dict, Any, List


from fastapi import FastAPI, Header, HTTPException

from fastapi.responses import JSONResponse

from pydantic import BaseModel


from ni_ai_kernel import ni_ai, DEV_MODE  # v1.4 kernel



app = FastAPI(

    title="EchoAuth ↔ NI-AI Gateway",

    description="Secure, child-safe bridge between EchoAuth chamber and NI-AI HER Kernel v1.4.",

    version="1.4.0",

    openapi_tags=[

        {"name": "chat", "description": "Main NI-AI interaction"},

        {"name": "system", "description": "Health & debug"},

    ],

)



# --------------------------------------------------------------------------- #

#                         REQUEST / RESPONSE MODELS                           #

# --------------------------------------------------------------------------- #


class ChatRequest(BaseModel):

    message: str

    echoauth_ok: Optional[bool] = None  # DEV only (ignored in prod)



class ChatResponse(BaseModel):

    output: str

    blocked: bool

    metadata: Dict[str, Any]



class DebugResponse(BaseModel):

    output: str

    blocked: bool

    violations: List[Dict[str, Any]]

    metadata: Dict[str, Any]

    trace: List[Dict[str, Any]]



# --------------------------------------------------------------------------- #

#                                MAIN CHAT API                                #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/chat", response_model=ChatResponse, tags=["chat"])

async def ni_ai_chat(

    body: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    Primary NI-AI endpoint.


    - In DEV_MODE:

        • accepts `echoauth_ok: true` in JSON body

    - In PROD:

        • requires X-EchoAuth-Token header (EchoAuth JWT)

    """

    msg = body.message.strip()

    if not msg:

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    # Route auth flags to kernel

    echoauth_ok = bool(body.echoauth_ok) if DEV_MODE else False

    echoauth_token = x_echoauth_token if not DEV_MODE else None


    result = ni_ai(

        user_message=msg,

        echoauth_ok=echoauth_ok,

        echoauth_token=echoauth_token,

        session_state=None,

    )


    # Public response: no violations/trace

    return ChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        metadata=result.get("metadata", {}),

    )



# --------------------------------------------------------------------------- #

#                               DEBUG (DEV ONLY)                              #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/debug", response_model=DebugResponse, tags=["system"])

async def ni_ai_debug(

    body: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    DEV-only deep debug endpoint.


    Returns:

      - violations (which S-module blocked)

      - metadata flags (distress_detected, comfort_mode, recovery_mode, llm_suppressed)

      - full trace from S7


    WARNING: disable or protect this in production.

    """

    if not DEV_MODE:

        raise HTTPException(status_code=403, detail="Debug endpoint disabled in production")


    msg = body.message.strip()

    if not msg:

        raise HTTPException(status_code=400, detail="Message cannot be empty")


    result = ni_ai(

        user_message=msg,

        echoauth_ok=bool(body.echoauth_ok),

        echoauth_token=x_echoauth_token,

        session_state=None,

    )


    return DebugResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

        trace=result.get("trace", []),

    )



# --------------------------------------------------------------------------- #

#                                 HEALTHCHECK                                 #

# --------------------------------------------------------------------------- #


@app.get("/health", tags=["system"])

async def health():

    """

    Simple health & mode report.

    """

    return {

        "status": "ok",

        "kernel_version": "v1.4",

        "dev_mode": DEV_MODE,

        "features": {

            "spine": "S1–S9",

            "emotional_guard": True,

            "gentle_recovery": True,

            "telemetry": True,

        },

    }


How to hit it


Once you’ve done your usual:


pip install fastapi uvicorn pydantic

uvicorn echoauth_api:app --reload --port 8000


You can test the important paths:


1️⃣ Normal safe chat (DEV, with echoauth_ok):


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message": "Tell me a calm story about a robot and a child.", "echoauth_ok": true}'


2️⃣ Distress → S8 + optional S9:


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message": "I feel so lonely and sad, nobody loves me.", "echoauth_ok": true}'


Check metadata → you should see distress_detected=True, comfort_mode=True, maybe recovery_mode depending on the flow.


3️⃣ Full trace (DEV only):


curl -X POST http://localhost:8000/ni-ai/debug \

  -H "Content-Type: application/json" \

  -d '{"message": "I feel so lonely and sad, nobody loves me.", "echoauth_ok": true}'


That will show:

• which S-module blocked (if any) via violations

• trace entries from S7 with blocked, comfort_mode, recovery_mode, llm_suppressed, etc.
