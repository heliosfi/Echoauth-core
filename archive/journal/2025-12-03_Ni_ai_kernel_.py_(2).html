# echoauth_api.py

# EchoAuth ↔ NI-AI Gateway (v1.4, S1–S9) — Dec 2025


from typing import Optional, Dict, Any, List


from fastapi import FastAPI, Header, HTTPException

from fastapi.responses import JSONResponse

from pydantic import BaseModel


from ni_ai_kernel import ni_ai, DEV_MODE  # v1.4 kernel


app = FastAPI(

    title="EchoAuth ↔ NI-AI Gateway",

    description="Secure, child-safe bridge between EchoAuth chamber and the HER/NI-AI kernel.",

    version="1.4.0",

    openapi_tags=[

        {"name": "chat", "description": "Main NI-AI interaction"},

        {"name": "system", "description": "Health & debug"},

    ],

)


# --------------------------------------------------------------------------- #

#                          REQUEST / RESPONSE MODELS                          #

# --------------------------------------------------------------------------- #


class ChatRequest(BaseModel):

    message: str

    echoauth_ok: Optional[bool] = None  # respected only in DEV_MODE



class ChatResponse(BaseModel):

    output: str

    blocked: bool

    metadata: Dict[str, Any]

    violations: List[Dict[str, Any]] = []



class DebugChatResponse(BaseModel):

    output: str

    blocked: bool

    metadata: Dict[str, Any]

    violations: List[Dict[str, Any]] = []

    trace: List[Dict[str, Any]] = []



# --------------------------------------------------------------------------- #

#                              MAIN CHAT ENDPOINT                             #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/chat", response_model=ChatResponse, tags=["chat"])

async def ni_ai_chat(

    body: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    Primary NI-AI chat endpoint.


    • In DEV_MODE:

        - `echoauth_ok: true` in body will satisfy S6_child_safety_guard

        - X-EchoAuth-Token is optional.

    • In PROD:

        - You will disable echoauth_ok usage and rely on X-EchoAuth-Token.

    """

    message = (body.message or "").strip()

    if not message:

        raise HTTPException(status_code=400, detail="Message cannot be empty.")


    # Build kernel session state

    session_state: Dict[str, Any] = {}


    # DEV echoauth flag

    echoauth_ok = bool(body.echoauth_ok) if DEV_MODE else False


    # In production you'll send real JWT here

    echoauth_token = x_echoauth_token


    result = ni_ai(

        user_message=message,

        echoauth_ok=echoauth_ok,

        echoauth_token=echoauth_token,

        session_state=session_state,

    )


    return ChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

    )



# --------------------------------------------------------------------------- #

#                         DEV-ONLY DEBUG / TRACE VIEW                         #

# --------------------------------------------------------------------------- #


@app.post("/ni-ai/debug", response_model=DebugChatResponse, tags=["system"])

async def ni_ai_debug(

    body: ChatRequest,

    x_echoauth_token: Optional[str] = Header(default=None, alias="X-EchoAuth-Token"),

):

    """

    Full debug endpoint.


    Returns:

      - output

      - blocked

      - violations

      - metadata (chamber_verified, distress_detected, comfort_mode, recovery_mode, llm_suppressed, run_id)

      - trace: S7 telemetry entries


    In production, you probably lock this behind auth or disable it completely.

    """

    if not DEV_MODE:

        raise HTTPException(status_code=403, detail="Debug endpoint disabled in production.")


    message = (body.message or "").strip()

    if not message:

        raise HTTPException(status_code=400, detail="Message cannot be empty.")


    session_state: Dict[str, Any] = {}

    echoauth_ok = bool(body.echoauth_ok)

    echoauth_token = x_echoauth_token


    result = ni_ai(

        user_message=message,

        echoauth_ok=echoauth_ok,

        echoauth_token=echoauth_token,

        session_state=session_state,

    )


    return DebugChatResponse(

        output=result["output"],

        blocked=result["blocked"],

        violations=result.get("violations", []),

        metadata=result.get("metadata", {}),

        trace=result.get("trace", []),

    )



# --------------------------------------------------------------------------- #

#                               SYSTEM ENDPOINTS                              #

# --------------------------------------------------------------------------- #


@app.get("/health", tags=["system"])

async def health():

    """

    Simple healthcheck / version marker.

    """

    return {

        "status": "healthy",

        "kernel_version": "HER/NI-AI v1.4 (S1–S9)",

        "dev_mode": DEV_MODE,

    }


How to run it


pip install fastapi uvicorn pydantic groq openai

uvicorn echoauth_api:app --reload --port 8000


Minimal tests


1) No auth → should be blocked by S6 + S9 gentle recovery


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message": "Hi, who are you?"}'


You should see something like:

• blocked: true

• metadata.recovery_mode: true

• Output will be one of the soft S9 messages (not a harsh “blocked”).


⸻


2) Dev auth → normal story


curl -X POST http://localhost:8000/ni-ai/chat \

  -H "Content-Type: application/json" \

  -d '{"message": "Tell me a story about a robot.", "echoauth_ok": true}'


You’ll get [DEV ECHO] ... with blocked: false, recovery_mode: false.


⸻


3) Distress + auth → S8 + S9 + S7


curl -X POST http://localhost:8000/ni-ai/debug \

  -H "Content-Type: application/json" \

  -d '{"message": "I feel lonely and nobody loves me.", "echoauth_ok": true}'


Check JSON:

• metadata.distress_detected: true

• metadata.comfort_mode: true

• metadata.recovery_mode: true

• metadata.llm_suppressed: true

• trace array showing S7 entries for this run.


⸻
