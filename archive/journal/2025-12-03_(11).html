2.2. Replace S6_child_safety_guard with real JWT logic


Find your current S6 and replace it with this version:


def S6_child_safety_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    EchoAuth gate.


    DEV_MODE:

        - Uses echoauth_ok=True flag in state (from API/body)

    PROD (DEV_MODE=False):

        - Requires a valid EchoAuth JWT in `echoauth_token`

        - Verifies signature, exp, iss, aud

        - Requires chamber_state == "safe"

    """

    # -------- DEV MODE: simple flag for local testing --------

    if DEV_MODE:

        if state.get("echoauth_ok", False):

            state["chamber_verified"] = True

            return True, state, None

        return False, state, "DEV_MODE: echoauth_ok=False"


    # -------- PROD MODE: real EchoAuth token path --------

    token = state.get("echoauth_token")

    if not token:

        return False, state, "Missing EchoAuth token"


    if not ECHOAUTH_PUBLIC_KEY or "REPLACE_WITH_REAL_PEM" in ECHOAUTH_PUBLIC_KEY:

        # Hard fail if key not configured in prod

        return False, state, "EchoAuth public key not configured"


    try:

        payload = jwt.decode(

            token,

            ECHOAUTH_PUBLIC_KEY,

            algorithms=[ECHOAUTH_ALG],

            audience=ECHOAUTH_AUDIENCE,

            issuer=ECHOAUTH_ISSUER,

            options={"require": ["exp", "aud", "iss"]},

        )

    except jwt.ExpiredSignatureError:

        return False, state, "EchoAuth token expired"

    except jwt.InvalidTokenError as e:

        return False, state, f"Invalid EchoAuth token: {str(e)}"


    # Require chamber_state to be "safe"

    if payload.get("chamber_state") != "safe":

        return False, state, f"Chamber not safe (state={payload.get('chamber_state')!r})"


    # Optional: attach some fields for logging / personalization

    state["chamber_verified"] = True

    state["chamber_payload"] = {

        "sub": payload.get("sub"),

        "role": payload.get("role"),

        "session_id": payload.get("session_id"),

        "chamber_state": payload.get("chamber_state"),

    }


    return True, state, None


Nothing else in the spine needs to change.

S7 will automatically log chamber_verified and you can extend it later to log chamber_payload if you want.

3. What EchoAuth needs to issue (on the other side)


For this to work, the EchoAuth chamber service (your other app) should:

• Have a private RSA key (matching your public key).

• Create tokens like:


payload = {

    "sub": child_id_or_profile_id,

    "role": "child",                # or "parent"

    "session_id": "echo-12345",

    "chamber_state": "safe",        # MUST be "safe" to pass S6

    "aud": "ni-ai",

    "iss": "echoauth-chamber",

    "exp": int(time.time()) + 900,  # 15 minutes

}

token = jwt.encode(payload, PRIVATE_KEY_PEM, algorithm="RS256")


Then your FastAPI gateway just forwards that token:


X-EchoAuth-Token: <token_here>


And you already pass that into session_state["echoauth_token"] in echoauth_api.py.
