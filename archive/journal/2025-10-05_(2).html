<script>

/* Subtle synchronization: brighten each node briefly as the pulse passes */

(function(){

  const nodes = [...document.querySelectorAll('.node')];

  const pulse = document.querySelector('.pulse');

  if(!nodes.length || !pulse) return;


  const brighten = (el) => {

    el.style.boxShadow = getComputedStyle(document.documentElement)

      .getPropertyValue('--shadow');

    el.style.borderColor = 'rgba(102,194,255,.45)';

    el.classList.add('active');

    setTimeout(()=> {

      el.style.boxShadow = '0 4px 18px rgba(0,0,0,.35)';

      el.style.borderColor = 'rgba(102,194,255,.18)';

      el.classList.remove('active');

    }, 950);

  };


  // timing map aligned with CSS keyframes travel (0→100%)

  const keyTimes = [12, 38, 63, 88]; // percentages where pulse meets each node

  let lastT = 0;


  function tick(ts){

    const d = (ts/1000) % 6.2; // same period as CSS animation

    const pct = (d/6.2)*100;


    keyTimes.forEach((k,i)=>{

      // trigger once as pct crosses key time moving forward

      if(lastT < k && pct >= k) brighten(nodes[i]);

      // handle wrapping from 100 -> 0

      if(lastT > pct && k <= pct) brighten(nodes[i]);

    });

    lastT = pct;

    requestAnimationFrame(tick);

  }

  requestAnimationFrame(tick);


  // keyboard focus shows tooltip

  nodes.forEach(n=>{

    n.addEventListener('focus',()=> n.classList.add('focus'));

    n.addEventListener('blur', ()=> n.classList.remove('focus'));

  });

})();

</script>
