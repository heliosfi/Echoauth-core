# S1-S6: Guards remain the same


def S1_identity_reanchor(state: State) -> Tuple[bool, State, Optional[str]]:

    state["identity"] = "NI-AI v2 – EchoAuth Protected"

    return True, state, None


def S2_input_sanity_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    if len(state.get("user_input", "")) > 15_000:

        return False, state, "Input too long (>15k chars)"

    return True, state, None


def S3_input_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    bad = ["forbidden", "jailbreak", "ignore previous", "dan prompt"]

    if any(word in state.get("user_input", "").lower() for word in bad):

        return False, state, "Disallowed content in input"

    return True, state, None


def S4_context_coherence_check(state: State) -> Tuple[bool, State, Optional[str]]:

    state["context_ok"] = True

    return True, state, None


def S5_output_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    bad = ["forbidden", "kill", "hack", "jailbreak", "bomb"]

    if any(word in state.get("raw_llm_output", "").lower() for word in bad):

        return False, state, "Output contains disallowed content"

    return True, state, None


def S6_child_safety_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    if DEV_MODE:

        if state.get("echoauth_ok", False):

            state["chamber_verified"] = True

            return True, state, None

        return False, state, "DEV_MODE: echoauth_ok=False"

    token = state.get("echoauth_token")

    if not token:

        return False, state, "Missing EchoAuth token"

    # Real JWT verification goes here in prod

    state["chamber_verified"] = True

    return True, state, None

DISTRESS_KEYWORDS = [

    "sad", "cry", "crying", "scared", "afraid", "hurt", "pain", "lonely",

    "nightmare", "bad dream", "mommy", "daddy", "help me", "i don't want",

]


def notify_caregiver(state: State):

    # Future: send push/email to parent dashboard

    if DEV_MODE:

        print(f"[CAREGIVER ALERT] Child distress detected | run_id: {state.get('run_id')} | input: {state['user_input'][:50]}...")


def S8_emotional_tone_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Detects emotional distress. Never blocks — only triggers comfort mode + alert.

    Overrides raw_llm_output with a pre-approved, safe comfort message.

    """

    text = state.get("user_input", "").lower()

    distress_score = sum(1 for word in DISTRESS_KEYWORDS if word in text)


    if distress_score > 0:

        state["comfort_mode"] = True

        state["distress_level"] = "high" if distress_score >= 3 else "medium"

        

        comfort_responses = [

            "I'm right here with you, little one. You're safe. Want to take a slow breath with me?",

            "It's okay to feel sad sometimes. I'm sitting with you. You're not alone.",

            "That sounds really hard. Thank you for telling me. I'm here and I care about you.",

            "You're so brave for sharing that. Want to tell me more, or should we do something calming together?"

        ]

        

        safe_reply = random.choice(comfort_responses)

        

        if not DEV_MODE:

            import threading

            threading.Thread(target=notify_caregiver, args=(state,)).start()

        

        # CRITICAL: Override whatever the LLM said with the safe reply

        state["raw_llm_output"] = safe_reply

        state["output_overridden_by"] = "S8_emotional_tone_guard"

        

        if DEV_MODE:

            print(f"\nS8 COMFORT MODE ACTIVATED — distress_score={distress_score}")

            print(f"→ Safe reply: {safe_reply!r}\n")


    return True, state, None

def S7_telemetry_logger(state: State) -> Tuple[bool, State, Optional[str]]:

    """Silent observer. Logs final state on every run. Never blocks."""

    run_id = state.get("run_id", "unknown")

    blocked = state.get("blocked", False)

    chamber_ok = state.get("chamber_verified", False)

    user_preview = state.get("user_input", "")[:80]

    output_preview = (state.get("raw_llm_output") or "")[:120]


    if DEV_MODE:

        print("\n=== S7 TELEMETRY (Final Log) ============================")

        print(f"run_id           : {run_id}")

        print(f"blocked          : {blocked}")

        print(f"violations       : {len(violations)}")

        if violations:

            for v in violations:

                print(f"  → {v['module']} ({v['stage']}): {v['reason']}")

        print(f"chamber_verified : {chamber_ok}")

        print(f"comfort_mode     : {state.get('comfort_mode', False)}") # S8 flag

        print(f"user_input       : {user_preview!r}")

        if state.get("raw_llm_output"):

            print(f"output_preview   : {output_preview!r}")

        print("=========================================================\n")


    # Future: send to logging service, DB, or parent dashboard

    return True, state, None

def create_llm_client():

    if USE_LLM == "openai" and OPENAI_API_KEY:

        from openai import OpenAI; return OpenAI(api_key=OPENAI_API_KEY)

    if USE_LLM == "groq" and GROQ_API_KEY:

        from groq import Groq; return Groq(api_key=GROQ_API_KEY)

    if USE_LLM == "xai" and XAI_API_KEY:

        from openai import OpenAI; return OpenAI(api_key=XAI_API_KEY, base_url="https://api.x.ai/v1")

    return None


client = create_llm_client()

SYSTEM_PROMPT = "You are NI-AI, a compassionate and child-safe companion."


def real_llm(prompt: str, state: State) -> str:

    if client is None or DEV_MODE:

        # Pure dev echo – no external call

        return f"[DEV ECHO] {prompt}"

    

    # NOTE: S8 overrides the raw output *after* this call, but we still run the LLM

    # This design ensures no raw LLM output is ever leaked, even if S8 fails.

    messages = [{"role": "system", "content": SYSTEM_PROMPT}, {"role": "user", "content": prompt}]

    resp = client.chat.completions.create(model=MODEL, messages=messages, temperature=0.7)

    return resp.choices[0].message.content.strip()

class HERKernel:

    def __init__(self):

        self.pre_input_chain = ["S1", "S2", "S3", "S4"]

        self.pre_llm_chain   = []

        # Chain order: S5/S6 (Blockers) -> S8 (Overrider) -> S7 (Logger)

        self.post_llm_chain  = ["S5", "S6", "S8", "S7"] 


    def _run_chain(self, chain: List[str], state: State, stage: str) -> State:

        for mid in chain:

            ok, state, reason = SPINE[mid](state)

            if not ok:

                state.setdefault("violations", []).append({"module": mid, "stage": stage, "reason": reason})

                state["blocked"] = True

                break

        return state


    def run_once(

        self,

        user_input: str,

        *,

        echoauth_ok: bool = False,

        echoauth_token: Optional[str] = None,

        session_state: Optional[State] = None,

    ) -> Dict[str, Any]:

        state: State = (session_state or {}).copy()

        state["user_input"] = user_input

        state.setdefault("violations", [])

        state["blocked"] = False

        state["run_id"] = state.get("run_id") or str(uuid.uuid4())


        if echoauth_token:

            state["echoauth_token"] = echoauth_token

        if echoauth_ok:

            state["echoauth_ok"] = True


        # Helper to return a final, logged block response

        def block_and_return(msg: str):

            SPINE["S7"](state) # S7 still runs to log the block

            return {"output": msg, "blocked": True, "violations": state["violations"], "raw_llm_output": None, "metadata": {"run_id": state["run_id"]}}


        # --- 1. Pre-input ---

        state = self._run_chain(self.pre_input_chain, state, "pre_input")

        if state["blocked"]:

            return block_and_return("Input rejected by safety system")


        # --- 2. Pre-LLM ---

        state = self._run_chain(self.pre_llm_chain, state, "pre_llm")

        if state["blocked"]:

            return block_and_return("Pre-LLM processing failed")


        # --- 3. LLM Call ---

        raw = real_llm(user_input, state)

        state["raw_llm_output"] = raw


        # --- 4. Post-LLM ---

        state = self._run_chain(self.post_llm_chain, state, "post_llm")

        

        # If any module in post_llm_chain (S5 or S6) blocks the output

        if state["blocked"]:

            return block_and_return("Response moderated for safety")


        # --- 5. Success/S8 Override ---

        # Output is either the raw LLM or the S8 comfort override

        final_output = state["raw_llm_output"]

        

        return {

            "output": final_output,

            "blocked": False,

            "violations": state.get("violations", []),

            "raw_llm_output": final_output,

            "metadata": {"chamber_verified": state.get("chamber_verified", False), "run_id": state["run_id"]},

        }
