db/session.py


from sqlalchemy import create_engine

from sqlalchemy.orm import sessionmaker, declarative_base


DATABASE_URL = "sqlite:///./echoauth.db"  # MVP: simple local file DB


engine = create_engine(

    DATABASE_URL,

    connect_args={"check_same_thread": False}

)


SessionLocal = sessionmaker(bind=engine, autoflush=False, autocommit=False)

Base = declarative_base()


db/models.py


from sqlalchemy import Column, Integer, String, Text, Float, ForeignKey, DateTime

from sqlalchemy.orm import relationship

from datetime import datetime

from .session import Base


class Caregiver(Base):

    __tablename__ = "caregivers"


    id = Column(Integer, primary_key=True, index=True)

    name = Column(String, nullable=False)

    email = Column(String, unique=True, index=True, nullable=False)

    hashed_password = Column(String, nullable=False)

    created_at = Column(DateTime, default=datetime.utcnow)


    children = relationship("Child", back_populates="caregiver")



class Child(Base):

    __tablename__ = "children"


    id = Column(Integer, primary_key=True, index=True)

    caregiver_id = Column(Integer, ForeignKey("caregivers.id"), nullable=False)

    name = Column(String, nullable=False)

    age_range = Column(String, nullable=True)  # e.g. "6–8"

    notes = Column(Text, nullable=True)

    created_at = Column(DateTime, default=datetime.utcnow)


    caregiver = relationship("Caregiver", back_populates="children")

    cues = relationship("CueDefinition", back_populates="child")



class CueDefinition(Base):

    """

    The caregiver-defined non-verbal “dictionary” for the child.

    Example: type=gesture, description='flapping hands at chest',

             meaning='excited but safe'

    """

    __tablename__ = "cue_definitions"


    id = Column(Integer, primary_key=True, index=True)

    child_id = Column(Integer, ForeignKey("children.id"), nullable=False)

    cue_type = Column(String, nullable=False)       # gesture / sound / behavior

    description = Column(Text, nullable=False)      # what it looks like

    meaning = Column(Text, nullable=False)          # what caregiver says it means

    created_at = Column(DateTime, default=datetime.utcnow)


    child = relationship("Child", back_populates="cues")



class CheckIn(Base):

    """

    One caregiver “Check In” event.

    """

    __tablename__ = "checkins"


    id = Column(Integer, primary_key=True, index=True)

    caregiver_id = Column(Integer, ForeignKey("caregivers.id"), nullable=False)

    child_id = Column(Integer, ForeignKey("children.id"), nullable=False)

    observed_cues = Column(Text, nullable=False)    # JSON string or text summary

    status = Column(String, nullable=False)         # success / failure

    result_label = Column(String, nullable=False)   # Safe / Needs Attention

    confidence = Column(Float, nullable=False)

    created_at = Column(DateTime, default=datetime.utcnow)



class NIAILog(Base):

    """

    Internal NI-AI log — private, not shown in UI.

    """

    __tablename__ = "ni_ai_logs"


    id = Column(Integer, primary_key=True, index=True)

    checkin_id = Column(Integer, ForeignKey("checkins.id"), nullable=False)

    winning_thread = Column(String, nullable=False)

    conflict_score = Column(Float, nullable=False)

    internal_notes = Column(Text, nullable=True)

    created_at = Column(DateTime, default=datetime.utcnow)

3. NI-AI MVP Core


core/ni_ai.py


from typing import Dict, Any, List

from dataclasses import dataclass


@dataclass

class NIContext:

    caregiver_id: int

    child_id: int

    cues: Dict[str, Any]          # e.g. {"gesture": "...", "sound": "..."}

    intent: str = "check_in"



@dataclass

class NIResult:

    status: str                   # "success" / "failure"

    result_label: str             # "Safe" / "Needs Attention"

    confidence: float             # 0.0 - 1.0

    winning_thread_id: str

    conflict_score: float

    notes: str = ""               # internal only



class ThreadLinkingEngine:

    def build_threads(self, ctx: NIContext) -> List[Dict[str, Any]]:

        # MVP: single candidate thread; later this can fan out.

        base_score = 0.9  # placeholder confidence

        return [{"id": "thread_1", "score": base_score}]



class CoherenceEngine:

    def evaluate_conflicts(self, threads: List[Dict[str, Any]]) -> float:

        # MVP: no conflict between threads (only one candidate).

        return 0.0



class LoadRegulationMonitor:

    def evaluate_load(self, ctx: NIContext) -> float:

        # MVP: static low load; later you can tie to cue complexity / child profile.

        return 0.2



class OutputCollapseEnforcer:

    def select_winner(self, threads: List[Dict[str, Any]]) -> Dict[str, Any]:

        return max(threads, key=lambda t: t.get("score", 0.0))



class NIEngine:

    """

    Orchestrates TLE, CoherenceEngine, LRM, and PNRE.

    Enforces: exactly ONE result per request.

    """

    def __init__(self):

        self.tle = ThreadLinkingEngine()

        self.coherence = CoherenceEngine()

        self.lrm = LoadRegulationMonitor()

        self.pnre = OutputCollapseEnforcer()


    def evaluate_checkin(self, ctx: NIContext) -> NIResult:

        threads = self.tle.build_threads(ctx)

        conflict_score = self.coherence.evaluate_conflicts(threads)

        load_score = self.lrm.evaluate_load(ctx)

        winner = self.pnre.select_winner(threads)

        base_score = winner.get("score", 0.0)


        # Simple rule: if score high and load OK → Safe, else Needs Attention

        if base_score >= 0.7 and load_score <= 0.7:

            return NIResult(

                status="success",

                result_label="Safe",

                confidence=base_score,

                winning_thread_id=winner["id"],

                conflict_score=conflict_score,

                notes="MVP: Safe path."

            )

        else:

            return NIResult(

                status="failure",

                result_label="Needs Attention",

                confidence=base_score,

                winning_thread_id=winner["id"],

                conflict_score=conflict_score,

                notes="MVP: Needs Attention path."

            )


This matches your rule:


One coherent answer exits; conflicts stay internal.

4. API Layer


api/main.py


from fastapi import FastAPI, Depends, HTTPException

from sqlalchemy.orm import Session

from pydantic import BaseModel

from typing import Dict, Any, List

import hashlib, json


from db.session import SessionLocal, engine, Base

from db import models

from core.ni_ai import NIEngine, NIContext


app = FastAPI(title="EchoAuth MVP")

Base.metadata.create_all(bind=engine)

ni_engine = NIEngine()


def get_db():

    db = SessionLocal()

    try:

        yield db

    finally:

        db.close()


# ---------- Schemas ----------


class CaregiverCreate(BaseModel):

    name: str

    email: str

    password: str


class ChildCreate(BaseModel):

    caregiver_id: int

    name: str

    age_range: str | None = None

    notes: str | None = None


class CueCreate(BaseModel):

    child_id: int

    cue_type: str

    description: str

    meaning: str


class CheckInCreate(BaseModel):

    caregiver_id: int

    child_id: int

    cues: Dict[str, Any]      # observed cues in this moment


class CheckInResponse(BaseModel):

    checkin_id: int

    status: str

    result: str

    confidence: float


class HistoryItem(BaseModel):

    id: int

    created_at: str

    result: str

    confidence: float


# ---------- Helpers ----------


def hash_password(pw: str) -> str:

    return hashlib.sha256(pw.encode("utf-8")).hexdigest()


# ---------- Routes ----------


@app.post("/caregivers/register")

def register_caregiver(payload: CaregiverCreate, db: Session = Depends(get_db)):

    existing = db.query(models.Caregiver).filter_by(email=payload.email).first()

    if existing:

        raise HTTPException(status_code=400, detail="Email already registered.")

    cg = models.Caregiver(

        name=payload.name,

        email=payload.email,

        hashed_password=hash_password(payload.password)

    )

    db.add(cg)

    db.commit()

    db.refresh(cg)

    return {"id": cg.id, "name": cg.name, "email": cg.email}


@app.post("/children/register")

def register_child(payload: ChildCreate, db: Session = Depends(get_db)):

    caregiver = db.query(models.Caregiver).get(payload.caregiver_id)

    if not caregiver:

        raise HTTPException(status_code=404, detail="Caregiver not found.")

    child = models.Child(

        caregiver_id=payload.caregiver_id,

        name=payload.name,

        age_range=payload.age_range,

        notes=payload.notes

    )

    db.add(child)

    db.commit()

    db.refresh(child)

    return {"id": child.id, "caregiver_id": child.caregiver_id, "name": child.name}


@app.post("/cues/define")

def define_cue(payload: CueCreate, db: Session = Depends(get_db)):

    child = db.query(models.Child).get(payload.child_id)

    if not child:

        raise HTTPException(status_code=404, detail="Child not found.")

    cue = models.CueDefinition(

        child_id=payload.child_id,

        cue_type=payload.cue_type,

        description=payload.description,

        meaning=payload.meaning

    )

    db.add(cue)

    db.commit()

    db.refresh(cue)

    return {"id": cue.id, "child_id": cue.child_id}


@app.post("/checkins/create", response_model=CheckInResponse)

def create_checkin(payload: CheckInCreate, db: Session = Depends(get_db)):

    caregiver = db.query(models.Caregiver).get(payload.caregiver_id)

    child = db.query(models.Child).get(payload.child_id)

    if not caregiver or not child:

        raise HTTPException(status_code=404, detail="Caregiver or child not found.")


    ctx = NIContext(

        caregiver_id=payload.caregiver_id,

        child_id=payload.child_id,

        cues=payload.cues

    )

    result = ni_engine.evaluate_checkin(ctx)


    checkin = models.CheckIn(

        caregiver_id=payload.caregiver_id,

        child_id=payload.child_id,

        observed_cues=json.dumps(payload.cues),

        status=result.status,

        result_label=result.result_label,

        confidence=result.confidence

    )

    db.add(checkin)

    db.commit()

    db.refresh(checkin)


    ni_log = models.NIAILog(

        checkin_id=checkin.id,

        winning_thread=result.winning_thread_id,

        conflict_score=result.conflict_score,

        internal_notes=result.notes

    )

    db.add(ni_log)

    db.commit()


    return CheckInResponse(

        checkin_id=checkin.id,

        status=result.status,

        result=result.result_label,

        confidence=result.confidence

    )


@app.get("/history/{child_id}", response_model=List[HistoryItem])

def get_history(child_id: int, db: Session = Depends(get_db)):

    checkins = (

        db.query(models.CheckIn)

        .filter_by(child_id=child_id)

        .order_by(models.CheckIn.created_at.desc())

        .all()

    )

    return [

        HistoryItem(

            id=c.id,

            created_at=c.created_at.isoformat(),

            result=c.result_label,

            confidence=c.confidence

        )

        for c in checkins

    ]
