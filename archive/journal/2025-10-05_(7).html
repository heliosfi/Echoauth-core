Love this. Binaural beats + AAC + EchoAuth is a tight, thoughtful combo. I reviewed your update and it’s very close—just a couple of “gotchas” to make it bullet-proof:


Quick fixes (important)

1. simpleaudio expects int16, but your waveforms are float32 in a few paths. Passing float32 with bytes_per_sample=2 can produce silence/garbage.

2. One f-string in /music has nested quotes that will raise a syntax error.

3. Add a clean stop and non-blocking playback pattern so the UI never blocks, plus a tiny status endpoint.


Below is a safe drop-in patch for app.py. It keeps your API the same, adds a /music_status, and fixes the two bugs.

🔧 Patch for app.py (replace your music helpers + endpoint with this)


# ---- top of file ----

import threading

import simpleaudio as sa  # pip install simpleaudio


# ---- globals for playback state ----

_play_lock = threading.Lock()

_play_obj = None

_is_playing = False


def _stop_audio_locked():

    """Stop current playback (must hold _play_lock)."""

    global _play_obj, _is_playing

    if _play_obj is not None:

        try:

            _play_obj.stop()

        except Exception:

            pass

        _play_obj = None

    _is_playing = False



# ====== AUDIO BUILDERS (return int16 arrays for simpleaudio) ======


def _to_int16(mono_f32: np.ndarray, volume: float) -> np.ndarray:

    """Clip -> scale -> int16 mono."""

    x = np.clip(mono_f32 * float(volume), -1.0, 1.0)

    return (x * 32767.0).astype(np.int16)


def _to_int16_stereo(stereo_f32: np.ndarray, volume: float) -> np.ndarray:

    """Clip -> scale -> int16 stereo (shape [N,2])."""

    x = np.clip(stereo_f32 * float(volume), -1.0, 1.0)

    return (x * 32767.0).astype(np.int16)


def generate_tone_f32(frequency: float, duration_sec: float = 5.0, sample_rate: int = 44100) -> np.ndarray:

    """Return mono float32 [-1,1]."""

    t = np.linspace(0, duration_sec, int(sample_rate * duration_sec), endpoint=False, dtype=np.float64)

    tone = np.sin(2 * np.pi * frequency * t)

    # Optionally add a gentle low harmonic for some presets (kept subtle)

    if frequency in (528, 396):

        tone += 0.25 * np.sin(2 * np.pi * 2 * frequency * t)

    return tone.astype(np.float32)


def generate_binaural_f32(base_freq: float, beat_freq: float, duration_sec: float = 5.0, sample_rate: int = 44100) -> np.ndarray:

    """Return stereo float32 [-1,1], shape [N,2]. Left=base, Right=base+beat."""

    t = np.linspace(0, duration_sec, int(sample_rate * duration_sec), endpoint=False, dtype=np.float64)

    left  = np.sin(2 * np.pi * base_freq * t)

    right = np.sin(2 * np.pi * (base_freq + beat_freq) * t)

    stereo = np.stack([left, right], axis=1).astype(np.float32)

    return stereo

@app.route("/music", methods=["POST"])

def music():

    """

    JSON:

      {

        "action": "start" | "stop",

        "music": "relax" | "focus" | "calm" | "alpha" | "delta",

        "duration": 5.0,      # seconds

        "volume": 0.5,        # 0.1..1.0 (sensory-friendly caps)

        "user_id": 1,

        "audio": "<base64 optional>"  # for loudness-based volume soften

      }

    """

    try:

        data = request.get_json(force=True)

        action = data.get("action", "start")

        volume = float(data.get("volume", 0.5))

        volume = max(0.1, min(1.0, volume))


        with _play_lock:

            if action == "stop":

                _stop_audio_locked()

                return jsonify({"status": "stopped"})


            # ---- START path ----

            from_base64 = data.get("audio")

            if from_base64:

                # Sensory adaptation: soften if environment is loud

                try:

                    env_audio = np.frombuffer(base64.b64decode(from_base64), dtype=np.float32)

                    loudness = float(np.abs(env_audio).mean())

                    volume = max(0.3, volume - 0.5 * loudness)

                except Exception:

                    pass


            # Stop anything currently playing

            _stop_audio_locked()


            music_key = data.get("music", "relax")

            cfg = MUSIC_OPTIONS.get(music_key, MUSIC_OPTIONS["relax"])

            duration = float(data.get("duration", 5.0))

            duration = max(1.0, min(60.0, duration))  # cap to 60s


            sr = 44100

            if cfg["type"] == "tone":

                mono_f32 = generate_tone_f32(float(cfg["frequency"]), duration, sr)

                mono_i16 = _to_int16(mono_f32, volume)

                _play_obj = sa.play_buffer(mono_i16, num_channels=1, bytes_per_sample=2, sample_rate=sr)


            else:  # "binaural"

                base = float(cfg["base_freq"])

                beat = float(cfg["beat_freq"])

                stereo_f32 = generate_binaural_f32(base, beat, duration, sr)

                stereo_i16 = _to_int16_stereo(stereo_f32, volume)

                _play_obj = sa.play_buffer(stereo_i16, num_channels=2, bytes_per_sample=2, sample_rate=sr)


            # Mark as playing and detach a watcher thread to reset flag when done

            global _is_playing

            _is_playing = True

            def _watch():

                try:

                    _play_obj.wait_done()

                finally:

                    with _play_lock:

                        _stop_audio_locked()

            threading.Thread(target=_watch, daemon=True).start()


            # Log record (fix nested f-string bug by building text in pieces)

            try:

                user_id = int(data.get("user_id", 1))

                if cfg["type"] == "tone":

                    desc_detail = f"{cfg['frequency']}Hz tone"

                else:

                    desc_detail = f"binaural base {cfg['base_freq']}Hz + {cfg['beat_freq']}Hz beat"

                desc = f"Played {cfg['type']}: {music_key} ({desc_detail}), vol={volume:.2f}, dur={duration:.1f}s"

                ts = time.strftime("%Y-%m-%d %H:%M:%S")

                enc = cipher.encrypt(desc.encode("utf-8"))

                conn.execute("INSERT INTO records VALUES (?, ?, ?)", (user_id, enc, ts))

                conn.commit()

            except Exception as e:

                app.logger.warning(f"Music log insert skipped: {e}")


            return jsonify({"status": "playing", "music": music_key, "volume": volume, "duration": duration})


    except Exception as e:

        app.logger.exception("Music endpoint error")

        return jsonify({"status": "error", "message": str(e)}), 400



@app.route("/music_status", methods=["GET"])

def music_status():

    with _play_lock:

        return jsonify({"playing": _is_playing})

Why these changes?

• Correct sample format: simpleaudio wants int16. We now always convert from float32 to int16, mono or stereo.

• No deadlocks / blocking: Playback runs in the background; we flip a flag when it finishes.

• Safer logging: The long f-string is split to avoid quote nesting errors.

• Sensory-friendly defaults: Volume capped and softened when the environment is loud (if you pass mic audio).
