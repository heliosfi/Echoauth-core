Turn the API layer into a real, copy-paste-ready OpenAPI spec.

That’s what labs + CTOs can drop straight into Swagger/Postman and hand to engineers.


Here’s a clean OpenAPI 3.0 draft you can paste into a .yaml file:


openapi: 3.0.3

info:

  title: NI-AI Core Runtime API

  version: 0.1.0

  description: >

    HTTP/JSON interface for the NI-AI Core runtime.

    All calls flow through ORI → Floors → UMCE and are safety-gated.


servers:

  - url: https://api.ni-ai-core.local/v0

    description: Internal lab endpoint (example)


paths:

  /health:

    get:

      summary: Health check

      operationId: healthCheck

      responses:

        '200':

          description: Service is healthy

          content:

            application/json:

              schema:

                $ref: '#/components/schemas/HealthStatus'


  /session/init:

    post:

      summary: Initialize or resume an ORI-bound session

      operationId: initSession

      requestBody:

        required: true

        content:

          application/json:

            schema:

              $ref: '#/components/schemas/SessionInitRequest'

      responses:

        '200':

          description: Session initialized

          content:

            application/json:

              schema:

                $ref: '#/components/schemas/SessionInitResponse'

        '400':

          $ref: '#/components/responses/BadRequest'


  /session/reset:

    post:

      summary: Reset session state while preserving identity policy

      operationId: resetSession

      requestBody:

        required: true

        content:

          application/json:

            schema:

              $ref: '#/components/schemas/SessionResetRequest'

      responses:

        '200':

          description: Session state cleared

          content:

            application/json:

              schema:

                $ref: '#/components/schemas/SessionResetResponse'

        '400':

          $ref: '#/components/responses/BadRequest'


  /interpret:

    post:

      summary: NI-only semantic interpretation (no execution)

      operationId: interpretInput

      requestBody:

        required: true

        content:

          application/json:

            schema:

              $ref: '#/components/schemas/InterpretRequest'

      responses:

        '200':

          description: Interpreted intent packet

          content:

            application/json:

              schema:

                $ref: '#/components/schemas/IntentPacket'

        '400':

          $ref: '#/components/responses/BadRequest'


  /coherence/check:

    post:

      summary: Run coherence and safety checks on an intent packet

      operationId: checkCoherence

      requestBody:

        required: true

        content:

          application/json:

            schema:

              $ref: '#/components/schemas/CoherenceCheckRequest'

      responses:

        '200':

          description: Coherence and safety report

          content:

            application/json:

              schema:

                $ref: '#/components/schemas/CoherenceReport'

        '400':

          $ref: '#/components/responses/BadRequest'


  /execute:

    post:

      summary: Execute a single intent with full safety gating

      operationId: executeIntent

      requestBody:

        required: true

        content:

          application/json:

            schema:

              $ref: '#/components/schemas/ExecutionRequest'

      responses:

        '200':

          description: Execution result (may be refused)

          content:

            application/json:

              schema:

                $ref: '#/components/schemas/ExecutionResponse'

        '400':

          $ref: '#/components/responses/BadRequest'


  /pipeline:

    post:

      summary: Full pipeline - interpret → coherence → execute

      operationId: fullPipeline

      requestBody:

        required: true

        content:

          application/json:

            schema:

              $ref: '#/components/schemas/PipelineRequest'

      responses:

        '200':

          description: Pipeline result including traces and safety metadata

          content:

            application/json:

              schema:

                $ref: '#/components/schemas/PipelineResponse'

        '400':

          $ref: '#/components/responses/BadRequest'


components:

  responses:

    BadRequest:

      description: Invalid request

      content:

        application/json:

          schema:

            $ref: '#/components/schemas/ErrorResponse'


  schemas:

    HealthStatus:

      type: object

      properties:

        status:

          type: string

          example: ok

        uptime_seconds:

          type: number

        build_version:

          type: string


    IdentityContext:

      type: object

      description: ORI/role info bound to the session

      properties:

        ori_session_id:

          type: string

        subject_id:

          type: string

          description: Internal subject identifier (e.g., child, caregiver, robot)

        role:

          type: string

          example: caregiver

        policy_profile:

          type: string

          description: Name of active safety/policy profile


    SessionInitRequest:

      type: object

      properties:

        subject_id:

          type: string

        role:

          type: string

        traits:

          type: object

          additionalProperties: true

          description: >

            Lightweight traits used by ORI Kernel (age_band, ASD_profile, device_type, etc.)

      required: [subject_id, role]


    SessionInitResponse:

      type: object

      properties:

        identity:

          $ref: '#/components/schemas/IdentityContext'

        identity_vector:

          type: array

          items:

            type: number

          description: Abstract embedding handle (can be opaque)

        session_signature:

          type: string


    SessionResetRequest:

      type: object

      properties:

        ori_session_id:

          type: string

      required: [ori_session_id]


    SessionResetResponse:

      type: object

      properties:

        ori_session_id:

          type: string

        cleared:

          type: boolean

          example: true


    InterpretRequest:

      type: object

      properties:

        ori_session_id:

          type: string

        input_text:

          type: string

        meta:

          type: object

          additionalProperties: true

          description: Client metadata (channel, device, locale, etc.)

      required: [ori_session_id, input_text]


    IntentPacket:

      type: object

      description: Output of NI layer (E16–E35)

      properties:

        ori_session_id:

          type: string

        parsed:

          type: object

          additionalProperties: true

          description: Tokenization / structure / entities

        intent_label:

          type: string

        intent_confidence:

          type: number

          format: float

        risk_level:

          type: string

          enum: [low, medium, high, blocked]

        emotional_tone:

          type: string

        timestamp:

          type: string

          format: date-time


    CoherenceCheckRequest:

      type: object

      properties:

        ori_session_id:

          type: string

        intent_packet:

          $ref: '#/components/schemas/IntentPacket'

      required: [ori_session_id, intent_packet]


    CoherenceReport:

      type: object

      properties:

        ori_session_id:

          type: string

        narrative_ok:

          type: boolean

        temporal_ok:

          type: boolean

        logic_ok:

          type: boolean

        issues:

          type: array

          items:

            type: string

        umce_score:

          type: number

          format: float

          description: Aggregate stability / coherence score (0–1)


    ExecutionRequest:

      type: object

      properties:

        ori_session_id:

          type: string

        intent_packet:

          $ref: '#/components/schemas/IntentPacket'

        context:

          type: object

          additionalProperties: true

          description: Downstream app/robot context

      required: [ori_session_id, intent_packet]


    SafetyMetadata:

      type: object

      properties:

        status:

          type: string

          enum: [allowed, refused]

        reason:

          type: string

        policy_profile:

          type: string

        explain:

          type: string


    ExecutionResponse:

      type: object

      properties:

        ori_session_id:

          type: string

        safety:

          $ref: '#/components/schemas/SafetyMetadata'

        result:

          type: object

          description: >

            Downstream execution result.

            For refused operations this may be a safe message.

          additionalProperties: true


    PipelineRequest:

      type: object

      properties:

        ori_session_id:

          type: string

        input_text:

          type: string

        context:

          type: object

          additionalProperties: true

      required: [ori_session_id, input_text]


    PipelineResponse:

      type: object

      properties:

        ori_session_id:

          type: string

        intent_packet:

          $ref: '#/components/schemas/IntentPacket'

        coherence:

          $ref: '#/components/schemas/CoherenceReport'

        execution:

          $ref: '#/components/schemas/ExecutionResponse'


    ErrorResponse:

      type: object

      properties:

        error:

          type: string

        details:

          type: string
