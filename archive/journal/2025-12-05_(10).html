# EchoAuth Prototype


Safety-governed neuro-support authentication system for autistic children,

caregivers, and clinical environments.


This repo contains the prototype implementation scaffold for:


- Sensor pipeline

- HER Kernel (emotional-safety engine)

- Governance Spine (S1–S23)

- EchoAuth Chamber (authentication + handoff logic)

- UI/API for caregiver/parent dashboards


## High-Level Build Order


1. Sensors + logging scaffold

2. HER Kernel (simulated signals)

3. Governance (S1–S23 core rules)

4. Chamber logic

5. UI + API

6. Integration tests and safety validation

logs/logger.py


import os

import logging

from config.settings import LOG_DIR, SYSTEM_NAME


os.makedirs(LOG_DIR, exist_ok=True)


def _build_logger(name: str, filename: str) -> logging.Logger:

    logger = logging.getLogger(name)

    logger.setLevel(logging.INFO)

    if not logger.handlers:

        fh = logging.FileHandler(os.path.join(LOG_DIR, filename))

        fmt = logging.Formatter(

            fmt=f"%(asctime)s | {SYSTEM_NAME} | %(name)s | %(levelname)s | %(message)s"

        )

        fh.setFormatter(fmt)

        logger.addHandler(fh)

    return logger


event_logger = _build_logger("event", "event.log")

safety_logger = _build_logger("safety", "safety.log")

chamber_logger = _build_logger("chamber", "chamber.log")

sensors/base_sensor.py


from abc import ABC, abstractmethod

from typing import Dict, Any


class BaseSensor(ABC):

    """

    Abstract base sensor.

    All concrete sensors (EDA, HR, accelerometer) should extend this.

    """


    @abstractmethod

    def read(self) -> Dict[str, Any]:

        """

        Return a single reading as a dict.

        Example: {"type": "hr", "value": 98, "timestamp": ...}

        """

        raise NotImplementedError

sensors/sensor_manager.py


"""

Coordinates multiple sensors and returns a combined reading.

"""


from typing import Dict, Any, List

from .base_sensor import BaseSensor

from .mock_sensors import MockHeartRateSensor, MockEDASensor, MockMotionSensor


class SensorManager:

    def __init__(self, sensors: List[BaseSensor] | None = None) -> None:

        if sensors is None:

            sensors = [

                MockHeartRateSensor(),

                MockEDASensor(),

                MockMotionSensor(),

            ]

        self.sensors = sensors


    def read_all(self) -> Dict[str, Any]:

        reading: Dict[str, Any] = {"raw": [], "timestamp": None}

        for sensor in self.sensors:

            data = sensor.read()

            reading["raw"].append(data)

            reading["timestamp"] = data.get("timestamp")

        return reading

her/preprocessing.py


"""

Preprocessing: raw sensor readings → normalized feature vector.

"""


from typing import Dict, Any

import statistics


def preprocess_sensor_payload(payload: Dict[str, Any]) -> Dict[str, float]:

    """

    payload: {"raw": [sensor_reading,...], "timestamp": ...}

    Returns: normalized feature dict: { "heart_rate": ..., "eda": ..., "motion": ... }

    """

    hr_vals = [r["value"] for r in payload["raw"] if r["type"] == "heart_rate"]

    eda_vals = [r["value"] for r in payload["raw"] if r["type"] == "eda"]

    motion_vals = [r["value"] for r in payload["raw"] if r["type"] == "motion"]


    features = {

        "heart_rate": statistics.fmean(hr_vals) if hr_vals else 0.0,

        "eda": statistics.fmean(eda_vals) if eda_vals else 0.0,

        "motion": statistics.fmean(motion_vals) if motion_vals else 0.0,

    }

    return features


her/her_kernel.py


"""

HER Kernel: emotional-safety engine.

Takes preprocessed features → EmotionalState object.

"""


from typing import Dict, Any

from .emotional_state import EmotionalState

from config.settings import HER_DEFAULT_THRESHOLDS


class HERKernel:

    def __init__(self, thresholds: Dict[str, float] | None = None) -> None:

        self.thresholds = thresholds or HER_DEFAULT_THRESHOLDS


    def classify(self, features: Dict[str, float]) -> EmotionalState:

        hr = features.get("heart_rate", 0.0)

        eda = features.get("eda", 0.0)

        motion = features.get("motion", 0.0)


        # Very simple baseline logic; to be upgraded by ML later.

        if hr >= self.thresholds["hr_red"] or eda >= self.thresholds["eda_red"]:

            return EmotionalState(state="Red", confidence=0.8, trigger="hr_or_eda_high", details=features)

        if hr >= self.thresholds["hr_amber"] or eda >= self.thresholds["eda_amber"]:

            return EmotionalState(state="Amber", confidence=0.7, trigger="moderate_stress", details=features)

        return EmotionalState(state="Green", confidence=0.9, trigger="stable", details=features)

governance/governance_engine.py


from typing import Tuple

from her.emotional_state import EmotionalState

from .system_state import SystemState

from .rules import S1_identity_check, S5_safety_priority_lock, Decision

from logs.logger import safety_logger


class GovernanceEngine:

    def __init__(self) -> None:

        self.system_state = SystemState()


    def evaluate(self, emotional_state: EmotionalState, context: dict | None = None) -> Tuple[Decision, SystemState]:

        """

        Run emotional_state + context through S-rules.

        Returns final decision + updated SystemState.

        """

        context = context or {}


        # Example rule sequence

        decision = "ALLOW"  # default


        # Identity rule

        id_decision = S1_identity_check(context)

        if id_decision == "DENY":

            decision = "DENY"


        # Safety priority

        safety_decision = S5_safety_priority_lock(emotional_state)

        if safety_decision == "OVERRIDE":

            decision = "OVERRIDE"

            self.system_state.mode = "OVERRIDE"

            self.system_state.reason = "Red emotional state"


        safety_logger.info(

            f"Governance decision={decision} state={emotional_state.state} reason={self.system_state.reason}"

        )

        return decision, self.system_state

"""

EchoAuth Chamber: uses HER + Governance to approve/deny handoffs.

"""


from typing import Tuple

from her.emotional_state import EmotionalState

from governance.governance_engine import GovernanceEngine

from .auth_models import HandoffContext

from logs.logger import chamber_logger


class ChamberEngine:

    def __init__(self, governance: GovernanceEngine) -> None:

        self.governance = governance


    def evaluate_handoff(self, ctx: HandoffContext, state: EmotionalState) -> Tuple[bool, str]:

        decision, system_state = self.governance.evaluate(

            emotional_state=state,

            context={"parent_id": ctx.parent.actor_id, "caregiver_id": ctx.caregiver.actor_id},

        )


        if decision == "DENY":

            chamber_logger.info("Handoff denied by governance")

            return False, "Handoff denied by governance"

        if decision == "OVERRIDE":

            chamber_logger.info("Handoff blocked: child in overload (Red state)")

            return False, "Handoff blocked: child overload"

        chamber_logger.info("Handoff approved")

        return True, "Handoff approved"

ui/api.py


from fastapi import FastAPI

from ui.schemas import EmotionalStateResponse, HandoffRequest

from sensors.sensor_manager import SensorManager

from her.preprocessing import preprocess_sensor_payload

from her.her_kernel import HERKernel

from governance.governance_engine import GovernanceEngine

from chamber.chamber_engine import ChamberEngine

from chamber.auth_models import Actor, HandoffContext


app = FastAPI(title="EchoAuth Prototype API")


sensor_manager = SensorManager()

her_kernel = HERKernel()

governance_engine = GovernanceEngine()

chamber_engine = ChamberEngine(governance_engine)


@app.get("/state", response_model=EmotionalStateResponse)

def get_emotional_state():

    payload = sensor_manager.read_all()

    features = preprocess_sensor_payload(payload)

    state = her_kernel.classify(features)

    return EmotionalStateResponse(

        state=state.state,

        confidence=state.confidence,

        trigger=state.trigger,

    )


@app.post("/handoff")

def evaluate_handoff(req: HandoffRequest):

    payload = sensor_manager.read_all()

    features = preprocess_sensor_payload(payload)

    state = her_kernel.classify(features)

    parent = Actor(actor_id=req.parent_id, role="PARENT")

    caregiver = Actor(actor_id=req.caregiver_id, role="CAREGIVER")

    ctx = HandoffContext(parent=parent, caregiver=caregiver, child_id=req.child_id)

    approved, message = chamber_engine.evaluate_handoff(ctx, state)

    return {"approved": approved, "message": message, "state": state.state}
