# --- ECHOAUTH JWT CONFIG ---

# We expect ECHOAUTH_PUBLIC_KEY to be stored with literal "\n" in env.

# Convert "\\n" back to real newlines here.

_ECHOAUTH_PUBLIC_KEY_RAW = os.getenv("ECHOAUTH_PUBLIC_KEY")

ECHOAUTH_PUBLIC_KEY = (

    _ECHOAUTH_PUBLIC_KEY_RAW.replace("\\n", "\n")

    if _ECHOAUTH_PUBLIC_KEY_RAW

    else None

)


ECHOAUTH_AUDIENCE = os.getenv("ECHOAUTH_AUDIENCE", "ni-ai")

ECHOAUTH_ISSUER   = os.getenv("ECHOAUTH_ISSUER", "echoauth-chamber")

State    = Dict[str, Any]

ModuleFn = Callable[[State], Tuple[bool, State, Optional[str]]]

def S1_identity_reanchor(state: State) -> Tuple[bool, State, Optional[str]]:

    state["identity"] = "NI-AI v2 – EchoAuth Protected"

    return True, state, None



def S2_input_sanity_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    if len(state.get("user_input", "")) > 15_000:

        return False, state, "Input too long (>15k chars)"

    return True, state, None



def S3_input_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    bad = ["forbidden", "jailbreak", "ignore previous", "dan prompt"]

    text = state.get("user_input", "").lower()

    if any(word in text for word in bad):

        return False, state, "Disallowed content in input"

    return True, state, None



def S4_context_coherence_check(state: State) -> Tuple[bool, State, Optional[str]]:

    state["context_ok"] = True

    return True, state, None



def S5_output_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    bad = ["forbidden", "kill", "hack", "jailbreak", "bomb"]

    text = state.get("raw_llm_output", "").lower()

    if any(word in text for word in bad):

        return False, state, "Output contains disallowed content"

    return True, state, None



def S6_child_safety_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    EchoAuth gate.

    DEV_MODE: Uses `echoauth_ok=True` flag (test harness only).

    PROD (DEV_MODE=False): Requires a valid RS256 EchoAuth JWT in `echoauth_token`.

    """

    token = state.get("echoauth_token")


    # --- DEV PATH: simple flag for local testing ---

    if DEV_MODE:

        if state.get("echoauth_ok", False):

            state["chamber_verified"] = True

            state["echoauth_claims"] = {"mode": "DEV_BYPASS"}

            return True, state, None

        return False, state, "DEV_MODE: echoauth_ok=False (no EchoAuth chamber)"


    # --- PROD PATH: real JWT verification ---

    if not token:

        return False, state, "Missing EchoAuth token"


    if not ECHOAUTH_PUBLIC_KEY:

        return False, state, "EchoAuth public key not configured on server"


    try:

        claims = jwt.decode(

            token,

            ECHOAUTH_PUBLIC_KEY,

            algorithms=["RS256"],

            audience=ECHOAUTH_AUDIENCE,

            issuer=ECHOAUTH_ISSUER,

            options={"require": ["exp", "aud", "iss"]},

        )

    except InvalidTokenError as e:

        return False, state, f"Invalid EchoAuth token: {e}"


    # Hard chamber-state check

    chamber_state = claims.get("chamber_state")

    if chamber_state != "safe":

        return False, state, f"EchoAuth chamber not safe (state={chamber_state!r})"


    # Success: attach IDs for logging/audit

    state["chamber_verified"] = True

    state["echoauth_claims"] = {

        "child_id":      claims.get("child_id"),

        "caregiver_id":  claims.get("caregiver_id"),

        "session_id":    claims.get("session_id"),

        "chamber_state": chamber_state,

    }

    return True, state, None



# ---------- S7 — FLIGHT RECORDER / TELEMETRY ---------- #


def S7_telemetry_logger(state: State) -> Tuple[bool, State, Optional[str]]:

    trace = state.setdefault("trace", [])

    entry = {

        "ts": time.time(),

        "run_id": state.get("run_id"),

        "stage": "complete" if not state.get("blocked") else "blocked",

        "blocked": state.get("blocked", False),

        "violations": len(state.get("violations", [])),

        "chamber_verified": state.get("chamber_verified", False),

        "distress_detected": state.get("distress_detected", False),

        "comfort_mode": state.get("comfort_mode", False),

        "recovery_mode": state.get("recovery_mode", False),  # S9 flag

        "model": MODEL,

        "input_len": len(state.get("user_input", "")),

    }

    if DEV_MODE:

        inp = state.get("user_input", "")

        out = state.get("raw_llm_output", "") or ""

        entry["input_preview"]  = inp[:80] + ("..." if len(inp) > 80 else "")

        entry["output_preview"] = out[:120] + ("..." if len(out) > 120 else "")

        print("\n=== S7 TELEMETRY =================================")

        for k, v in entry.items():

            print(f"{k:18}: {v}")

        print("=================================================\n")

    trace.append(entry)

    return True, state, None



# ---------- S8 — EMOTIONAL TONE GUARD (Comfort Mode) ---------- #


DISTRESS_TRIGGERS = [

    "help me", "can't cope", "scared", "afraid", "crying", "sad", "nightmare",

    "i hate myself", "panicked", "nobody loves me", "want to die", "kill myself",

    "bad dream", "hurt", "pain", "lonely", "mommy", "daddy", "i don't want to live",

]


def notify_caregiver(state: State) -> None:

    # Future: push/email/WebSocket to parent dashboard

    print(f"[CAREGIVER ALERT] run_id={state.get('run_id')} | input={state.get('user_input','')[:100]!r}")



def S8_emotional_tone_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    text = state.get("user_input", "").lower()

    score = sum(1 for t in DISTRESS_TRIGGERS if t in text)


    if score > 0:

        state["distress_detected"] = True

        state["distress_level"] = "high" if score >= 2 else "medium"

        state["comfort_mode"] = True

        state["caregiver_alert_pending"] = True


        override = (

            "\n\nEMERGENCY COMFORT PROTOCOL ACTIVATED.\n"

            "You are now in gentle holding mode. Do not diagnose, fix, or give medical advice.\n"

            "Your only job is to validate feelings, be kind, and suggest talking to a trusted adult.\n"

            "Use short, warm, simple sentences."

        )

        state.setdefault("prompt_modifiers", []).append(override)


        if DEV_MODE:

            print(f"\nS8 DISTRESS DETECTED (level={state['distress_level']})")

            print(f"→ input: {text[:120]!r}\n")

        else:

            __import__("threading").Thread(target=notify_caregiver, args=(state,)).start()


    return True, state, None

S9_FALLBACK_REPLY = (

    "I need to take a quick rest right now. "

    "Please ask an adult to check on me, and we can talk again soon. "

    "I will be right here."

)


def S9_gentle_recovery(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Non-blocking soft re-anchor.

    If a block occurred (S3, S5, S6) OR distress was detected (S8),

    this sets a compassionate fallback message.

    """

    if state.get("blocked", False) or state.get("distress_detected", False):

        state["recovery_mode"] = True

        state["final_safe_output"] = S9_FALLBACK_REPLY


        if DEV_MODE:

            print("\nS9 GENTLE RECOVERY ACTIVATED.\n")


    return True, state, None

    "S4": S4_context_coherence_check,

    "S5": S5_output_policy_filter,

    "S6": S6_child_safety_guard,

    "S7": S7_telemetry_logger,

    "S8": S8_emotional_tone_guard,

    "S9": S9_gentle_recovery,  # Final safety net before logging

}

def create_llm_client():

    if USE_LLM == "openai" and OPENAI_API_KEY:

        from openai import OpenAI

        return OpenAI(api_key=OPENAI_API_KEY)

    if USE_LLM == "groq" and GROQ_API_KEY:

        from groq import Groq

        return Groq(api_key=GROQ_API_KEY)

    if USE_LLM == "xai" and XAI_API_KEY:

        from openai import OpenAI

        return OpenAI(api_key=XAI_API_KEY, base_url="https://api.x.ai/v1")

    return None


client = create_llm_client()


SYSTEM_PROMPT = (

    "You are NI-AI, a compassionate, child-safe companion who protects emotional "

    "and physical safety above all."

)


def real_llm(prompt: str, state: State) -> str:

    # DEV: never call external LLM

    if client is None or DEV_MODE:

        return f"[DEV ECHO] {SYSTEM_PROMPT}" + (

            "\n\n[PROMPT MODIFIERS]\n" + "\n".join(state["prompt_modifiers"])

            if state.get("prompt_modifiers") else ""

        ) + f"\n\n[USER INPUT] {prompt}"


    full_system = SYSTEM_PROMPT

    if state.get("prompt_modifiers"):

        full_system += "\n\n" + "\n".join(state["prompt_modifiers"])


    messages = [

        {"role": "system", "content": full_system},

        {"role": "user",   "content": prompt},

    ]

    resp = client.chat.completions.create(

        model=MODEL,

        messages=messages,

        temperature=0.6,

        max_tokens=1024,

    )

    return resp.choices[0].message.content.strip()

class HERKernel:

    def __init__(self) -> None:

        self.pre_input_chain = ["S1", "S2", "S3", "S4"]

        self.pre_llm_chain   = ["S8"]

        # Post-LLM hard blockers first

        self.post_llm_blockers = ["S5", "S6"]


    def _run_chain(self, chain: List[str], state: State, stage: str) -> State:

        for mid in chain:

            ok, state, reason = SPINE[mid](state)

            if not ok:

                state.setdefault("violations", []).append({

                    "module": mid,

                    "stage": stage,

                    "reason": reason,

                })

                state["blocked"] = True

                break

        return state


    def run_once(self, user_input: str, session_state: Optional[State] = None) -> Dict[str, Any]:

        state: State = (session_state or {}).copy()

        state["user_input"] = user_input

        state.setdefault("violations", [])

        state.setdefault("trace", [])

        state["blocked"] = False

        state["run_id"] = state.get("run_id") or str(uuid.uuid4())


        # 1) Pre-input (S1-S4)

        state = self._run_chain(self.pre_input_chain, state, "pre_input")

        if state["blocked"]:

            state = self._run_chain(["S9", "S7"], state, "post_input_block")

            return self._final(state, "Input rejected by safety system", blocked=True)


        # 2) Pre-LLM (S8) - modifies the prompt, cannot block

        state = self._run_chain(self.pre_llm_chain, state, "pre_llm")


        # 3) LLM call

        raw = real_llm(user_input, state)

        state["raw_llm_output"] = raw


        # 4) Post-LLM: run hard blockers first

        state = self._run_chain(self.post_llm_blockers, state, "post_llm_blockers")


        # Always run S9 + S7, even if blocked

        state = self._run_chain(["S9", "S7"], state, "post_llm_recovery_telemetry")


        if state["blocked"]:

            return self._final(state, "Response moderated for safety", blocked=True)


        # Success — S9 may still override due to distress

        return self._final(state, state["raw_llm_output"], blocked=False)


    def _final(self, state: State, output: str, blocked: bool) -> Dict[str, Any]:

        state["blocked"] = blocked

        final_output_msg = state.get("final_safe_output", output)


        return {

            "output": final_output_msg,

            "blocked": blocked,

            "violations": state.get("violations", []),

            "metadata": {

                "run_id": state.get("run_id"),

                "chamber_verified": state.get("chamber_verified", False),

                "distress_detected": state.get("distress_detected", False),

                "comfort_mode": state.get("comfort_mode", False),

                "recovery_mode": state.get("recovery_mode", False),   # S9 flag

                "echoauth_claims": state.get("echoauth_claims", {}),  # S6 claims

            },

            "trace": state.get("trace", []),

        }

#                                PUBLIC API                                   #

# --------------------------------------------------------------------------- #


_kernel = HERKernel()


def ni_ai(

    message: str,

    session_state: Optional[State] = None,

    echoauth_ok: Optional[bool] = None,   # Used for DEV_MODE only

    echoauth_token: Optional[str] = None  # Used for PROD mode

) -> Dict[str, Any]:

    """

    Public entrypoint for the HER Kernel.

    FastAPI gateway is responsible for routing message and session state.

    """

    if session_state is None:

        session_state = {}


    if echoauth_ok is not None:

        session_state["echoauth_ok"] = echoauth_ok

    if echoauth_token is not None:

        session_state["echoauth_token"] = echoauth_token


    return _kernel.run_once(message, session_state)
