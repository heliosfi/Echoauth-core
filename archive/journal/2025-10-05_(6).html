# app.py

# Flask app with EchoAuth, AAC (TTS), music (tones + binaural beats), and records

from flask import Flask, request, jsonify, render_template

import numpy as np

import sqlite3

import time

import base64

import os

import threading

from cryptography.fernet import Fernet

from echoauth_chamber import verify_user_once  # From echoauth_chamber.py

import pyttsx3  # For text-to-speech

import simpleaudio as sa  # For music playback

import torch

import logging


app = Flask(__name__)


# Setup logging

logging.basicConfig(filename='echoauth.log', level=logging.WARNING)


# Encryption key (generate once and store securely)

key = Fernet.generate_key()

cipher = Fernet(key)


# SQLite setup for records

conn = sqlite3.connect('records.db', check_same_thread=False)

conn.execute('CREATE TABLE IF NOT EXISTS records (user_id INTEGER, note BLOB, date TEXT)')

conn.commit()


# Initialize text-to-speech

tts_engine = pyttsx3.init()

voices = tts_engine.getProperty('voices')

for voice in voices:

    if 'female' in voice.name.lower():  # Softer voice for sensory comfort

        tts_engine.setProperty('voice', voice.id)

        break


# Mock AAC symbols

AAC_SYMBOLS = {

    "hello": "Say hello",

    "help": "Need help",

    "happy": "I am happy",

    "food": "I want food",

    "calm": "I need calm"

}


# Music tones and binaural beats

MUSIC_TONES = {

    "relax_528": {"frequency": 528.0, "mode": "sine"},      # Relaxation tone

    "focus_432": {"frequency": 432.0, "mode": "sine"},      # Focus tone

    "ground_396": {"frequency": 396.0, "mode": "sine"},     # Grounding tone

    "binaural_10": {"frequency": 440.0, "mode": "binaural", "beat_hz": 10.0}  # 10Hz alpha beat

}


# Global player state

_play_lock = threading.Lock()

_play_obj = None


def analyze_sound(audio: np.ndarray) -> float:

    """Analyze audio loudness for sensory adaptation."""

    audio_t = torch.from_numpy(audio).abs()

    return float(audio_t.mean())


def _make_tone(freq_hz: float, duration_sec: float = 5.0, sample_rate: int = 44100, volume: float = 0.5) -> np.ndarray:

    """Mono sine wave, int16 for simpleaudio."""

    t = np.linspace(0, duration_sec, int(sample_rate * duration_sec), endpoint=False)

    wave = np.sin(2 * np.pi * freq_hz * t)

    wave = (volume * wave).clip(-1.0, 1.0)

    return (wave * 32767).astype(np.int16)


def _make_binaural(carrier_hz: float, beat_hz: float = 10.0, duration_sec: float = 5.0, sample_rate: int = 44100, volume: float = 0.5) -> np.ndarray:

    """Return stereo int16: left=carrier, right=carrier+beat."""

    n = int(sample_rate * duration_sec)

    t = np.arange(n) / sample_rate

    left = np.sin(2 * np.pi * carrier_hz * t)

    right = np.sin(2 * np.pi * (carrier_hz + beat_hz) * t)

    left = (volume * left).clip(-1.0, 1.0)

    right = (volume * right).clip(-1.0, 1.0)

    stereo = np.stack([left, right], axis=1)  # (N,2), float

    return (stereo * 32767).astype(np.int16)


def _stop_audio_locked():

    global _play_obj

    if _play_obj is not None:

        try:

            _play_obj.stop()

        except Exception:

            pass

        _play_obj = None


@app.route('/')

def home():

    return render_template('index.html', symbols=AAC_SYMBOLS, music=MUSIC_TONES)


@app.route('/auth', methods=['POST'])

def auth():

    try:

        if os.environ.get('ECHOAUTH_MOCK') == '1':

            return jsonify({'status': 'authenticated', 'score': 0.85, 'user_id': 1})

        frame_data = base64.b64decode(request.form['frame'])

        audio_data = base64.b64decode(request.form['audio'])

        frame = np.frombuffer(frame_data, dtype=np.uint8).reshape((480, 640, 3))

        audio = np.frombuffer(audio_data, dtype=np.float32)

        result = verify_user_once(frame, audio)

        if result['decision'] == 'accept':

            return jsonify({'status': 'authenticated', 'score': result['score'], 'user_id': 1})

        return jsonify({'status': 'denied', 'score': result['score']}), 401

    except Exception as e:

        logging.error(f"Auth failed: {e}")

        return jsonify({'status': 'error', 'message': str(e)}), 400


@app.route('/add_record', methods=['POST'])

def add_record():

    try:

        data = request.json

        user_id = data['user_id']

        note = data['note']

        date = time.strftime('%Y-%m-%d %H:%M:%S')

        encrypted_note = cipher.encrypt(note.encode())

        conn.execute('INSERT INTO records VALUES (?, ?, ?)', (user_id, encrypted_note, date))

        conn.commit()

        return jsonify({'status': 'ok'})

    except Exception as e:

        logging.error(f"Add record failed: {e}")

        return jsonify({'status': 'error', 'message': str(e)}), 400


@app.route('/get_records', methods=['GET'])

def get_records():

    try:

        user_id = request.args.get('user_id')

        cur = conn.execute('SELECT note, date FROM records WHERE user_id = ?', (user_id,))

        rows = cur.fetchall()

        decrypted = [{'note': cipher.decrypt(note).decode(), 'date': date} for note, date in rows]

        return jsonify(decrypted)

    except Exception as e:

        logging.error(f"Get records failed: {e}")

        return jsonify({'status': 'error', 'message': str(e)}), 400


@app.route('/communicate', methods=['POST'])

def communicate():

    try:

        data = request.json

        symbol = data['symbol']

        rate = data.get('rate', 150)

        volume = data.get('volume', 0.8)

        audio_data = data.get('audio', '')

        if audio_data:

            audio = np.frombuffer(base64.b64decode(audio_data), dtype=np.float32)

            loudness = analyze_sound(audio)

            volume = max(0.3, volume - loudness * 0.5)

        tts_engine.setProperty('rate', rate)

        tts_engine.setProperty('volume', volume)

        message = AAC_SYMBOLS.get(symbol, "Unknown")

        tts_engine.say(message)

        tts_engine.runAndWait()

        return jsonify({'status': 'ok', 'spoken': message})

    except Exception as e:

        logging.error(f"Communication failed: {e}")

        return jsonify({'status': 'error', 'message': str(e)}), 400


@app.route('/music', methods=['POST'])

def music():

    """

    JSON: {"action": "start" | "stop", "tone": str, "duration": float, "volume": float}

    """

    try:

        data = request.get_json(force=True)

        action = data.get('action', 'start')

        volume = float(data.get('volume', 0.5))

        volume = max(0.1, min(1.0, volume))

        audio_data = data.get('audio', '')  # For sensory adaptation

        if audio_data:

            audio = np.frombuffer(base64.b64decode(audio_data), dtype=np.float32)

            loudness = analyze_sound(audio)

            volume = max(0.3, volume - loudness * 0.5)


        with _play_lock:

            if action == 'stop':

                _stop_audio_locked()

                return jsonify({'status': 'stopped'})


            tone_key = data.get('tone', 'relax_528')

            cfg = MUSIC_TONES.get(tone_key, MUSIC_TONES['relax_528'])

            duration = float(data.get('duration', 5.0))

            duration = max(1.0, min(60.0, duration))


            _stop_audio_locked()


            mode = cfg.get('mode', 'sine')

            if mode == 'binaural':

                carrier = float(cfg.get('frequency', 440.0))

                beat = float(cfg.get('beat_hz', 10.0))

                stereo_i16 = _make_binaural(carrier, beat, duration, 44100, volume)

                global _play_obj

                _play_obj = sa.play_buffer(stereo_i16, num_channels=2, bytes_per_sample=2, sample_rate=44100)

            else:

                freq = float(cfg.get('frequency', 528.0))

                mono_i16 = _make_tone(freq, duration, 44100, volume)

                _play_obj = sa.play_buffer(mono_i16, num_channels=1, bytes_per_sample=2, sample_rate=44100)


            try:

                user_id = int(data.get('user_id', 1))

                desc = f"Played tone '{tone_key}' vol={volume:.2f} dur={duration:.1f}s"

                ts = time.strftime('%Y-%m-%d %H:%M:%S')

                enc = cipher.encrypt(desc.encode('utf-8'))

                conn.execute('INSERT INTO records VALUES (?, ?, ?)', (user_id, enc, ts))

                conn.commit()

            except Exception as e:

                app.logger.warning(f'Music log insert skipped: {e}')


            return jsonify({'status': 'playing', 'tone': tone_key, 'volume': volume, 'duration': duration})

    except Exception as e:

        app.logger.exception('Music endpoint error')

        return jsonify({'status': 'error', 'message': str(e)}), 400


if __name__ == '__main__':

    app.run(debug=True, host='0.0.0.0', port=5000)
