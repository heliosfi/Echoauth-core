1️⃣ Extend the DB models for review flags


In db/models.py add fields to CheckIn:


from sqlalchemy import Boolean  # add this import at the top


class CheckIn(Base):

    __tablename__ = "checkins"


    id = Column(Integer, primary_key=True, index=True)

    caregiver_id = Column(Integer, ForeignKey("caregivers.id"), nullable=False)

    child_id = Column(Integer, ForeignKey("children.id"), nullable=False)

    observed_cues = Column(Text, nullable=False)

    result_label = Column(String, nullable=False)  # "Safe" / "Needs Attention"

    confidence = Column(Float, nullable=False)

    status = Column(String, nullable=False)        # "success" / "error"


    # 🔥 NEW:

    needs_review = Column(Boolean, default=False)  # automatic flag

    review_status = Column(String, default="none") # "none" / "pending" / "reviewed"

    reviewer_notes = Column(Text, nullable=True)


    created_at = Column(DateTime, default=datetime.utcnow)


    child = relationship("Child", back_populates="checkins")


For a fresh MVP DB you can just delete echoauth.db and let SQLAlchemy recreate with the new columns.

3️⃣ Endpoint: list flagged check-ins (for caregiver / clinician)


Add this to api/main.py:


from typing import List


@app.get("/checkins/flagged", response_model=List[dict])

def list_flagged_checkins(caregiver_id: int, db: Session = Depends(get_db)):

    """

    Returns all check-ins for this caregiver that need manual review.

    No internal NI-AI logs are exposed here.

    """

    q = db.query(models.CheckIn).filter(

        models.CheckIn.caregiver_id == caregiver_id,

        models.CheckIn.needs_review == True,

        models.CheckIn.review_status == "pending"

    ).order_by(models.CheckIn.created_at.desc())


    results = []

    for c in q.all():

        results.append({

            "checkin_id": c.id,

            "child_id": c.child_id,

            "result_label": c.result_label,

            "confidence": c.confidence,

            "created_at": c.created_at.isoformat(),

            "review_status": c.review_status

        })

    return results


This is the list your caregiver / therapist UI can show like:


“These are situations where EchoAuth wants your eyes on it.”

4️⃣ Endpoint: manual review + notes


class ReviewUpdate(BaseModel):

    reviewer_notes: str

    mark_resolved: bool = True


@app.post("/checkins/{checkin_id}/review")

def review_checkin(checkin_id: int, payload: ReviewUpdate, db: Session = Depends(get_db)):

    checkin = db.query(models.CheckIn).filter(

        models.CheckIn.id == checkin_id

    ).first()


    if not checkin:

        raise HTTPException(status_code=404, detail="Check-in not found")


    checkin.reviewer_notes = payload.reviewer_notes


    if payload.mark_resolved:

        checkin.needs_review = False

        checkin.review_status = "reviewed"

    else:

        checkin.review_status = "pending"


    db.commit()

    db.refresh(checkin)


    return {

        "checkin_id": checkin.id,

        "needs_review": checkin.needs_review,

        "review_status": checkin.review_status

    }


This is where a caregiver, therapist, or you can say:

• “I checked this event, here is what was actually happening.”

• “This was a meltdown from noise.”

• “This was a dangerous situation, we intervened.”


The notes stay inside the system, not on the child UI.

async function loadFlagged() {

  try {

    const cgId = document.getElementById('flagCaregiverId').value;

    const data = await fetch(

      document.getElementById('backendUrl').value +

      '/checkins/flagged?caregiver_id=' + encodeURIComponent(cgId)

    ).then(r => r.json());

    showStatus('Flagged check-ins:\n' + JSON.stringify(data, null, 2));

  } catch (e) {

    showStatus('Error: ' + e.message);

  }

}
