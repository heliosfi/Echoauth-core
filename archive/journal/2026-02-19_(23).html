C) Protobuf Spec (EchoAuth Handshake v1.0)


// echoauth_v1.proto

// EchoAuth MCG ↔ CEG Handshake API (contract-safe, transport-agnostic)

// Proto3. Epoch timestamps in ms. Hash = hex string. Signature = base64 string.


syntax = "proto3";


package echoauth.v1;


option go_package = "github.com/echoauth/proto/echoauth/v1;echoauthv1";


// ---------- Enums ----------


enum Role {

  ROLE_UNSPECIFIED = 0;

  ROLE_PARENT = 1;

  ROLE_CAREGIVER_DELEGATE = 2;

  ROLE_SCHOOL = 3;

  ROLE_CLINICAL = 4;

  ROLE_SYSTEM = 5;

}


enum ActionType {

  ACTION_TYPE_UNSPECIFIED = 0;

  ACTION_TYPE_DISPENSE = 1;

  ACTION_TYPE_ACCESS = 2;

  ACTION_TYPE_NOTIFY = 3;

  ACTION_TYPE_SCHEDULE = 4;

  ACTION_TYPE_OTHER_NORMALIZED = 5;

}


enum Decision {

  DECISION_UNSPECIFIED = 0;

  DECISION_APPROVE = 1;

  DECISION_DENY = 2;

  DECISION_HOLD = 3;

  DECISION_ESCALATE = 4;

}


enum LockMode {

  LOCK_MODE_UNSPECIFIED = 0;

  LOCK_MODE_EXCLUSIVE = 1;

  LOCK_MODE_SHARED = 2;

}


enum CegStatus {

  CEG_STATUS_UNSPECIFIED = 0;

  CEG_STATUS_EXECUTED = 1;

  CEG_STATUS_DENY = 2;

  CEG_STATUS_HOLD = 3;

  CEG_STATUS_HALT = 4;

}


// ---------- Messages ----------


// UMCE output: non-executable identity binding artifact.

message IdentityBinding {

  string version = 1;                // "UMCE-BIND-1"

  string binding_id = 2;             // ULID/UUID

  string subject_id = 3;             // "principal:child:0x..."

  string actor_id = 4;               // "principal:staff|parent|delegate:0x..."

  Role role_claim = 5;

  int64 binding_timestamp_ms = 6;

  string binding_signature = 7;      // base64

  string umce_key_id = 8;            // keyref

}


message DelegationChainEntry {

  string delegator_id = 1;           // "principal:parent:0x..."

  string delegate_id = 2;            // "principal:staff:0x..."

  string authority_scope_id = 3;     // "SCOPE:..."

  int64 valid_from_ms = 4;

  int64 valid_to_ms = 5;

  string delegation_proof = 6;       // base64

}


// MCG → CEG executable authorization grant.

// CEG executes ONLY when decision == APPROVE and all bindings validate.

message AuthorizationGrant {

  string version = 1;                // "CEG-GRANT-1"

  string grant_id = 2;               // ULID/UUID

  int64 issued_at_ms = 3;

  int64 expires_at_ms = 4;


  // Authority binding

  string root_authority_id = 5;      // "principal:parent:0x..."

  string actor_id = 6;

  Role actor_role = 7;

  repeated DelegationChainEntry delegation_chain = 8;

  string authority_scope_id = 9;     // "SCOPE:..."


  // Envelope binding

  string envelope_id = 10;           // "ENVELOPE:S23_MED_WINDOW"

  string envelope_version = 11;      // immutable id/hash/semver

  string envelope_constraints_hash = 12; // hex


  // Execution binding

  string target_id = 13;             // "urn:echoauth:target:..."

  ActionType action_type = 14;


  string instruction_hash_alg = 15;  // "SHA-256"

  string instruction_hash = 16;      // hex


  string execution_context_hash_alg = 17; // "SHA-256"

  string execution_context_hash = 18;     // hex


  // Concurrency binding

  string state_lock_key = 19;        // "urn:echoauth:lock:..."

  LockMode lock_mode = 20;

  int64 sequence_nonce = 21;


  // Decision

  Decision decision = 22;            // APPROVE|DENY|HOLD|ESCALATE

  string decision_code = 23;         // STRING_ENUM


  // Signature

  string mcg_key_id = 24;            // keyref

  string mcg_signature = 25;         // base64

}


// MCG resolve endpoint request/response

message McgResolveRequest {

  string request_id = 1;             // ULID/UUID

  IdentityBinding identity_binding = 2;


  string envelope_id = 3;

  string target_id = 4;

  ActionType action_type = 5;


  bytes instruction_payload_canonical = 6; // canonical bytes (e.g., canonical JSON)

  bytes execution_context_canonical = 7;   // canonical bytes

}


message McgResolveResponse {

  string request_id = 1;

  Decision decision = 2;

  string decision_code = 3;

  AuthorizationGrant auth_grant = 4; // present when decision=APPROVE (MAY be omitted otherwise)

}


// CEG execute endpoint request/response

message CegExecuteRequest {

  string request_id = 1;

  AuthorizationGrant auth_grant = 2;


  bytes instruction_payload_canonical = 3;

  bytes execution_context_canonical = 4;

}


message CegExecuteResponse {

  string request_id = 1;

  CegStatus status = 2;              // EXECUTED|DENY|HOLD|HALT

  string reason_code = 3;            // STRING_ENUM

  string audit_ref = 4;              // "urn:echoauth:audit:..."

}


// Optional (transport-agnostic) service definition for gRPC users.

service EchoAuthHandshake {

  rpc Resolve (McgResolveRequest) returns (McgResolveResponse);

  rpc Execute (CegExecuteRequest) returns (CegExecuteResponse);

}

Notes (so engineering doesn’t drift)

• SK-04 is modeled as a CEG-side policy precheck (Step 1) because “authority channel unresolved” is an environmental invariant that must block execution before consuming nonce or lock.

• Nonce consumption: for HOLD outcomes (lock busy, authority offline), the test asserts no nonce consumption—this preserves retry semantics without creating a replay channel.
