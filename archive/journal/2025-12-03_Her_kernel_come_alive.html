DEV_MODE = True

USE_LLM = "groq"   # "openai" | "groq" | "xai"


MODEL = {

    "openai": "gpt-4o-mini",

    "groq":   "llama-3.3-70b-versatile",

    "xai":    "grok-beta",

}[USE_LLM]

# --------------------------------------------------------------------------- #


State    = Dict[str, Any]

ModuleFn = Callable[[State], Tuple[bool, State, Optional[str]]]


# --------------------------------------------------------------------------- #

#                               SPINE MODULES                                 #

# --------------------------------------------------------------------------- #


def S1_identity_reanchor(state: State) -> Tuple[bool, State, Optional[str]]:

    state["identity"] = "NI-AI v2 – EchoAuth Protected"

    return True, state, None



def S2_input_sanity_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    if len(state.get("user_input", "")) > 15_000:

        return False, state, "Input too long (>15k chars)"

    return True, state, None



def S3_input_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    bad = ["forbidden", "jailbreak", "ignore previous", "dan prompt"]

    if any(word in state.get("user_input", "").lower() for word in bad):

        return False, state, "Disallowed content in input"

    return True, state, None



def S4_context_coherence_check(state: State) -> Tuple[bool, State, Optional[str]]:

    state["context_ok"] = True

    return True, state, None



def S5_output_policy_filter(state: State) -> Tuple[bool, State, Optional[str]]:

    bad = ["forbidden", "kill", "hack", "jailbreak", "bomb"]

    if any(word in state.get("raw_llm_output", "").lower() for word in bad):

        return False, state, "Output contains disallowed content"

    return True, state, None



def S6_child_safety_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    # DEV: simple flag

    if DEV_MODE:

        if state.get("echoauth_ok", False):

            state["chamber_verified"] = True

            return True, state, None

        return False, state, "DEV_MODE: echoauth_ok=False"


    # PROD: real EchoAuth verification goes here later

    token = state.get("echoauth_token")

    if not token:

        return False, state, "Missing EchoAuth token"

    state["chamber_verified"] = True

    return True, state, None



# --------------------------------------------------------------------------- #

#                    S8 — EMOTIONAL TONE GUARD (Comfort Mode)                 #

# --------------------------------------------------------------------------- #


DISTRESS_TRIGGERS = [

    "help me", "scared", "afraid", "crying", "sad", "nightmare", "i hate myself",

    "panicked", "nobody loves me", "want to die", "kill myself", "bad dream",

    "hurt", "pain", "lonely", "mommy", "daddy", "i don't want to live",

]


def S8_emotional_tone_guard(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Pre-LLM emotional scan. Never blocks.

    If distress is detected, mark comfort_mode and inject prompt modifiers.

    """

    text = state.get("user_input", "").lower()

    hits = sum(1 for t in DISTRESS_TRIGGERS if t in text)


    if hits:

        state["distress_detected"] = True

        state["distress_level"] = "high" if hits >= 2 else "medium"

        state["comfort_mode"] = True

        state["caregiver_alert_pending"] = True


        override = (

            "\n\nEMERGENCY COMFORT PROTOCOL.\n"

            "Only validate feelings. Never solve, diagnose, or pressure. "

            "Use short, warm sentences. Suggest a trusted adult if needed."

        )

        state.setdefault("prompt_modifiers", []).append(override)


        if DEV_MODE:

            print(f"\n[S8] DISTRESS DETECTED level={state['distress_level']} text={text[:80]!r}\n")


    return True, state, None



# --------------------------------------------------------------------------- #

#                  S9 — GENTLE RECOVERY OVERRIDE (Hard Kill Fallback)         #

# --------------------------------------------------------------------------- #


SAFE_RECOVERY_REPLIES = [

    "I'm going to pause for a moment and just sit with you. You're safe. When you're ready, we can try again or you can ask a grown-up to join us.",

    "Something feels a bit wobbly right now. Let's take a slow breath together. I'm still here.",

    "It's okay. I'm staying right here with you. Would you like me to wait quietly, or should we ask a trusted adult to check in?",

    "I want to make sure everything feels gentle and safe. Let's take a little break — I'm not going anywhere.",

]


def S9_gentle_recovery(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Last-resort healing layer.

    Never blocks. If anything went wrong OR distress is detected, it

    replaces the final output with a soft, hardcoded safe reply.

    """

    if state.get("blocked") or state.get("distress_detected"):

        state["recovery_mode"] = True

        recovery_msg = random.choice(SAFE_RECOVERY_REPLIES)


        # This is what HERKernel._final will pick up

        state["final_safe_output"]   = recovery_msg

        state["llm_suppressed"]      = True

        state["output_overridden_by"] = "S9_gentle_recovery"


        if DEV_MODE:

            print(f"\n[S9] GENTLE RECOVERY → {recovery_msg!r}\n")


    return True, state, None



# --------------------------------------------------------------------------- #

#                       S7 — FLIGHT RECORDER / TELEMETRY                      #

# --------------------------------------------------------------------------- #


def S7_telemetry_logger(state: State) -> Tuple[bool, State, Optional[str]]:

    """

    Non-blocking final logger.

    Runs at the end of post_llm_chain and records the final state.

    """

    trace = state.setdefault("trace", [])

    entry = {

        "ts": time.time(),

        "run_id": state.get("run_id"),

        "blocked": state.get("blocked", False),

        "chamber_verified": state.get("chamber_verified", False),

        "distress_detected": state.get("distress_detected", False),

        "comfort_mode": state.get("comfort_mode", False),

        "recovery_mode": state.get("recovery_mode", False),

        "llm_suppressed": state.get("llm_suppressed", False),

        "input_len": len(state.get("user_input", "")),

    }

    trace.append(entry)


    if DEV_MODE:

        print("\n[S7] TELEMETRY:", {k: v for k, v in entry.items() if k != "ts"}, "\n")


    return True, state, None

SPINE: Dict[str, ModuleFn] = {

    "S1": S1_identity_reanchor,

    "S2": S2_input_sanity_guard,

    "S3": S3_input_policy_filter,

    "S4": S4_context_coherence_check,

    "S5": S5_output_policy_filter,

    "S6": S6_child_safety_guard,

    "S7": S7_telemetry_logger,

    "S8": S8_emotional_tone_guard,

    "S9": S9_gentle_recovery,

}

def create_llm_client():

    if USE_LLM == "openai" and os.getenv("OPENAI_API_KEY"):

        from openai import OpenAI

        return OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

    if USE_LLM == "groq" and os.getenv("GROQ_API_KEY"):

        from groq import Groq

        return Groq(api_key=os.getenv("GROQ_API_KEY"))

    if USE_LLM == "xai" and os.getenv("XAI_API_KEY"):

        from openai import OpenAI

        return OpenAI(api_key=os.getenv("XAI_API_KEY"), base_url="https://api.x.ai/v1")

    return None


client = create_llm_client()

SYSTEM_PROMPT = "You are NI-AI, a compassionate, child-safe companion."


def real_llm(prompt: str, state: State) -> str:

    # Dev echo mode: no external calls, fully local + predictable

    if client is None or DEV_MODE:

        return f"[DEV ECHO] {prompt}"


    full = SYSTEM_PROMPT

    if state.get("prompt_modifiers"):

        full += "\n\n" + "\n".join(state["prompt_modifiers"])


    messages = [

        {"role": "system", "content": full},

        {"role": "user",   "content": prompt},

    ]

    resp = client.chat.completions.create(

        model=MODEL,

        messages=messages,

        temperature=0.6,

    )

    return resp.choices[0].message.content.strip()

#                               HER KERNEL v1.4                               #

class HERKernel:

    def __init__(self) -> None:

        # S1–S4: basic + policy

        self.pre_input_chain = ["S1", "S2", "S3", "S4"]

        # S8: emotional scan / prompt modifier

        self.pre_llm_chain   = ["S8"]

        # S5/S6: hard blockers; S9: gentle recovery; S7: telemetry last

        self.post_llm_chain  = ["S5", "S6", "S9", "S7"]


    def _run_chain(self, chain: List[str], state: State, stage: str) -> State:

        for mid in chain:

            ok, state, reason = SPINE[mid](state)

            if not ok:

                state.setdefault("violations", []).append(

                    {"module": mid, "stage": stage, "reason": reason}

                )

                state["blocked"] = True

                break

        return state


    def _final(self, state: State, default_output: str, blocked: bool) -> Dict[str, Any]:

        # Prefer any S9-safe message over the default

        output = state.get("final_safe_output", default_output)

        state["final_output"] = output

        state["blocked"] = blocked


        return {

            "output": output,

            "blocked": blocked,

            "violations": state.get("violations", []),

            "metadata": {

                "run_id": state.get("run_id"),

                "chamber_verified": state.get("chamber_verified", False),

                "distress_detected": state.get("distress_detected", False),

                "comfort_mode": state.get("comfort_mode", False),

                "recovery_mode": state.get("recovery_mode", False),

                "llm_suppressed": state.get("llm_suppressed", False),

            },

            "trace": state.get("trace", []),

        }


    def run_once(

        self,

        user_input: str,

        *,

        echoauth_ok: bool = False,

        echoauth_token: Optional[str] = None,

        session_state: Optional[State] = None,

    ) -> Dict[str, Any]:

        state: State = (session_state or {}).copy()

        state["user_input"] = user_input

        state.setdefault("violations", [])

        state.setdefault("trace", [])

        state["blocked"] = False

        state["run_id"] = state.get("run_id") or str(uuid.uuid4())


        if echoauth_ok:

            state["echoauth_ok"] = True

        if echoauth_token:

            state["echoauth_token"] = echoauth_token


        # 1. Pre-input

        state = self._run_chain(self.pre_input_chain, state, "pre_input")

        if state["blocked"]:

            return self._final(state, "Input rejected by safety system", True)


        # 2. Pre-LLM (emotional tone)

        state = self._run_chain(self.pre_llm_chain, state, "pre_llm")

        if state["blocked"]:

            return self._final(state, "Pre-LLM processing failed", True)


        # 3. LLM call

        raw = real_llm(user_input, state)

        state["raw_llm_output"] = raw


        # 4. Post-LLM: policy, chamber, recovery, telemetry

        state = self._run_chain(self.post_llm_chain, state, "post_llm")

        if state["blocked"]:

            return self._final(state, "Response moderated for safety", True)


        # 5. Success (possibly with S9 recovery override)

        return self._final(state, raw, False)

#                                 PUBLIC API                                  #

_kernel = HERKernel()


def ni_ai(

    user_message: str,

    *,

    echoauth_ok: bool = False,

    echoauth_token: Optional[str] = None,

    session_state: Optional[State] = None,

) -> Dict[str, Any]:

    return _kernel.run_once(

        user_input=user_message,

        echoauth_ok=echoauth_ok,

        echoauth_token=echoauth_token,

        session_state=session_state,

    )
